44  	@GET
45  	@Path("/describe/modeshape")
46  	public Response describeModeshape() throws JsonGenerationException,
47  			JsonMappingException, IOException, RepositoryException {
48  		final Session session = repo.login();
49  		logger.debug("Repository name: "
50  				+ repo.getDescriptor(Repository.REP_NAME_DESC));
51  		final Builder<String, Object> repoproperties = builder();
52  		for (final String key : repo.getDescriptorKeys()) {
53  			if (repo.getDescriptor(key) != null)
54  				repoproperties.put(key, repo.getDescriptor(key));
55  		}
56  
57  		// add in node namespaces
58  		final NamespaceRegistry reg = session.getWorkspace()
59  				.getNamespaceRegistry();
60  		final Builder<String, String> namespaces = builder();
61  		for (final String prefix : reg.getPrefixes()) {
62  			namespaces.put(prefix, reg.getURI(prefix));
63  		}
64  		repoproperties.put("node.namespaces", namespaces.build());
65  
66  		// add in node types
67  		final NodeTypeManager ntmanager = (NodeTypeManager) session
68  				.getWorkspace().getNodeTypeManager();
69  		final Builder<String, String> nodetypes = builder();
70  		NodeTypeIterator i = ntmanager.getAllNodeTypes();
71  		while (i.hasNext()) {
72  			NodeType nt = i.nextNodeType();
73  			nodetypes.put(nt.getName(), nt.toString());
74  		}
75  		repoproperties.put("node.types", nodetypes.build());
76  		Map<String, Object> props = repoproperties.build();
77  		session.logout();
78  		return ok(mapper.writerWithType(Map.class).writeValueAsString(props))
79  				.build();
80  	}
