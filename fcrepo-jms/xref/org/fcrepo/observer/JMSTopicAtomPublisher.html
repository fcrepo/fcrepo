65  	@Subscribe
66  	public void publishJCREvent(Event jcrEvent) throws JMSException,
67  			RepositoryException, IOException {
68  
69  		Entry entry = abdera.newEntry();
70  
71  		entry.setTitle(operationsMappings.getFedoraMethodType(jcrEvent), TEXT)
72  				.setBaseUri("http://localhost:8080/rest");
73  
74  		// assume that the PID is the last section of the node path
75  		String path = jcrEvent.getPath();
76  		String pid = path.substring(path.lastIndexOf('/') + 1, path.length());
77  		entry.addCategory("xsd:string", pid, "fedora-types:pid");
78  
79  		StringWriter writer = new StringWriter();
80  		entry.writeTo(writer);
81  		String atomMessage = writer.toString();
82  		producer.send(jmsSession.createTextMessage(atomMessage));
83  
84  		logger.debug("Put event: \n" + atomMessage + "\n onto JMS.");
85  	}

113 	// maps JCR mutations to Fedora Classic API method types
114 	final private class OperationsMappings {

116 		// this actor will never mutate the state of the repo,
117 		// so we keep the session live for efficiency
118 		private javax.jcr.Session session;
119 
120 		public String getFedoraMethodType(Event jcrEvent)
121 				throws PathNotFoundException, RepositoryException {
122 
123 			// we need to know if this is an object or a datastream
124 			Set<NodeType> nodeTypes = Sets.newHashSet(session.getNode(
125 					jcrEvent.getPath()).getMixinNodeTypes());
126 
127 			// Now we can select from the combination of JCR Event type
128 			// and resource type to determine a Fedora Classic API method
129 			Integer eventType = jcrEvent.getType();
130 
131 			if (any(nodeTypes, isObjectNodeType)) {
132 				switch (eventType) {
133 				case NODE_ADDED:
134 					return "ingest";
135 				case NODE_REMOVED:
136 					return "purgeObject";
137 				case PROPERTY_ADDED:
138 					return "modifyObject";
139 				case PROPERTY_CHANGED:
140 					return "modifyObject";
141 				case PROPERTY_REMOVED:
142 					return "modifyObject";
143 				}
144 			}
145 			if (any(nodeTypes, isDatastreamNodeType)) {
146 				switch (eventType) {
147 				case NODE_ADDED:
148 					return "addDatastream";
149 				case NODE_REMOVED:
150 					return "purgeDatastream";
151 				case PROPERTY_ADDED:
152 					return "modifyDatastream";
153 				case PROPERTY_CHANGED:
154 					return "modifyDatastream";
155 				case PROPERTY_REMOVED:
156 					return "modifyDatastream";
157 				}
158 			}
159 			return null;
160 		}
161 
162 		private Predicate<NodeType> isObjectNodeType = new Predicate<NodeType>() {
163 			@Override
164 			public boolean apply(NodeType type) {
165 				return type.getName().equals("fedora:object");
166 			}
167 		};
168 
169 		private Predicate<NodeType> isDatastreamNodeType = new Predicate<NodeType>() {
170 			@Override
171 			public boolean apply(NodeType type) {
172 				return type.getName().equals("fedora:datastream");
173 			}
174 		};
