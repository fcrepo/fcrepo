494     private void validateHTML(final String path) throws Exception {
495         final HttpGet getMethod = new HttpGet(serverAddress + path);
496         getMethod.addHeader("Accept", "text/html");
497         final HttpResponse response = client.execute(getMethod);
498         assertEquals(OK.getStatusCode(), response.getStatusLine()
499                 .getStatusCode());
500         final String content = EntityUtils.toString(response.getEntity());
501         logger.debug("Retrieved HTML view:\n" + content);
502 
503         final HtmlParser htmlParser = new HtmlParser(ALLOW);
504         htmlParser.setDoctypeExpectation(NO_DOCTYPE_ERRORS);
505         htmlParser.setErrorHandler(new HTMLErrorHandler());
506         htmlParser.setContentHandler(new TreeBuilder());
507         try (final InputStream htmlStream =
508                 new ByteArrayInputStream(content.getBytes())) {
509             htmlParser.parse(new InputSource(htmlStream));
510         }
511         logger.info("HTML found to be valid.");
512     }
