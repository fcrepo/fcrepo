180     @Test
181     public void testGetFixity()
182         throws BinaryStoreException, IOException, NoSuchAlgorithmException {
183         LowLevelCacheEntry ispnEntry =
184                 new LocalBinaryStoreEntry(mockStore, testKey);
185         final byte[] bytes = new byte[]{0,1,2,3,4};
186         when(mockStore.getInputStream(testKey)).thenAnswer(
187                 new Answer<InputStream>() {
188 
189                     @Override
190                     public InputStream answer(InvocationOnMock invocation)
191                             throws Throwable {
192                         return new ByteArrayInputStream(bytes);
193                     }
194 
195                 }
196         );
197         final MessageDigest d = MessageDigest.getInstance("SHA-1");
198         byte[] digested = d.digest(bytes);
199         URI testCS = ContentDigest.asURI("SHA-1", digested);
200         System.out.println(testCS);
201         FixityResult actual = ispnEntry.checkFixity(testCS, bytes.length);
202         assertEquals(1, actual.status.size());
203         assertEquals(actual.status.iterator().next().toString(),
204                 true, actual.status.contains(FixityState.SUCCESS));
205 
206         // report the wrong size
207         actual = ispnEntry.checkFixity(testCS, bytes.length + 1);
208         assertEquals(1, actual.status.size());
209         assertEquals(actual.status.iterator().next().toString(),
210                 true, actual.status.contains(FixityState.BAD_SIZE));
211         // break the digest
212         digested[0] += 9;
213         testCS = ContentDigest.asURI("SHA-1", digested);
214         actual = ispnEntry.checkFixity(testCS, bytes.length);
215         assertEquals(1, actual.status.size());
216         assertEquals(actual.status.iterator().next().toString(),
217                 true, actual.status.contains(FixityState.BAD_CHECKSUM));
218         // report the wrong size and the wrong digest
219         actual = ispnEntry.checkFixity(testCS, bytes.length + 1);
220         assertEquals(2, actual.status.size());
221         assertEquals(true, actual.status.contains(FixityState.BAD_CHECKSUM));
222         assertEquals(true, actual.status.contains(FixityState.BAD_SIZE));
223     }
