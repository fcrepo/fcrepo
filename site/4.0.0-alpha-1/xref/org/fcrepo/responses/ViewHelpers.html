66      /**
67       * Return an iterator of Quads that match the given subject and predicate
68       * 
69       * @param dataset
70       * @param subject

72       * @return

74      public Iterator<Quad> getObjects(final DatasetGraph dataset,
75              final Node subject, final Resource predicate) {
76          return dataset.find(Node.ANY, subject, predicate.asNode(), Node.ANY);
77      }

79      /**
80       * Get the canonical title of a subject from the graph
81       * 
82       * @param dataset
83       * @param subject
84       * @return
85       */
86      public String getObjectTitle(final DatasetGraph dataset,
87              final Node subject) {
88  
89          Property[] properties =
90                  new Property[] {RdfLexicon.RDFS_LABEL, RdfLexicon.DC_TITLE};
91  
92          for (Property p : properties) {
93              final Iterator<Quad> objects = getObjects(dataset, subject, p);
94  
95              if (objects.hasNext()) {
96                  return objects.next().getObject().getLiteralValue().toString();
97              }
98          }
99  
100         if (subject.isURI()) {
101             return subject.getURI();
102         } else if (subject.isBlank()) {
103             return subject.getBlankNodeLabel();
104         } else {
105             return subject.toString();
106         }
107 
108     }

110     /**
111      * Get the string version of the object that matches the given subject and
112      * predicate
113      * 
114      * @param dataset
115      * @param subject
116      * @param predicate
117      * @return
118      */
119     public String getObjectsAsString(final DatasetGraph dataset,
120             final Node subject, final Resource predicate) {
121         final Iterator<Quad> iterator = getObjects(dataset, subject, predicate);
122 
123         if (iterator.hasNext()) {
124             final Node object = iterator.next().getObject();
125 
126             if (object.isLiteral()) {
127                 final String s = object.getLiteralValue().toString();
128 
129                 if (s.isEmpty()) {
130                     return "<empty>";
131                 } else {
132                     return s;
133                 }
134             } else {
135                 return "&lt;<a href=\"" + object.getURI() + "\">" +
136                         object.getURI() + "</a>&gt;";
137             }
138         } else {
139             return "";
140         }
141     }

143     /**
144      * Generate url -> local name breadcrumbs for a given node's tree
145      * 
146      * @param uriInfo
147      * @param subject
148      * @return
149      */
150     public Map<String, String> getNodeBreadcrumbs(final UriInfo uriInfo,
151             final Node subject) {
152         final String topic = subject.getURI();
153 
154         LOGGER.trace("Generating breadcrumbs for subject {}", subject);
155         final ImmutableMap.Builder<String, String> builder =
156                 ImmutableMap.builder();
157 
158         final String baseUri = uriInfo.getBaseUri().toString();
159 
160         if (!topic.startsWith(baseUri)) {
161             LOGGER.trace("Topic wasn't part of our base URI {}", baseUri);
162             return builder.build();
163         }
164 
165         final String salientPath = topic.substring(baseUri.length());
166 
167         final String[] split = salientPath.split("/");
168 
169         StringBuilder cumulativePath = new StringBuilder();
170 
171         for (final String path : split) {
172 
173             if (path.isEmpty()) {
174                 continue;
175             }
176 
177             cumulativePath.append(path);
178 
179             final String uri =
180                     uriInfo.getBaseUriBuilder().path(cumulativePath.toString())
181                             .build().toString();
182 
183             LOGGER.trace("Adding breadcrumb for path segment {} => {}", path,
184                     uri);
185 
186             builder.put(uri, path);
187 
188             cumulativePath.append("/");
189 
190         }
191 
192         return builder.build();
193 
194     }

196     /**
197      * Sort a Iterator of Quads alphabetically by its subject, predicate, and
198      * object
199      * 
200      * @param model
201      * @param it
202      * @return
203      */
204     public List<Quad> getSortedTriples(final Model model,
205             final Iterator<Quad> it) {
206         return Ordering.from(new QuadOrdering(model)).sortedCopy(
207                 ImmutableList.copyOf(it));
208     }

210     /**
211      * Get the namespace prefix (or the namespace URI itself, if no prefix is
212      * available) from a prefix mapping
213      * 
214      * @param mapping
215      * @param namespace
216      * @return
217      */
218     public String getNamespacePrefix(final PrefixMapping mapping,
219             final String namespace) {
220         final String nsURIPrefix = mapping.getNsURIPrefix(namespace);
221 
222         if (nsURIPrefix == null) {
223             return namespace;
224         } else {
225             return nsURIPrefix + ":";
226         }
227     }

229     /**
230      * Get a prefix preamble appropriate for a SPARQL-UPDATE query from a prefix
231      * mapping object
232      * 
233      * @param mapping
234      * @return
235      */
236     public String getPrefixPreamble(final PrefixMapping mapping) {
237         StringBuilder sb = new StringBuilder();
238 
239         final Map<String, String> nsPrefixMap = mapping.getNsPrefixMap();
240 
241         for (Map.Entry<String, String> entry : nsPrefixMap.entrySet()) {
242             sb.append("PREFIX " + entry.getKey() + ": <" + entry.getValue() +
243                     ">\n");
244         }
245 
246         sb.append("\n");
247         return sb.toString();
248     }
