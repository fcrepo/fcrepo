6   package org.fcrepo.http.commons.domain;
7   
8   import static javax.ws.rs.core.Variant.mediaTypes;
9   import static org.apache.jena.riot.WebContent.contentTypeJSONLD;
10  import static org.apache.jena.riot.WebContent.contentTypeN3;
11  import static org.apache.jena.riot.WebContent.contentTypeN3Alt2;
12  import static org.apache.jena.riot.WebContent.contentTypeNTriples;
13  import static org.apache.jena.riot.WebContent.contentTypeRDFXML;
14  import static org.apache.jena.riot.WebContent.contentTypeTurtle;
15  
16  import java.util.List;
17  
18  import javax.ws.rs.core.MediaType;
19  import javax.ws.rs.core.Variant;
20  
21  /**
22   * This is a convenience class carrying the various RDF content-type values as
23   * Strings and MediaTypes, after the fashion of the constants available on
24   * javax.ws.rs.core.MediaType
25   *
26   * @author ba2213
27   */
28  public abstract class RDFMediaType extends MediaType {
29  
30      private static final String CHARSET_UTF8 = ";charset=utf-8";
31  
32      public static final String N3 = contentTypeN3;
33  
34      public static final MediaType N3_TYPE = typeFromString(N3);
35  
36      public static final String N3_WITH_CHARSET = N3 + CHARSET_UTF8;
37  
38      public static final String N3_ALT2 = contentTypeN3Alt2;
39  
40      public static final MediaType N3_ALT2_TYPE = typeFromString(N3_ALT2);
41  
42      public static final String N3_ALT2_WITH_CHARSET = N3_ALT2 + CHARSET_UTF8;
43  
44      public static final String TURTLE = contentTypeTurtle;
45  
46      public static final MediaType TURTLE_TYPE = typeFromString(TURTLE);
47  
48      public static final String TURTLE_WITH_CHARSET = TURTLE + CHARSET_UTF8;
49  
50      public static final String RDF_XML = contentTypeRDFXML;
51  
52      public static final MediaType RDF_XML_TYPE = typeFromString(RDF_XML);
53  
54      public static final String NTRIPLES = contentTypeNTriples;
55  
56      public static final MediaType NTRIPLES_TYPE = typeFromString(NTRIPLES);
57  
58      public final static String JSON_LD = contentTypeJSONLD;
59  
60      public final static MediaType JSON_LD_TYPE = typeFromString(JSON_LD);
61  
62      public final static String TEXT_PLAIN_WITH_CHARSET = TEXT_PLAIN + CHARSET_UTF8;
63  
64      public final static String TEXT_HTML_WITH_CHARSET = TEXT_HTML + CHARSET_UTF8;
65  
66      public static final String APPLICATION_LINK_FORMAT = "application/link-format";
67  
68      public static final List<Variant> POSSIBLE_RDF_VARIANTS = mediaTypes(
69              RDF_XML_TYPE, TURTLE_TYPE, N3_TYPE, N3_ALT2_TYPE, NTRIPLES_TYPE,
70              TEXT_PLAIN_TYPE, JSON_LD_TYPE).add().build();
71  
72      public static final String[] POSSIBLE_RDF_RESPONSE_VARIANTS_STRING = {
73          TURTLE_WITH_CHARSET, N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES,
74          TEXT_PLAIN_WITH_CHARSET, JSON_LD };
75  
76      private static MediaType typeFromString(final String type) {
77          return new MediaType(type.split("/")[0], type.split("/")[1]);
78      }
79  
80  }
