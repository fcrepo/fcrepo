6   package org.fcrepo.kernel.api.utils;
7   
8   import static org.apache.jena.graph.NodeFactory.createURI;
9   
10  import org.apache.jena.graph.Node;
11  import org.apache.jena.graph.Triple;
12  
13  /**
14   * Utility for remapping subjects in rdf triples.
15   *
16   * @author bbpennel
17   */
18  public class SubjectMappingUtil {
19  
20      private SubjectMappingUtil() {
21          // Empty constructor for static utility class
22      }
23  
24      /**
25       * Maps the subject of t from resourceUri to destinationUri to produce a new Triple.
26       * If the triple does not have the subject resourceUri, then the triple is unchanged.
27       *
28       * @param t triple to be remapped.
29       * @param resourceUri resource subject uri to be remapped.
30       * @param destinationUri subject uri for the resultant triple.
31       * @return triple with subject remapped to destinationUri or the original subject.
32       */
33      public static Triple mapSubject(final Triple t, final String resourceUri, final String destinationUri) {
34          final Node destinationNode = createURI(destinationUri);
35          return mapSubject(t, resourceUri, destinationNode);
36      }
37  
38      /**
39       * Maps the subject of t from resourceUri to destinationNode to produce a new Triple.
40       * If the triple does not have the subject resourceUri, then the triple is unchanged.
41       *
42       * @param t triple to be remapped.
43       * @param resourceUri resource subject uri to be remapped.
44       * @param destinationNode subject node for the resultant triple.
45       * @return triple with subject remapped to destinationNode or the original subject.
46       */
47      public static Triple mapSubject(final Triple t, final String resourceUri, final Node destinationNode) {
48          final Node tripleSubj = t.getSubject();
49          final String tripleSubjUri = tripleSubj.getURI();
50          final Node subject;
51          if (tripleSubjUri.equals(resourceUri)) {
52              subject = destinationNode;
53          } else if (tripleSubjUri.startsWith(resourceUri)) {
54              // If the subject begins with the originating resource uri, such as a hash uri, then rebase
55              // the portions of the subject after the resource uri to the destination uri.
56              final String suffix = tripleSubjUri.substring(resourceUri.length());
57              subject = createURI(destinationNode.getURI() + suffix);
58          } else {
59              subject = t.getSubject();
60          }
61          return new Triple(subject, t.getPredicate(), t.getObject());
62      }
63  }
