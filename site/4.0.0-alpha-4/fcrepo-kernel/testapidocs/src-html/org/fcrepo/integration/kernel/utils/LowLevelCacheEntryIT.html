062    }

134    @Test
135    public void testGetExternalIdentifierWithInfinispan() throws Exception {
136
137        final EmbeddedCacheManager cm =
138                new DefaultCacheManager("config/testing/infinispan-basic.xml");
139
140        final CacheStore ispn =
141                cm.getCache("FedoraRepository").getAdvancedCache()
142                        .getComponentRegistry().getComponent(
143                                CacheLoaderManager.class).getCacheStore();
144        final LowLevelCacheEntry cs =
145                new CacheStoreEntry(ispn, "FedoraRepository", new BinaryKey("asd"));
146        assertEquals(
147                "/org.infinispan.loaders.file." +
148                "FileCacheStore:FedoraRepository:org.infinispan." +
149                "loaders.file.FileCacheStore:target/FedoraRepository/storage",
150                cs.getExternalIdentifier());
151    }
152
153    @Test
154    public void testModifyingCacheStores() throws Exception {
155
156        final EmbeddedCacheManager cm =
157                new DefaultCacheManager(
158                        "config/testing/infinispan-chained.xml");
159
160        final CacheStore ispn =
161                cm.getCache("FedoraRepository").getAdvancedCache()
162                        .getComponentRegistry().getComponent(
163                                CacheLoaderManager.class).getCacheStore();
164
165        assert ispn instanceof ChainingCacheStore;
166
167        final BinaryKey key = new BinaryKey("123");
168
169        final ChainingCacheStore chained_store = (ChainingCacheStore) ispn;
170
171        final LinkedHashMap<CacheStore, CacheStoreConfiguration> stores =
172                chained_store.getStores();
173
174        final LowLevelCacheEntry cs =
175                new CacheStoreEntry((CacheStore) stores.keySet()
176                        .toArray()[0], "FedoraRepository", key);
177        final LowLevelCacheEntry cs2 =
178                new CacheStoreEntry((CacheStore) stores.keySet()
179                        .toArray()[1], "FedoraRepository", key);
180
181        cs.storeValue(new ByteArrayInputStream("123456".getBytes()));
182
183        cs2.storeValue(new ByteArrayInputStream("asdfg".getBytes()));
184
185        Thread.sleep(1000);
186
187        final String v1 = IOUtils.toString(cs.getInputStream());
188        final String v2 = IOUtils.toString(cs2.getInputStream());
189
190        assertEquals("Found the wrong value in our cache store", "123456", v1);
191        assertEquals("Found the wrong value in our cache store", "asdfg", v2);
192
193    }
