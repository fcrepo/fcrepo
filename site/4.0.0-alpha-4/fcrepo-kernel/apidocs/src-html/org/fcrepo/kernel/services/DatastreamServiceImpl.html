126    /**
127     * Create a new Datastream node in the JCR store
128     *
129     *
130     * @param session the jcr session to use
131     * @param dsPath the absolute path to put the datastream
132     * @param contentType the mime-type for the requestBodyStream
133     * @param originalFileName the original file name for the input stream
134     * @param requestBodyStream binary payload for the datastream
135     * @param checksum the digest for the binary payload (as urn:sha1:xyz)   @return
136     * @throws RepositoryException
137     * @throws InvalidChecksumException
138     */
139    @Override
140    public Node createDatastreamNode(final Session session,
141                                     final String dsPath, final String contentType,
142                                     final String originalFileName, final InputStream requestBodyStream,
143                                     final URI checksum)
144        throws RepositoryException, InvalidChecksumException {
145
146        final Datastream ds = createDatastream(session, dsPath);
147        ds.setContent(requestBodyStream, contentType, checksum,
148                         originalFileName, getStoragePolicyDecisionPoint());
149        return ds.getNode();
150    }

192    /**
193     * Get the fixity results for the datastream as a RDF Dataset
194     *
195     * @param subjects
196     * @param datastream
197     * @return
198     * @throws RepositoryException
199     */
200    @Override
201    public RdfStream getFixityResultsModel(final GraphSubjects subjects,
202            final Datastream datastream) throws RepositoryException {
203        final Collection<FixityResult> blobs = runFixityAndFixProblems(datastream);
204
205        return JcrRdfTools.withContext(subjects,
206                datastream.getNode().getSession()).getJcrTriples(
207                datastream.getNode(), blobs).topic(
208                subjects.getGraphSubject(datastream.getNode()).asNode());
209    }

211    /**
212     * Run the fixity check on the datastream and attempt to automatically
213     * correct failures if additional copies of the bitstream are available
214     *
215     * @param datastream
216     * @return
217     * @throws RepositoryException
218     */
219    @Override
220    public Collection<FixityResult> runFixityAndFixProblems(
221            final Datastream datastream) throws RepositoryException {
222
223        Set<FixityResult> fixityResults;
224        Set<FixityResult> goodEntries;
225        final URI digestUri = datastream.getContentDigest();
226        final long size = datastream.getContentSize();
227
228        fixityCheckCounter.inc();
229
230        final Timer.Context context = timer.time();
231
232        try {
233            fixityResults =
234                    copyOf(getFixity(datastream.getNode().getNode(JCR_CONTENT),
235                            digestUri, size));
236
237            goodEntries =
238                    copyOf(filter(fixityResults, new Predicate<FixityResult>() {
239
240                        @Override
241                        public boolean apply(
242                                final FixityResult input) {
243                            return input.matches(size, digestUri);
244                        }
245                    }));
246
247        } finally {
248            context.stop();
249        }
250
251        if (goodEntries.isEmpty()) {
252            LOGGER.error("ALL COPIES OF " + datastream.getNode().getPath() +
253                             " HAVE FAILED FIXITY CHECKS.");
254            return fixityResults;
255        }
256
257        final CacheEntry anyGoodCacheEntry =
258                goodEntries.iterator().next().getEntry();
259
260        final Set<FixityResult> badEntries =
261                difference(fixityResults, goodEntries);
262
263        for (final FixityResult result : badEntries) {
264            try {
265                // we can safely cast to a LowLevelCacheEntry here, since
266                // other entries have to be filtered out before
267                final LowLevelCacheEntry lle = (LowLevelCacheEntry) result.getEntry();
268                lle.storeValue(anyGoodCacheEntry.getInputStream());
269                final FixityResult newResult =
270                        result.getEntry().checkFixity(digestUri, size);
271                if (newResult.isSuccess()) {
272                    result.setRepaired();
273                    fixityRepairedCounter.inc();
274                } else {
275                    fixityErrorCounter.inc();
276                }
277            } catch (final IOException e) {
278                LOGGER.warn("Exception repairing low-level cache entry: {}", e);
279            }
280        }
281
282        return fixityResults;
283    }

285    /**
286     * Get the fixity results for this datastream's bitstream, and compare it
287     * against the given checksum and size.
288     *
289     * @param resource
290     * @param dsChecksum -the checksum and algorithm represented as a URI
291     * @param dsSize
292     * @return
293     * @throws RepositoryException
294     */
295    @Override
296    public Collection<FixityResult> getFixity(final Node resource,
297            final URI dsChecksum, final long dsSize) throws RepositoryException {
298        LOGGER.debug("Checking resource: " + resource.getPath());
299
300        final Binary bin = resource.getProperty(JCR_DATA).getBinary();
301
302        if (bin instanceof ExternalBinaryValue) {
303            return ImmutableSet.of(new ProjectedCacheEntry(bin, resource.getPath())
304                                    .checkFixity(dsChecksum, dsSize));
305
306        } else if (bin instanceof InMemoryBinaryValue) {
307            return ImmutableSet.of(new BinaryCacheEntry(bin, resource.getPath())
308                                    .checkFixity(dsChecksum, dsSize));
309        } else {
310            final Function<LowLevelCacheEntry, FixityResult> checkCacheFunc =
311                getCheckCacheFixityFunction(dsChecksum, dsSize);
312            return llStoreService.transformLowLevelCacheEntries(resource,
313                                                                   checkCacheFunc);
314        }
315
316    }
