132     /**
133      * Retrieve the node profile
134      *
135      * @param pathList
136      * @param offset with limit, control the pagination window of details for
137      *        child nodes
138      * @param limit with offset, control the pagination window of details for
139      *        child nodes
140      * @param request
141      * @param uriInfo
142      * @return
143      * @throws RepositoryException
144      */
145     @GET
146     @Produces({TURTLE, N3, N3_ALT2, RDF_XML, NTRIPLES, APPLICATION_XML, TEXT_PLAIN, TURTLE_X,
147             TEXT_HTML, APPLICATION_XHTML_XML})
148     public RdfStream describe(@PathParam("path") final List<PathSegment> pathList,
149             @QueryParam("offset") @DefaultValue("0") final int offset,
150             @QueryParam("limit") @DefaultValue("-1") final int limit,
151             @QueryParam("non-member-properties") final String nonMemberProperties,
152             @Context final Request request,
153             @Context final HttpServletResponse servletResponse,
154             @Context final UriInfo uriInfo) throws RepositoryException {
155         final String path = toPath(pathList);
156         LOGGER.trace("Getting profile for: {}", path);
157 
158         final FedoraResource resource = nodeService.getObject(session, path);
159 
160         final EntityTag etag = new EntityTag(resource.getEtagValue());
161         final Date date = resource.getLastModifiedDate();
162         final Date roundedDate = new Date();
163         if (date != null) {
164             roundedDate.setTime(date.getTime() - date.getTime() % 1000);
165         }
166         final ResponseBuilder builder =
167             request.evaluatePreconditions(roundedDate, etag);
168         if (builder != null) {
169             final CacheControl cc = new CacheControl();
170             cc.setMaxAge(0);
171             cc.setMustRevalidate(true);
172             // here we are implicitly emitting a 304
173             // the exception is not an error, it's genuinely
174             // an exceptional condition
175             throw new WebApplicationException(builder.cacheControl(cc)
176                     .lastModified(date).tag(etag).build());
177         }
178         final HttpGraphSubjects subjects =
179             new HttpGraphSubjects(session, this.getClass(), uriInfo);
180 
181         final int realLimit;
182         if (nonMemberProperties != null && limit == NO_LIMIT) {
183             realLimit = NO_MEMBER_PROPERTIES;
184         } else {
185             realLimit = limit;
186         }
187 
188         final RdfStream rdfStream =
189             resource.getTriples(subjects).concat(
190                     resource.getHierarchyTriples(subjects)).session(session)
191                     .topic(subjects.getGraphSubject(resource.getNode())
192                             .asNode());
193         if (realLimit != NO_MEMBER_PROPERTIES) {
194             final Node firstPage =
195                 createURI(uriInfo.getRequestUriBuilder().replaceQueryParam(
196                         "offset", 0).replaceQueryParam("limit", limit).build()
197                         .toString().replace("&", "&amp;"));
198             final Node nextPage =
199                 createURI(uriInfo.getRequestUriBuilder().replaceQueryParam(
200                         "offset", offset + limit).replaceQueryParam("limit",
201                         limit).build().toString().replace("&", "&amp;"));
202             rdfStream.concat(
203                     create(subjects.getContext().asNode(), NEXT_PAGE.asNode(),
204                             nextPage),
205                     create(subjects.getContext().asNode(), FIRST_PAGE.asNode(),
206                             firstPage)).limit(realLimit).skip(offset);
207 
208             servletResponse.addHeader("Link", firstPage + ";rel=\"first\"");
209         }
210 
211 
212         if (!etag.getValue().isEmpty()) {
213             servletResponse.addHeader("ETag", etag.toString());
214         }
215 
216         if (resource.getLastModifiedDate() != null) {
217             servletResponse.addDateHeader("Last-Modified", resource
218                     .getLastModifiedDate().getTime());
219         }
220         servletResponse.addHeader("Accept-Patch", contentTypeSPARQLUpdate);
221         servletResponse.addHeader("Link", LDP_NAMESPACE + "Resource;rel=\"type\"");
222 
223         addResponseInformationToStream(resource, rdfStream, uriInfo,
224                 subjects);
225 
226         return rdfStream;
227 
228 
229     }

231     /**
232      * Update an object using SPARQL-UPDATE
233      *
234      * @param pathList
235      * @return 201
236      * @throws RepositoryException
237      * @throws org.fcrepo.kernel.exception.InvalidChecksumException
238      * @throws IOException
239      */
240     @PATCH
241     @Consumes({contentTypeSPARQLUpdate})
242     @Timed
243     public Response updateSparql(@PathParam("path")
244             final List<PathSegment> pathList,
245             @Context
246             final UriInfo uriInfo,
247             final InputStream requestBodyStream,
248             @Context final Request request)
249         throws RepositoryException, IOException {
250 
251         final String path = toPath(pathList);
252         LOGGER.debug("Attempting to update path: {}", path);
253 
254         try {
255 
256             if (requestBodyStream != null) {
257 
258                 final FedoraResource resource =
259                         nodeService.getObject(session, path);
260 
261 
262                 final EntityTag etag = new EntityTag(resource.getEtagValue());
263                 final Date date = resource.getLastModifiedDate();
264                 final Date roundedDate = new Date();
265 
266                 if (date != null) {
267                     roundedDate.setTime(date.getTime() - date.getTime() % 1000);
268                 }
269 
270                 final ResponseBuilder builder =
271                     request.evaluatePreconditions(roundedDate, etag);
272 
273                 if (builder != null) {
274                     throw new WebApplicationException(builder.build());
275                 }
276 
277                 final Dataset properties = resource.updatePropertiesDataset(new HttpGraphSubjects(
278                         session, FedoraNodes.class, uriInfo), IOUtils
279                         .toString(requestBodyStream));
280 
281 
282                 final Model problems = properties.getNamedModel(PROBLEMS_MODEL_NAME);
283                 if (!problems.isEmpty()) {
284                     LOGGER.info(
285                                    "Found these problems updating the properties for {}: {}",
286                                    path, problems);
287                     return status(FORBIDDEN).entity(problems.toString())
288                             .build();
289 
290                 }
291 
292                 session.save();
293                 versionService.nodeUpdated(resource.getNode());
294 
295                 return status(SC_NO_CONTENT).build();
296             }
297             return status(SC_BAD_REQUEST).entity(
298                     "SPARQL-UPDATE requests must have content!").build();
299 
300         } finally {
301             session.logout();
302         }
303     }

306      * Replace triples with triples from a new model
307      * @param pathList
308      * @param uriInfo
309      * @param requestContentType
310      * @param requestBodyStream
311      * @return
312      * @throws Exception

314     @PUT

316     @Timed
317     public Response createOrReplaceObjectRdf(
318             @PathParam("path") final List<PathSegment> pathList,
319             @Context final UriInfo uriInfo,
320             @HeaderParam("Content-Type")
321             final MediaType requestContentType,
322             final InputStream requestBodyStream,
323             @Context
324             final Request request) throws RepositoryException, ParseException,
325             IOException, InvalidChecksumException, URISyntaxException {
326         final String path = toPath(pathList);
327         LOGGER.debug("Attempting to replace path: {}", path);
328         try {
329 
330             if (!nodeService.exists(session, path)) {
331                 return createObject(pathList, null, null, null,
332                         requestContentType, null, uriInfo, requestBodyStream);
333             }
334 
335             final FedoraResource resource =
336                 nodeService.getObject(session, path);
337 
338             final Date date = resource.getLastModifiedDate();
339             final Date roundedDate = new Date();
340 
341 
342             final EntityTag etag = new EntityTag(resource.getEtagValue());
343 
344             if (date != null) {
345                 roundedDate.setTime(date.getTime() - date.getTime() % 1000);
346             }
347 
348             final ResponseBuilder builder =
349                 request.evaluatePreconditions(roundedDate, etag);
350 
351             if (builder != null) {
352                 throw new WebApplicationException(builder.build());
353             }
354 
355             final HttpGraphSubjects graphSubjects = new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
356 
357             if (requestContentType != null && requestBodyStream != null)  {
358                 final String contentType = requestContentType.toString();
359 
360                 final String format = contentTypeToLang(contentType).getName()
361                                           .toUpperCase();
362 
363                 final Model inputModel = createDefaultModel()
364                                              .read(requestBodyStream,
365                                                       graphSubjects.getGraphSubject(resource.getNode()).toString(),
366                                                       format);
367 
368                 resource.replaceProperties(graphSubjects, inputModel);
369             }
370 
371             session.save();
372             versionService.nodeUpdated(resource.getNode());
373 
374             return status(SC_NO_CONTENT).build();
375         } finally {
376             session.logout();
377         }

380     /**
381      * Creates a new object.
382      *
383      * @param pathList
384      * @return 201
385      * @throws Exception
386      */
387     @POST
388     @Timed
389     public Response createObject(@PathParam("path")
390             final List<PathSegment> pathList,
391             @QueryParam("mixin")
392             final String mixin,
393             @QueryParam("checksum")
394             final String checksum,
395             @HeaderParam("Content-Disposition") final String contentDisposition,
396             @HeaderParam("Content-Type")
397             final MediaType requestContentType,
398             @HeaderParam("Slug")
399             final String slug,
400             @Context
401             final UriInfo uriInfo, final InputStream requestBodyStream)
402         throws RepositoryException, ParseException, IOException,
403                    InvalidChecksumException, URISyntaxException {
404 
405         final String newObjectPath;
406         final String path = toPath(pathList);
407 
408 
409         if (nodeService.exists(session, path)) {
410             final String pid;
411 
412             if (slug != null) {
413                 pid = slug;
414             }  else {
415                 pid = pidMinter.mintPid();
416             }
417 
418             newObjectPath = path + "/" + pid;
419         } else {
420             newObjectPath = path;
421         }
422 
423         LOGGER.debug("Attempting to ingest with path: {}", newObjectPath);
424 
425         try {
426             if (nodeService.exists(session, newObjectPath)) {
427                 return status(SC_CONFLICT).entity(
428                         path + " is an existing resource!").build();
429             }
430 
431             final URI checksumURI;
432 
433             if (checksum != null && !checksum.equals("")) {
434                 checksumURI = new URI(checksum);
435             } else {
436                 checksumURI = null;
437             }
438 
439             final HttpGraphSubjects subjects = new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
440 
441             final String objectType;
442 
443             if (mixin != null) {
444                 objectType = mixin;
445             } else {
446                 if (requestContentType != null) {
447                     final String s = requestContentType.toString();
448                     if (s.equals(contentTypeSPARQLUpdate) || contentTypeToLang(s) != null) {
449                         objectType = FEDORA_OBJECT;
450                     } else {
451                         objectType = FEDORA_DATASTREAM;
452                     }
453                 } else {
454                     objectType = FEDORA_OBJECT;
455                 }
456             }
457 
458             final FedoraResource result;
459 
460             switch (objectType) {
461                 case FEDORA_OBJECT:
462                     result = objectService.createObject(session, newObjectPath);
463                     break;
464                 case FEDORA_DATASTREAM:
465                     result = datastreamService.createDatastream(session, newObjectPath);
466                     break;
467                 default:
468                     throw new WebApplicationException(clientError().entity(
469                             "Unknown object type " + objectType).build());
470             }
471 
472             if (requestBodyStream != null) {
473 
474                 final MediaType contentType =
475                     requestContentType != null ? requestContentType
476                         : APPLICATION_OCTET_STREAM_TYPE;
477 
478                 final String contentTypeString = contentType.toString();
479 
480                 if (contentTypeString.equals(contentTypeSPARQLUpdate)) {
481                     result.updatePropertiesDataset(subjects, IOUtils.toString(requestBodyStream));
482                 } else if (contentTypeToLang(contentTypeString) != null) {
483 
484                     final Lang lang = contentTypeToLang(contentTypeString);
485 
486                     if (lang == null) {
487                         throw new WebApplicationException(notAcceptable().entity(
488                                 "Invalid Content type " + contentType).build());
489                     }
490 
491                     final String format = lang.getName()
492                                               .toUpperCase();
493 
494                     final Model inputModel =
495                         createDefaultModel()
496                                                  .read(requestBodyStream,
497                                                           subjects.getGraphSubject(result.getNode()).toString(),
498                                                           format);
499 
500                     result.replaceProperties(subjects, inputModel);
501                 } else if (result instanceof Datastream) {
502 
503                     final String originalFileName;
504 
505                     if (contentDisposition != null) {
506                         final ContentDisposition disposition = new ContentDisposition(contentDisposition);
507                         originalFileName = disposition.getFileName();
508                     } else {
509                         originalFileName = null;
510                     }
511 
512                     datastreamService.createDatastreamNode(session,
513                             newObjectPath, contentTypeString, originalFileName, requestBodyStream, checksumURI);
514 
515                 }
516             }
517 
518             session.save();
519             versionService.nodeUpdated(result.getNode());
520 
521             LOGGER.debug("Finished creating {} with path: {}", mixin, newObjectPath);
522 
523             final URI location;
524             if (result.hasContent()) {
525                 location = new URI(subjects.getGraphSubject(result.getNode().getNode(JcrConstants.JCR_CONTENT))
526                                                  .getURI());
527             } else {
528                 location = new URI(subjects.getGraphSubject(result.getNode())
529                                                  .getURI());
530             }
531 
532             return created(location).entity(location.toString()).build();
533 
534         } finally {
535             session.logout();
536         }
537     }

608     /**
609      * Copies an object from one path to another
610      */
611     @COPY
612     @Timed
613     public Response copyObject(@PathParam("path") final List<PathSegment> path,
614                                @HeaderParam("Destination") final String destinationUri)
615         throws RepositoryException, URISyntaxException {
616 
617         try {
618 
619             final GraphSubjects subjects =
620                 new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
621 
622             if (!nodeService.exists(session, toPath(path))) {
623                 return status(SC_CONFLICT).entity("The source path does not exist").build();
624             }
625 
626             final String destination =
627                 subjects.getPathFromGraphSubject(ResourceFactory.createResource(destinationUri));
628 
629             if (destination == null) {
630                 return status(SC_BAD_GATEWAY).entity("Destination was not a valid resource path").build();
631             }
632 
633             nodeService.copyObject(session, toPath(path), destination);
634             session.save();
635             versionService.nodeUpdated(session, destination);
636             return created(new URI(destinationUri)).build();
637         } catch (final ItemExistsException e) {
638             throw new WebApplicationException(e,
639                 status(SC_PRECONDITION_FAILED).entity("Destination resource already exists").build());
640         } catch (final PathNotFoundException e) {
641             throw new WebApplicationException(e, status(SC_CONFLICT).entity(
642                     "There is no node that will serve as the parent of the moved item")
643                     .build());
644         } finally {
645             session.logout();
646         }
647 
648     }
