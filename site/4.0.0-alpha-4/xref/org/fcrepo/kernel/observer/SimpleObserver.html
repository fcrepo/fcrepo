48  /**
49   * Simple JCR EventListener that filters JCR Events through a Fedora EventFilter
50   * and puts the resulting stream onto the internal Fedora EventBus as a stream
51   * of FedoraEvents.
52   *
53   * @author eddies
54   * @author ajs6f
55   * @date Feb 7, 2013
56   */
57  public class SimpleObserver implements EventListener {
58  
59      private static final Logger LOGGER = getLogger(SimpleObserver.class);
60  
61      /**
62       * A simple counter of events that pass through this observer
63       */
64      static final Counter EVENT_COUNTER = getMetrics().counter(name(SimpleObserver.class, "onEvent"));
65  
66      static final Integer EVENT_TYPES = NODE_ADDED + NODE_REMOVED + NODE_MOVED + PROPERTY_ADDED + PROPERTY_CHANGED
67              + PROPERTY_REMOVED;
68  
69      @Inject
70      private Repository repository;
71  
72      @Inject
73      private EventBus eventBus;
74  
75      @Inject
76      private InternalExternalEventMapper eventMapper;
77  
78      @Inject
79      private EventFilter eventFilter;
80  
81      // THIS SESSION SHOULD NOT BE USED TO LOOK UP NODES
82      // it is used only to register and deregister this observer to the JCR
83      private Session session;
84  
85      /**
86       * Register this observer with the JCR event listeners
87       *
88       * @throws RepositoryException
89       */
90      @PostConstruct
91      public void buildListener() throws RepositoryException {
92          LOGGER.debug("Constructing an observer for JCR events...");
93          session = repository.login();
94          session.getWorkspace().getObservationManager()
95                  .addEventListener(this, EVENT_TYPES, "/", true, null, null, false);
96          session.save();
97      }
98  
99      /**
100      * logout of the session
101      *
102      * @throws RepositoryException
103      */
104     @PreDestroy
105     public void stopListening() throws RepositoryException {
106         LOGGER.debug("Destroying an observer for JCR events...");
107         session.getWorkspace().getObservationManager().removeEventListener(this);
108         session.logout();
109     }
110 
111     /**
112      * Filter JCR events and transform them into our own FedoraEvents.
113      *
114      * @param events
115      */
116     @Override
117     public void onEvent(final javax.jcr.observation.EventIterator events) {
118         Session lookupSession = null;
119         try {
120             lookupSession = repository.login();
121             @SuppressWarnings("unchecked")
122             final Iterator<Event> filteredEvents = filter(events, eventFilter.getFilter(lookupSession));
123             final Iterator<FedoraEvent> publishableEvents = eventMapper.apply(filteredEvents);
124             while (publishableEvents.hasNext()) {
125                 eventBus.post(publishableEvents.next());
126                 EVENT_COUNTER.inc();
127             }
128         } catch (final RepositoryException ex) {
129             throw propagate(ex);
130         } finally {
131             if (lookupSession != null) {
132                 lookupSession.logout();
133             }
134         }
135     }
