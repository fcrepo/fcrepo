018package org.fcrepo.http.commons.exceptionhandlers;
019
020import static javax.ws.rs.core.Response.status;
021import static javax.ws.rs.core.Response.Status.BAD_REQUEST;
022import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
023import static org.slf4j.LoggerFactory.getLogger;
024
025import javax.ws.rs.core.Response;
026import javax.ws.rs.ext.ExceptionMapper;
027import javax.ws.rs.ext.Provider;
028
029import org.slf4j.Logger;
030
031import com.fasterxml.jackson.core.JsonParseException;
032
033/**
034 * If an injected JSON resource fails to parse, return an HTTP 400 Bad Request.
035 *
036 * @author Esme Cowles
037 * @since 2014-05-21
038 */
039@Provider
040public class JsonParseExceptionMapper implements
041        ExceptionMapper<JsonParseException>, ExceptionDebugLogging {
042
043    private static final Logger LOGGER = getLogger(JsonParseExceptionMapper.class);
044
045    @Override
046    public Response toResponse(final JsonParseException e) {
047        debugException(this, e, LOGGER);
048        return status(BAD_REQUEST).entity(e.getMessage()).type(TEXT_PLAIN_WITH_CHARSET).build();
049    }
050}
