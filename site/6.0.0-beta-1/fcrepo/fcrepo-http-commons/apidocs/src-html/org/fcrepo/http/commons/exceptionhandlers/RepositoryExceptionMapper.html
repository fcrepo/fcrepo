018package org.fcrepo.http.commons.exceptionhandlers;
019
020import org.fcrepo.kernel.api.exception.RepositoryException;
021import org.slf4j.Logger;
022
023import javax.ws.rs.core.Response;
024import javax.ws.rs.ext.ExceptionMapper;
025import javax.ws.rs.ext.Provider;
026
027import static javax.ws.rs.core.Response.Status.BAD_REQUEST;
028import static javax.ws.rs.core.Response.serverError;
029import static javax.ws.rs.core.Response.status;
030import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
031import static org.slf4j.LoggerFactory.getLogger;
032
033/**
034 * Provide a quasi-useful stacktrace when a generic RepositoryException is caught
035 *
036 * @author awoods
037 */
038@Provider
039public class RepositoryExceptionMapper implements
040                                       ExceptionMapper<RepositoryException>, ExceptionDebugLogging {
041
042    private static final Logger LOGGER = getLogger(RepositoryExceptionMapper.class);
043
044    @Override
045    public Response toResponse(final RepositoryException e) {
046        LOGGER.error("Caught a repository exception", e);
047
048        if (e.getMessage().matches("Error converting \".+\" from String to a Name")) {
049            return status(BAD_REQUEST).entity(e.getMessage()).type(TEXT_PLAIN_WITH_CHARSET).build();
050        }
051
052        return serverError().entity(e.getMessage()).type(TEXT_PLAIN_WITH_CHARSET).build();
053    }
054}
