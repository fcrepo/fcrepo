018package org.fcrepo.kernel.api.operations;
019
020
021import org.fcrepo.config.ServerManagedPropsMode;
022import org.fcrepo.kernel.api.identifiers.FedoraId;
023
024/**
025 * Factory for operations on rdf sources
026 *
027 * @author bbpennel
028 */
029public interface RdfSourceOperationFactory extends ResourceOperationFactory {
030
031    /**
032     * Get a builder for an operation to create an RDF source
033     *
034     * @param rescId id of the resource targeted by the operation
035     * @param interactionModel interaction model for the resource being created
036     * @param serverManagedPropsMode server managed props mode
037     * @return new builder
038     */
039    CreateRdfSourceOperationBuilder createBuilder(FedoraId rescId,
040                                                  String interactionModel,
041                                                  ServerManagedPropsMode serverManagedPropsMode);
042
043    /**
044     * Get a builder for an operation to update an RDF source
045     *
046     * @param rescId id of the resource targeted by the operation
047     * @param serverManagedPropsMode server managed props mode
048     * @return new builder
049     */
050    RdfSourceOperationBuilder updateBuilder(FedoraId rescId, final ServerManagedPropsMode serverManagedPropsMode);
051
052}
