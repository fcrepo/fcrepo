94      private enum State {
95          COMMIT_NOT_STARTED(true),
96          PREPARE_STARTED(false),
97          PREPARED(true),
98          PREPARE_FAILED(true),
99          COMMIT_STARTED(false),
100         COMMITTED(true),
101         COMMIT_FAILED(true),
102         ROLLING_BACK(false),
103         ROLLED_BACK(false),
104         ROLLBACK_FAILED(false);
105 
106         final boolean rollbackAllowed;
107 
108         State(final boolean rollbackAllowed) {
109             this.rollbackAllowed = rollbackAllowed;
110         }

114     /**
115      * Constructor
116      *
117      * @param sessionId            session id.
118      * @param fedoraOcflIndex      the index
119      * @param objectSessionFactory the session factory
120      */
121     protected OcflPersistentStorageSession(final String sessionId,
122                                            final FedoraToOcflObjectIndex fedoraOcflIndex,
123                                            final OcflObjectSessionFactory objectSessionFactory,
124                                            final ReindexService reindexService) {
125         this.sessionId = sessionId;
126         this.fedoraOcflIndex = fedoraOcflIndex;
127         this.objectSessionFactory = objectSessionFactory;
128         this.reindexSerivce = reindexService;
129         this.sessionsToRollback = new HashMap<>();
130 
131         if (sessionId == null) {
132             // The read-only session is never closed, so it needs to periodically expire object sessions
133             this.sessionMap = Caffeine.newBuilder()
134                     .maximumSize(512)
135                     .expireAfterAccess(10, TimeUnit.MINUTES)
136                     .<String, OcflObjectSession>build()
137                     .asMap();
138         } else {
139             this.sessionMap = new ConcurrentHashMap<>();
140         }
141 
142         //load the persister list if empty
143         persisterList.add(new CreateRdfSourcePersister(this.fedoraOcflIndex));
144         persisterList.add(new UpdateRdfSourcePersister(this.fedoraOcflIndex));
145         persisterList.add(new CreateNonRdfSourcePersister(this.fedoraOcflIndex));
146         persisterList.add(new UpdateNonRdfSourcePersister(this.fedoraOcflIndex));
147         persisterList.add(new DeleteResourcePersister(this.fedoraOcflIndex));
148         persisterList.add(new CreateVersionPersister(this.fedoraOcflIndex));
149         persisterList.add(new PurgeResourcePersister(this.fedoraOcflIndex));
150         persisterList.add(new ReindexResourcePersister(this.reindexSerivce));
151 
152     }

166     @Override

171     @Override
172     public void persist(final ResourceOperation operation) throws PersistentStorageException {
173         actionNeedsWrite();
174         ensureCommitNotStarted();
175 
176         try {
177             phaser.register();
178 
179             //resolve the persister based on the operation
180             final var persister = persisterList.stream().filter(p -> p.handle(operation)).findFirst().orElse(null);
181 
182             if (persister == null) {
183                 throw new UnsupportedOperationException(format("The %s is not yet supported", operation.getClass()));
184             }
185 
186             //perform the operation
187             persister.persist(this, operation);
188 
189         } finally {
190             phaser.arriveAndDeregister();
191         }
192 
193     }

195     private void ensureCommitNotStarted() throws PersistentSessionClosedException {
196         if (!state.equals(State.COMMIT_NOT_STARTED)) {
197             throw new PersistentSessionClosedException(
198                     String.format("Storage session %s is already closed", sessionId));
199         }
200     }
201 
202     private void ensurePrepared() throws PersistentSessionClosedException {
203         if (!state.equals(State.PREPARED)) {
204             throw new PersistentStorageException(
205                     String.format("Storage session %s cannot be committed because it is not in the correct state: %s",
206                             sessionId, state));
207         }
208     }

210     OcflObjectSession findOrCreateSession(final String ocflId) {
211         return this.sessionMap.computeIfAbsent(ocflId, key -> {
212             return new FcrepoOcflObjectSessionWrapper(this.objectSessionFactory.newSession(key));
213         });
214     }

216     @Override
217     public ResourceHeaders getHeaders(final FedoraId identifier, final Instant version)
218             throws PersistentStorageException {
219 
220         ensureCommitNotStarted();
221 
222         final FedoraOcflMapping mapping = getFedoraOcflMapping(identifier);
223         final OcflObjectSession objSession = findOrCreateSession(mapping.getOcflObjectId());
224 
225         final var versionId = resolveVersionNumber(objSession, identifier, version);
226         final var headers = objSession.readHeaders(identifier.getResourceId(), versionId);
227 
228         return new ResourceHeadersAdapter(headers).asKernelHeaders();
229     }

231     private FedoraOcflMapping getFedoraOcflMapping(final FedoraId identifier) throws PersistentStorageException {
232         try {
233             return fedoraOcflIndex.getMapping(sessionId, identifier);
234         } catch (final FedoraOcflMappingNotFoundException e) {
235             throw new PersistentItemNotFoundException(e.getMessage());
236         }

239     @Override
240     public RdfStream getTriples(final FedoraId identifier, final Instant version)
241             throws PersistentStorageException {
242         ensureCommitNotStarted();
243 
244         try (final InputStream is = getBinaryContent(identifier, version)) {
245             final Model model = createDefaultModel();
246             RDFDataMgr.read(model, is, OcflPersistentStorageUtils.getRdfFormat().getLang());
247             final FedoraId topic = resolveTopic(identifier);
248             return DefaultRdfStream.fromModel(createURI(topic.getFullId()), model);
249         } catch (final IOException ex) {
250             throw new PersistentStorageException(format("unable to read %s ;  version = %s", identifier, version), ex);
251         }
252     }

254     @Override
255     public List<Instant> listVersions(final FedoraId fedoraIdentifier) throws PersistentStorageException {
256         final var mapping = getFedoraOcflMapping(fedoraIdentifier);
257         final var objSession = findOrCreateSession(mapping.getOcflObjectId());

259         return objSession.listVersions(fedoraIdentifier.getResourceId()).stream()

264     @Override
265     public InputStream getBinaryContent(final FedoraId identifier, final Instant version)
266             throws PersistentStorageException {
267         ensureCommitNotStarted();
268 
269         final var mapping = getFedoraOcflMapping(identifier);
270         final var objSession = findOrCreateSession(mapping.getOcflObjectId());
271 
272         final var versionNumber = resolveVersionNumber(objSession, identifier, version);
273 
274         return objSession.readContent(identifier.getResourceId(), versionNumber)
275                 .getContentStream()
276                 .orElseThrow(() -> new PersistentItemNotFoundException("No binary content found for resource "
277                         + identifier.getFullId()));

280     @Override
281     public synchronized void prepare() {
282         ensureCommitNotStarted();
283         if (isReadOnly()) {
284             // No changes to commit.
285             return;
286         }
287 
288         this.state = State.PREPARE_STARTED;
289         LOGGER.debug("Starting storage session {} prepare for commit", sessionId);
290 
291         if (this.phaser.getRegisteredParties() > 0) {
292             this.phaser.awaitAdvance(0);
293         }
294 
295         LOGGER.trace("All persisters are complete in session {}", sessionId);
296 
297         try {
298             fedoraOcflIndex.commit(sessionId);
299             state = State.PREPARED;
300         } catch (RuntimeException e) {
301             state = State.PREPARE_FAILED;
302             throw new PersistentStorageException(String.format("Failed to prepare storage session <%s> for commit",
303                     sessionId), e);
304         }
305     }
306 
307     @Override
308     public synchronized void commit() throws PersistentStorageException {
309         ensurePrepared();
310         if (isReadOnly()) {
311             // No changes to commit.
312             return;
313         }
314 
315         this.state = State.COMMIT_STARTED;
316         LOGGER.debug("Starting storage session {} commit", sessionId);
317 
318         // order map for testing
319         final var sessions = new TreeMap<>(sessionMap);
320         commitObjectSessions(sessions);
321 
322         LOGGER.debug("Committed storage session {}", sessionId);
323     }
324 
325     private void commitObjectSessions(final Map<String, OcflObjectSession> sessions)
326             throws PersistentStorageException {
327         this.sessionsToRollback = new HashMap<>(sessionMap.size());
328 
329         for (final var entry : sessions.entrySet()) {
330             final var id = entry.getKey();
331             final var session = entry.getValue();
332             try {
333                 session.commit();
334                 sessionsToRollback.put(id, session);
335             } catch (final Exception e) {
336                 this.state = State.COMMIT_FAILED;
337                 throw new PersistentStorageException(String.format("Failed to commit object <%s> in session <%s>",
338                         id, sessionId), e);
339             }
340         }
341 
342         state = State.COMMITTED;
343     }

345     @Override
346     public void rollback() throws PersistentStorageException {
347         if (isReadOnly()) {
348             // No changes to rollback
349             return;
350         }
351 
352         if (!state.rollbackAllowed) {
353             throw new PersistentStorageException("This session cannot be rolled back in this state: " + state);
354         }
355 
356         final boolean commitWasStarted = this.state != State.COMMIT_NOT_STARTED;
357 
358         this.state = State.ROLLING_BACK;
359         LOGGER.debug("Rolling back storage session {}", sessionId);
360 
361         if (!commitWasStarted) {
362             //if the commit had not been started at the time this method was invoked
363             //we must ensure that all persist operations are complete before we close any
364             //ocfl object sessions. If the commit had been started then this synchronization step
365             //will have already occurred and is thus unnecessary.
366             if (this.phaser.getRegisteredParties() > 0) {
367                 try {
368                     this.phaser.awaitAdvanceInterruptibly(0, AWAIT_TIMEOUT, MILLISECONDS);
369                 } catch (final InterruptedException | TimeoutException e) {
370                     throw new PersistentStorageException(
371                             "Waiting for operations to complete took too long, rollback failed");
372                 }
373             }
374         }
375 
376         closeUncommittedSessions();
377 
378         if (commitWasStarted) {
379             rollbackCommittedSessions();
380         }
381 
382         this.state = State.ROLLED_BACK;
383         LOGGER.trace("Successfully rolled back storage session {}", sessionId);
384     }

386     /**
387      * Resolve an instant to a version
388      *
389      * @param objSession session
390      * @param fedoraId the FedoraId of the resource
391      * @param version version time
392      * @return name of version
393      * @throws PersistentStorageException thrown if version not found
394      */
395     private String resolveVersionNumber(final OcflObjectSession objSession,
396                                        final FedoraId fedoraId,
397                                        final Instant version)
398             throws PersistentStorageException {
399         if (version != null) {
400             final var versions = objSession.listVersions(fedoraId.getResourceId());
401             // reverse order so that the most recent version is matched first
402             Collections.reverse(versions);
403             return versions.stream()
404                     .filter(vd -> vd.getCreated().equals(version))
405                     .map(OcflVersionInfo::getVersionNumber)
406                     .findFirst()
407                     .orElseThrow(() -> {
408                         return new PersistentItemNotFoundException(format(
409                                 "There is no version in %s with a created date matching %s",
410                                 fedoraId, version));
411                     });
412         }
413 
414         return null;
415     }
416 
417     private void closeUncommittedSessions() {
418         this.sessionMap.entrySet().stream()
419                 .filter(entry -> !sessionsToRollback.containsKey(entry.getKey()))
420                 .map(Map.Entry::getValue)
421                 .forEach(OcflObjectSession::abort);
422     }
423 
424     private void rollbackCommittedSessions() throws PersistentStorageException {
425         final List<String> rollbackFailures = new ArrayList<>(this.sessionsToRollback.size());
426 
427         for (final var entry : this.sessionsToRollback.entrySet()) {
428             final var id = entry.getKey();
429             final var session = entry.getValue();
430 
431             try {
432                 session.rollback();
433             } catch (final Exception e) {
434                 rollbackFailures.add(String.format("Failed to rollback object <%s> in session <%s>: %s",
435                         id, session.sessionId(), e.getMessage()));
436             }
437         }
438 
439         try {
440             fedoraOcflIndex.rollback(sessionId);
441         } catch (final Exception e) {
442             rollbackFailures.add(String.format("Failed to rollback OCFL index updates in transaction <%s>: %s",
443                     sessionId, e.getMessage()));
444         }
445 
446         //throw an exception if any sessions could not be rolled back.
447         if (rollbackFailures.size() > 0) {
448             state = State.ROLLBACK_FAILED;
449             final StringBuilder builder = new StringBuilder()
450                     .append("Unable to rollback storage session ")
451                     .append(sessionId)
452                     .append(" completely due to the following errors: \n");
453 
454             for (final String failures : rollbackFailures) {
455                 builder.append("\t").append(failures).append("\n");
456             }
457 
458             throw new PersistentStorageException(builder.toString());
459         }
460     }

462     /**
463      * Check if we are in a read-only session.
464      *
465      * @return whether we are read-only (ie. no transaction).
466      */
467     private boolean isReadOnly() {
468         return this.sessionId == null;
469     }
470 
471     /**
472      * Utility to throw exception if trying to perform write operation on read-only session.
473      */
474     private void actionNeedsWrite() throws PersistentStorageException {
475         if (isReadOnly()) {
476             throw new PersistentStorageException("Session is read-only");
477         }
478     }

480     /**
481      * Returns the RDF topic to be returned for a given resource identifier
482      * For example:  passing info:fedora/resource1/fcr:metadata would return
483      *  info:fedora/resource1 since  info:fedora/resource1 would be the expected
484      *  topic.
485      * @param fedoraIdentifier The fedora identifier
486      * @return The resolved topic
487      */
488     private FedoraId/FedoraId.html#FedoraId">FedoraId resolveTopic(final FedoraId fedoraIdentifier) {
489         if (fedoraIdentifier.isDescription()) {
490             return fedoraIdentifier.asBaseId();
491         } else {
492             return fedoraIdentifier;
493         }
494     }
