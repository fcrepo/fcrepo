38      private final ResourceHeadersImpl kernelHeaders;
39      private final org.fcrepo.storage.ocfl.ResourceHeaders.Builder storageHeaders;
40  
41      /**
42       * Default constructor
43       */
44      public ResourceHeadersAdapter() {
45          kernelHeaders = new ResourceHeadersImpl();
46          storageHeaders = org.fcrepo.storage.ocfl.ResourceHeaders.builder();
47      }
48  
49      /**
50       * @param storageHeaders storage headers to adapt
51       */
52      public ResourceHeadersAdapter(final org.fcrepo.storage.ocfl.ResourceHeaders storageHeaders) {
53          this.storageHeaders = org.fcrepo.storage.ocfl.ResourceHeaders
54                  .builder(Objects.requireNonNull(storageHeaders, "storageHeaders cannot be null"));
55          this.kernelHeaders = new ResourceHeadersImpl();
56  
57          if (storageHeaders.getId() != null) {
58              kernelHeaders.setId(FedoraId.create(storageHeaders.getId()));
59          }
60          if (storageHeaders.getParent() != null) {
61              kernelHeaders.setParent(FedoraId.create(storageHeaders.getParent()));
62          }
63          if (storageHeaders.getArchivalGroupId() != null) {
64              kernelHeaders.setArchivalGroupId(FedoraId.create(storageHeaders.getArchivalGroupId()));
65          }
66          kernelHeaders.setArchivalGroup(storageHeaders.isArchivalGroup());
67          kernelHeaders.setContentPath(storageHeaders.getContentPath());
68          kernelHeaders.setContentSize(storageHeaders.getContentSize());
69          kernelHeaders.setCreatedBy(storageHeaders.getCreatedBy());
70          kernelHeaders.setCreatedDate(storageHeaders.getCreatedDate());
71          kernelHeaders.setDeleted(storageHeaders.isDeleted());
72          kernelHeaders.setDigests(new ArrayList<>(storageHeaders.getDigests()));
73          this.storageHeaders.withDigests(kernelHeaders.getDigests());
74          kernelHeaders.setExternalHandling(storageHeaders.getExternalHandling());
75          kernelHeaders.setExternalUrl(storageHeaders.getExternalUrl());
76          kernelHeaders.setFilename(storageHeaders.getFilename());
77          kernelHeaders.setInteractionModel(storageHeaders.getInteractionModel());
78          kernelHeaders.setLastModifiedBy(storageHeaders.getLastModifiedBy());
79          kernelHeaders.setLastModifiedDate(storageHeaders.getLastModifiedDate());
80          kernelHeaders.setMementoCreatedDate(storageHeaders.getMementoCreatedDate());
81          kernelHeaders.setMimeType(storageHeaders.getMimeType());
82          kernelHeaders.setObjectRoot(storageHeaders.isObjectRoot());
83          kernelHeaders.setStateToken(storageHeaders.getStateToken());
84          kernelHeaders.setHeadersVersion(storageHeaders.getHeadersVersion());
85      }
86  
87      /**
88       * @param kernelHeaders kernel headers to adapt
89       */
90      public ResourceHeadersAdapter(final ResourceHeadersImpl kernelHeaders) {
91          this.kernelHeaders = Objects.requireNonNull(kernelHeaders, "kernelHeaders cannot be null");
92          this.storageHeaders = org.fcrepo.storage.ocfl.ResourceHeaders.builder();
93  
94          if (kernelHeaders.getId() != null) {
95              storageHeaders.withId(kernelHeaders.getId().getFullId());
96          }
97          if (kernelHeaders.getParent() != null) {
98              storageHeaders.withParent(kernelHeaders.getParent().getFullId());
99          }
100         if (kernelHeaders.getArchivalGroupId() != null) {
101             storageHeaders.withArchivalGroupId(kernelHeaders.getArchivalGroupId().getFullId());
102         }
103         storageHeaders.withArchivalGroup(kernelHeaders.isArchivalGroup());
104         storageHeaders.withContentPath(kernelHeaders.getContentPath());
105         storageHeaders.withContentSize(kernelHeaders.getContentSize());
106         storageHeaders.withCreatedBy(kernelHeaders.getCreatedBy());
107         storageHeaders.withCreatedDate(kernelHeaders.getCreatedDate());
108         storageHeaders.withDeleted(kernelHeaders.isDeleted());
109         storageHeaders.withDigests(kernelHeaders.getDigests());
110         storageHeaders.withExternalHandling(kernelHeaders.getExternalHandling());
111         storageHeaders.withExternalUrl(kernelHeaders.getExternalUrl());
112         storageHeaders.withFilename(kernelHeaders.getFilename());
113         storageHeaders.withInteractionModel(kernelHeaders.getInteractionModel());
114         storageHeaders.withLastModifiedBy(kernelHeaders.getLastModifiedBy());
115         storageHeaders.withLastModifiedDate(kernelHeaders.getLastModifiedDate());
116         storageHeaders.withMementoCreatedDate(kernelHeaders.getMementoCreatedDate());
117         storageHeaders.withMimeType(kernelHeaders.getMimeType());
118         storageHeaders.withObjectRoot(kernelHeaders.isObjectRoot());
119         storageHeaders.withStateToken(kernelHeaders.getStateToken());
120         storageHeaders.withHeadersVersion(kernelHeaders.getHeadersVersion());
121     }

137     private String idToString(final FedoraId id) {
138         if (id == null) {
139             return null;
140         }
141         return id.getFullId();
142     }
