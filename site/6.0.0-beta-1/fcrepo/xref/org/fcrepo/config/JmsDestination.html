19  package org.fcrepo.config;
20  
21  /**
22   * Indicates if a JMS topic or queue is the destination
23   *
24   * @author pwinckles
25   */
26  public enum JmsDestination {
27  
28      TOPIC("topic"),
29      QUEUE("queue");
30  
31      private final String value;
32  
33      JmsDestination(final String value) {
34          this.value = value;
35      }
36  
37      public String getValue() {
38          return value;
39      }
40  
41      public static JmsDestination fromString(final String value) {
42          for (final var destination : values()) {
43              if (destination.value.equalsIgnoreCase(value)) {
44                  return destination;
45              }
46          }
47          throw new IllegalArgumentException("Unknown JMS destination: " + value);
48      }
49  
50      @Override
51      public String toString() {
52          return value;
53      }
54  
55  }
