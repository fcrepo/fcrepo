18  package org.fcrepo.http.commons.exceptionhandlers;
19  
20  import static org.fcrepo.kernel.api.RdfLexicon.CONSTRAINED_BY;
21  
22  import javax.servlet.ServletContext;
23  import javax.ws.rs.core.Link;
24  import javax.ws.rs.core.UriInfo;
25  import javax.ws.rs.ext.ExceptionMapper;
26  
27  import org.fcrepo.kernel.api.exception.ConstraintViolationException;
28  
29  /**
30   * Abstract class for constraint violation subclasses
31   *
32   * @author whikloj
33   * @since 2015-06-24
34   * @param <T> Throwable subclass of ConstraintViolationException
35   */
36  public abstract class ConstraintExceptionMapper<T extends ConstraintViolationException> implements ExceptionMapper<T> {
37  
38      /**
39       * Where the RDF exception files sit.
40       */
41      private static final String CONSTRAINT_DIR = "/static/constraints/";
42  
43      /**
44       * Creates a constrainedBy link header with the appropriate RDF URL for the exception.
45       *
46       * @param e ConstraintViolationException Exception which implements the buildContraintUri method.
47       * @param context ServletContext ServletContext that we're running in.
48       * @param uriInfo UriInfo UriInfo from the ExceptionMapper.
49       * @return Link A http://www.w3.org/ns/ldp#constrainedBy link header
50       */
51      public static Link buildConstraintLink(final ConstraintViolationException e,
52                                             final ServletContext context,
53                                             final UriInfo uriInfo) {
54          return buildConstraintLink(e.getClass(), context, uriInfo);
55      }
56  
57      /**
58       * Creates a constrainedBy link header with the appropriate RDF URL for the exception.
59       *
60       * @param clazz the class of the exception to build the link for.
61       * @param context ServletContext ServletContext that we're running in.
62       * @param uriInfo UriInfo UriInfo from the ExceptionMapper.
63       * @return Link A http://www.w3.org/ns/ldp#constrainedBy link header
64       */
65      public static Link buildConstraintLink(final Class<? extends ConstraintViolationException> clazz,
66                                             final ServletContext context,
67                                             final UriInfo uriInfo) {
68          String path = context.getContextPath();
69          if (path.equals("/")) {
70              path = "";
71          }
72  
73          final String constraintURI = uriInfo == null ? "" : String.format("%s://%s%s%s%s.rdf",
74                  uriInfo.getBaseUri().getScheme(), uriInfo.getBaseUri().getAuthority(), path,
75                  CONSTRAINT_DIR, clazz.getName().substring(clazz.getName().lastIndexOf('.') + 1));
76          return Link.fromUri(constraintURI).rel(CONSTRAINED_BY.getURI()).build();
77      }
78  
79  }
