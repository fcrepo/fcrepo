18  package org.fcrepo.persistence.ocfl;

20  
21  import org.fcrepo.config.FedoraPropsConfig;
22  import org.fcrepo.config.OcflPropsConfig;
23  import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
24  import org.fcrepo.kernel.api.identifiers.FedoraId;
25  import org.fcrepo.kernel.api.operations.RdfSourceOperation;
26  import org.fcrepo.kernel.api.operations.RdfSourceOperationFactory;
27  import org.fcrepo.kernel.api.operations.VersionResourceOperationFactory;
28  import org.fcrepo.persistence.api.PersistentStorageSession;
29  import org.fcrepo.persistence.api.exceptions.PersistentItemNotFoundException;
30  import org.fcrepo.persistence.api.exceptions.PersistentStorageException;
31  import org.fcrepo.persistence.ocfl.api.IndexBuilder;
32  import org.fcrepo.persistence.ocfl.impl.OcflPersistentSessionManager;
33  import org.slf4j.Logger;
34  import org.slf4j.LoggerFactory;
35  import org.springframework.stereotype.Component;
36  
37  import javax.annotation.PostConstruct;
38  import javax.inject.Inject;
39  
40  import static org.fcrepo.kernel.api.RdfLexicon.BASIC_CONTAINER;
41  
42  /**
43   * This class is responsible for initializing the repository on start-up.
44   *
45   * @author dbernstein
46   */
47  @Component
48  public class RepositoryInitializer {
49  
50      private static final Logger LOGGER = LoggerFactory.getLogger(RepositoryInitializer.class);
51  
52      @Inject
53      private OcflPersistentSessionManager sessionManager;
54  
55      @Inject
56      private RdfSourceOperationFactory operationFactory;
57  
58      @Inject
59      private IndexBuilder indexBuilder;
60  
61      @Inject
62      private VersionResourceOperationFactory versionResourceOperationFactory;
63  
64      @Inject
65      private OcflPropsConfig config;
66  
67      @Inject
68      private FedoraPropsConfig fedoraPropsConfig;
69  
70      /**
71       * Initializes the repository
72       */
73      @PostConstruct
74      public void initialize() {
75          //check that the root is initialized
76          final PersistentStorageSession session = this.sessionManager.getSession("initializationSession" +
77                  System.currentTimeMillis());
78  
79          indexBuilder.rebuildIfNecessary();
80  
81          final var root = FedoraId.getRepositoryRootId();
82  
83          try {
84              try {
85                  session.getHeaders(root, null);
86              } catch (final PersistentItemNotFoundException e) {
87                  LOGGER.info("Repository root ({}) not found. Creating...", root);
88                  final RdfSourceOperation operation = this.operationFactory.createBuilder(root,
89                          BASIC_CONTAINER.getURI(), fedoraPropsConfig.getServerManagedPropsMode())
90                          .parentId(root).build();
91  
92                  session.persist(operation);
93  
94                  //if auto versioning is not enabled, be sure to create an immutable version
95                  if (!config.isAutoVersioningEnabled()) {
96                      final var versionOperation = this.versionResourceOperationFactory.createBuilder(root).build();
97                      session.persist(versionOperation);
98                  }
99                  session.prepare();
100                 session.commit();
101 
102                 LOGGER.info("Successfully created repository root ({}).", root);
103             }
104 
105         } catch (final PersistentStorageException ex) {
106             throw new RepositoryRuntimeException(ex.getMessage(), ex);
107         }
108     }
109 
110 }
