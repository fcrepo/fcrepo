061/**
062 * Service that does the reindexing for one OCFL object.
063 * @author whikloj
064 */
065@Component
066public class ReindexService {
067
068    @Inject
069    private PersistentStorageSessionManager persistentStorageSessionManager;
070
071    @Inject
072    private OcflObjectSessionFactory ocflObjectSessionFactory;
073
074    @Autowired
075    @Qualifier("ocflIndex")
076    private FedoraToOcflObjectIndex ocflIndex;
077
078    @Autowired
079    @Qualifier("containmentIndex")
080    private ContainmentIndex containmentIndex;
081
082    @Autowired
083    @Qualifier("searchIndex")
084    private SearchIndex searchIndex;
085
086    @Autowired
087    @Qualifier("referenceService")
088    private ReferenceService referenceService;
089
090    @Inject
091    private MembershipService membershipService;
092
093    @Inject
094    private ObjectValidator objectValidator;
095
096    @Inject
097    private FedoraPropsConfig config;
098
099    private static final Logger LOGGER = getLogger(ReindexService.class);
100
101    private int membershipPageSize = 500;
102
103    public void indexOcflObject(final String txId, final String ocflId) {
104        LOGGER.debug("Indexing ocflId {} in transaction {}", ocflId, txId);
105
106        objectValidator.validate(ocflId, config.isRebuildFixityCheck());
107
108        try (final var session = ocflObjectSessionFactory.newSession(ocflId)) {
109            final var rootId = new AtomicReference<FedoraId>();
110            final var fedoraIds = new ArrayList<FedoraId>();
111            final var headersList = new ArrayList<ResourceHeaders>();
112
113            session.streamResourceHeaders().forEach(storageHeaders -> {
114                final var headers = new ResourceHeadersAdapter(storageHeaders);
115
116                final var fedoraId = headers.getId();
117                fedoraIds.add(fedoraId);
118                if (headers.isArchivalGroup() || headers.isObjectRoot()) {
119                    rootId.set(fedoraId);
120                }
121
122                if (!fedoraId.isRepositoryRoot()) {
123                    var parentId = headers.getParent();
124
125                    if (headers.getParent() == null) {
126                        if (headers.isObjectRoot()) {
127                            parentId = FedoraId.getRepositoryRootId();
128                        } else {
129                            throw new IllegalStateException(
130                                    String.format("Resource %s must have a parent defined", fedoraId.getFullId()));
131                        }
132                    }
133                    final var created = headers.getCreatedDate();
134                    if (!headers.isDeleted()) {
135                        if (!headers.getInteractionModel().equals(NON_RDF_SOURCE.toString())) {
136                            final Optional<InputStream> content = session.readContent(fedoraId.getFullId())
137                                    .getContentStream();
138                            if (content.isPresent()) {
139                                try (final var stream = content.get()) {
140                                    final RdfStream rdf = parseRdf(fedoraId, stream);
141                                    this.referenceService.updateReferences(txId, fedoraId, null, rdf);
142                                } catch (final IOException e) {
143                                    LOGGER.warn("Content stream for {} closed prematurely, inbound references skipped.",
144                                            fedoraId.getFullId());
145                                    throw new RepositoryRuntimeException(e.getMessage(), e);
146                                }
147                            }
148                        }
149
150                        this.containmentIndex.addContainedBy(txId, parentId, fedoraId, created, null);
151                        headersList.add(headers.asKernelHeaders());
152                    } else {
153                        final var deleted = headers.getLastModifiedDate();
154                        this.containmentIndex.addContainedBy(txId, parentId, fedoraId, created, deleted);
155                    }
156                }
157            });
158
159            if (rootId.get() == null) {
160                throw new IllegalStateException(String.format("Failed to find the root resource in object " +
161                        "identified by %s. Please ensure that the object ID you are attempting to index " +
162                        "refers to a corresponding valid Fedora-flavored object in the OCFL repository. Additionally " +
163                        "be sure that the object ID corresponds with the object root resource (as opposed to child " +
164                        "resources within the object).", ocflId));
165            }
166
167            fedoraIds.forEach(fedoraIdentifier -> {
168                final var rootFedoraIdentifier = rootId.get();
169                ocflIndex.addMapping(txId, fedoraIdentifier, rootFedoraIdentifier, ocflId);
170                LOGGER.debug("Rebuilt fedora-to-ocfl object index entry for {}", fedoraIdentifier);
171            });
172
173            headersList.forEach(headers -> {
174                searchIndex.addUpdateIndex(txId, headers);
175                LOGGER.debug("Rebuilt searchIndex for {}", headers.getId());
176            });
177        }
178    }

180    /**
181     * Remove persistent sessions for a transaction to avoid memory leaks.
182     * @param transactionId the transaction id.
183     */
184    public void cleanupSession(final String transactionId) {
185        persistentStorageSessionManager.removeSession(transactionId);
186    }
187
188    /**
189     * Set the membership page size.
190     * @param pageSize the new page size.
191     */
192    public void setMembershipPageSize(final int pageSize) {
193        membershipPageSize = pageSize;
194    }
195
196    /**
197     * Reset all the indexes.
198     */
199    public void reset() {
200        ocflIndex.reset();
201        containmentIndex.reset();
202        searchIndex.reset();
203        referenceService.reset();
204        membershipService.reset();
205    }
206
207    /**
208     * Commit the records added from transaction.
209     * @param transactionId the id of the transaction.
210     */
211    public void commit(final String transactionId) {
212        try {
213            LOGGER.debug("Performing commit of transaction {}", transactionId);
214            containmentIndex.commitTransaction(transactionId);
215            ocflIndex.commit(transactionId);
216            referenceService.commitTransaction(transactionId);
217            LOGGER.debug("Finished commit of transaction {}", transactionId);
218        } catch (final RuntimeException e) {
219            rollback(transactionId);
220            throw e;
221        }
222    }
223
224    /**
225     * Quietly rollback all changes to the various indexes.
226     * @param transactionId the transaction to rollback.
227     */
228    public void rollback(final String transactionId) {
229        execQuietly("Failed to reset searchIndex", () -> {
230            searchIndex.reset();
231            return null;
232        });
233
234        execQuietly("Failed to rollback containment index transaction " + transactionId, () -> {
235            containmentIndex.rollbackTransaction(transactionId);
236            return null;
237        });
238        execQuietly("Failed to rollback OCFL index transaction " + transactionId, () -> {
239            ocflIndex.rollback(transactionId);
240            return null;
241        });
242
243        execQuietly("Failed to rollback the reference index transaction " + transactionId, () -> {
244            referenceService.rollbackTransaction(transactionId);
245            return null;
246        });
247
248        execQuietly("Failed to rollback membership index transaction " + transactionId, () -> {
249            membershipService.rollbackTransaction(transactionId);
250            return null;
251        });
252    }

254    /**
255     * Index all membership properties by querying for Direct containers, and then
256     * trying population of the membership index for each one
257     * @param txId the transaction id.
258     */
259    public void indexMembership(final String txId) {
260        LOGGER.debug("Starting indexMembership for transaction {}", txId);
261        final var fields = List.of(Condition.Field.FEDORA_ID);
262        final var conditions = List.of(Condition.fromEnums(Condition.Field.RDF_TYPE, Condition.Operator.EQ,
263                RdfLexicon.DIRECT_CONTAINER.getURI()));
264        int offset = 0;
265
266        try {
267            int numResults;
268            do {
269                final var params = new SearchParameters(fields, conditions, membershipPageSize,
270                        offset, Condition.Field.FEDORA_ID, "asc");
271
272                final var searchResult = searchIndex.doSearch(params);
273                final var resultList = searchResult.getItems();
274                numResults = resultList.size();
275
276                resultList.stream()
277                        .map(entry -> FedoraId.create((String) entry.get(Condition.Field.FEDORA_ID.toString())))
278                        .forEach(containerId -> membershipService.populateMembershipHistory(txId, containerId));
279
280                // Results are paged, so step through pages until we reach the last one
281                offset += membershipPageSize;
282            } while (numResults == membershipPageSize);
283
284        } catch (final InvalidQueryException e) {
285            throw new RepositoryRuntimeException("Failed to repopulate membership history", e);
286        }
287        membershipService.commitTransaction(txId);
288        LOGGER.debug("Finished indexMembership for transaction {}", txId);
289    }

291    /**
292     * Executes the closure, capturing all exceptions, and logging them as errors.
293     *
294     * @param failureMessage what to print if the closure fails
295     * @param callable closure to execute
296     */
297    private void execQuietly(final String failureMessage, final Callable<Void> callable) {
298        try {
299            callable.call();
300        } catch (final Exception e) {
301            LOGGER.error(failureMessage, e);
302        }
303    }
304
305    /**
306     * Parse the inputstream from a Rdf resource to a RDFstream.
307     *
308     * @param fedoraIdentifier the resource identifier.
309     * @param inputStream the inputstream.
310     * @return an RdfStream of the resource triples.
311     */
312    private static RdfStream parseRdf(final FedoraId fedoraIdentifier, final InputStream inputStream) {
313        final Model model = createDefaultModel();
314        RDFDataMgr.read(model, inputStream, getRdfFormat().getLang());
315        final FedoraId topic = (fedoraIdentifier.isDescription() ? fedoraIdentifier.asBaseId() : fedoraIdentifier);
316        return DefaultRdfStream.fromModel(createURI(topic.getFullId()), model);
317    }
