41  /**
42   * A very simple abstraction to prevent event-driven machinery downstream from the repository from relying directly
43   * on a JCR interface {@link Event}. Can represent either a single JCR event or several.
44   *
45   * @author ajs6f
46   * @since Feb 19, 2013
47   */
48  public class FedoraEvent {
49  
50      public static final String NODE_TYPE_KEY = "fedora:nodeTypeKey";
51  
52      private Event e;
53  
54      private Set<Integer> eventTypes = new HashSet<>();
55      private Set<String> eventProperties = new HashSet<>();
56  
57      /**
58       * Wrap a JCR Event with our FedoraEvent decorators
59       *
60       * @param e
61       */
62      public FedoraEvent(final Event e) {
63          checkArgument(e != null, "null cannot support a FedoraEvent!");
64          this.e = e;
65      }
66  
67      /**
68       * @return the event types of the underlying JCR {@link Event}s
69       */
70      public Set<Integer> getTypes() {
71          return eventTypes != null ? union(singleton(e.getType()), eventTypes) : singleton(e.getType());
72      }
73  
74      /**
75       * @param type
76       * @return this object for continued use
77       */
78      public FedoraEvent addType(final Integer type) {
79          eventTypes.add(type);
80          return this;
81      }
82  
83      /**
84       * @return the property names of the underlying JCR property {@linkEvent}s
85      **/
86      public Set<String> getProperties() {
87          return eventProperties;
88      }
89  
90      /**
91       * Add a property name to this event
92       * @param property property name
93       * @return this object for continued use
94      **/
95      public FedoraEvent addProperty( final String property ) {
96          eventProperties.add(property);
97          return this;
98      }
99  
100     /**
101      * @return the path of the underlying JCR {@link Event}s
102      */
103     public String getPath() throws RepositoryException {
104         if (e.getType() == PROPERTY_ADDED   ||
105             e.getType() == PROPERTY_CHANGED ||
106             e.getType() == PROPERTY_REMOVED) {
107             return e.getPath().substring(0, e.getPath().lastIndexOf("/"));
108         } else {
109             return e.getPath();
110         }
111     }
112 
113     /**
114      * @return the user ID of the underlying JCR {@link Event}s
115      */
116     public String getUserID() {
117         return e.getUserID();
118     }
119 
120     /**
121      * @return the node identifer of the underlying JCR {@link Event}s
122      */
123     public String getIdentifier() throws RepositoryException {
124         return e.getIdentifier();
125     }
126 
127     /**
128      * @return the info map of the underlying JCR {@link Event}s
129      */
130     @SuppressWarnings("unchecked")
131     public Map<Object, Object> getInfo() throws RepositoryException {
132         return new HashMap<>(e.getInfo());
133     }
134 
135     /**
136      * @return the user data of the underlying JCR {@link Event}s
137      */
138     public String getUserData() throws RepositoryException {
139         return e.getUserData();
140     }
141 
142     /**
143      * @return the date of the underlying JCR {@link Event}s
144      */
145     public long getDate() throws RepositoryException {
146         return e.getDate();
147     }
148 
149     @Override
150     public String toString() {
151         try {
152             return toStringHelper(this).add("Event types:",
153                     Joiner.on(',').join(Iterables.transform(getTypes(), new Function<Integer, String>() {
154 
155                         @Override
156                         public String apply(final Integer type) {
157                             return EventType.valueOf(type).getName();
158                         }
159                     }))).add("Event properties:",
160                     Joiner.on(',').join(eventProperties)).add("Path:", getPath()).add("Date: ",
161                     getDate()).add("Info:", getInfo()).toString();
162         } catch (final RepositoryException e) {
163             throw propagate(e);
164         }
165     }
166 }
