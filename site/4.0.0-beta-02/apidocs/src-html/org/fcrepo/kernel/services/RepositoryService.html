016package org.fcrepo.kernel.services;
017
018import java.io.File;
019import java.util.Map;
020
021import javax.jcr.RepositoryException;
022import javax.jcr.Session;
023
024import org.fcrepo.kernel.rdf.IdentifierTranslator;
025import org.fcrepo.kernel.utils.iterators.RdfStream;
026import org.modeshape.jcr.api.Problems;
027
028import com.hp.hpl.jena.query.Dataset;
029import com.hp.hpl.jena.rdf.model.Resource;
030
031/**
032 * @author bbpennel
033 * @since Feb 20, 2014
034 */
035public interface RepositoryService extends Service {
036
037    /**
038     * Calculate the total size of all the binary properties in the repository
039     *
040     * @return size in bytes
041     */
042    Long getRepositorySize();
043
044    /**
045     * Calculate the number of objects in the repository
046     *
047     * @return number of objects in the repository
048     */
049    Long getRepositoryObjectCount();
050
051    /**
052     * Get a map of JCR prefixes to their URI namespaces
053     *
054     * @param session
055     * @return map of JCR prefixes to their URI namespaces
056     * @throws RepositoryException
057     */
058    Map<String, String> getRepositoryNamespaces(final Session session) throws RepositoryException;
059
060    /**
061     * Serialize the JCR namespace information as an RDF Dataset
062     *
063     * @param session
064     * @return JCR namespace information as an RDF Dataset
065     * @throws RepositoryException
066     */
067    Dataset getNamespaceRegistryDataset(final Session session, final IdentifierTranslator idTranslator)
068        throws RepositoryException;
069
070    /**
071     * Serialize the JCR namespace information as an {@link RdfStream}
072     *
073     * @param session
074     * @return JCR namespace information as an {@link RdfStream}
075     * @throws RepositoryException
076     */
077    RdfStream getNamespaceRegistryStream(final Session session, final IdentifierTranslator idTranslator)
078        throws RepositoryException;
079
080    /**
081     * Perform a full-text search on the whole repository and return the
082     * information as an RDF Dataset
083     *
084     * @param subjectFactory
085     * @param searchSubject RDF resource to use as the subject of the search
086     * @param session
087     * @param terms
088     * @param limit
089     * @param offset
090     * @return full text search results as an RDF Dataset
091     * @throws RepositoryException
092     */
093    Dataset searchRepository(IdentifierTranslator subjectFactory, Resource searchSubject, Session session, String terms,
094            int limit, long offset) throws RepositoryException;
095
096    /**
097     * This method backups up a running repository
098     *
099     * @param session
100     * @param backupDirectory
101     * @return problems
102     * @throws RepositoryException
103     */
104    Problems backupRepository(Session session, File backupDirectory) throws RepositoryException;
105
106    /**
107     * This methods restores the repository from a backup
108     *
109     * @param session
110     * @param backupDirectory
111     * @return problems
112     * @throws RepositoryException
113     */
114    Problems restoreRepository(Session session, File backupDirectory) throws RepositoryException;
115
116}
