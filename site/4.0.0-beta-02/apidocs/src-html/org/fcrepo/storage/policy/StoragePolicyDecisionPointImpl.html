016package org.fcrepo.storage.policy;
017
018import static org.slf4j.LoggerFactory.getLogger;
019
020import org.fcrepo.kernel.services.policy.StoragePolicyDecisionPoint;
021import org.fcrepo.kernel.services.policy.StoragePolicy;
022import org.slf4j.Logger;
023
024import javax.jcr.Node;
025
026import java.util.ArrayList;
027import java.util.List;
028
029/**
030 * Service that evaluates a set of storage policies for an object and provides
031 * storage hints for a binary stream
032 *
033 * @author cbeer
034 * @since Apr 25, 2013
035 */
036public class StoragePolicyDecisionPointImpl implements StoragePolicyDecisionPoint {
037
038    private static final Logger LOGGER = getLogger(StoragePolicyDecisionPointImpl.class);
039
040    private List<StoragePolicy> policies;
041
042    /**
043     * Initialize the policy storage machinery
044     */
045    public StoragePolicyDecisionPointImpl() {
046        LOGGER.debug("Initializing binary StoragePolicyDecisionPointImpl");
047        policies = new ArrayList<>();
048    }
049
050    /*
051     * (non-Javadoc)
052     * @see org.fcrepo.kernel.services.policy.StoragePolicyDecisionPoint#addPolicy(StoragePolicy)
053     */
054    @Override
055    public void addPolicy(final org.fcrepo.kernel.services.policy.StoragePolicy p) {
056        policies.add(p);
057    }
058
059    /*
060     * (non-Javadoc)
061     * @see org.fcrepo.kernel.services.policy.StoragePolicyDecisionPoint#evaluatePolicies(javax.jcr.Node)
062     */
063    @Override
064    public String evaluatePolicies(final Node n) {
065        for (final StoragePolicy p : policies) {
066            final String h = p.evaluatePolicy(n);
067            if (h != null) {
068                return h;
069            }
070        }
071        return null;
072    }
073
074    /*
075     * (non-Javadoc)
076     * @see org.fcrepo.kernel.services.policy.StoragePolicyDecisionPoint#addPolicy(StoragePolicy)
077     */
078    @Override
079    public void removePolicy(final StoragePolicy p) {
080        policies.remove(p);
081    }
082
083    /*
084     * (non-Javadoc)
085     * @see org.fcrepo.kernel.services.policy.StoragePolicyDecisionPoint#addPolicy(StoragePolicy)
086     */
087    @Override
088    public void removeAll() {
089        policies.clear();
090    }
091
092    /**
093     * For now: simple contains implementation
094     *
095     * @param p
096     */
097    @Override
098    public boolean contains(final StoragePolicy p) {
099        return policies.contains(p);
100    }
101
102    /*
103     * (non-Javadoc)
104     * @see org.fcrepo.kernel.services.policy.StoragePolicyDecisionPoint#setPolicies(java.util.List)
105     */
106    @Override
107    public void setPolicies(final List<StoragePolicy> policies) {
108        this.policies = policies;
109    }
110
111    /*
112     * (non-Javadoc)
113     * @see java.lang.Object#toString()
114     */
115    @Override
116    public String toString() {
117        return "policies=" + policies;
118    }
119
120    /**
121     * @return policies size
122     */
123    @Override
124    public int size() {
125        return policies.size();
126    }
127
128    @Override
129    public boolean isEmpty() {
130        return size() == 0;
131    }
132}
