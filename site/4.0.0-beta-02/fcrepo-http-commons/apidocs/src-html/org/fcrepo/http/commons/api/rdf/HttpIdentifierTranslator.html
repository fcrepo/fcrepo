054/**
055 * Translate JCR paths to URLs.  There are a few types of translations
056 * that occur as part of this implementation:
057 * <ul>
058 *     <li>jcr:content is replaced with fcr:content</li>
059 *     <li>information about the transaction is added</li>
060 * </ul>
061 *
062 */
063/**
064 * @author barmintor
065 * @author ajs6f
066 * @since Apr 2, 2014
067 */
068public class HttpIdentifierTranslator extends SpringContextAwareIdentifierTranslator {
069
070    private static final Logger LOGGER = getLogger(HttpIdentifierTranslator.class);
071    public static final String WORKSPACE_PREFIX = "workspace:";
072    public static final String TX_PREFIX = "tx:";
073
074    protected final UriBuilder uriBuilder;
075
076    private final String basePath;
077
078    private final int pathIx;
079
080    private final URI context;
081
082    private final Session session;
083
084    private final String defaultWorkspace;
085
086    private final Function<Repository, String> getDefaultWorkspace = new GetDefaultWorkspace();
087    private final Class<?> relativeTo;
088    private final UriInfo uris;
089    private final boolean canonical;

091    /**
092     * Build HTTP graph subjects relative to the given JAX-RS resource, using the UriInfo provided.
093     *
094     * The session may provide additional information (e.g. workspaces) that need to be
095     * taken into account.
096     *
097     * @param session
098     * @param relativeTo
099     * @param uris
100     */
101    public HttpIdentifierTranslator(final Session session, final Class<?> relativeTo, final UriInfo uris) {
102        this(session, relativeTo, uris, false);
103    }
104
105    /**
106     * Build HTTP graph subjects relative to the given JAX-RS resource, using the UriInfo provided.
107     *
108     * The session may provide additional information (e.g. workspaces) that need to be
109     * taken into account.
110     *
111     * @param session
112     * @param relativeTo
113     * @param uris
114     * @param canonical generate canonical URIs for resources
115     */
116    public HttpIdentifierTranslator(final Session session,
117                                    final Class<?> relativeTo,
118                                    final UriInfo uris,
119                                    final boolean canonical) {
120        this.session = session;
121        this.relativeTo = relativeTo;
122        this.uris = uris;
123        this.canonical = canonical;
124        this.context = uris.getRequestUri();
125        this.uriBuilder = uris.getBaseUriBuilder().clone().path(relativeTo);
126        String normalizedBasePath = uriBuilder.build("").toString();
127        if (!normalizedBasePath.endsWith("/")) {
128            normalizedBasePath = normalizedBasePath + "/";
129        }
130        this.basePath = normalizedBasePath;
131        this.pathIx = normalizedBasePath.length() - 1;
132        this.defaultWorkspace = getDefaultWorkspace.apply(session.getRepository());
133        LOGGER.debug("Resolving graph subjects to a base URI of \"{}\"",
134                normalizedBasePath);
135    }
136
137    /**
138     * Get the canonical IdentifierTranslator (e.g. one that ignores transactions and other transient states)
139     * @param canonical
140     * @return the canonical IdentifierTranslator
141     */
142    public HttpIdentifierTranslator getCanonical(final boolean canonical) {
143        return new HttpIdentifierTranslator(session, relativeTo, uris, canonical);
144    }
145
146    /**
147     * Is the current IdentifierTranslator canonical?
148     * @return true if the current IdentifierTranslator is canonical
149     */
150    public boolean isCanonical() {
151        return canonical || getCurrentTransactionId(session) == null;
152    }

154    @Override
155    public Resource getSubject(final String absPath) throws RepositoryException {
156        resetTranslationChain();
157        try {
158            LOGGER.debug("Creating RDF subject from identifier: {}", decode(absPath, "UTF-8"));
159            return doForward(decode(absPath, "UTF-8"));
160        } catch ( UnsupportedEncodingException ex ) {
161            LOGGER.warn("Required encoding (UTF-8) not supported, trying undecoded path",ex);
162            return doForward(absPath);
163        }
164    }
165
166    @Override
167    public Resource getContext() {
168        return createResource(context.toString());
169    }

171    private static String getResourceURI(final Resource subject) {
172        if (!subject.isURIResource()) {
173            LOGGER.debug("RDF resource {} was not a URI resource: returning null.", subject);
174            return null;
175        }
176        return subject.getURI();
177    }
178
179    @Override
180    public String getPathFromSubject(final Resource subject) throws RepositoryException {
181        resetTranslationChain();
182        return doBackward(subject);
183    }
184
185    /**
186     * Gets a path from the graph subject's URI.  This method does the heavy
187     * lifting for getNodeFromGraphSubject and getPathFromGraphSubject in a way
188     * that's tied to a more generic URI rather than a rdf Resource.
189     */
190    protected String getPathFromGraphSubject(@NotNull final String subjectUri) throws RepositoryException {
191
192        if (!isFedoraGraphSubject(subjectUri)) {
193            LOGGER.debug("RDF resource {} was not a URI resource with our expected basePath {}, returning null.",
194                    subjectUri, basePath);
195            return null;
196        }
197
198        final StringBuilder pathBuilder = new StringBuilder();
199        final String absPath;
200        final String[] pathSegments =
201                subjectUri.substring(pathIx).split("/");
202
203        for (final String segment : pathSegments) {
204            if (segment.startsWith(TX_PREFIX)) {
205                if (!canonical) {
206                    final String tx = segment.substring(TX_PREFIX.length());
207                    final String currentTxId = getCurrentTransactionId(session);
208
209                    if (currentTxId == null || !tx.equals(currentTxId)) {
210                        throw new RepositoryException("Subject is not in this transaction");
211                    }
212                }
213            } else if (segment.startsWith(WORKSPACE_PREFIX)) {
214                final String workspace = segment.substring(WORKSPACE_PREFIX.length());
215                if (!session.getWorkspace().getName().equals(workspace)) {
216                    throw new RepositoryException("Subject is not in this workspace");
217                }
218            } else if (segment.equals(FCR_CONTENT)) {
219                pathBuilder.append("/");
220                pathBuilder.append(JCR_CONTENT);
221            } else {
222                if (!segment.isEmpty()) {
223                    pathBuilder.append("/");
224                    pathBuilder.append(segment);
225                }
226            }
227        }
228
229        if (pathBuilder.length() == 0) {
230            absPath = "/";
231        } else {
232            absPath = pathBuilder.toString();
233        }
234
235        if (isValidJcrPath(absPath)) {
236            return absPath;
237        }
238        return null;
239    }
240
241    private boolean isValidJcrPath(final String absPath) {
242        try {
243            String pathToValidate = absPath;
244            final String txId = getCurrentTransactionId(session);
245            if (!canonical && txId != null) {
246                final String txIdWithSlash = "/" + TX_PREFIX + txId;
247                /* replace the first occurrence of tx within the path */
248                pathToValidate = replaceOnce(absPath, txIdWithSlash, EMPTY);
249                LOGGER.debug("removed {} from URI {}. Path for JCR validation is now: {}", txIdWithSlash, absPath,
250                        pathToValidate);
251            }
252            session.getValueFactory().createValue(pathToValidate, PATH);
253            return true;
254        } catch (final ValueFormatException e) {
255            LOGGER.trace("Unable to validate JCR path: ", e);
256            return false;
257        } catch (final RepositoryException e) {
258            throw new RepositoryRuntimeException("Unable to validate JCR path: ", e);
259        }
260    }
261
262    @Override
263    public boolean isFedoraGraphSubject(final Resource subject) {
264        return subject.isURIResource() && isFedoraGraphSubject(subject.getURI());
265    }
266
267    private boolean isFedoraGraphSubject(final String subjectUri) {
268        return subjectUri.startsWith(basePath) && isValidJcrPath(subjectUri.substring(pathIx));
269    }
270
271    private Map<String, String> getPathMap(final String absPath) {
272        // the path param value doesn't start with a slash
273        String path = absPath.substring(1);
274        if (session != null) {
275            final Workspace workspace = session.getWorkspace();
276
277            final String txId = getCurrentTransactionId(session);
278
279            if (!canonical && txId != null) {
280                path = TX_PREFIX + txId + "/" + path;
281            } else if (workspace != null && !workspace.getName().equals(defaultWorkspace)) {
282                path = WORKSPACE_PREFIX + workspace.getName() + "/" + path;
283            }
284        }
285
286        return singletonMap("path", path);
287    }
288
289    /* (non-Javadoc)
290     * @see org.fcrepo.kernel.identifiers.ExternalIdentifierConverter#doRdfForward(java.lang.String)
291     */
292    @Override
293    protected Resource doRdfForward(final String inputId) {
294        final URI result = uriBuilder.buildFromMap(getPathMap(inputId));
295        return createResource(result.toString());
296    }
297
298    /* (non-Javadoc)
299     * @see org.fcrepo.kernel.identifiers.ExternalIdentifierConverter#doRdfBackward(com.hp.hpl.jena.rdf.model.Resource)
300     */
301    @Override
302    protected String doRdfBackward(final Resource subject) {
303        final String subjectUri = getResourceURI(subject);
304        if (subjectUri == null) {
305            return null;
306        }
307        try {
308            return getPathFromGraphSubject(subjectUri);
309        } catch (final RepositoryException e) {
310            throw propagate(e);
311        }
312    }
313
314
315    protected void resetTranslationChain() {
316        if (translationChain == null) {
317            if (getTranslationChain() != null) {
318                setTranslationChain(getTranslationChain());
319            } else {
320                setTranslationChain(minimalTranslationChain);
321            }
322        }
323    }
324
325    private static final List<InternalIdentifierConverter> minimalTranslationChain =
326        singletonList((InternalIdentifierConverter) new NamespaceConverter());
327
328    @Override
329    public String getBaseUri() {
330        return uris.getBaseUri().toString();
331    }
