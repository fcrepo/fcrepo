066    /**
067     * Predicate for determining whether this {@link Node} is a Fedora resource.
068     */
069    public static Predicate<Node> isFedoraResource = new Predicate<Node>() {
070
071        @Override
072        public boolean apply(final Node node) {
073            checkNotNull(node, "null cannot be a Fedora object!");
074            try {
075                return map(node.getMixinNodeTypes(), nodetype2name).contains(
076                        FEDORA_RESOURCE);
077            } catch (final RepositoryException e) {
078                throw propagate(e);
079            }
080        }
081    };

083    /**
084     * Predicate for determining whether this {@link Node} is a frozen node
085     * (a part of the system version history).
086     */
087    public static Predicate<Node> isFrozen = new Predicate<Node>() {
088
089        @Override
090        public boolean apply(final Node node) {
091            checkNotNull(node, "null cannot be a Frozen node!");
092            try {
093                return node.getPrimaryNodeType().getName().equals(FROZEN_NODE);
094            } catch (final RepositoryException e) {
095                throw propagate(e);
096            }
097        }
098    };

133    /**
134     * Translates a JCR {@link Value} to its {@link String} expression.
135     */
136    public static Function<Value, String> value2string =
137        new Function<Value, String>() {
138
139            @Override
140            public String apply(final Value v) {
141                try {
142                    checkNotNull(v, "null has no appropriate "

144                    return v.getString();
145                } catch (final RepositoryException e) {
146                    throw propagate(e);
147                }
148            }

151    /**
152     * Constructs an {@link Iterator} of {@link Value}s from any
153     * {@link Property}, multi-valued or not.
154     */
155    public static Function<Property, Iterator<Value>> property2values =
156        new Function<Property, Iterator<Value>>() {
157
158            @Override
159            public Iterator<Value> apply(final Property p) {
160                try {
161                    if (p.isMultiple()) {
162                        LOGGER.debug("Found multi-valued property: {}", p);
163                        return Iterators.forArray(p.getValues());
164                    }
165                    LOGGER.debug("Found single-valued property: {}", p);
166                    return Iterators.forArray(p.getValue());
167                } catch (final RepositoryException e) {
168                    throw propagate(e);
169                } catch (final Exception e) {
170                    throw propagate(e);
171                }
172            }
173        };

209    /**
210     * Check if a property is a reference property.
211     */
212    public static Predicate<Property> isReferenceProperty =
213        new Predicate<Property>() {
214
215            @Override
216            public boolean apply(final Property p) {
217                try {
218                    return (p.getType() == REFERENCE || p.getType() == WEAKREFERENCE)
219                        && p.getName().endsWith(REFERENCE_PROPERTY_SUFFIX);
220                } catch (final RepositoryException e) {
221                    throw propagate(e);
222                }
223            }
224        };
225
226    /**
227    * Check whether a property is an internal property that should be suppressed
228    * from external output.
229    */
230    public static Predicate<Property> isInternalProperty =
231        new Predicate<Property>() {
232
233            @Override
234            public boolean apply(final Property p) {
235                return isReferenceProperty.apply(p) || isBinaryContentProperty.apply(p)
236                        || isProtectedAndShouldBeHidden.apply(p);
237            }
238        };
239
240    /**
241     * Check whether a property is protected (ie, cannot be modified directly) but
242     * is not one we've explicitly chosen to include.
243     */
244    public static Predicate<Property> isProtectedAndShouldBeHidden =
245        new Predicate<Property>() {
246
247            @Override
248            public boolean apply(final Property p) {
249                try {
250                    if (!p.getDefinition().isProtected()) {
251                        return false;
252                    } else if (p.getParent().isNodeType(FROZEN_NODE)) {
253                        // everything on a frozen node is protected
254                        // but we wish to display it anyway and there's
255                        // another mechanism in place to make clear that
256                        // things cannot be edited.
257                        return false;
258                    } else {
259                        final String name = p.getName();
260                        for (String exposedName : EXPOSED_PROTECTED_JCR_TYPES) {
261                            if (name.equals(exposedName)) {
262                                return false;
263                            }
264                        }
265                        return true;
266                    }
267                } catch (RepositoryException e) {
268                    throw propagate(e);
269                }
270            }
271        };

273    /**
274     * Check if a node is "internal" and should not be exposed e.g. via the REST
275     * API
276     */
277    public static Predicate<Node> isInternalNode = new Predicate<Node>() {
278
279        @Override
280        public boolean apply(final Node n) {
281            checkNotNull(n, "null is neither internal nor not internal!");
282            try {
283                final NodeType primaryNodeType = n.getPrimaryNodeType();
284                return primaryNodeType != null
285                        && primaryNodeType.isNodeType("mode:system");
286            } catch (final RepositoryException e) {
287                throw propagate(e);
288            }
289        }
290    };

309    /**
310     * Get the property definition information (containing type and multi-value
311     * information)
312     *
313     * @param node the node to use for inferring the property definition
314     * @param propertyName the property name to retrieve a definition for
315     * @return a JCR PropertyDefinition, if available, or null
316     * @throws javax.jcr.RepositoryException
317     */
318    public static PropertyDefinition getDefinitionForPropertyName(final Node node,
319        final String propertyName) throws RepositoryException {
320
321        final PropertyDefinition[] propertyDefinitions =
322            node.getPrimaryNodeType().getPropertyDefinitions();
323        LOGGER.debug("Looking for property name: {}", propertyName);
324        for (final PropertyDefinition p : propertyDefinitions) {
325            LOGGER.debug("Checking property: {}", p.getName());
326            if (p.getName().equals(propertyName)) {
327                return p;
328            }
329        }
330
331        for (final NodeType nodeType : node.getMixinNodeTypes()) {
332            for (final PropertyDefinition p : nodeType.getPropertyDefinitions()) {
333                if (p.getName().equals(propertyName)) {
334                    return p;
335                }
336            }
337        }
338        return null;
339    }

341    /**
342     * Convenience method for transforming arrays into {@link Collection}s
343     * through a mapping {@link Function}.
344     *
345     * @param input A Collection<F>.
346     * @param f A Function<F,T>.
347     * @return An ImmutableSet copy of input after transformation by f
348     */
349    public static <F, T> Collection<T> map(final F[] input,
350            final Function<F, T> f) {
351        return transform(copyOf(input), f);
352    }

354    /**
355     * @param date Instance of java.util.Date.
356     * @return the lexical form of the XSD dateTime value, e.g.
357     *         "2006-11-13T09:40:55.001Z".
358     */
359    public static String convertDateToXSDString(final long date) {
360        final DateTime dt = new DateTime(date, DateTimeZone.UTC);
361        return FMT.print(dt);
362    }

378    /**
379     * Check if the property contains the given string value
380     * 
381     * @param p
382     * @param value
383     * @return true if the property contains the given string value
384     */
385    public static boolean propertyContains(final Property p, final String value) throws RepositoryException {
386
387        if (p == null) {
388            return false;
389        }
390
391        if (p.isMultiple()) {
392            return contains(transform(forArray(p.getValues()), value2string), value);
393        }
394        return value.equals(p.getString());
395
396    }
