018package org.fcrepo.kernel.modeshape.utils;
019
020import java.io.InputStream;
021import java.security.DigestInputStream;
022import java.security.MessageDigest;
023
024import org.apache.commons.io.input.CountingInputStream;
025
026/**
027 * An InputStream wrapper that calculates the size and digest
028 * while reading from the stream.
029 * @author Chris Beer
030 * @since Mar 12, 2013
031 */
032public class FixityInputStream extends CountingInputStream {
033
034    /**
035     * Creates a <code>FilterInputStream</code> by assigning the
036     * argument <code>in</code> to the field <code>this.in</code>
037     * so as to remember it for later use.
038     *
039     * @param in the underlying input stream, or <code>null</code> if
040     *           this instance is to be created without an underlying stream.
041     * @param digest the given digest
042     */
043    public FixityInputStream(final InputStream in, final MessageDigest digest) {
044        super(new DigestInputStream(in, digest));
045    }
046
047    /**
048     * Retrieve the calculated digest for the input stream
049     * @return digest for this input stream
050     */
051    public MessageDigest getMessageDigest() {
052        return ((DigestInputStream) in).getMessageDigest();
053    }
054
055}
