043    public void testCreateTransaction() throws Exception {
044        /* create a tx */
045        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
046        final HttpResponse response = execute(createTx);
047        assertEquals(201, response.getStatusLine().getStatusCode());
048
049        final String location = response.getFirstHeader("Location").getValue();

051        logger.info("Got location {}", location);
052        assertTrue(

054                compile("tx:[0-9a-f-]+$").matcher(location).find());

068    @Test
069    public void testCreateAndTimeoutTransaction() throws Exception {
070
071        /* create a short-lived tx */
072        final long testTimeout =
073                Math.min(500, TransactionService.REAP_INTERVAL / 2);
074        System.setProperty(Transaction.TIMEOUT_SYSTEM_PROPERTY, Long
075                .toString(testTimeout));
076
077        /* create a tx */
078        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
079        final HttpResponse response = execute(createTx);
080        assertEquals(201, response.getStatusLine().getStatusCode());
081
082        final String location = response.getFirstHeader("Location").getValue();
083
084        final HttpGet getWithinTx = new HttpGet(location);
085        HttpResponse resp = execute(getWithinTx);
086        IOUtils.toString(resp.getEntity().getContent());
087        assertEquals(200, resp.getStatusLine().getStatusCode());
088
089        int statusCode = 0;
090
091        Thread.sleep(TransactionService.REAP_INTERVAL * 2);
092        final HttpGet getWithExpiredTx = new HttpGet(location);
093        resp = execute(getWithExpiredTx);
094        IOUtils.toString(resp.getEntity().getContent());
095        statusCode = resp.getStatusLine().getStatusCode();
096
097        try {
098            assertEquals("Transaction did not expire", 410, statusCode);
099        } finally {
100            System.setProperty(Transaction.TIMEOUT_SYSTEM_PROPERTY, Long
101                    .toString(Transaction.DEFAULT_TIMEOUT));
102            System.clearProperty("fcrepo4.tx.timeout");
103        }
104    }

106    @Test
107    public void testCreateDoStuffAndRollbackTransaction() throws Exception {
108        /* create a tx */
109        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
110
111        final HttpResponse response = execute(createTx);
112        assertEquals(201, response.getStatusLine().getStatusCode());
113
114        final String txLocation =
115                response.getFirstHeader("Location").getValue();
116
117        /* create a new object inside the tx */
118        final HttpPost postNew =
119                new HttpPost(txLocation + "/object-in-tx-rollback");
120        HttpResponse resp = execute(postNew);
121        assertEquals(201, resp.getStatusLine().getStatusCode());
122
123        /* fetch the created tx from the endpoint */
124        final HttpGet getTx =
125                new HttpGet(txLocation + "/object-in-tx-rollback");
126        getTx.setHeader("Accept", "application/n3");
127        resp = execute(getTx);
128        assertEquals(
129                "Expected to find our object within the scope of the transaction",
130                200, resp.getStatusLine().getStatusCode());
131
132        final GraphStore graphStore =
133                TestHelpers.parseTriples(resp.getEntity().getContent());
134        logger.debug(graphStore.toString());
135
136        assertTrue(graphStore.toDataset().asDatasetGraph().contains(Node.ANY,
137                NodeFactory.createURI(txLocation + "/object-in-tx-rollback"),
138                Node.ANY, Node.ANY));
139
140        /* fetch the created tx from the endpoint */
141        final HttpGet getObj =
142                new HttpGet(serverAddress + "/object-in-tx-rollback");
143        resp = execute(getObj);
144        assertEquals(
145                "Expected to not find our object within the scope of the transaction",
146                404, resp.getStatusLine().getStatusCode());
147
148        /* and rollback */
149        final HttpPost rollbackTx =
150                new HttpPost(txLocation + "/fcr:tx/fcr:rollback");
151        resp = execute(rollbackTx);
152
153        assertEquals(204, resp.getStatusLine().getStatusCode());
154
155    }

157    @Test
158    public void testCreateDoStuffAndCommitTransaction() throws Exception {
159        /* create a tx */
160
161        final String objectInTxCommit = randomUUID().toString();
162
163        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
164
165        final HttpResponse response = execute(createTx);
166        assertEquals(201, response.getStatusLine().getStatusCode());
167
168        final String txLocation =
169                response.getFirstHeader("Location").getValue();
170
171        /* create a new object inside the tx */
172        final HttpPost postNew =
173                new HttpPost(txLocation + "/" + objectInTxCommit);
174        HttpResponse resp = execute(postNew);
175        assertEquals(201, resp.getStatusLine().getStatusCode());
176
177        /* fetch the created tx from the endpoint */
178        final HttpGet getTx = new HttpGet(txLocation + "/" + objectInTxCommit);
179        getTx.setHeader("Accept", "application/n3");
180        resp = execute(getTx);
181        assertEquals(
182                "Expected to find our object within the scope of the transaction",
183                200, resp.getStatusLine().getStatusCode());
184
185        final GraphStore graphStore =
186            parseTriples(resp.getEntity().getContent());
187        logger.debug(graphStore.toString());
188
189        assertTrue(graphStore.toDataset().asDatasetGraph().contains(ANY,
190                createURI(txLocation + "/" + objectInTxCommit), ANY, ANY));
191
192        /* fetch the object-in-tx outside of the tx */
193        final HttpGet getObj =
194                new HttpGet(serverAddress + "/" + objectInTxCommit);
195        resp = execute(getObj);
196        assertEquals(
197                "Expected to not find our object within the scope of the transaction",
198                404, resp.getStatusLine().getStatusCode());
199
200        /* and rollback */
201        final HttpPost commitTx =
202                new HttpPost(txLocation + "/fcr:tx/fcr:commit");
203        resp = execute(commitTx);
204
205        assertEquals(204, resp.getStatusLine().getStatusCode());
206
207        /* fetch the object-in-tx outside of the tx after it has been committed */
208        final HttpGet getObjCommitted =
209                new HttpGet(serverAddress + "/" + objectInTxCommit);
210        resp = execute(getObjCommitted);
211        assertEquals(
212                "Expected to  find our object after the transaction was committed",
213                200, resp.getStatusLine().getStatusCode());
214
215    }
