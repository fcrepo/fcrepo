043    @Test
044    public void testMultipleDatastreams() throws Exception {
045        final HttpPost createObjMethod =
046                postObjMethod("FedoraDatastreamsTest7");
047        assertEquals(201, getStatus(createObjMethod));
048
049        final HttpPost createDS1Method =
050                postDSMethod("FedoraDatastreamsTest7", "ds1",
051                        "marbles for everyone");
052        assertEquals(201, getStatus(createDS1Method));
053        final HttpPost createDS2Method =
054                postDSMethod("FedoraDatastreamsTest7", "ds2",
055                        "marbles for no one");
056        assertEquals(201, getStatus(createDS2Method));
057
058        final HttpGet getDSesMethod =
059                new HttpGet(serverAddress + "FedoraDatastreamsTest7");
060        getDSesMethod.addHeader("Accept", "text/n3");
061        final HttpResponse response = client.execute(getDSesMethod);
062        assertEquals(200, response.getStatusLine().getStatusCode());
063        final GraphStore result =
064                TestHelpers.parseTriples(response.getEntity().getContent());
065        logger.debug("Received triples: \n{}", result.toString());
066        final String subjectURI =
067                serverAddress + "FedoraDatastreamsTest7";
068
069        assertTrue("Didn't find the first datastream! ", result.contains(ANY,
070                createURI(subjectURI), ANY, createURI(subjectURI + "/ds1")));
071        assertTrue("Didn't find the second datastream! ", result.contains(ANY,
072                createURI(subjectURI), ANY, createURI(subjectURI + "/ds2")));
073    }

075    @Test
076    public void testModifyMultipleDatastreams() throws Exception {
077        final HttpPost objMethod = postObjMethod("FedoraDatastreamsTest8");
078
079        assertEquals(201, getStatus(objMethod));
080
081        final HttpPost createDSVOIDMethod =
082                postDSMethod("FedoraDatastreamsTest8", "ds_void",
083                        "marbles for everyone");
084        assertEquals(201, getStatus(createDSVOIDMethod));
085
086        final HttpPost post =
087                new HttpPost(serverAddress +
088                        "FedoraDatastreamsTest8/fcr:datastreams?delete=ds_void");
089
090        final MultipartEntity multiPartEntity = new MultipartEntity();
091        multiPartEntity.addPart("ds1", new StringBody("asdfg"));
092        multiPartEntity.addPart("ds2", new StringBody("qwerty"));
093
094        post.setEntity(multiPartEntity);
095
096        final HttpResponse postResponse = client.execute(post);
097
098        assertEquals(201, postResponse.getStatusLine().getStatusCode());
099
100        final HttpGet getDSesMethod =
101                new HttpGet(serverAddress + "FedoraDatastreamsTest8");
102        getDSesMethod.addHeader("Accept", "text/n3");
103        final HttpResponse response = client.execute(getDSesMethod);
104        assertEquals(200, response.getStatusLine().getStatusCode());
105
106        final String subjectURI =
107                serverAddress + "FedoraDatastreamsTest8";
108        final GraphStore result =
109                parseTriples(response.getEntity().getContent());
110        assertTrue("Didn't find the first datastream! ", result.contains(ANY,
111                createURI(subjectURI), ANY, createURI(subjectURI + "/ds1")));
112        assertTrue("Didn't find the second datastream! ", result.contains(ANY,
113                createURI(subjectURI), ANY, createURI(subjectURI + "/ds2")));
114        assertFalse("Found the deleted datastream! ", result.contains(Node.ANY,
115                createURI(subjectURI), ANY, createURI(subjectURI + "/ds_void")));
116
117    }

121
122        final HttpPost objMethod = postObjMethod("FedoraDatastreamsTest9");
123        assertEquals(201, getStatus(objMethod));
124        final HttpPost post =
125                new HttpPost(serverAddress +
126                        "FedoraDatastreamsTest9/fcr:datastreams/");
127
128        final MultipartEntity multiPartEntity = new MultipartEntity();
129        multiPartEntity.addPart("ds1", new StringBody("asdfg"));
130        multiPartEntity.addPart("ds2", new StringBody("qwerty"));
131
132        post.setEntity(multiPartEntity);
133
134        final HttpResponse postResponse = client.execute(post);
135        assertEquals(201, postResponse.getStatusLine().getStatusCode());
136
137        // TODO: we should actually evaluate the multipart response for the
138        // things we're expecting
139        final HttpGet getDSesMethod =
140                new HttpGet(serverAddress +
141                        "FedoraDatastreamsTest9/fcr:datastreams");
142        final HttpResponse response = client.execute(getDSesMethod);
143        assertEquals(200, response.getStatusLine().getStatusCode());
144        final String content = EntityUtils.toString(response.getEntity());
145
146        assertTrue("Didn't find the first datastream!",
147                compile("asdfg", DOTALL).matcher(content).find());
148        assertTrue("Didn't find the second datastream!", compile("qwerty",
149                DOTALL).matcher(content).find());

189        execute(postObjMethod("FedoraDatastreamsTest12"));
190        final HttpPost method1 =
191                postDSMethod("FedoraDatastreamsTest12", "ds1", "foo1");
192        assertEquals(201, getStatus(method1));
193        final HttpPost method2 =
194                postDSMethod("FedoraDatastreamsTest12", "ds2", "foo2");
195        assertEquals(201, getStatus(method2));
196
197        final HttpDelete dmethod =
198                new HttpDelete(serverAddress +
199                        "FedoraDatastreamsTest12/fcr:datastreams?dsid=ds1&dsid=ds2");
200        assertEquals(204, getStatus(dmethod));
201
202        final HttpGet method_test_get1 =
203                new HttpGet(serverAddress +
204                        "FedoraDatastreamsTest12/ds1");
205        assertEquals(404, getStatus(method_test_get1));
206        final HttpGet method_test_get2 =
207                new HttpGet(serverAddress +
208                        "FedoraDatastreamsTest12/ds2");
209        assertEquals(404, getStatus(method_test_get2));
