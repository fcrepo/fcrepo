188    @Test
189    public void testGetFixity() throws BinaryStoreException, IOException,
190            NoSuchAlgorithmException {
191        final LowLevelCacheEntry ispnEntry =
192                new LocalBinaryStoreEntry(mockStore, testKey);
193        final byte[] bytes = new byte[] {0, 1, 2, 3, 4};
194        when(mockStore.getInputStream(testKey)).thenAnswer(
195                new Answer<InputStream>() {
196
197                    @Override
198                    public InputStream
199                            answer(final InvocationOnMock invocation)
200                                    throws Throwable {
201                        return new ByteArrayInputStream(bytes);
202                    }
203
204                });
205        final MessageDigest d = MessageDigest.getInstance("SHA-1");
206        final byte[] digested = d.digest(bytes);
207        URI testCS = ContentDigest.asURI("SHA-1", digested);
208        LOGGER.debug(testCS.toString());
209        FixityResult actual = ispnEntry.checkFixity(testCS, bytes.length);
210        assertEquals(1, actual.status.size());
211        assertEquals(actual.status.iterator().next().toString(), true,
212                actual.status.contains(SUCCESS));
213
214        // report the wrong size
215        actual = ispnEntry.checkFixity(testCS, bytes.length + 1);
216        assertEquals(1, actual.status.size());
217        assertEquals(actual.status.iterator().next().toString(), true,
218                actual.status.contains(BAD_SIZE));
219        // break the digest
220        digested[0] += 9;
221        testCS = ContentDigest.asURI("SHA-1", digested);
222        actual = ispnEntry.checkFixity(testCS, bytes.length);
223        assertEquals(1, actual.status.size());
224        assertEquals(actual.status.iterator().next().toString(), true,
225                actual.status.contains(BAD_CHECKSUM));
226        // report the wrong size and the wrong digest
227        actual = ispnEntry.checkFixity(testCS, bytes.length + 1);
228        assertEquals(2, actual.status.size());
229        assertEquals(true, actual.status.contains(BAD_CHECKSUM));
230        assertEquals(true, actual.status.contains(BAD_SIZE));
231    }
