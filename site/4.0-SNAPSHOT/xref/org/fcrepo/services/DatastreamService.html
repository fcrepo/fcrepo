44      public static Node createDatastreamNode(final Session session,
45              final String dsPath, final String contentType,
46              final InputStream requestBodyStream) throws RepositoryException,
47              IOException {
48  
49          final Node ds = jcrTools.findOrCreateNode(session, dsPath, NT_FILE);
50          ds.addMixin(FEDORA_DATASTREAM);
51  
52          final Node contentNode =
53                  jcrTools.findOrCreateChild(ds, JCR_CONTENT, NT_RESOURCE);
54          logger.debug("Created content node at path: " + contentNode.getPath());
55          /*
56           * This next line of code deserves explanation. If we chose for the
57           * simpler line:
58           * Property dataProperty = contentNode.setProperty(JCR_DATA,
59           * requestBodyStream);
60           * then the JCR would not block on the stream's completion, and we would
61           * return to the requester before the mutation to the repo had actually
62           * completed. So instead we use createBinary(requestBodyStream), because
63           * its contract specifies:
64           * "The passed InputStream is closed before this method returns either
65           * normally or because of an exception."
66           * which lets us block and not return until the job is done! The simpler
67           * code may still be useful to us for an asynchronous method that we
68           * develop later.
69           */
70          Property dataProperty =
71                  contentNode.setProperty(JCR_DATA, session.getValueFactory()
72                          .createBinary(requestBodyStream));
73          logger.debug("Created data property at path: " + dataProperty.getPath());
74  
75          ds.setProperty("fedora:contentType", contentType);
76  
77          ds.addMixin(FEDORA_OWNED);
78          ds.setProperty("fedora:ownerId", session.getUserID());
79  
80          if (!ds.hasProperty("fedora:created")) {
81              ds.setProperty("fedora:created", Calendar.getInstance());
82          }
83          ds.setProperty("jcr:lastModified", Calendar.getInstance());
84  
85          // TODO: I guess we should also have the PID + DSID..
86          ds.setProperty(DC_IDENTIFER, new String[] {ds.getIdentifier(),
87                  ds.getParent().getName() + "/" + ds.getName()});
88  
89          return ds;
90      }

92      public static Node getDatastreamNode(final String pid, final String dsId)
93              throws PathNotFoundException, RepositoryException {
94          logger.trace("Executing getDatastreamNode() with pid: " + pid +
95                  " and dsId: " + dsId);
96          final Node objNode = getObjectNode(pid);
97          logger.trace("Retrieved object node: " + objNode.getName());
98          final Node dsNode = objNode.getNode(dsId);
99          logger.trace("Retrieved datastream node: " + dsNode.getName());
100         return dsNode;
101     }

103     public static Datastream getDatastream(final String pid, final String dsId)
104             throws PathNotFoundException, RepositoryException {
105         return new Datastream(getObjectNode(pid).getNode(dsId));
