37  @Path("/rss")
38  public class RSSPublisher extends AbstractResource {
39  
40      private static final Integer FEED_LENGTH = 10;
41  
42      private static final String FEED_TYPE = "rss_2.0";
43  
44      private static final String FEED_TITLE = "What's happening in Fedora";
45  
46      private static final String FEED_DESCRIPTION = FEED_TITLE;
47  
48      @Inject
49      EventBus eventBus;
50  
51      private BlockingQueue<Event> feedQueue = new ArrayBlockingQueue<Event>(
52              FEED_LENGTH);
53  
54      private SyndFeed feed = new SyndFeedImpl();
55  
56      @GET
57      @Path("")
58      @Produces("application/rss+xml")
59      public StreamSource getFeed() throws FeedException {
60          feed.setLink(uriInfo.getBaseUri().toString());
61          feed.setEntries(transform(copyOf(feedQueue), event2entry));
62          // TODO ought to make this stream, not go through a string
63          return new StreamSource(new ByteArrayInputStream(new SyndFeedOutput()
64                  .outputString(feed).getBytes()));
65      }
66  
67      private Function<Event, SyndEntry> event2entry =
68              new Function<Event, SyndEntry>() {
69  
70                  @Override
71                  public SyndEntry apply(Event event) {
72                      SyndEntry entry = new SyndEntryImpl();
73                      try {
74                          entry.setTitle(event.getIdentifier());
75                          entry.setLink(event.getPath());
76                          entry.setPublishedDate(new DateTime(event.getDate())
77                                  .toDate());
78                          SyndContent description = new SyndContentImpl();
79                          description.setType("text/plain");
80                          description.setValue(getEventType(event.getType())
81                                  .toString());
82                          entry.setDescription(description);
83                      } catch (RepositoryException e) {
84                          throw new SystemFailureException(e);
85                      }
86                      return entry;
87                  }
88  
89              };
90  
91      @PostConstruct
92      public void initialize() {
93          eventBus.register(this);
94          feed.setFeedType(FEED_TYPE);
95          feed.setTitle(FEED_TITLE);
96          feed.setDescription(FEED_DESCRIPTION);
97      }
98  
99      @Subscribe
100     public void newEvent(Event event) {
101         if (feedQueue.remainingCapacity() > 0) {
102             feedQueue.offer(event);
103         } else {
104             feedQueue.poll();
105             feedQueue.offer(event);
106         }
107     }
