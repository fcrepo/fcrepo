65      @PostConstruct
66      public void initialize() throws LoginException, NoSuchWorkspaceException,
67              RepositoryException {
68  
69          final Session session = repo.login();
70          session.getWorkspace().getNamespaceRegistry().registerNamespace("test",
71                  "info:fedora/test");
72          Node objects = jcrTools.findOrCreateNode(session, "/objects");
73          objects.setProperty("size", 0L);
74          session.save();
75          session.logout();
76      }

78      protected synchronized Response deleteResource(final Node resource)
79              throws RepositoryException {
80  
81          logger.debug("Attempting to delete resource at path: " +
82                  resource.getPath());
83          final Session session = resource.getSession();
84  
85          try {
86              resource.remove();
87              session.save();
88          } finally {
89              session.logout();
90          }
91          return noContent().build();
92  
93      }

95      public static Long getNodePropertySize(Node node)
96              throws RepositoryException {
97          Long size = 0L;
98          PropertyIterator i = node.getProperties();
99          while (i.hasNext()) {
100             Property p = i.nextProperty();
101             if (p.isMultiple()) {
102                 for (Value v : copyOf(p.getValues())) {
103                     size = size + v.getBinary().getSize();
104                 }
105             } else {
106                 size = size + p.getBinary().getSize();
107             }
108         }
109         return size;
110     }

112     /**
113      * Alter the total repository size.
114      * 
115      * @param change
116      *            the amount by which to [de|in]crement the total repository
117      *            size
118      * @param session
119      *            the javax.jcr.Session in which the originating mutation is
120      *            occurring
121      * @throws PathNotFoundException
122      * @throws RepositoryException
123      */
124     protected void updateRepositorySize(Long change, Session session)
125             throws PathNotFoundException, RepositoryException {
126         logger.debug("updateRepositorySize called with change quantity: " +
127                 change);
128         Property sizeProperty = session.getNode("/objects").getProperty("size");
129         Long previousSize = sizeProperty.getLong();
130         logger.debug("Previous repository size: " + previousSize);
131         synchronized (sizeProperty) {
132             sizeProperty.setValue(previousSize + change);
133             session.save();
134         }
135         logger.debug("Current repository size: " + sizeProperty.getLong());
136     }

138     protected Long getRepositorySize(Session session)
139             throws ValueFormatException, PathNotFoundException,
140             RepositoryException {
141         return session.getNode("/objects").getProperty("size").getLong();
142     }
