100     /**
101      * Creates a new object.
102      * 
103      * @param pid
104      * @return 201
105      * @throws RepositoryException
106      */
107     @POST
108     @Path("/{pid}")
109     public Response ingest(@PathParam("pid")
110     final String pid) throws RepositoryException {
111 
112         logger.debug("Attempting to ingest with pid: " + pid);
113 
114         final Session session = repo.login();
115         try {
116             final Node obj = createObjectNode(session, pid);
117             session.save();
118             /*
119              * we save before updating the repo size because the act of
120              * persisting session state creates new system-curated nodes and
121              * properties which contribute to the footprint of this resource
122              */
123             updateRepositorySize(getObjectSize(obj), session);
124             // now we save again to persist the repo size
125             session.save();
126             logger.debug("Finished ingest with pid: " + pid);
127             return created(uriInfo.getAbsolutePath()).entity(pid).build();
128 
129         } finally {
130             session.logout();
131         }
132     }

134     /**
135      * Returns an object profile.
136      * 
137      * @param pid
138      * @return 200
139      * @throws RepositoryException
140      * @throws IOException
141      */
142     @GET
143     @Path("/{pid}")
144     @Produces({TEXT_XML, APPLICATION_JSON})
145     public ObjectProfile getObject(@PathParam("pid")
146     final String pid) throws RepositoryException, IOException {
147 
148         final Node obj = getObjectNode(pid);
149         final ObjectProfile objectProfile = new ObjectProfile();
150 
151         objectProfile.pid = pid;
152         if (obj.hasProperty(DC_TITLE)) {
153             Property dcTitle = obj.getProperty(DC_TITLE);
154             if (!dcTitle.isMultiple())
155                 objectProfile.objLabel = obj.getProperty(DC_TITLE).getString();
156             else {
157                 objectProfile.objLabel =
158                         on('/').join(map(dcTitle.getValues(), value2string));
159             }
160         }
161         objectProfile.objOwnerId =
162                 obj.getProperty("fedora:ownerId").getString();
163         objectProfile.objCreateDate =
164                 obj.getProperty("jcr:created").getString();
165         objectProfile.objLastModDate =
166                 obj.getProperty("jcr:lastModified").getString();
167         objectProfile.objSize = getObjectSize(obj);
168         objectProfile.objItemIndexViewURL =
169                 uriInfo.getAbsolutePathBuilder().path("datastreams").build();
170         objectProfile.objState = A;
171         objectProfile.objModels = map(obj.getMixinNodeTypes(), nodetype2name);
172         return objectProfile;
173 
174     }

198     static Long getObjectSize(Node obj) throws RepositoryException {
199         return getNodePropertySize(obj) + getObjectDSSize(obj);
200     }

202     /**
203      * @param obj
204      * @return object's datastreams' total size in bytes
205      * @throws RepositoryException
206      */
207     private static Long getObjectDSSize(Node obj) throws RepositoryException {
208         Long size = 0L;
209         NodeIterator i = obj.getNodes();
210         while (i.hasNext()) {
211             Node ds = i.nextNode();
212             size = size + getNodePropertySize(ds);
213             size = size + getContentSize(ds);
214         }
215         return size;
216     }
