51                                           InvalidChecksumException {
52          Session session = repo.login();
53          objectService.createObject(session, "/testDatastreamObject");
54          datastreamService
55              .createDatastreamNode(session,

57                                    "application/octet-stream",
58                                    new ByteArrayInputStream("asdf".getBytes()));
59          session.save();
60          session.logout();
61          session = repo.login();
62          final Datastream ds =
63              datastreamService
64              .getDatastream(session,
