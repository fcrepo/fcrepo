16  package org.fcrepo.jms.headers;
17  
18  import static com.google.common.base.Strings.isNullOrEmpty;
19  import static java.util.stream.Collectors.joining;
20  import static org.fcrepo.kernel.api.RdfLexicon.REPOSITORY_NAMESPACE;
21  import static org.slf4j.LoggerFactory.getLogger;
22  
23  import java.io.IOException;
24  import java.util.Set;
25  import javax.jms.JMSException;
26  import javax.jms.Message;
27  
28  import org.fcrepo.jms.observer.JMSEventMessageFactory;
29  import org.fcrepo.kernel.api.observer.EventType;
30  import org.fcrepo.kernel.api.observer.FedoraEvent;
31  
32  import org.slf4j.Logger;
33  
34  import com.fasterxml.jackson.databind.ObjectMapper;
35  import com.fasterxml.jackson.databind.JsonNode;
36  
37  /**
38   * Generates JMS {@link Message}s composed entirely of headers, based entirely
39   * on information found in the {@link FedoraEvent} that triggers publication.
40   *
41   * @author ajs6f
42   * @author escowles
43   * @since Dec 2, 2013
44   */
45  public class DefaultMessageFactory implements JMSEventMessageFactory {
46  
47      public static final String JMS_NAMESPACE = "org.fcrepo.jms.";
48  
49      public static final String TIMESTAMP_HEADER_NAME = JMS_NAMESPACE
50              + "timestamp";
51  
52      public static final String IDENTIFIER_HEADER_NAME = JMS_NAMESPACE
53              + "identifier";
54  
55      public static final String EVENT_TYPE_HEADER_NAME = JMS_NAMESPACE
56              + "eventType";
57  
58      public static final String BASE_URL_HEADER_NAME = JMS_NAMESPACE
59              + "baseURL";
60  
61      public static final String PROPERTIES_HEADER_NAME = JMS_NAMESPACE
62              + "properties";
63  
64      public static final String USER_HEADER_NAME = JMS_NAMESPACE + "user";
65      public static final String USER_AGENT_HEADER_NAME = JMS_NAMESPACE + "userAgent";
66      public static final String EVENT_ID_HEADER_NAME = JMS_NAMESPACE + "eventID";
67  
68      private String baseURL;
69      private String userAgent;
70  
71      @Override
72      public Message getMessage(final FedoraEvent event,
73          final javax.jms.Session jmsSession) throws JMSException {
74  
75          final Message message = jmsSession.createMessage();
76          message.setLongProperty(TIMESTAMP_HEADER_NAME, event.getDate());
77  
78          // extract baseURL and userAgent from event UserData
79          try {
80              final String userdata = event.getUserData();
81              if (!isNullOrEmpty(userdata)) {
82                  final ObjectMapper mapper = new ObjectMapper();
83                  final JsonNode json = mapper.readTree(userdata);
84                  String url = json.get("baseURL").asText();
85                  while (url.endsWith("/")) {
86                      url = url.substring(0, url.length() - 1);
87                  }
88                  this.baseURL = url;
89                  this.userAgent = json.get("userAgent").asText();
90                  LOGGER.debug("MessageFactory baseURL: {}, userAgent: {}", baseURL, userAgent);
91  
92              } else {
93                  LOGGER.warn("MessageFactory event UserData is empty!");
94              }
95  
96          } catch ( final IOException ex ) {
97              LOGGER.warn("Error setting baseURL or userAgent", ex);
98          }
99  
100         message.setStringProperty(IDENTIFIER_HEADER_NAME, event.getPath());
101         message.setStringProperty(EVENT_TYPE_HEADER_NAME, getEventURIs(event
102                 .getTypes()));
103         message.setStringProperty(BASE_URL_HEADER_NAME, baseURL);
104         message.setStringProperty(USER_HEADER_NAME, event.getUserID());
105         message.setStringProperty(USER_AGENT_HEADER_NAME, userAgent);
106         message.setStringProperty(PROPERTIES_HEADER_NAME, String.join(",", event.getProperties()));
107         message.setStringProperty(EVENT_ID_HEADER_NAME, event.getEventID());
108 
109         LOGGER.trace("getMessage() returning: {}", message);
110         return message;
111     }
112 
113     private static String getEventURIs(final Set<EventType> types) {
114         final String uris = types.stream()
115                                  .map(EventType::toString)
116                                  .map(REPOSITORY_NAMESPACE::concat)
117                                  .collect(joining(","));
118 
119         LOGGER.debug("Constructed event type URIs: {}", uris);
120         return uris;
121     }
122 
123     private static final Logger LOGGER = getLogger(DefaultMessageFactory.class);
124 
125 }
