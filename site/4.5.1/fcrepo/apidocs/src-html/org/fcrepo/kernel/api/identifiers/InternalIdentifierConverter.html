016package org.fcrepo.kernel.api.identifiers;
017
018import com.google.common.base.Converter;
019
020/**
021 * Translates internal {@link String} identifiers to internal {@link String}
022 * identifiers.
023 *
024 * @author ajs6f
025 * @since Apr 1, 2014
026 */
027public abstract class InternalIdentifierConverter extends Converter<String, String> {
028
029    /*
030     * (non-Javadoc)
031     * @see com.google.common.base.Converter#doForward(java.lang.Object)
032     */
033    @Override
034    protected String doForward(final String a) {
035        return a;
036    }
037
038    /*
039     * (non-Javadoc)
040     * @see com.google.common.base.Converter#doBackward(java.lang.Object)
041     */
042    @Override
043    protected String doBackward(final String b) {
044        return b;
045    }
046
047}
