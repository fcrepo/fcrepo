18  package org.fcrepo.kernel.impl.operations;
19  
20  import org.fcrepo.kernel.api.Transaction;
21  import org.fcrepo.kernel.api.identifiers.FedoraId;
22  import org.fcrepo.kernel.api.operations.ResourceOperationBuilder;
23  
24  
25  /**
26   * Builder for operations to reindex a resource
27   *
28   * @author dbernstein
29   */
30  public class ReindexResourceOperationBuilder extends AbstractResourceOperationBuilder
31          implements ResourceOperationBuilder {
32  
33      /**
34       * Construct the builder
35       *
36       * @param tx the transaction
37       * @param rescId identifier of the resource to reindex
38       */
39      public ReindexResourceOperationBuilder(final Transaction tx, final FedoraId rescId) {
40          super(tx, rescId);
41      }
42  
43      @Override
44      public ReindexResourceOperationImpl build() {
45          final var operation = new ReindexResourceOperationImpl(transaction, rescId);
46          operation.setUserPrincipal(userPrincipal);
47          operation.setTransaction(transaction);
48          return operation;
49      }
50  }
