018package org.fcrepo.kernel.impl.operations;
019
020import static org.fcrepo.kernel.api.RdfLexicon.NON_RDF_SOURCE;
021
022import java.io.InputStream;
023import java.net.URI;
024
025import org.fcrepo.kernel.api.Transaction;
026import org.fcrepo.kernel.api.identifiers.FedoraId;
027import org.fcrepo.kernel.api.operations.CreateResourceOperation;
028
029/**
030 * Operation for creating a new non-rdf source
031 *
032 * @author bbpennel
033 */
034public class CreateNonRdfSourceOperation extends AbstractNonRdfSourceOperation implements CreateResourceOperation {
035
036    private FedoraId parentId;
037
038    /**
039     * Constructor for external content.
040     *
041     * @param rescId the internal identifier.
042     * @param externalContentURI the URI of the external content.
043     * @param externalHandling the type of external content handling (REDIRECT, PROXY)
044     */
045    protected CreateNonRdfSourceOperation(final Transaction transaction, final FedoraId rescId,
046                                          final URI externalContentURI, final String externalHandling) {
047        super(transaction, rescId, externalContentURI, externalHandling);
048    }
049
050    /**
051     * Constructor for internal binaries.
052     *
053     * @param transaction the transaction
054     * @param rescId the internal identifier.
055     * @param content the stream of the content.
056     */
057    protected CreateNonRdfSourceOperation(final Transaction transaction, final FedoraId rescId,
058                                          final InputStream content) {
059        super(transaction, rescId, content);
060    }
061
062    @Override
063    public String getInteractionModel() {
064        return NON_RDF_SOURCE.toString();
065    }
066
067    @Override
068    public boolean isArchivalGroup() {
069        return false;
070    }
071
072    @Override
073    public FedoraId getParentId() {
074        return parentId;
075    }
076
077    /**
078     * @param parentId the parentId to set
079     */
080    public void setParentId(final FedoraId parentId) {
081        this.parentId = parentId;
082    }
083
084}
