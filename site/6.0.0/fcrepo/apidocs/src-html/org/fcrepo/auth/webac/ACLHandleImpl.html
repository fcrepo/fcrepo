018package org.fcrepo.auth.webac;
019
020import java.util.List;
021
022import org.fcrepo.kernel.api.auth.ACLHandle;
023import org.fcrepo.kernel.api.auth.WebACAuthorization;
024import org.fcrepo.kernel.api.models.FedoraResource;
025
026/**
027 * A simple class connecting an URI pointing to an ACL to a {@link FedoraResource} that points to that ACL.
028 *
029 * @author ajs6f
030 */
031public class ACLHandleImpl implements ACLHandle {
032
033    private final FedoraResource resource;
034
035    private final List<WebACAuthorization> authorizations;
036    /**
037     * Default constructor.
038     *
039     * @param resource the requested FedoraResource
040     * @param authorizations any authorizations associated with the uri
041     */
042    public ACLHandleImpl(final FedoraResource resource, final List<WebACAuthorization> authorizations) {
043        this.resource = resource;
044        this.authorizations = authorizations;
045    }
046
047    @Override
048    public FedoraResource getResource() {
049        return resource;
050    }
051
052    @Override
053    public List<WebACAuthorization> getAuthorizations() {
054        return authorizations;
055    }
056}
