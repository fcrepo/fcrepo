18  package org.fcrepo.auth.webac;
19  
20  import java.util.List;
21  
22  import org.fcrepo.kernel.api.auth.ACLHandle;
23  import org.fcrepo.kernel.api.auth.WebACAuthorization;
24  import org.fcrepo.kernel.api.models.FedoraResource;
25  
26  /**
27   * A simple class connecting an URI pointing to an ACL to a {@link FedoraResource} that points to that ACL.
28   *
29   * @author ajs6f
30   */
31  public class ACLHandleImpl implements ACLHandle {
32  
33      private final FedoraResource resource;
34  
35      private final List<WebACAuthorization> authorizations;
36      /**
37       * Default constructor.
38       *
39       * @param resource the requested FedoraResource
40       * @param authorizations any authorizations associated with the uri
41       */
42      public ACLHandleImpl(final FedoraResource resource, final List<WebACAuthorization> authorizations) {
43          this.resource = resource;
44          this.authorizations = authorizations;
45      }
46  
47      @Override
48      public FedoraResource getResource() {
49          return resource;
50      }
51  
52      @Override
53      public List<WebACAuthorization> getAuthorizations() {
54          return authorizations;
55      }
56  }
