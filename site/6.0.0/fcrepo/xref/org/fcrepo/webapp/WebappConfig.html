19  package org.fcrepo.webapp;
20  
21  import java.util.Optional;
22  import java.util.concurrent.ExecutorService;
23  import java.util.concurrent.Executors;
24  import java.util.concurrent.TimeUnit;
25  
26  import javax.inject.Inject;
27  
28  import org.fcrepo.config.FedoraPropsConfig;
29  import org.fcrepo.http.api.ExternalContentHandlerFactory;
30  import org.fcrepo.http.api.ExternalContentPathValidator;
31  import org.fcrepo.kernel.api.auth.ACLHandle;
32  import org.fcrepo.kernel.api.rdf.RdfNamespaceRegistry;
33  
34  import org.apache.http.conn.HttpClientConnectionManager;
35  import org.apache.http.impl.conn.PoolingHttpClientConnectionManager;
36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  import org.springframework.context.annotation.Bean;
39  import org.springframework.context.annotation.Configuration;
40  import org.springframework.scheduling.annotation.EnableScheduling;
41  import org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler;
42  
43  import com.github.benmanes.caffeine.cache.Cache;
44  import com.github.benmanes.caffeine.cache.Caffeine;
45  import com.google.common.eventbus.AsyncEventBus;
46  import com.google.common.eventbus.EventBus;
47  
48  /**
49   * Spring config for the webapp
50   *
51   * @author pwinckles
52   */
53  @Configuration
54  @EnableScheduling
55  public class WebappConfig {
56  
57      private static final Logger LOGGER = LoggerFactory.getLogger(WebappConfig.class);
58  
59      @Inject
60      private FedoraPropsConfig fedoraPropsConfig;
61  
62      /**
63       * Task scheduler used for cleaning up transactions
64       *
65       * @return scheduler
66       */
67      @Bean
68      public ThreadPoolTaskScheduler taskScheduler() {
69          final var scheduler = new ThreadPoolTaskScheduler();
70          scheduler.setPoolSize(1);
71          scheduler.setThreadNamePrefix("ScheduledTask");
72          return scheduler;
73      }
74  
75      /**
76       * HTTP connection manager
77       *
78       * @return connection manager
79       */
80      @Bean
81      public HttpClientConnectionManager connectionManager() {
82          return new PoolingHttpClientConnectionManager();
83      }
84  
85      /**
86       * Fedora's lightweight internal event bus. Currently memory-resident.
87       *
88       * @param propsConfig config
89       * @return event bus
90       */
91      @Bean
92      public EventBus eventBus(final FedoraPropsConfig propsConfig) {
93          return new AsyncEventBus(eventBusExecutor(propsConfig));
94      }
95  
96      /**
97       * @param propsConfig config
98       * @return executor intended to be used by the Guava event bus
99       */
100     @Bean
101     public ExecutorService eventBusExecutor(final FedoraPropsConfig propsConfig) {
102         LOGGER.debug("Event bus threads: {}", propsConfig);
103         return Executors.newFixedThreadPool(propsConfig.getEventBusThreads());
104     }
105 
106     /**
107      * Configuration of namespace prefixes
108      *
109      * @param propsConfig config properties
110      * @return rdf namespace registry
111      */
112     @Bean(initMethod = "init", destroyMethod = "shutdown")
113     public RdfNamespaceRegistry rdfNamespaceRegistry(final FedoraPropsConfig propsConfig) {
114         final var registry = new RdfNamespaceRegistry();
115         registry.setConfigPath(propsConfig.getNamespaceRegistry());
116         registry.setMonitorForChanges(true);
117         return registry;
118     }
119 
120     /**
121      * External content configuration
122      *
123      * @param propsConfig config properties
124      * @return external content path validator
125      */
126     @Bean(initMethod = "init", destroyMethod = "shutdown")
127     public ExternalContentPathValidator externalContentPathValidator(final FedoraPropsConfig propsConfig) {
128         final var validator = new ExternalContentPathValidator();
129         validator.setConfigPath(propsConfig.getExternalContentAllowed());
130         validator.setMonitorForChanges(true);
131         return validator;
132     }
133 
134     @Bean
135     public ExternalContentHandlerFactory externalContentHandlerFactory(final ExternalContentPathValidator validator) {
136         final var factory = new ExternalContentHandlerFactory();
137         factory.setValidator(validator);
138         return factory;
139     }
140 
141     /**
142      * Used to cache the effective ACL location and authorizations for a given resource.
143      *
144      * @return the cache
145      */
146     @Bean
147     public Cache<String, Optional<ACLHandle>> authHandleCache() {
148         return Caffeine.newBuilder().weakValues()
149                 .expireAfterAccess(fedoraPropsConfig.getWebacCacheTimeout(), TimeUnit.MINUTES)
150                 .maximumSize(fedoraPropsConfig.getWebacCacheSize()).build();
151     }
152 }
