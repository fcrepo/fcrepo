018package org.fcrepo.http.commons.exceptionhandlers;
019
020import org.fcrepo.kernel.api.exception.ServerManagedPropertyException;
021
022import org.slf4j.Logger;
023
024import javax.servlet.ServletContext;
025import javax.ws.rs.core.Context;
026import javax.ws.rs.core.Link;
027import javax.ws.rs.core.Response;
028import javax.ws.rs.core.UriInfo;
029import javax.ws.rs.ext.Provider;
030
031import static javax.ws.rs.core.Response.Status.CONFLICT;
032import static javax.ws.rs.core.Response.status;
033import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
034import static org.slf4j.LoggerFactory.getLogger;
035
036/**
037 * @author cabeer
038 * @author whikloj
039 * @since 10/1/14
040 */
041@Provider
042public class ServerManagedPropertyExceptionMapper extends ConstraintExceptionMapper<ServerManagedPropertyException>
043    implements ExceptionDebugLogging {
044
045    private static final Logger LOGGER =
046            getLogger(ServerManagedPropertyExceptionMapper.class);
047
048    @Context
049    private UriInfo uriInfo;
050
051    @Context
052    private ServletContext context;
053
054    @Override
055    public Response toResponse(final ServerManagedPropertyException e) {
056        debugException(this, e, LOGGER);
057        final Link link = buildConstraintLink(e, context, uriInfo);
058        final String msg = e.getMessage();
059        return status(CONFLICT).entity(msg).links(link).type(TEXT_PLAIN_WITH_CHARSET).build();
060    }
061}
