018package org.fcrepo.http.commons.api.rdf;
019
020import javax.ws.rs.core.UriInfo;
021
022import org.apache.jena.rdf.model.Resource;
023import org.fcrepo.kernel.api.models.FedoraResource;
024import org.fcrepo.kernel.api.identifiers.IdentifierConverter;
025
026import org.apache.jena.rdf.model.Model;
027
028/**
029 * Helper to generate an RDF model for a FedoraResourceImpl that (likely) creates
030 * relations from our resource to other HTTP components
031 *
032 * @author awoods
033 */
034public interface UriAwareResourceModelFactory {
035
036    /**
037     * Given a resource, the UriInfo and a way to generate graph subjects,
038     * create a model with triples to inject into an RDF response for the
039     * resource (e.g. to add HATEOAS links)
040     *
041     * @param resource the resource
042     * @param uriInfo the uri info
043     * @param idTranslator the id translator
044     * @return model containing triples for the given resource
045     */
046    Model createModelForResource(final FedoraResource resource,
047            final UriInfo uriInfo, final IdentifierConverter<Resource,FedoraResource> idTranslator);
048}
