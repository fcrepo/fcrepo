215     /**
216      * Create a resource at a specified path, or replace triples with provided RDF.
217      * @param requestContentType
218      * @param requestBodyStream
219      * @return 204
220      */
221     @PUT
222     @Consumes
223     @Timed
224     public Response createOrReplaceObjectRdf(
225             @HeaderParam("Content-Type") final MediaType requestContentType,
226             @ContentLocation final InputStream requestBodyStream,
227             @QueryParam("checksum") final String checksum,
228             @HeaderParam("Content-Disposition") final ContentDisposition contentDisposition,
229             @HeaderParam("If-Match") final String ifMatch)
230             throws InvalidChecksumException, MalformedRdfException {
231 
232         final FedoraResource resource;
233         final Response.ResponseBuilder response;
234 
235         final String path = toPath(translator(), externalPath);
236 
237         final MediaType contentType = getSimpleContentType(requestContentType);
238 
239         if (nodeService.exists(session, path)) {
240             resource = resource();
241             response = noContent();
242         } else {
243             final MediaType effectiveContentType
244                     = requestBodyStream == null || requestContentType == null ? null : contentType;
245             resource = createFedoraResource(path, effectiveContentType, contentDisposition);
246 
247             final URI location = getUri(resource);
248 
249             response = created(location).entity(location.toString());
250         }
251 
252         if (httpConfiguration.putRequiresIfMatch() && StringUtils.isBlank(ifMatch) && !resource.isNew()) {
253             throw new ClientErrorException("An If-Match header is required", 428);
254         }
255 
256         evaluateRequestPreconditions(request, servletResponse, resource, session);
257 
258         final RdfStream resourceTriples;
259 
260         if (resource.isNew()) {
261             resourceTriples = new RdfStream();
262         } else {
263             resourceTriples = getResourceTriples();
264         }
265 
266         if (resource instanceof FedoraBinary) {
267             replaceResourceBinaryWithStream((FedoraBinary) resource,
268                     requestBodyStream, contentDisposition, requestContentType, checksum);
269         } else if (isRdfContentType(contentType.toString())) {
270             try {
271                 replaceResourceWithStream(resource, requestBodyStream, contentType, resourceTriples);
272             } catch (final RiotException e) {
273                 throw new BadRequestException("RDF was not parsable", e);
274             }
275         } else if (!resource.isNew()) {
276             boolean emptyRequest = true;
277             try {
278                 emptyRequest = requestBodyStream.read() == -1;
279             } catch (IOException ex) {
280                 LOGGER.debug("Error checking for request body content", ex);
281             }
282 
283             if (requestContentType == null && emptyRequest) {
284                 throw new ClientErrorException("Resource Already Exists", CONFLICT);
285             } else {
286                 throw new ClientErrorException("Invalid Content Type " + requestContentType, UNSUPPORTED_MEDIA_TYPE);
287             }
288         }
289 
290         try {
291             session.save();
292         } catch (final RepositoryException e) {
293             throw new RepositoryRuntimeException(e);
294         }
295 
296         addCacheControlHeaders(servletResponse, resource, session);
297 
298         addResourceLinkHeaders(resource);
299 
300         return response.build();
301 
302     }

304     /**
305      * Update an object using SPARQL-UPDATE
306      *
307      * @return 201
308      * @throws RepositoryException
309      * @throws IOException
310      */
311     @PATCH
312     @Consumes({contentTypeSPARQLUpdate})
313     @Timed
314     public Response updateSparql(@ContentLocation final InputStream requestBodyStream)
315             throws IOException, MalformedRdfException {
316 
317         if (null == requestBodyStream) {
318             throw new BadRequestException("SPARQL-UPDATE requests must have content!");
319         }
320 
321         if (resource() instanceof FedoraBinary) {
322             throw new BadRequestException(resource() + " is not a valid object to receive a PATCH");
323         }
324 
325         try {
326             final String requestBody = IOUtils.toString(requestBodyStream);
327             if (isBlank(requestBody)) {
328                 throw new BadRequestException("SPARQL-UPDATE requests must have content!");
329             }
330 
331             evaluateRequestPreconditions(request, servletResponse, resource(), session);
332 
333             final RdfStream resourceTriples;
334 
335             if (resource().isNew()) {
336                 resourceTriples = new RdfStream();
337             } else {
338                 resourceTriples = getResourceTriples();
339             }
340 
341             patchResourcewithSparql(resource(), requestBody, resourceTriples);
342 
343             try {
344                 session.save();
345             } catch (final RepositoryException e) {
346                 throw new RepositoryRuntimeException(e);
347             }
348 
349             addCacheControlHeaders(servletResponse, resource(), session);
350 
351             return noContent().build();
352 
353         } catch ( final RuntimeException ex ) {
354             final Throwable cause = ex.getCause();
355             if ( cause != null && cause instanceof PathNotFoundException) {
356                 // the sparql update referred to a repository resource that doesn't exist
357                 throw new BadRequestException(cause.getMessage());
358             }
359             throw ex;
360         }
361     }

363     /**
364      * Creates a new object.
365      *
366      * application/octet-stream;qs=1001 is a workaround for JERSEY-2636, to ensure
367      * requests without a Content-Type get routed here.
368      *
369      * @return 201
370      */
371     @POST
372     @Consumes({MediaType.APPLICATION_OCTET_STREAM + ";qs=1001", MediaType.WILDCARD})
373     @Timed
374     public Response createObject(@QueryParam("checksum") final String checksum,
375                                  @HeaderParam("Content-Disposition") final ContentDisposition contentDisposition,
376                                  @HeaderParam("Content-Type") final MediaType requestContentType,
377                                  @HeaderParam("Slug") final String slug,
378                                  @ContentLocation final InputStream requestBodyStream)
379             throws InvalidChecksumException, IOException, MalformedRdfException {
380 
381         if (!(resource() instanceof Container)) {
382             throw new ClientErrorException("Object cannot have child nodes", CONFLICT);
383         }
384 
385         final MediaType contentType = getSimpleContentType(requestContentType);
386 
387         final String contentTypeString = contentType.toString();
388 
389         final String newObjectPath = mintNewPid(slug);
390 
391         LOGGER.debug("Attempting to ingest with path: {}", newObjectPath);
392 
393         final MediaType effectiveContentType
394                 = requestBodyStream == null || requestContentType == null ? null : contentType;
395         final FedoraResource result = createFedoraResource(
396                 newObjectPath,
397                 effectiveContentType,
398                 contentDisposition);
399 
400         final RdfStream resourceTriples;
401 
402         if (result.isNew()) {
403             resourceTriples = new RdfStream();
404         } else {
405             resourceTriples = getResourceTriples();
406         }
407 
408         if (requestBodyStream == null) {
409             LOGGER.trace("No request body detected");
410         } else {
411             LOGGER.trace("Received createObject with a request body and content type \"{}\"", contentTypeString);
412 
413             if ((result instanceof Container)
414                     && isRdfContentType(contentTypeString)) {
415                 replaceResourceWithStream(result, requestBodyStream, contentType, resourceTriples);
416             } else if (result instanceof FedoraBinary) {
417                 LOGGER.trace("Created a datastream and have a binary payload.");
418 
419                 replaceResourceBinaryWithStream((FedoraBinary) result,
420                         requestBodyStream, contentDisposition, requestContentType, checksum);
421 
422             } else if (contentTypeString.equals(contentTypeSPARQLUpdate)) {
423                 LOGGER.trace("Found SPARQL-Update content, applying..");
424                 patchResourcewithSparql(result, IOUtils.toString(requestBodyStream), resourceTriples);
425             } else {
426                 if (requestBodyStream.read() != -1) {
427                     throw new ClientErrorException("Invalid Content Type " + contentTypeString, UNSUPPORTED_MEDIA_TYPE);
428                 }
429             }
430         }
431 
432         try {
433             session.save();
434         } catch (final RepositoryException e) {
435             throw new RepositoryRuntimeException(e);
436         }
437 
438         LOGGER.debug("Finished creating resource with path: {}", newObjectPath);
439 
440         addCacheControlHeaders(servletResponse, result, session);
441 
442         final URI location = getUri(result);
443 
444         addResourceLinkHeaders(result, true);
445 
446         return created(location).entity(location.toString()).build();
447 
448     }

473     private void addOptionsHttpHeaders() {
474         final String options;
475 
476         if (resource() instanceof FedoraBinary) {
477             options = "DELETE,HEAD,GET,PUT,OPTIONS";
478 
479         } else if (resource() instanceof NonRdfSourceDescription) {
480             options = "MOVE,COPY,DELETE,POST,HEAD,GET,PUT,PATCH,OPTIONS";
481             servletResponse.addHeader("Accept-Patch", contentTypeSPARQLUpdate);
482 
483         } else if (resource() instanceof Container) {
484             options = "MOVE,COPY,DELETE,POST,HEAD,GET,PUT,PATCH,OPTIONS";
485             servletResponse.addHeader("Accept-Patch", contentTypeSPARQLUpdate);
486 
487             final String rdfTypes = TURTLE + "," + N3 + ","
488                     + N3_ALT2 + "," + RDF_XML + "," + NTRIPLES;
489             servletResponse.addHeader("Accept-Post", rdfTypes + "," + MediaType.MULTIPART_FORM_DATA
490                     + "," + contentTypeSPARQLUpdate);
491         } else {
492             options = "";
493         }
494 
495         addResourceLinkHeaders(resource());
496 
497         servletResponse.addHeader("Allow", options);
498     }

504     private void addResourceLinkHeaders(final FedoraResource resource, final boolean includeAnchor) {
505         if (resource instanceof NonRdfSourceDescription) {
506             final URI uri = getUri(((NonRdfSourceDescription) resource).getDescribedResource());
507             final Link link = Link.fromUri(uri).rel("describes").build();
508             servletResponse.addHeader("Link", link.toString());
509         } else if (resource instanceof FedoraBinary) {
510             final URI uri = getUri(((FedoraBinary) resource).getDescription());
511             final Link.Builder builder = Link.fromUri(uri).rel("describedby");
512 
513             if (includeAnchor) {
514                 builder.param("anchor", getUri(resource).toString());
515             }
516             servletResponse.addHeader("Link", builder.build().toString());
517         }

522     private String getRequestedObjectType(final MediaType requestContentType,
523                                           final ContentDisposition contentDisposition) {
524 
525         if (requestContentType != null) {
526             final String s = requestContentType.toString();
527             if (!s.equals(contentTypeSPARQLUpdate) && !isRdfContentType(s) || s.equals(TEXT_PLAIN)) {
528                 return FEDORA_BINARY;
529             }
530         }
531 
532         if (contentDisposition != null && contentDisposition.getType().equals("attachment")) {
533             return FEDORA_BINARY;
534         }
535 
536         return FEDORA_CONTAINER;
537     }

539     private FedoraResource createFedoraResource(final String path,
540                                                 final MediaType requestContentType,
541                                                 final ContentDisposition contentDisposition) {
542         final String objectType = getRequestedObjectType(requestContentType, contentDisposition);
543 
544         final FedoraResource result;
545 
546         if (objectType.equals(FEDORA_BINARY)) {
547             result = binaryService.findOrCreate(session, path);
548         } else {
549             result = containerService.findOrCreate(session, path);
550         }
551 
552         return result;

560     private String mintNewPid(final String slug) {
561         String pid;
562 
563         if (slug != null && !slug.isEmpty()) {
564             pid = slug;
565         } else {
566             pid = pidMinter.mintPid();
567         }
568         // reverse translate the proffered or created identifier
569         LOGGER.trace("Using external identifier {} to create new resource.", pid);
570         LOGGER.trace("Using prefixed external identifier {} to create new resource.", uriInfo.getBaseUri() + "/"
571                 + pid);
572 
573         final URI newResourceUri = uriInfo.getAbsolutePathBuilder().clone().path(FedoraLdp.class)
574                 .resolveTemplate("path", pid, false).build();
575 
576         pid = translator().asString(createResource(newResourceUri.toString()));
577         try {
578             pid = URLDecoder.decode(pid, "UTF-8");
579         } catch (UnsupportedEncodingException e) {
580             // noop
581         }
582         // remove leading slash left over from translation
583         LOGGER.trace("Using internal identifier {} to create new resource.", pid);
584 
585         if (nodeService.exists(session, pid)) {
586             LOGGER.trace("Resource with path {} already exists; minting new path instead", pid);
587             return mintNewPid(null);
588         }
589 
590         return pid;
591     }
