54  /**
55   * PID minter that uses an external REST service to mint PIDs.
56   *
57   * @author escowles
58   * @since 04/28/2014
59   */
60  public class HttpPidMinter implements PidMinter {
61  
62      private static final Logger log = getLogger(HttpPidMinter.class);
63      protected final String url;
64      protected final String method;
65      protected final String username;
66      protected final String password;
67      private final String regex;
68      private XPathExpression xpath;
69  
70      protected HttpClient client;
71  
72      /**
73       * Create a new HttpPidMinter.
74       * @param url The URL for the minter service.  This is the only required argument -- all
75       *    other parameters can be blank.
76       * @param method The HTTP method (POST, PUT or GET) used to generate a new PID (POST will
77       *    be used if the method is blank.
78       * @param username If not blank, use this username to connect to the minter service.
79       * @param password If not blank, use this password used to connect to the minter service.
80       * @param regex If not blank, use this regular expression used to remove unwanted text from the
81       *    minter service response.  For example, if the response text is "/foo/bar:baz" and the
82       *    desired identifier is "baz", then the regex would be ".*:".
83       * @param xpath If not blank, use this XPath expression used to extract the desired identifier
84       *    from an XML minter response.
85      **/
86      public HttpPidMinter( final String url, final String method, final String username,
87          final String password, final String regex, final String xpath ) {
88  
89          checkArgument( !isBlank(url), "Minter URL must be specified!" );
90  
91          this.url = url;
92          this.method = method;
93          this.username = username;
94          this.password = password;
95          this.regex = regex;
96          if ( xpath != null ) {
97              try {
98                  this.xpath = XPathFactory.newInstance().newXPath().compile(xpath);
99              } catch ( XPathException ex ) {
100                 log.warn("Error parsing xpath ({}): {}", xpath, ex );
101             }
102         }
103         this.client = buildClient();
104     }
105 
106     /**
107      * Setup authentication in httpclient.
108     **/
109     protected HttpClient buildClient() {
110         HttpClientBuilder builder = HttpClientBuilder.create().useSystemProperties().setConnectionManager(
111             new PoolingHttpClientConnectionManager());
112         if (!isBlank(username) && !isBlank(password)) {
113             final URI uri = URI.create(url);
114             final CredentialsProvider credsProvider = new BasicCredentialsProvider();
115             credsProvider.setCredentials(new AuthScope(uri.getHost(), uri.getPort()),
116                 new UsernamePasswordCredentials(username, password));
117             builder = builder.setDefaultCredentialsProvider(credsProvider);
118         }
119         return builder.build();
120     }
121 
122     /**
123      * Instantiate a request object based on the method variable.
124     **/
125     private HttpUriRequest minterRequest() {
126         if ( method != null && method.equalsIgnoreCase("GET") ) {
127             return new HttpGet(url);
128         } else if ( method != null && method.equalsIgnoreCase("PUT") ) {
129             return new HttpPut(url);
130         } else {
131             return new HttpPost(url);
132         }
133     }
134 
135     /**
136      * Remove unwanted text from the minter service response to produce the desired identifer.
137      * Override this method for processing more complex than a simple regex replacement.
138     **/
139     protected String responseToPid( final String responseText ) throws Exception {
140         log.debug("responseToPid({})", responseText);
141         if ( !isBlank(regex) ) {
142             return responseText.replaceFirst(regex,"");
143         } else if ( xpath != null ) {
144             return xpath( responseText, xpath );
145         } else {
146             return responseText;
147         }
148     }
149 
150     /**
151      * Extract the desired identifier value from an XML response using XPath
152     **/
153     private static String xpath( final String xml, final XPathExpression xpath ) throws Exception {
154         final DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
155         final Document doc = builder.parse(new ByteArrayInputStream(xml.getBytes()));
156         return xpath.evaluate(doc);
157     }
158 
159     /**
160      * Mint a unique identifier using an external HTTP API.
161      * @return The generated identifier.
162      */
163     @Timed
164     @Override
165     public String mintPid() {
166         try {
167             log.debug("mintPid()");
168             final HttpResponse resp = client.execute( minterRequest() );
169             return responseToPid( EntityUtils.toString(resp.getEntity()) );
170         } catch ( IOException ex ) {
171             log.warn("Error minting pid from {}: {}", url, ex);
172             throw new RuntimeException("Error minting pid", ex);
173         } catch ( Exception ex ) {
174             log.warn("Error processing minter response", ex);
175             throw new RuntimeException("Error processing minter response", ex);
176         }
177     }
