16  package org.fcrepo.http.commons.session;
17  
18  import static org.fcrepo.kernel.impl.services.TransactionServiceImpl.isInTransaction;
19  import static org.modeshape.jcr.SessionUtils.unregisterObservationManager;
20  import static org.slf4j.LoggerFactory.getLogger;
21  
22  import javax.inject.Inject;
23  import javax.jcr.Session;
24  import javax.servlet.http.HttpServletRequest;
25  import javax.ws.rs.ext.Provider;
26  
27  import org.glassfish.hk2.api.Factory;
28  import org.glassfish.jersey.process.internal.RequestScoped;
29  import org.slf4j.Logger;
30  
31  /**
32   * Provide a JCR session within the current request context
33   *
34   * @author awoods
35   */
36  @Provider
37  @RequestScoped
38  public class SessionProvider implements Factory<Session> {
39  
40      @Inject
41      SessionFactory sessionFactory;
42  
43      private HttpServletRequest request;
44  
45      /**
46       * Create a new session provider for a request
47       * @param request
48       */
49      @Inject
50      public SessionProvider(final HttpServletRequest request) {
51          this.request = request;
52      }
53  
54      private static final Logger LOGGER = getLogger(SessionProvider.class);
55  
56      @Override
57      public Session provide() {
58          final Session session = sessionFactory.getSession(request);
59          LOGGER.trace("Providing new session {}", session);
60          return session;
61      }
62  
63      @Override
64      public void dispose(final Session session) {
65          LOGGER.trace("Disposing session {}", session);
66  
67          if (session.isLive() && !isInTransaction(session)) {
68              unregisterObservationManager(session);
69              session.logout();
70          }
71      }
72  }
