016package org.fcrepo.serialization;
017
018import java.io.IOException;
019import java.io.InputStream;
020import java.io.OutputStream;
021
022import javax.jcr.Repository;
023import javax.jcr.RepositoryException;
024import javax.jcr.Session;
025
026import org.fcrepo.kernel.api.models.FedoraResource;
027import org.fcrepo.kernel.api.exception.InvalidChecksumException;
028import org.springframework.beans.factory.annotation.Autowired;
029
030/**
031 * Auto-wire some helpful services for the FedoraObjectSerializer
032 *
033 * @author cbeer
034 */
035public abstract class BaseFedoraObjectSerializer implements
036        FedoraObjectSerializer {
037
038    @Autowired
039    protected Repository repo;
040
041    @Override
042    public abstract void serialize(final FedoraResource obj,
043                                   final OutputStream out,
044                                   final boolean skipBinary,
045                                   final boolean recurse) throws RepositoryException, IOException,
046            InvalidSerializationFormatException;
047
048    @Override
049    public abstract void deserialize(final Session session, final String path,
050            final InputStream stream) throws IOException, RepositoryException,
051            InvalidChecksumException, InvalidSerializationFormatException;
052
053}
