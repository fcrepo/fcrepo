018package org.fcrepo.http.commons.exceptionhandlers;
019
020import static javax.ws.rs.core.Response.status;
021import static javax.ws.rs.core.Response.Status.BAD_REQUEST;
022import static org.fcrepo.kernel.api.RdfLexicon.CONSTRAINED_BY;
023import static org.slf4j.LoggerFactory.getLogger;
024
025import javax.ws.rs.core.Link;
026import javax.ws.rs.core.Response;
027import javax.ws.rs.ext.ExceptionMapper;
028import javax.ws.rs.ext.Provider;
029
030import org.fcrepo.kernel.api.exception.MalformedRdfException;
031
032import org.apache.commons.codec.binary.Base64;
033import org.slf4j.Logger;
034
035/**
036 * @author cabeer
037 * @since 9/30/14
038 */
039@Provider
040public class MalformedRdfExceptionMapper implements
041        ExceptionMapper<MalformedRdfException>, ExceptionDebugLogging {
042
043    private static final Logger LOGGER =
044            getLogger(MalformedRdfExceptionMapper.class);
045
046    private static final int REASONABLE_LENGTH = 500;
047
048    @Override
049    public Response toResponse(final MalformedRdfException e) {
050        debugException(this, e, LOGGER);
051        final Link link = Link.fromUri(getConstraintUri(e)).rel(CONSTRAINED_BY.getURI()).build();
052        final String msg = e.getMessage();
053        if (msg.matches(".*org.*Exception: .*")) {
054            return status(BAD_REQUEST).entity(msg.replaceAll("org.*Exception: ", "")).links(link).build();
055        }
056        return status(BAD_REQUEST).entity(msg).links(link).build();
057    }
058
059    private static String getConstraintUri(final MalformedRdfException e) {
060        final int index = Math.min(e.getMessage().length(), REASONABLE_LENGTH);
061        if (index > e.getMessage().length()) {
062            LOGGER.debug("Truncating Link header to {} characters.", REASONABLE_LENGTH);
063        }
064
065        return "data:text/plain;base64," + Base64.encodeBase64String(e.getMessage().substring(0, index).getBytes());
066    }
067}
