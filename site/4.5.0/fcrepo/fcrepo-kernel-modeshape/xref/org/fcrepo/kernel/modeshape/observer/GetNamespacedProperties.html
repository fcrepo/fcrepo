16  package org.fcrepo.kernel.modeshape.observer;
17  
18  import java.util.function.Function;
19  
20  import org.fcrepo.kernel.api.observer.FedoraEvent;
21  
22  import org.slf4j.Logger;
23  
24  import javax.jcr.NamespaceRegistry;
25  import javax.jcr.RepositoryException;
26  import javax.jcr.Session;
27  
28  import static com.hp.hpl.jena.rdf.model.ResourceFactory.createProperty;
29  import static org.fcrepo.kernel.api.RdfLexicon.jcrProperties;
30  import static org.fcrepo.kernel.modeshape.utils.NamespaceTools.getNamespaceRegistry;
31  import static org.fcrepo.kernel.api.RdfLexicon.REPOSITORY_NAMESPACE;
32  import static org.slf4j.LoggerFactory.getLogger;
33  
34  /**
35   * @author Andrew Woods
36   * @author ajs6f
37   * @since 11/22/14
38   */
39  public class GetNamespacedProperties implements Function<FedoraEvent, FedoraEvent> {
40  
41      private static final Logger LOGGER = getLogger(SimpleObserver.class);
42  
43      private final Session session;
44  
45      /**
46       * Constructor
47       *
48       * @param session used to get NamespaceRegistry
49       */
50      public GetNamespacedProperties(final Session session) {
51          this.session = session;
52      }
53  
54      @Override
55      public FedoraEvent apply(final FedoraEvent evt) {
56          final NamespaceRegistry namespaceRegistry = getNamespaceRegistry(session);
57  
58          final FedoraEvent event = new FedoraEventImpl(evt);
59          for (final String property : evt.getProperties()) {
60              final String[] parts = property.split(":", 2);
61              if (parts.length == 2) {
62                  final String prefix = parts[0];
63                  if ("jcr".equals(prefix)) {
64                      if (jcrProperties.contains(createProperty(REPOSITORY_NAMESPACE + parts[1]))) {
65                          event.addProperty(REPOSITORY_NAMESPACE + parts[1]);
66                      } else {
67                          LOGGER.debug("Swallowing jcr property: {}", property);
68                      }
69                  } else {
70                      try {
71                          event.addProperty(namespaceRegistry.getURI(prefix) + parts[1]);
72                      } catch (final RepositoryException ex) {
73                          LOGGER.debug("Prefix could not be dereferenced using the namespace registry: {}", property);
74                          event.addProperty(property);
75                      }
76                  }
77              } else {
78                  event.addProperty(property);
79              }
80          }
81          evt.getTypes().forEach(event::addType);
82          return event;
83      }
84  
85  }
