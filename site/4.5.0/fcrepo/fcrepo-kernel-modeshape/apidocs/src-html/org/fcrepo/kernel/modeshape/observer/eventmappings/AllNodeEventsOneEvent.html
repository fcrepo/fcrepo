016package org.fcrepo.kernel.modeshape.observer.eventmappings;
017
018import static com.google.common.collect.Multimaps.index;
019import static org.modeshape.jcr.api.JcrConstants.JCR_CONTENT;
020import static org.slf4j.LoggerFactory.getLogger;
021import static org.fcrepo.kernel.modeshape.utils.UncheckedFunction.uncheck;
022import static java.util.Arrays.asList;
023import static javax.jcr.observation.Event.PROPERTY_ADDED;
024import static javax.jcr.observation.Event.PROPERTY_CHANGED;
025import static javax.jcr.observation.Event.PROPERTY_REMOVED;
026
027import java.util.Iterator;
028import java.util.List;
029import java.util.function.Function;
030
031import javax.jcr.RepositoryException;
032import javax.jcr.observation.Event;
033
034import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
035import org.fcrepo.kernel.api.observer.FedoraEvent;
036import org.fcrepo.kernel.api.observer.eventmappings.InternalExternalEventMapper;
037import org.fcrepo.kernel.modeshape.observer.FedoraEventImpl;
038
039import org.slf4j.Logger;
040
041import com.google.common.collect.Multimap;
042
043/**
044 * Maps all JCR {@link Event}s concerning one JCR node to one {@link FedoraEvent}. Adds the types of those JCR events
045 * together to calculate the final type of the emitted FedoraEvent.
046 * TODO stop aggregating events in the heap, if possible
047 *
048 * @author ajs6f
049 * @since Feb 27, 2014
050 */
051public class AllNodeEventsOneEvent implements InternalExternalEventMapper {
052
053    private static final List<Integer> PROPERTY_EVENT_TYPES = asList(PROPERTY_ADDED, PROPERTY_CHANGED,
054            PROPERTY_REMOVED);
055
056    private final static Logger LOGGER = getLogger(AllNodeEventsOneEvent.class);
057
058    /**
059     * Extracts an identifier from a JCR {@link Event} by building an id from nodepath and user to collapse multiple
060     * events from repository mutations
061     */
062    private static final Function<Event, String> EXTRACT_NODE_ID = uncheck(ev -> {
063            final FedoraEvent event = new FedoraEventImpl(ev);
064            final String id = event.getPath() + "-" + event.getUserID();
065            LOGGER.debug("Sorting an event by identifier: {}", id);
066            return id;
067    });
068
069    @Override
070    public Iterator<FedoraEvent> apply(final Iterator<Event> events) {
071        return new FedoraEventIterator(events);
072    }
073
074    private static class FedoraEventIterator implements Iterator<FedoraEvent> {
075
076        // JCR events in a Multimap keyed by identifier
077        private final Multimap<String, Event> sortedEvents;
078
079        private final Iterator<String> ids;
080
081        public FedoraEventIterator(final Iterator<Event> events) {
082            sortedEvents = index(events, EXTRACT_NODE_ID::apply);
083            ids = sortedEvents.keySet().iterator();
084        }
085
086        @Override
087        public boolean hasNext() {
088            return ids.hasNext();
089        }
090
091        @Override
092        public FedoraEvent next() {
093            final Iterator<Event> nodeSpecificEvents = sortedEvents.get(ids.next()).iterator();
094            // we can safely call next() immediately on nodeSpecificEvents because if there was no event at all, there
095            // would appear no entry in our Multimap under this key
096            final Event firstEvent = nodeSpecificEvents.next();
097
098            final FedoraEvent fedoraEvent = new FedoraEventImpl(firstEvent);
099
100            addProperty(fedoraEvent, firstEvent);
101            while (nodeSpecificEvents.hasNext()) {
102                // add the event type and property name to the event we are building up to emit
103                // we could aggregate other information here if that seems useful
104                final Event otherEvent = nodeSpecificEvents.next();
105                fedoraEvent.addType(FedoraEventImpl.valueOf(otherEvent.getType()));
106                addProperty(fedoraEvent, otherEvent);
107            }
108            return fedoraEvent;
109        }
110
111        @Override
112        public void remove() {
113            // the underlying Multimap is immutable anyway
114            throw new UnsupportedOperationException();
115        }
116
117        private static void addProperty( final FedoraEvent fedoraEvent, final Event ev ) {
118            try {
119                if ( ev.getPath().contains(JCR_CONTENT)) {
120                    fedoraEvent.addProperty("fedora:hasContent");
121                }
122                if (PROPERTY_EVENT_TYPES.contains(ev.getType())) {
123                    final String eventPath = ev.getPath();
124                    fedoraEvent.addProperty(eventPath.substring(eventPath.lastIndexOf('/') + 1));
125                } else {
126                    LOGGER.trace("Not adding non-event property: {}, {}", fedoraEvent, ev);
127                }
128            } catch (final RepositoryException e) {
129                throw new RepositoryRuntimeException(e);
130            }
131        }
132    }
133}
