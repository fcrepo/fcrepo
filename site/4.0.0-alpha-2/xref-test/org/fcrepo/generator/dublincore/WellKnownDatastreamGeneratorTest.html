54      public void testGetStreamPresent() throws Exception {
55          final String dsid = "foo";
56          testObj.setWellKnownDsid(dsid);
57          final Node mockNode = mock(Node.class);
58          final Node mockDS = mock(Node.class);
59          final Node mockCN = mock(Node.class);
60          final Binary mockB = mock(Binary.class);
61          final Property mockD = mock(Property.class);
62          when(mockNode.hasNode(dsid)).thenReturn(true);
63          when(mockNode.getNode(dsid)).thenReturn(mockDS);
64          when(mockDS.getNode(JCR_CONTENT)).thenReturn(mockCN);
65          when(mockCN.getProperty(JCR_DATA)).thenReturn(mockD);
66          when(mockD.getBinary()).thenReturn(mockB);
67          testObj.getStream(mockNode);
68          verify(mockNode).getNode(dsid);
69      }
