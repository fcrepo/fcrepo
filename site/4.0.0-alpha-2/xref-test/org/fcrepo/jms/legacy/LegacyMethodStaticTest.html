46      @Test
47      public void testCanParse() throws JMSException {
48          // Should the static tests be broken out into a separate test class, so
49          // we can use PowerMock with better scope?
50          boolean result;
51          Message mockYes = mock(Message.class);
52          when(mockYes.getJMSType()).thenReturn(EntryFactory.FORMAT);
53          when(mockYes.getStringProperty("methodName")).thenReturn("ingest");
54          result = LegacyMethod.canParse(mockYes);
55          assertEquals(true, result);
56          Message mockNoFormat = mock(Message.class);
57          when(mockNoFormat.getJMSType()).thenReturn("crazyType");
58          when(mockNoFormat.getStringProperty("methodName")).thenReturn("ingest");
59          result = LegacyMethod.canParse(mockNoFormat);
60          assertEquals(false, result);
61          Message mockNoMessage = mock(Message.class);
62          when(mockNoMessage.getJMSType()).thenReturn(EntryFactory.FORMAT);
63          when(mockNoMessage.getStringProperty("methodName")).thenReturn(
64                  "destroyEverything");
65          result = LegacyMethod.canParse(mockNoMessage);
66          assertEquals(false, result);
67      }

69      @Test
70      public void testObjectToString() {
71          String test = null;
72          String testType = null;
73          assertEquals("null", LegacyMethod.objectToString(test, testType));
74          testType = "fedora-types:ArrayOfString";
75          test = "fo\"o";
76          assertEquals("[UNSUPPORTED" + testType + "]", LegacyMethod
77                  .objectToString(test, testType));
78          testType = "fedora-types:RelationshipTuple";
79          assertEquals("[UNSUPPORTED" + testType + "]", LegacyMethod
80                  .objectToString(test, testType));
81          testType = "xsd:boolean";
82          assertEquals(test, LegacyMethod.objectToString(test, testType));
83          testType = "xsd:nonNegativeInteger";
84          assertEquals(test, LegacyMethod.objectToString(test, testType));
85          testType = "xsd:string";
86          assertEquals("fo'o", LegacyMethod.objectToString(test, testType));

89      @Test
90      public void testGetReturnValue() throws RepositoryException {
91          String testName = "test:nodeName";
92          long testDate = new Date().getTime();
93          String testXSDDate = FedoraTypesUtils.convertDateToXSDString(testDate);
94          Node mockNode = mock(Node.class);
95          Event mockEvent = mock(Event.class);
96          when(mockEvent.getDate()).thenReturn(testDate);
97          when(mockEvent.getType()).thenReturn(NODE_ADDED);
98          when(mockNode.getName()).thenReturn(testName);
99          assertEquals(testName, LegacyMethod.getReturnValue(mockEvent, mockNode));
100         when(mockEvent.getType()).thenReturn(NODE_REMOVED);
101         assertEquals(testXSDDate, LegacyMethod.getReturnValue(mockEvent,
102                 mockNode));
103         when(mockEvent.getType()).thenReturn(PROPERTY_ADDED);
104         assertEquals(testXSDDate, LegacyMethod.getReturnValue(mockEvent,
105                 mockNode));
106         when(mockEvent.getType()).thenReturn(PROPERTY_CHANGED);
107         assertEquals(testXSDDate, LegacyMethod.getReturnValue(mockEvent,
108                 mockNode));
109         when(mockEvent.getType()).thenReturn(PROPERTY_REMOVED);
110         assertEquals(testXSDDate, LegacyMethod.getReturnValue(mockEvent,
111                 mockNode));
