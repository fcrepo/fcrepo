112     /**
113      * POST to store nodeType and hint
114      * 
115      * @param request For now, follows pattern: mix:mimeType image/tiff
116      *        store-hint
117      * @return status
118      */
119 
120     @POST
121     @Consumes(APPLICATION_FORM_URLENCODED)
122     @Timed
123     public Response post(final @PathParam("path") String path,
124                          final String request) throws Exception {
125         LOGGER.debug("POST Received request param: {}", request);
126         Response.ResponseBuilder response;
127 
128         if (!path.equalsIgnoreCase(POLICY_RESOURCE)) {
129             response = Response.status(405);
130             return response.entity(
131                     "POST method not allowed on " + getUriInfo().getAbsolutePath() +
132                             ", try /policies/fcr:storagepolicy")
133                     .build();
134         }
135 
136         final JcrTools jcrTools = getJcrTools();
137         final String[] str = StringUtils.split(request); // simple split for now
138         validateArgs(str.length);
139         try {
140             final Node node =
141                 jcrTools.findOrCreateNode(session,
142                     "/fedora:system/fedora:storage_policy", "test");
143             if (isValidNodeTypeProperty(session, str[0]) ||
144                 isValidConfigurationProperty(str[0])) {
145                 node.setProperty(str[0], new String[] {str[1] + ":" + str[2]});
146 
147                 // TODO (for now) instantiate PolicyType based on mix:mimeType
148                 StoragePolicy policy = newPolicyInstance(str[0], str[1], str[2]);
149                 // TODO (for now) based on object comparison using equals()
150                 if (storagePolicyDecisionPoint.contains(policy)) {
151                     throw new StoragePolicyTypeException("Property already exists!");
152                 }
153                 storagePolicyDecisionPoint.addPolicy(policy);
154                 session.save();
155                 LOGGER.debug("Saved PDS hint {}", request);
156 
157                 response = Response.created(getUriInfo().getBaseUriBuilder()
158                                                     .path(str[0])
159                                                     .path("fcr:storagepolicy")
160                                                     .build());
161             } else {
162                 throw new StoragePolicyTypeException(
163                         "Invalid property type specified: " + str[0]);
164             }
165         } catch (final Exception e) {
166             response = Response.serverError().entity(e.getMessage());
167         } finally {
168             session.logout();
169         }
170 
171         return response.build();
172     }

174     /**
175      * For nodeType n or runtime property p get org.fcrepo.binary.StoragePolicy
176      * implementation. Note: Signature might need to change, or a more
177      * sophisticated method used, as implementation evolves.
178      * 
179      * @param propertyType
180      * @param itemType
181      * @param value
182      * @return
183      * @throws StoragePolicyTypeException
184      */
185     protected StoragePolicy newPolicyInstance(final String propertyType,
186         final String itemType, final String value) throws StoragePolicyTypeException {
187 
188         switch (propertyType) {
189             case NodeType.MIX_MIMETYPE:
190             case "mix:mimeType":
191                 return new MimeTypeStoragePolicy(itemType, value);
192             default:
193                 throw new StoragePolicyTypeException("Mapping not found");
194         }
195     }

248     private Response getAllStoragePolicies() throws Exception {
249         if (storagePolicyDecisionPoint == null ||
250             storagePolicyDecisionPoint.size() == 0) {

252         }
253         return Response.ok(storagePolicyDecisionPoint.toString()).build();

256     private Response getStoragePolicy(final String nodeType) {
257         LOGGER.debug("Get storage policy for: {}", nodeType);
258         Response.ResponseBuilder response;
259         try {
260             final Node node =
261                     jcrTools.findOrCreateNode(session,
262                                               "/fedora:system/fedora:storage_policy",
263                                               "test");
264 
265             Property prop = node.getProperty(nodeType);
266             if (null == prop) {
267                 throw new PathNotFoundException("StoragePolicy not found: " + nodeType);
268             }
269 
270             Value[] values = prop.getValues();
271             if (values != null && values.length > 0) {
272                 response = Response.ok(values[0].getString());
273             } else {
274                 throw new PathNotFoundException("StoragePolicy not found: " + nodeType);
275             }
276 
277         } catch (PathNotFoundException e) {
278             response = Response.status(404).entity(e.getMessage());
279         } catch (Exception e) {
280             response = Response.serverError().entity(e.getMessage());
281 
282         } finally {
283             session.logout();
284         }
285         return response.build();
286     }

289      * Verifies whether node type is valid
290      * 
291      * @param session
292      * @param type
293      * @return
294      * @throws RepositoryException
295      */
296     private boolean isValidNodeTypeProperty(final Session session,
297         final String type) throws RepositoryException {
298         try {
299             return session.getWorkspace().getNodeTypeManager()
300                 .getNodeType(type).getName().equals(type);
301         } catch (NoSuchNodeTypeException e) {
302             LOGGER.debug("No corresponding Node type found for: {}", type);
303             return false;
304         }
305     }
