017package org.fcrepo.http.commons.responses;
018
019import static com.google.common.collect.ImmutableList.of;
020import static org.slf4j.LoggerFactory.getLogger;
021
022import java.io.IOException;
023import java.io.OutputStream;
024import java.lang.annotation.Annotation;
025import java.lang.reflect.Type;
026
027import javax.ws.rs.WebApplicationException;
028import javax.ws.rs.core.MediaType;
029import javax.ws.rs.core.MultivaluedMap;
030import javax.ws.rs.ext.MessageBodyWriter;
031import javax.ws.rs.ext.Provider;
032
033import org.slf4j.Logger;
034import org.springframework.stereotype.Component;
035
036import com.hp.hpl.jena.query.QueryExecution;
037import com.hp.hpl.jena.query.ResultSet;
038import com.hp.hpl.jena.sparql.resultset.ResultsFormat;
039
040/**
041 * Helper for writing QueryExecutions results out in a variety
042 * of serialization formats.
043 *
044 */
045@Provider
046@Component
047public class QueryExecutionProvider implements MessageBodyWriter<QueryExecution> {
048
049    private static final Logger logger = getLogger(QueryExecutionProvider.class);
050
051    @Override
052    public void writeTo(final QueryExecution qexec, final Class<?> type,
053            final Type genericType, final Annotation[] annotations,
054            final MediaType mediaType,
055            final MultivaluedMap<String, Object> httpHeaders,
056            final OutputStream entityStream) throws IOException,
057        WebApplicationException {
058
059        logger.debug("Writing a response for: {} with MIMEtype: {}", qexec,
060                mediaType);
061
062        // add standard headers
063        httpHeaders.put("Content-type", of((Object) mediaType.toString()));
064
065        try {
066            final ResultSet results = qexec.execSelect();
067            new ResultSetStreamingOutput(results, mediaType).write(entityStream);
068        } finally {
069            qexec.close();
070        }
071    }
072
073    @Override
074    public boolean isWriteable(final Class<?> type, final Type genericType,
075            final Annotation[] annotations, final MediaType mediaType) {
076
077        // we can return a result for any MIME type that Jena can serialize
078        final Boolean appropriateResultType = ResultSetStreamingOutput.getResultsFormat(mediaType) != ResultsFormat.FMT_UNKNOWN;
079        return appropriateResultType &&
080                (QueryExecution.class.isAssignableFrom(type) || QueryExecution.class
081                        .isAssignableFrom(genericType.getClass()));
082    }
083
084    @Override
085    public long getSize(final QueryExecution rdf, final Class<?> type,
086            final Type genericType, final Annotation[] annotations,
087            final MediaType mediaType) {
088        // we don't know in advance how large the result might be
089        return -1;
090    }
091
092}
