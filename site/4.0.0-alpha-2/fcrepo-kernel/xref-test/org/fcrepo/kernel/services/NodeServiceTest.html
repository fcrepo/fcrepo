58      public void setUp() throws RepositoryException {

60          testObj = new NodeService();
61          when(mockSession.getRootNode()).thenReturn(mockRoot);
62      }

64      @SuppressWarnings("unchecked")
65      @Test
66      public void testGetObjectNames() throws RepositoryException {
67          final String objPath = "";
68          when(mockObjNode.getName()).thenReturn("foo");
69          when(mockObjNode.isNodeType("nt:folder")).thenReturn(true);
70          final NodeIterator mockIter = mock(NodeIterator.class);
71          when(mockIter.hasNext()).thenReturn(true, false);
72          when(mockIter.nextNode()).thenReturn(mockObjNode).thenThrow(
73                  IndexOutOfBoundsException.class);
74          when(mockRoot.getNodes()).thenReturn(mockIter);
75          when(mockSession.getNode(objPath)).thenReturn(mockRoot);
76          final Set<String> actual = testObj.getObjectNames(mockSession, "");
77          verify(mockSession).getNode(objPath);
78          assertEquals(1, actual.size());
79          assertEquals("foo", actual.iterator().next());
80      }
81  
82      @Test

84          final String objPath = "foo";
85          final Node mockObjectsNode = mock(Node.class);
86          when(mockSession.getRootNode()).thenReturn(mockRoot);
87          when(mockRoot.getNode("objects")).thenReturn(mockObjectsNode);
88          when(mockSession.getNode(objPath)).thenReturn(mockObjNode);
89          mockStatic(ServiceHelpers.class);
