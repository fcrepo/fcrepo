016package org.fcrepo.integration.auth.oauth.api;
017
018import java.util.Collection;
019import java.util.Collections;
020import java.util.List;
021
022import javax.xml.bind.annotation.XmlElement;
023import javax.xml.bind.annotation.XmlElements;
024import javax.xml.bind.annotation.XmlRootElement;
025
026import org.fcrepo.integration.auth.oauth.api.bind.ContextParam;
027import org.fcrepo.integration.auth.oauth.api.bind.FilterMapping;
028import org.fcrepo.integration.auth.oauth.api.bind.ServletMapping;
029import org.fcrepo.integration.auth.oauth.api.bind.Displayable;
030import org.fcrepo.integration.auth.oauth.api.bind.Filter;
031import org.fcrepo.integration.auth.oauth.api.bind.Listener;
032import org.fcrepo.integration.auth.oauth.api.bind.Servlet;
033
034import com.google.common.base.Predicate;
035import com.google.common.collect.Collections2;
036
037@XmlRootElement(namespace = "http://java.sun.com/xml/ns/javaee",
038        name = "web-app")
039public class WebAppConfig extends Displayable {
040
041    @XmlElements(value = {@XmlElement(
042            namespace = "http://java.sun.com/xml/ns/javaee",
043            name = "context-param")})
044    List<ContextParam> contextParams;
045
046    @XmlElements(
047            value = {@XmlElement(
048                    namespace = "http://java.sun.com/xml/ns/javaee",
049                    name = "listener")})
050    List<Listener> listeners;
051
052    @XmlElements(value = {@XmlElement(
053            namespace = "http://java.sun.com/xml/ns/javaee", name = "servlet")})
054    List<Servlet> servlets;
055
056    @XmlElements(value = {@XmlElement(
057            namespace = "http://java.sun.com/xml/ns/javaee", name = "filter")})
058    List<Filter> filters;
059
060    @XmlElements(value = {@XmlElement(
061            namespace = "http://java.sun.com/xml/ns/javaee",
062            name = "servlet-mapping")})
063    List<ServletMapping> servletMappings;
064
065    @XmlElements(value = {@XmlElement(
066            namespace = "http://java.sun.com/xml/ns/javaee",
067            name = "filter-mapping")})
068    List<FilterMapping> filterMappings;
069
070    public Collection<ServletMapping> servletMappings(String servletName) {
071        return Collections2
072                .filter(servletMappings, new SMapFilter(servletName));
073    }
074
075    public Collection<FilterMapping> filterMappings(String filterName) {
076        return Collections2.filter(filterMappings, new FMapFilter(filterName));
077    }
078
079    @SuppressWarnings("unchecked")
080    public Collection<ContextParam> contextParams() {
081        return (contextParams != null) ? contextParams : Collections.EMPTY_LIST;
082    }
083
084    private static class SMapFilter implements Predicate<ServletMapping> {
085
086        String servletName;
087
088        SMapFilter(String sName) {
089            servletName = sName;
090        }
091
092        @Override
093        public boolean apply(ServletMapping input) {
094            return (servletName == null) ? input.servletName() == null
095                    : servletName.equals(input.servletName());
096        }
097
098    }
099
100    private static class FMapFilter implements Predicate<FilterMapping> {
101
102        String filterName;
103
104        FMapFilter(String sName) {
105            filterName = sName;
106        }
107
108        @Override
109        public boolean apply(FilterMapping input) {
110            return (filterName == null) ? input.filterName() == null
111                    : filterName.equals(input.filterName());
112        }
113
114    }
115}
