76      /**
77       * Used to get an attribute. If one of those values isn't being asked for,
78       * or if the types are wrong, then an empty bag is returned.
79       *
80       * @param attributeType
81       *        the datatype of the attributes to find, which must be time, date,
82       *        or dateTime for this module to resolve a value
83       * @param attributeId
84       *        the identifier of the attributes to find, which must be one of the
85       *        three ENVIRONMENT_* fields for this module to resolve a value
86       * @param issuer
87       *        the issuer of the attributes, or null if unspecified
88       * @param subjectCategory
89       *        the category of the attribute or null, which ignored since this
90       *        only handles non-subjects
91       * @param context
92       *        the representation of the request data
93       * @param designatorType
94       *        the type of designator, which must be ENVIRONMENT_TARGET for this
95       *        module to resolve a value
96       * @return the result of attribute retrieval, which will be a bag with a
97       *         single attribute, an empty bag, or an error
98       */
99      @Override
100     public EvaluationResult findAttribute(URI attributeType,
101                                           URI attributeId,
102                                           URI issuer,
103                                           URI subjectCategory,
104                                           EvaluationCtx context,
105                                           int designatorType) {
106 
107         String resourceId = context.getResourceId().encode();
108         if (resourceId == null || resourceId.equals("")) {
109             String pid = PolicyFinderModule.getPid(context);
110             if (pid != null) {
111                 resourceId = "info:fedora/" + pid;
112             }
113         }
114         if (logger.isDebugEnabled()) {
115             logger.debug("RIAttributeFinder: [" + attributeType.toString() + "] "
116                     + attributeId + ", rid=" + resourceId);
117         }
118 
119         if (resourceId == null || resourceId.equals("")) {
120             return new EvaluationResult(BagAttribute
121                     .createEmptyBag(attributeType));
122         }
123 
124         if (resourceId.equals("/FedoraRepository")) {
125             return new EvaluationResult(BagAttribute
126                                         .createEmptyBag(attributeType));
127         }
128 
129         // figure out which attribute we're looking for
130         String attrName = attributeId.toString();
131 
132         // we only know about registered attributes from Spring config
133         if (!m_attributes.containsKey(designatorType)) {
134             if (logger.isDebugEnabled()) {
135                 logger.debug("Does not know about designatorType: "
136                         + designatorType);
137             }
138             return new EvaluationResult(BagAttribute
139                     .createEmptyBag(attributeType));
140         }
141 
142         Set<String> allowedAttributes =
143             m_attributes.get(designatorType).keySet();
144         if (!allowedAttributes.contains(attrName)) {
145             if (logger.isDebugEnabled()) {
146                 logger.debug("Does not know about attribute: " + attrName);
147             }
148             return new EvaluationResult(BagAttribute
149                     .createEmptyBag(attributeType));
150         }
151 
152         EvaluationResult result = null;
153         try {
154             result = getEvaluationResult(resourceId, attrName, designatorType, attributeType);
155         } catch (Exception e) {
156             logger.error("Error finding attribute: " + e.getMessage(), e);
157             return new EvaluationResult(BagAttribute
158                     .createEmptyBag(attributeType));
159         }
160 
161         return result;
162     }

164     /**
165      *
166      * @param resourceID - the hierarchical XACML resource ID
167      * @param attribute - attribute to get - this is a URI that maps to a Fedora relationship name
168      * @param type
169      * @return
170      * @throws AttributeFinderException
171      */
172     private EvaluationResult getEvaluationResult(String resourceID,
173                                                  String attribute,
174                                                  int designatorType,
175                                                  URI type)
176             throws AttributeFinderException {
177 
178         // split up the path of the hierarchical resource id
179         String resourceParts[] = resourceID.split("/");
180         Set<String> results = null;
181 
182         // either the last part is the pid, or the last-but one is the pid and the last is the datastream
183         // if we have a pid, we query on that, if we have a datastream we query on the datastream
184         String pid;
185         if (resourceParts.length > 1) {
186             if (resourceParts[resourceParts.length - 1].contains(":")) { // ends with a pid, we have pid only
187                 pid = resourceParts[resourceParts.length - 1];
188             } else { // datastream
189                 pid = resourceParts[resourceParts.length - 2];
190             }
191         } else {
192             // eg /FedoraRepository, not a valid path to PID or PID/DS
193             logger.debug("Resource ID not valid path to PID or datastream: " + resourceID);
194             return new EvaluationResult(BagAttribute.createEmptyBag(type));
195         }
196 
197         logger.debug("Getting attribute for resource " + resourceID);
198 
199         try{
200             SubjectNode snode = new SimpleURIReference(new URI(resourceID));
201             PredicateNode pnode = new SimpleURIReference(new URI(attribute));
202             DOReader reader = m_doManager.getReader(false, getContext(), pid);
203             Set<RelationshipTuple> triples = reader.getRelationships(snode, pnode, null);
204             results = new HashSet<String>();
205 
206             for (RelationshipTuple triple:triples){
207                 results.add(triple.object);
208             }
209         }
210         catch (Exception e){
211             logger.warn("Error retreiving triples in attributeFinder",e);
212         }
213 
214         if (results == null || results.isEmpty()) {
215             return new EvaluationResult(BagAttribute.createEmptyBag(type));
216         }
217 
218 
219         Set<AttributeValue> bagValues = new HashSet<AttributeValue>();
220         logger.debug("Attribute values found: " + results.size());
221         for (String s : results) {
222             AttributeValue attributeValue = null;
223             try {
224                 attributeValue = m_attributeFactory.createValue(type, s);
225             } catch (Exception e) {
226                 logger.error("Error creating attribute: " + e.getMessage(), e);
227                 continue;
228             }
229 
230             bagValues.add(attributeValue);
231 
232             if (logger.isDebugEnabled()) {
233                 logger.debug("AttributeValue found: [" + type.toASCIIString()
234                              + "] " + s);
235             }
236 
237         }
238 
239         BagAttribute bag = new BagAttribute(type, bagValues);
240         return new EvaluationResult(bag);
241 
242     }
