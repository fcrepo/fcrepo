94      @Override
95      public void init() throws PolicyStoreException, FileNotFoundException {
96          if (log.isDebugEnabled()) {
97              Runtime runtime = Runtime.getRuntime();
98              log.debug("Total memory: " + runtime.totalMemory() / 1024);
99              log.debug("Free memory: " + runtime.freeMemory() / 1024);
100             log.debug("Max memory: " + runtime.maxMemory() / 1024);
101         }
102         super.init();
103         // if no pid namespace was specified, use the default specified in fedora.fcfg
104         if (pidNamespace.equals("")) {
105             pidNamespace = fedoraServer.getModule("org.fcrepo.server.storage.DOManager").getParameter("pidNamespace");
106         }
107 
108         // check control group was supplied
109         if (datastreamControlGroup.equals("")) {
110             throw new PolicyStoreException("No control group for policy datastreams was specified in FedoraPolicyStore configuration");
111         }
112         if (validateSchema) {
113             String schemaLocation = schemaLocations.get(XACML20_POLICY_NS);
114             if ( schemaLocation == null) {
115                 throw new PolicyStoreException("Configuration error - no policy schema specified");
116             }
117             try{
118             String serverHome =
119                     fedoraServer.getHomeDir().getCanonicalPath() + File.separator;
120 
121                 String schemaPath =
122                     ((schemaLocation)
123                             .startsWith(File.separator) ? "" : serverHome)
124                             + schemaLocation;
125                 FileInputStream in = new FileInputStream(schemaPath);
126                 PolicyParser policyParser = new PolicyParser(in);
127                 ValidationUtility.setFeslPolicyParser(policyParser);
128             } catch (IOException ioe) {
129                 throw new PolicyStoreException(ioe.getMessage(),ioe);
130             } catch (SAXException se) {
131                 throw new PolicyStoreException(se.getMessage(),se);
132             }
133         }
134     }

206     /*
207      * (non-Javadoc)
208      * @see
209      * org.fcrepo.server.security.xacml.pdp.data.PolicyDataManager#addPolicy
210      * (java.lang.String, java.lang.String)
211      */
212     @Override
213     public String addPolicy(String document, String name)
214     throws PolicyStoreException {
215 
216         String policyName;
217 
218         if (name == null || name.equals("")) {
219             // no policy name, derive from document
220             // (note: policy ID is mandatory according to schema)
221             try {
222                 policyName = utils.getPolicyName(document);
223             } catch (MelcoePDPException e) {
224                 throw new PolicyStoreException("Could not get policy name from policy", e);
225             }
226             // if name from document contains pid separator, escape it
227             if (name.contains(":")) {
228                 name = name.replace(":", PID_SEPARATOR_ESCAPED);
229             }
230 
231         } else {
232             policyName = name;
233         }
234         String pid = this.getPID(policyName);
235 
236         ObjectProfile objectProfile = null;
237         try {
238             objectProfile =
239                 this.apiAService.getObjectProfile(getContext(), pid, null);
240         } catch (ObjectNotInLowlevelStorageException e) {
241         } catch (ServerException e) {
242             throw new PolicyStoreException("Add: error getting object profile for "
243                                                  + pid
244                                                  + " - "
245                                                  + e.getMessage(),
246                                                  e);
247         }
248 
249         if (objectProfile != null) {
250             // object exists, check state
251             if (objectProfile.objectState != "D") {
252                 throw new PolicyStoreException("Add:  attempting to add policy "
253                                                      + pid + " but it already exists");
254             }
255             // deleted object:  set state to active and do an update instead
256             try {
257                 this.apiMService
258                 .modifyObject(getContext(),
259                               pid,
260                               "A",
261                               objectProfile.objectLabel,
262                               objectProfile.objectOwnerId,
263                 "Fedora policy manager:  Adding policy by activating deleted object",null);
264             } catch (ServerException e) {
265                 throw new PolicyStoreException("Add: " + e.getMessage(),
266                                                      e);
267             }
268             this.updatePolicy(policyName, document);
269             return pid;
270         } else {
271             // create new object
272 
273             // if control group is M - managed - we need a temp location for the datastream
274             String dsLocationOrContent = null;
275             if (datastreamControlGroup.equals("M")) {
276                 try {
277                     ByteArrayInputStream is =
278                         new ByteArrayInputStream(document.getBytes("UTF-8"));
279                     dsLocationOrContent =
280                         apiMService.putTempStream(getContext(), is);
281                 } catch (Exception e) {
282                     throw new PolicyStoreException("Add: error generating temp datastream location - "
283                                                          + e
284                                                          .getMessage(),
285                                                          e);
286                 }
287             } else {
288                 dsLocationOrContent = document;
289             }
290 
291             try {
292                 return apiMService
293                 .ingest(getContext(),
294                         new ByteArrayInputStream(getFOXMLPolicyTemplate(pid,
295                                                                         "XACML policy " + policyName,
296                                                                         contentModel,
297                                                                         collection,
298                                                                         collectionRelationship,
299                                                                         dsLocationOrContent,
300                                                                         datastreamControlGroup)
301                                                                         .getBytes("UTF-8")),
302                                                                         "Fedora Policy Manager creating policy",
303                                                                         Constants.FOXML1_1.uri,
304                                                                         "UTF-8",
305                                                                         "");
306             } catch (Exception e) {
307                 throw new PolicyStoreException("Add: error ingesting "
308                                                      + pid + " - " + e.getMessage(), e);
309             }
310 
311         }
312     }

334                                           null);
335         } catch (ServerException e) {
336             throw new PolicyStoreException("Delete: error deleting policy "

346     /*
347      * (non-Javadoc)
348      * @see
349      * org.fcrepo.server.security.xacml.pdp.data.PolicyDataManager#updatePolicy
350      * (java.lang.String, java.lang.String)
351      */
352     @Override
353     public boolean updatePolicy(String name, String newDocument)
354     throws PolicyStoreException {
355 
356         String pid = this.getPID(name);
357 
358         if (!contains(name)) {
359             throw new PolicyStoreException("Update:  policy " + pid
360                                                  + " not found");
361         }
362 
363         if (datastreamControlGroup.equals("X")) {
364             // inline, modify by value
365             try {
366                 this.apiMService
367                 .modifyDatastreamByValue(getContext(),
368                                          pid,
369                                          FESL_POLICY_DATASTREAM,
370                                          null,
371                                          null,
372                                          null,
373                                          null,
374                                          new ByteArrayInputStream(newDocument
375                                                                   .getBytes("UTF-8")),
376                                                                   "DISABLED",
377                                                                   null,
378                                                                   "Modifying policy " + pid,
379                                                                   null);
380             } catch (Exception e) {
381                 throw new PolicyStoreException("Update:  error modifying datastream by value for "
382                                                      + pid
383                                                      + " - "
384                                                      + e.getMessage(),
385                                                      e);
386             }
387 
388         } else if (datastreamControlGroup.equals("M")) {
389             // managed, generate temp location, modify by reference
390             String dsLocation = null;
391             try {
392                 ByteArrayInputStream is =
393                     new ByteArrayInputStream(newDocument.getBytes("UTF-8"));
394                 dsLocation = apiMService.putTempStream(getContext(), is);
395             } catch (Exception e) {
396                 throw new PolicyStoreException("Update: error generating temp datastream location - "
397                                                      + e.getMessage(),
398                                                      e);
399             }
400             try {
401                 apiMService.modifyDatastreamByReference(getContext(),
402                                                         pid,
403                                                         FESL_POLICY_DATASTREAM,
404                                                         null,
405                                                         null,
406                                                         null,
407                                                         null,
408                                                         dsLocation,
409                                                         "DISABLED",
410                                                         null,
411                                                         "Modifying policy "
412                                                         + pid,
413                                                         null);
414             } catch (ServerException e) {
415                 throw new PolicyStoreException("Update:  error modifying datastream by reference for "
416                                                      + pid
417                                                      + " - "
418                                                      + e.getMessage(),
419                                                      e);
420             }
421 
422         } else {
423             throw new PolicyStoreException("Update:  Invalid datastream control group "
424                                                  + datastreamControlGroup + " - use M or X");
425         }
426 
427         return true;
428     }

458     /**
459      * Check if the policy identified by policyName exists.
460      *
461      * @param policyName
462      * @return true iff the policy store contains a policy identified as
463      *         policyName
464      * @throws PolicyStoreException
465      */
466     @Override
467     public boolean contains(String policyName)
468     throws PolicyStoreException {
469 
470         // search for policy - active and inactive
471         String pid = this.getPID(policyName);
472 
473         ObjectProfile objectProfile = null;
474         try {
475             objectProfile =
476                 this.apiAService.getObjectProfile(getContext(), pid, null);
477         } catch (ObjectNotInLowlevelStorageException e) {
478         } catch (ServerException e) {
479             throw new PolicyStoreException("Add: error getting object profile for "
480                                                  + pid
481                                                  + " - "
482                                                  + e.getMessage(),
483                                                  e);
484 
485         }
486 
487         if (objectProfile == null) {
488             // no object found
489             return false;
490         } else {
491             if (objectProfile.objectState.equals("A")
492                     || objectProfile.objectState.equals("I")) {
493                 // active or inactive object found - policy exists
494                 return true;
495             } else {
496                 // deleted object - policy does not exist
497                 return false;
498             }
499         }
500     }

533     /**
534      * Given a policy name (corresponds to Policy identifier in XACML), generate a PID
535      *
536      * If the name already contains a pid namespace, use that, otherwise use the default
537      *
538      * @param name
539      * @return normalized name (a PID)
540      * @throws PolicyStoreException
541      * @throws MalformedPIDException
542      */
543     private String getPID(String name) throws PolicyStoreException {
544 
545         String pid;
546         // only bootstrap policies specify the PID namespace, all others follow the config
547         if (name.startsWith(FESL_BOOTSTRAP_POLICY_NAMESPACE + ":")) {
548             pid = name;
549         } else {
550             // TODO: would be nice to have the PID class contain a method for this
551             // (could be used as a generic method instead of the from-filename etc methods)
552 
553             // name might contain non-legal PID characters so encode them
554             StringBuffer out = new StringBuffer();
555             for (int i = 0; i < name.length(); i++) {
556                 char c = name.charAt(i);
557                 // valid pid characters
558                 if (isAlphaNum(c) || c == '-' || c == '.' || c == '~'
559                     || c == '_') {
560                 out.append(c);
561                 } else {
562                     // do each byte
563                     // FIXME: percent-encoding causing various issues
564                     // (not least with web admin client)
565                     out.append("_");
566                     /*
567                     byte[] bytes;
568                     try {
569                         bytes = Character.toString(c).getBytes("UTF-8");
570                     } catch (UnsupportedEncodingException e) {
571                         // should never happen
572                         throw new RuntimeException(e);
573                     }
574                     for (byte b : bytes ) {
575                         // percent-encode the byte
576                         out.append("%");
577                         out.append(hexChar[(b >>> 4) & 0xf]);
578                         out.append(hexChar[b & 0xf]);
579                     }
580                     */
581                 }
582             }
583             pid = pidNamespace + ":" + out.toString();
584         }
585 
586         try {
587             // just in case...
588             return PID.normalize(pid);
589         } catch (MalformedPIDException e) {
590             throw new PolicyStoreException("Invalid policy name '" + name
591                                                  + "'.  Could not create a valid PID from this name: " + e.getMessage(), e);
592         }
593 
594     }

608     /**
609      * Generate FOXML for a new policy object
610      * @param pid
611      * @param label
612      * @param contentModel
613      * @param policyOrLocation
614      * @param controlGroup
615      * @return
616      * @throws PolicyStoreException
617      */
618     private static String getFOXMLPolicyTemplate(String pid,
619                                                  String label,
620                                                  String contentModel,
621                                                  String collection,
622                                                  String collectionRelationship,
623                                                  String policyOrLocation,
624                                                  String controlGroup)
625     throws PolicyStoreException {
626         StringBuilder foxml = new StringBuilder();
627         foxml.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
628         foxml.append("<foxml:digitalObject VERSION=\"1.1\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"\n");
629         foxml.append("    xmlns:foxml=\"info:fedora/fedora-system:def/foxml#\"\n");
630         foxml.append("           xsi:schemaLocation=\"" + Constants.FOXML.uri
631                      + " " + Constants.FOXML1_1.xsdLocation + "\"");
632         foxml.append("\n           PID=\"" + StreamUtility.enc(pid)
633                          + "\">\n");
634         foxml.append("  <foxml:objectProperties>\n");
635         foxml.append("    <foxml:property NAME=\"info:fedora/fedora-system:def/model#state\" VALUE=\"A\"/>\n");
636         foxml.append("    <foxml:property NAME=\"info:fedora/fedora-system:def/model#label\" VALUE=\""
637                 + StreamUtility.enc(label) + "\"/>\n");
638         foxml.append("  </foxml:objectProperties>\n");
639 
640 
641         // RELS-EXT specifying content model - if present, collection relationship if present
642         // but not for bootstrap policies
643         if (!pid.startsWith(FESL_BOOTSTRAP_POLICY_NAMESPACE + ":")) {
644             if (!contentModel.equals("") || !collection.equals("")) {
645                 foxml.append("<foxml:datastream ID=\"RELS-EXT\" CONTROL_GROUP=\"X\">");
646                 foxml.append("<foxml:datastreamVersion FORMAT_URI=\"info:fedora/fedora-system:FedoraRELSExt-1.0\" ID=\"RELS-EXT.0\" MIMETYPE=\"application/rdf+xml\" LABEL=\"RDF Statements about this object\">");
647                 foxml.append("  <foxml:xmlContent>");
648                 foxml.append("   <rdf:RDF xmlns:rdf=\"http://www.w3.org/1999/02/22-rdf-syntax-ns#\" xmlns:rdfs=\"http://www.w3.org/2000/01/rdf-schema#\" xmlns:fedora-model=\"info:fedora/fedora-system:def/model#\" xmlns:rel=\"info:fedora/fedora-system:def/relations-external#\">");
649                 foxml.append("      <rdf:Description rdf:about=\"" + "info:fedora/"
650                              + StreamUtility.enc(pid) + "\">");
651                 if (!contentModel.equals("")) {
652                     foxml.append("        <fedora-model:hasModel rdf:resource=\""
653                                  + StreamUtility.enc(contentModel) + "\"/>");
654                 }
655                 if (!collection.equals("")) {
656                     foxml.append("        <rel:" + StreamUtility.enc(collectionRelationship) + " rdf:resource=\""
657                                  + StreamUtility.enc(collection) + "\"/>");
658                 }
659                 foxml.append("       </rdf:Description>");
660                 foxml.append("      </rdf:RDF>");
661                 foxml.append("    </foxml:xmlContent>");
662                 foxml.append("  </foxml:datastreamVersion>");
663                 foxml.append("</foxml:datastream>");
664             }
665         }
666         // the POLICY datastream
667         foxml.append("<foxml:datastream ID=\"" + FESL_POLICY_DATASTREAM
668                      + "\" CONTROL_GROUP=\"" + controlGroup + "\">");
669         foxml.append("<foxml:datastreamVersion ID=\"POLICY.0\" MIMETYPE=\"text/xml\" LABEL=\"XACML policy datastream\">");
670         if (controlGroup.equals("M")) {
671             foxml.append("  <foxml:contentLocation REF=\"" + policyOrLocation
672                          + "\" TYPE=\"" + org.fcrepo.server.storage.types.Datastream.DS_LOCATION_TYPE_URL + "\"/>");
673 
674         } else if (controlGroup.equals("X")) {
675             foxml.append("  <foxml:xmlContent>");
676             foxml.append(policyOrLocation);
677             foxml.append("    </foxml:xmlContent>");
678 
679         } else {
680             throw new PolicyStoreException("Generating new object XML:  Invalid control group: "
681                                                  + controlGroup + " - use X or M.");
682         }
683         foxml.append("  </foxml:datastreamVersion>");
684         foxml.append("</foxml:datastream>");
685         foxml.append("</foxml:digitalObject>");
686 
687         return foxml.toString();
688     }
