63      /* Assure that listMethods works as advertised */
64      @Test
65      public void testListMethods() throws Exception {
66          FedoraAPIAMTOM apia = m_client.getAPIAMTOM();
67          ObjectMethodsDef[] methods;
68  
69          methods = filterMethods(apia.listMethods(OBJECT_1_PID, null).toArray(new ObjectMethodsDef[0]));
70  
71          assertEquals("Wrong number of methods", 2, methods.length);
72          assertNotSame("SDeps are not distinct", methods[0]
73                  .getServiceDefinitionPID(), methods[1]
74                  .getServiceDefinitionPID());
75          assertNotSame("Methods are not distinct",
76                        methods[0].getMethodName(),
77                        methods[1].getMethodName());
78  
79          /* Order may not be deterministic, but that's OK */
80          if (methods[0].getServiceDefinitionPID().equals(SDEF_1_PID)) {
81              assertEquals(methods[1].getServiceDefinitionPID(), SDEF_2_PID);
82              assertEquals(methods[0].getMethodName(), SDEF_1_METHOD);
83              assertEquals(methods[1].getMethodName(), SDEF_2_METHOD);
84          } else if (methods[0].getServiceDefinitionPID().equals(SDEF_2_PID)) {
85              assertEquals(methods[1].getServiceDefinitionPID(), SDEF_1_PID);
86              assertEquals(methods[0].getMethodName(), SDEF_2_METHOD);
87              assertEquals(methods[1].getMethodName(), SDEF_1_METHOD);
88          }
89      }
