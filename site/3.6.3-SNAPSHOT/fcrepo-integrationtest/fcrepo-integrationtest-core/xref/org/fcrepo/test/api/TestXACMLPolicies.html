91      public void testXACMLUnmodifiableContentModel() throws Exception {
92  
93          // test policy disallows modifyObject for test:RestrictedCModel
94          final String unrestrictedCModel = "test:UnrestrictedCModel";
95          final String restrictedCModel = "test:RestrictedCModel";
96  
97          final String hasUnrestricted = "test:HasUnrestrictedCModel";
98          addTestObject(hasUnrestricted, null, unrestrictedCModel);
99  
100         final String hasRestricted = "test:HasRestrictedCModel";
101         addTestObject(hasRestricted, null, restrictedCModel);
102 
103         final String hasUnrestrictedAndRestricted =
104                 "test:HasUnrestrictedAndRestrictedCModel";
105         addTestObject(hasUnrestrictedAndRestricted,
106                       null,
107                       unrestrictedCModel,
108                       restrictedCModel);
109 
110         final String hasRestrictedAndUnrestricted =
111                 "test:HasRestrictedAndUnrestrictedCModel";
112         addTestObject(hasRestrictedAndUnrestricted,
113                       null,
114                       restrictedCModel,
115                       unrestrictedCModel);
116 
117         try {
118             assertTrue(canWrite(admin, hasUnrestricted,"testXACMLUnmodifiableContentModel"));
119             assertFalse(canWrite(admin, hasRestricted,"testXACMLUnmodifiableContentModel"));
120             assertFalse(canWrite(admin, hasUnrestrictedAndRestricted,"testXACMLUnmodifiableContentModel"));
121             assertFalse(canWrite(admin, hasRestrictedAndUnrestricted,"testXACMLUnmodifiableContentModel"));
122         } finally {
123             removeTestObject(hasUnrestricted);
124             removeTestObject(hasRestricted);
125             removeTestObject(hasUnrestrictedAndRestricted);
126             removeTestObject(hasRestrictedAndUnrestricted);
127         }

439                                   Class args[],

447             Method func = apim1.getClass().getMethod(functionToTest, args);
448             Object result = func.invoke(apim1, parms);
449             fail("Illegal access allowed");

451             Throwable cause = ite.getCause();
452             if (cause instanceof SOAPFaultException) {
453                 SOAPFaultException af =
454                         (SOAPFaultException) cause;
455                 System.out.println("    Reason = "
456                         + af.getMessage());
457                 assertTrue(af.getMessage(),
458                            af.getMessage().contains("Authorization Denied"));
459                 System.out.println("Access denied correctly");
460             } else {
461                 System.out.println("Got exception: "
462                         + cause.getClass().getName());
463                 fail("Illegal access dis-allowed for some other reason: " +
464                      cause.getClass().getName());
465             }
466         } catch (IOException ioe) {
467             System.out.println("    Reason = " + ioe.getMessage()/*
468                                                                   * .substring(ioe
469                                                                   * .
470                                                                   * getMessage()
471                                                                   * .
472                                                                   * lastIndexOf(
473                                                                   * "["))
474                                                                   */);
475             assertTrue(ioe.getMessage().contains("[403 Forbidden]"));

477             // exception was expected, all is A-OK
478         } catch (Exception ae) {
479             String message = "Some other exception: " + ae.toString();

625     public boolean isAPIAAuthzOn() throws IOException {
626         File installProperties =
627                 new File(FEDORA_HOME, "install/install.properties");
628         BufferedReader prop = null;
629         try {
630             prop = new BufferedReader(new FileReader(installProperties));
631             String line = null;
632             while ((line = prop.readLine()) != null) {
633                 if (line.startsWith("apia.auth.required")) {
634                     if (line.equals("apia.auth.required=true")) {
