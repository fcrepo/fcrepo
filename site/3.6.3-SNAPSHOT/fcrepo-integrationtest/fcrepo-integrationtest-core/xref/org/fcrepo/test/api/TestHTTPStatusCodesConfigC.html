397     private static FedoraClient getClient(boolean validUser,
398                                           boolean validPass,
399                                           boolean authorized) throws Exception {
400         if (validUser) {
401             if (validPass) {
402                 System.out
403                         .println("Using Fedora Client with valid user, valid pass");
404                 if (authorized) {
405                     if (CLIENT_VALID_USER_VALID_PASS == null) {
406                         CLIENT_VALID_USER_VALID_PASS = getFedoraClient();
407                     }
408                     return CLIENT_VALID_USER_VALID_PASS;
409                 } else {
410                     if (CLIENT_VALID_USER_VALID_PASS_UNAUTHORIZED == null) {
411                         CLIENT_VALID_USER_VALID_PASS_UNAUTHORIZED =
412                                 getFedoraClient(getBaseURL(),
413                                                 "untrustedUser",
414                                                 "password");
415                     }
416                     return CLIENT_VALID_USER_VALID_PASS_UNAUTHORIZED;
417                 }
418             } else {
419                 System.out
420                         .println("Using Fedora Client with valid user, bogus pass");
421                 if (CLIENT_VALID_USER_BOGUS_PASS == null) {
422                     CLIENT_VALID_USER_BOGUS_PASS =
423                             getFedoraClient(getBaseURL(),
424                                             getUsername(),
425                                             "bogus");
426                 }
427                 return CLIENT_VALID_USER_BOGUS_PASS;
428             }
429         } else {
430             System.out.println("Using Fedora Client with bogus user");
431             if (CLIENT_BOGUS_USER == null) {
432                 CLIENT_BOGUS_USER =
433                         getFedoraClient(getBaseURL(), "bogus", "bogus");
434             }
435             return CLIENT_BOGUS_USER;
436         }

498     private static void copyFile(File sourceFile, File destFile)
499             throws Exception {
500         FileInputStream in = new FileInputStream(sourceFile);
501         FileOutputStream out = new FileOutputStream(destFile);
502         byte[] buf = new byte[4096];
503         int len;
504         try {
505             while ((len = in.read(buf)) > 0) {
506                 out.write(buf, 0, len);
507             }
508         } finally {
509             in.close();
510             out.close();
511         }

572     private static void checkGetCode(FedoraClient client,
573                                      String requestPath,
574                                      String errorMessage,
575                                      int expectedCode) throws Exception {
576         HttpInputStream in = client.get(requestPath, false);
577         try {
578             int gotCode = in.getStatusCode();
579             assertEquals(errorMessage + " (" + requestPath + ")",
580                          expectedCode,
581                          gotCode);
582             if (expectedCode != 200) {
583                 String expectedString = "Fedora: " + expectedCode + " ";
584                 BufferedReader reader =
585                         new BufferedReader(new InputStreamReader(in));
586                 boolean foundExpectedString = false;
587                 String line = reader.readLine();
588                 while (line != null) {
589                     if (line.indexOf(expectedString) != -1) {
590                         foundExpectedString = true;
591                     }
592                     line = reader.readLine();
593                 }
594                 assertTrue("HTTP status code was correct (" + expectedCode
595                         + "), but body did not contain " + "the string \""
596                         + expectedString + "\"", foundExpectedString);
597             }
598         } finally {
599             in.close();
600         }
601     }

622     private static int getUploadCode(FedoraClient client,
623                                      String url,
624                                      File file,
625                                      String partName) throws Exception {
626         PostMethod post = null;
627         try {
628             post = new PostMethod(url);
629             post.setDoAuthentication(true);
630             post.getParams().setParameter("Connection", "Keep-Alive");
631             post.setContentChunked(true);
632             Part[] parts = {new FilePart(partName, file)};
633             post.setRequestEntity(new MultipartRequestEntity(parts, post
634                     .getParams()));
635             int responseCode = client.getHttpClient().executeMethod(post);
636             if (responseCode > 299 && responseCode < 400) {
637                 String location = post.getResponseHeader("location").getValue();
638                 System.out.println("Redirected to " + location);
639                 return getUploadCode(client, location, file, partName);
640             } else {
641                 return responseCode;
642             }
643         } finally {
644             if (post != null) {
645                 post.releaseConnection();
646             }
647         }
648     }
