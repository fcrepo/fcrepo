360     private static String getAccessErrorMessage(String subjectPolicy, String type, String verb, String[] mismatches){
361         StringBuilder sb = new StringBuilder();
362         sb.append(subjectPolicy).append(": Access ").append(verb).append(" for ").append(type).append("[");
363         for (int i=0; i<mismatches.length; i++){
364             sb.append(mismatches[i]);
365             if (i < mismatches.length - 1) sb.append(',');
366         }
367         sb.append(']');
368         return sb.toString();

373     // utility class for performing object and datastream access tests on a range of pids
374     // representing the results in an [object | datastream] / [allowed | denied] / set of matching pids
375     // structure
376     class PermissionTest {
377         private final EntityPerms m_object = new EntityPerms();
378         private final EntityPerms m_datastream = new EntityPerms();
379         private final int m_first;
380         private final int m_last;
381         private final String m_pidns;
382         private final String m_dsid;
383 
384         PermissionTest(int first, int last, String pidNamespace, String dsid) throws Exception {
385             m_first = first;
386             m_last = last;
387             m_pidns = pidNamespace;
388             m_dsid = dsid;
389             for (int i = m_first; i <= m_last; i++) {
390                 String pid = m_pidns + ":" + i;
391 
392                 // test object access
393                 String url = "/fedora/objects/" + pid + "?format=xml";
394                 try {
395                     String response = httpUtils.get(url);
396                     // if we got here, it was allowed, so...
397                     m_object.allowed().add(pid);
398                 } catch (AuthorizationDeniedException e) {
399                     // access was denied
400                     m_object.denied().add(pid);
401                 }
402                 // test datastream access
403                 if (!m_dsid.equals("")) {
404                     url = "/fedora/objects/" + pid + "/datastreams/" + m_dsid + "?format=xml";
405                     try {
406                         String response = httpUtils.get(url);
407                         // if we got here, it was allowed, so...
408                         m_datastream.allowed().add(pid);
409                     } catch (AuthorizationDeniedException e) {
410                         // access was denied
411                         m_datastream.denied().add(pid);
412                     }
413                     // Now check that the datastreams are being filtered correctly from listing
414                     url = "/fedora/objects?resultFormat=xml&query=type%7Evideo";
415                     String response = httpUtils.get(url);
416                     Matcher matcher = Pattern.compile("<pid>(.*)<\\/pid>").matcher(response);
417                     while(matcher.find()){
418                         m_object.listed().add(matcher.group(1));
419                     }
420                 }
421             }
422             // sanity check - allowed + denied = number of objects tested
423             if (m_object.allowed().size() + m_object.denied().size() != pidCount()) {
424                 fail("Error in checking permissions - total of allowed and denied objects does not equal number of objects tested");
425                 throw new RuntimeException("Should not happen");
426             }
427             if (!m_dsid.equals("")) {
428                 if (m_datastream.allowed().size() + m_datastream.denied().size() != pidCount()) {
429                     fail("Error in checking permissions - total of allowed and denied datastreams does not equal number of object datastreams tested");
430                     throw new RuntimeException("Also should not happen");
431                 }
432             }
433         }
434 
435         public EntityPerms object() {
436             return m_object;
437         }
438         public EntityPerms datastream() {
439             return m_datastream;
440         }
441         public int pidCount() {
442             return m_last - m_first + 1;
443         }
444     }

462     // holds a set of pids with utility method for comparing to string array
463     class Perms extends HashSet<String>{
464         private static final long serialVersionUID =
465                 3747931619024146008L;
466 
467         public boolean containsAll(String[] items) {
468             return this.containsAll(Arrays.asList(items));
469         }
470         public boolean containsOnly(String[] items) {
471             return ((this.size() == items.length) && containsAll(items));
472         }
473         public boolean containsAny(String [] items) {
474             boolean result = false;
475             for (String item:items) { result |= contains(item);}
476             return result;
477         }
478         /*
479          * returns a string representation of difference between the set members and the supplied array.
480          * Return empty string "" if items match
481          */
482         public String[] mismatch(String[] items, boolean indicate) {
483             ArrayList<String> res = new ArrayList<String>();
484             if (containsOnly(items))
485                 return res.toArray(new String[0]); // they match
486 
487             // expected items not present in this set
488             for (String item : items) {
489                 if (!contains(item)){
490                     if (indicate) res.add("-" + item);
491                     else res.add(item);
492                 }
493             }
494 
495             // items in set not present in supplied array
496             List<String> asList = Arrays.asList(items);
497             for (String item : this) {
498                 if (!asList.contains(item)){
499                     if (indicate) res.add("+" + item);
500                     else res.add(item);
501                 }
502             }
503             return res.toArray(new String[0]);
504         }
