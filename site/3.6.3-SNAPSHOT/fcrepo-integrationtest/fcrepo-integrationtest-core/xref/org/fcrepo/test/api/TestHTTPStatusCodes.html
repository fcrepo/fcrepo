380     private static FedoraClient getClient(boolean validUser,
381                                           boolean validPass,
382                                           boolean authorized) throws Exception {
383         if (validUser) {
384             if (validPass) {
385                 System.out
386                         .println("Using Fedora Client with valid user, valid pass");
387                 if (authorized) {
388                     if (CLIENT_VALID_USER_VALID_PASS == null) {
389                         CLIENT_VALID_USER_VALID_PASS = getFedoraClient();
390                     }
391                     return CLIENT_VALID_USER_VALID_PASS;
392                 } else {
393                     if (CLIENT_VALID_USER_VALID_PASS_UNAUTHORIZED == null) {
394                         CLIENT_VALID_USER_VALID_PASS_UNAUTHORIZED =
395                                 getFedoraClient(getBaseURL(),
396                                                 "untrustedUser",
397                                                 "password");
398                     }
399                     return CLIENT_VALID_USER_VALID_PASS_UNAUTHORIZED;
400                 }
401             } else {
402                 System.out
403                         .println("Using Fedora Client with valid user, bogus pass");
404                 if (CLIENT_VALID_USER_BOGUS_PASS == null) {
405                     CLIENT_VALID_USER_BOGUS_PASS =
406                             getFedoraClient(getBaseURL(),
407                                             getUsername(),
408                                             "bogus");
409                 }
410                 return CLIENT_VALID_USER_BOGUS_PASS;
411             }
412         } else {
413             System.out.println("Using Fedora Client with bogus user");
414             if (CLIENT_BOGUS_USER == null) {
415                 CLIENT_BOGUS_USER =
416                         getFedoraClient(getBaseURL(), "bogus", "bogus");
417             }
418             return CLIENT_BOGUS_USER;
419         }
420     }

479     private static void copyFile(File sourceFile, File destFile)
480             throws Exception {
481         FileInputStream in = new FileInputStream(sourceFile);
482         FileOutputStream out = new FileOutputStream(destFile);
483         byte[] buf = new byte[4096];
484         int len;
485         try {
486             while ((len = in.read(buf)) > 0) {
487                 out.write(buf, 0, len);
488             }
489         } finally {
490             in.close();
491             out.close();
492         }
493     }

541     private static void checkGetCode(FedoraClient client,
542                                      String requestPath,
543                                      String errorMessage,
544                                      int expectedCode) throws Exception {
545         HttpInputStream in = client.get(requestPath, false);
546         try {
547             int gotCode = in.getStatusCode();
548             assertEquals(errorMessage + " (" + requestPath + ")",
549                          expectedCode,
550                          gotCode);
551             if (expectedCode != 200) {
552                 String expectedString = "Fedora: " + expectedCode + " ";
553                 BufferedReader reader =
554                         new BufferedReader(new InputStreamReader(in));
555                 boolean foundExpectedString = false;
556                 String line = reader.readLine();
557                 while (line != null) {
558                     if (line.indexOf(expectedString) != -1) {
559                         foundExpectedString = true;
560                     }
561                     line = reader.readLine();
562                 }
563                 assertTrue("HTTP status code was correct (" + expectedCode
564                         + "), but body did not contain " + "the string \""
565                         + expectedString + "\"", foundExpectedString);
566             }
567         } finally {
568             in.close();
569         }
570     }

573                                         String partName,
574                                         String errorMessage,
575                                         int expectedCode) throws Exception {

577         try {

581                                   getBaseURL() + "/management/upload",
582                                   file,
583                                   partName);

586         } finally {

588         }

591     private static int getUploadCode(FedoraClient client,
592                                      String url,
593                                      File file,
594                                      String partName) throws Exception {
595         PostMethod post = null;
596         try {
597             post = new PostMethod(url);
598             post.setDoAuthentication(true);
599             post.getParams().setParameter("Connection", "Keep-Alive");
600             post.setContentChunked(true);
601             Part[] parts = {new FilePart(partName, file)};
602             post.setRequestEntity(new MultipartRequestEntity(parts, post
603                     .getParams()));
604             int responseCode = client.getHttpClient().executeMethod(post);
605             if (responseCode > 299 && responseCode < 400) {
606                 String location = post.getResponseHeader("location").getValue();
607                 System.out.println("Redirected to " + location);
608                 return getUploadCode(client, location, file, partName);
609             } else {
610                 return responseCode;
611             }
612         } finally {
613             if (post != null) {
614                 post.releaseConnection();
615             }
616         }
617     }
