7   import java.util.Comparator;
8   import java.util.Vector;

10  /**
11   * ColumnComparator.
12   *

14   * Claude Duguay, originally made available at

17   * @author Claude Duguay
18   * @author Chris Wilper

20  public class ColumnComparator
21          implements Comparator {
22  
23      protected int index;
24  
25      protected boolean ascending;
26  
27      public ColumnComparator(int index, boolean ascending) {
28          this.index = index;
29          this.ascending = ascending;
30      }
31  
32      public int compare(Object one, Object two) {
33          if (one instanceof Vector && two instanceof Vector) {
34              Vector vOne = (Vector) one;
35              Vector vTwo = (Vector) two;
36              Object oOne = vOne.elementAt(index);
37              Object oTwo = vTwo.elementAt(index);
38              if (oOne instanceof Comparable && oTwo instanceof Comparable) {
39                  Comparable cOne = (Comparable) oOne;
40                  Comparable cTwo = (Comparable) oTwo;
41                  if (ascending) {
42                      return cOne.compareTo(cTwo);
43                  } else {
44                      return cTwo.compareTo(cOne);
45                  }
46              }
47          }
48          return 1;
49      }
50  }
