5   package org.fcrepo.client;
6   
7   import java.io.ByteArrayOutputStream;
8   import java.io.PrintStream;
9   
10  /**
11   * A PrintStream that sends its output to Administrator.WATCH_AREA, the
12   * JTextArea of the Tools->Advanced->STDOUT/STDERR window. This is used for
13   * redirecting System.out/err output to the UI.
14   * 
15   * @author Chris Wilper
16   */
17  public class WatchPrintStream
18          extends PrintStream {
19  
20      /** Output is buffered here until a call to println(String) */
21      private final ByteArrayOutputStream m_out;
22  
23      public WatchPrintStream(ByteArrayOutputStream out) {
24          super(out);
25          m_out = out;
26      }
27  
28      /**
29       * Every time this is called, the buffer is cleared an output is sent to the
30       * JTextArea.
31       */
32      @Override
33      public void println(String str) {
34          super.println(str);
35          if (Administrator.WATCH_AREA != null) {
36              String buf = m_out.toString();
37              m_out.reset();
38              Administrator.WATCH_AREA.append(buf);
39          }
40      }
41  
42  }
