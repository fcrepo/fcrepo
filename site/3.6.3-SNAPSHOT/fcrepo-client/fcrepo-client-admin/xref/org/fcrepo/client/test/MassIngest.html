36      public MassIngest(AutoIngestor ingestor,
37                        File templateFile,
38                        File dictFile,
39                        String format,
40                        int numTimes)
41              throws Exception {
42          // load the template file into two parts... with splitter=##SPLITTER##
43          BufferedReader in = new BufferedReader(new FileReader(templateFile));
44          String nextLine = "";
45          StringBuffer startBuffer = new StringBuffer();
46          StringBuffer endBuffer = new StringBuffer();
47          boolean seenSplitter = false;
48          while (nextLine != null) {
49              nextLine = in.readLine();
50              if (nextLine != null) {
51                  if (!seenSplitter) {
52                      if (nextLine.startsWith("##SPLITTER##")) {
53                          seenSplitter = true;
54                      } else {
55                          startBuffer.append(nextLine + "\n");
56                      }
57                  } else {
58                      endBuffer.append(nextLine + "\n");
59                  }
60              }
61          }
62          in.close();
63          String start = startBuffer.toString();
64          String end = endBuffer.toString();
65          RandomDCMetadataFactory dcFactory =
66                  new RandomDCMetadataFactory(dictFile);
67          for (int i = 0; i < numTimes; i++) {
68              String xml = start + dcFactory.get(2, 13) + end;
69              String pid =
70                      ingestor
71                              .ingestAndCommit(new ByteArrayInputStream(xml
72                                                       .getBytes("UTF-8")),
73                                               format,
74                                               "part of massingest of "
75                                                       + numTimes
76                                                       + " auto-generated objects.");
77              int t = i + 1;
78              System.out.println(pid + " " + t + "/" + numTimes);
79          }
80  
81      }

89      public static void main(String[] args) throws Exception {
90          System.setProperty("java.awt.headless", "true");
91          try {
92              if (args.length < 9 || args.length > 10) {
93                  MassIngest.showUsage("You must provide nine or ten arguments.");
94              } else {
95                  String hostName = args[0];
96                  int portNum = Integer.parseInt(args[1]);
97                  String username = args[2];
98                  String password = args[3];
99                  File dictFile = new File(args[5]);
100                 String format = args[6];
101                 // third arg==file... must exist
102                 File f = new File(args[4]);
103                 String protocol = args[8];
104                 String context = Constants.FEDORA_DEFAULT_APP_CONTEXT;
105                 if (args.length == 10 && !args[9].equals("")){
106                     context = args[9];
107                 }
108 
109 
110                 // ******************************************
111                 // NEW: use new client utility class
112                 String baseURL =
113                         protocol + "://" + hostName + ":" + portNum + "/"
114                                 + context;
115                 FedoraClient fc = new FedoraClient(baseURL, username, password);
116                 APIA = fc.getAPIAMTOM();
117                 APIM = fc.getAPIMMTOM();
118                 //*******************************************
119                 AutoIngestor autoIngestor = new AutoIngestor(APIA, APIM);
120 
121                 new MassIngest(autoIngestor, f, dictFile, format, Integer
122                         .parseInt(args[7]));
123             }
124         } catch (Exception e) {
125             MassIngest.showUsage(e.getClass().getName()
126                     + " - "
127                     + (e.getMessage() == null ? "(no detail provided)" : e
128                             .getMessage()));
129         }
