26      public AutoBatchBuildIngest(String objectTemplate,
27                                  String objectSpecificDir,
28                                  String objectDir,
29                                  String logFile,
30                                  String logFormat,
31                                  String objectFormat,
32                                  String host,
33                                  String port,
34                                  String username,
35                                  String password,
36                                  String protocol,
37                                  String context)
38              throws Exception {
39  
40          batchProperties.setProperty("merge-objects", "yes");
41          batchProperties.setProperty("ingest", "yes");
42          batchProperties.setProperty("template", objectTemplate);
43          batchProperties.setProperty("specifics", objectSpecificDir);
44          batchProperties.setProperty("objects", objectDir);
45          batchProperties.setProperty("ingested-pids", logFile);
46          batchProperties.setProperty("pids-format", logFormat);
47          batchProperties.setProperty("object-format", objectFormat);
48          batchProperties.setProperty("server-fqdn", host);
49          batchProperties.setProperty("server-port", port);
50          batchProperties.setProperty("server-context", context);
51          batchProperties.setProperty("username", username);
52          batchProperties.setProperty("password", password);
53          batchProperties.setProperty("server-protocol", protocol);
54  
55          BatchTool batchTool = new BatchTool(batchProperties, null, null);
56          batchTool.prep();
57          batchTool.process();
58      }

60      public static final void main(String[] args) throws Exception {
61          System.setProperty("java.awt.headless", "true");
62          boolean errors = false;
63          String context = "/" + Constants.FEDORA_DEFAULT_APP_CONTEXT;
64          String objectFormat = null;
65          if (args.length == 9 || args.length == 10) {
66              if (!new File(args[0]).exists() && !new File(args[0]).isFile()) {
67                  System.out.println("Specified object template file path: \""
68                          + args[0] + "\" does not exist.");
69                  errors = true;
70              }
71              if (!new File(args[1]).isDirectory()) {
72                  System.out.println("Specified object specific directory: \""
73                          + args[1] + "\" is not directory.");
74                  errors = true;
75              }
76              if (!new File(args[2]).isDirectory()) {
77                  System.out.println("Specified object directory: \"" + args[2]
78                          + "\" is not a directory.");
79                  errors = true;
80              }
81              if (!args[4].equals("xml") && !args[4].equals("text")) {
82                  System.out
83                          .println("Format for log file must must be either: \""
84                                  + "\"xml\"  or  \"txt\"");
85                  errors = true;
86              }
87              String[] server = args[5].split(":");
88              if (server.length != 2) {
89                  System.out.println("Specified server name does not specify "
90                          + "port number: \"" + args[5] + "\" .");
91                  errors = true;
92              }
93  
94              if (!args[8].equals("http") && !args[8].equals("https")) {
95                  System.out.println("Protocl must be either: \""
96                          + "\"http\"  or  \"https\"");
97                  errors = true;
98              }
99  
100             if (args.length == 10 && !args[9].equals("")){
101                 context = "/" + args[9];
102             }
103             // Verify format of template file to see if it is a METS or FOXML template
104             BufferedReader br = new BufferedReader(new FileReader(args[0]));
105             String line;
106             while ((line = br.readLine()) != null) {
107                 System.out.println(line);
108                 if (line.indexOf("<foxml:") != -1) {
109                     objectFormat = FOXML1_1.uri;
110                     break;
111                 }
112                 if (line.indexOf("<METS:") != -1) {
113                     objectFormat = METS_EXT1_1.uri;
114                     break;
115                 }
116             }
117             br.close();
118             br = null;
119 
120             if (objectFormat == null) {
121                 errors = true;
122             }
123             if (!errors) {
124                 System.out.println("\n*** Format of template files is: "
125                         + objectFormat + " . Generated objects will be in "
126                         + objectFormat + " format.\n");
127                 AutoBatchBuildIngest autoBatch =
128                         new AutoBatchBuildIngest(args[0],
129                                                  args[1],
130                                                  args[2],
131                                                  args[3],
132                                                  args[4],
133                                                  objectFormat,
134                                                  server[0],
135                                                  server[1],
136                                                  args[6],
137                                                  args[7],
138                                                  args[8],
139                                                  context);
140             }
141         } else {
142             if (objectFormat == null && (args.length == 10 || args.length == 9)) {
143                 System.out.println("\nUnknown format for template file.\n"
144                         + "Template file must either be METS or FOXML.\n");
145             } else {
146                 System.out.println("\n**** Wrong Number of Arguments *****\n");
147                 System.out
148                         .println("AutoBatchBuildIngest requires 9 arguments.");
149                 System.out.println("(1) - full path to object template file");
150                 System.out
151                         .println("(2) - full path to object specific directory");
152                 System.out.println("(3) - full path to object directory");
153                 System.out.println("(4) - full path to log file");
154                 System.out.println("(5) - format of log file (xml or text)\n");
155                 System.out
156                         .println("(6) - host name and port of Fedora server (host:port)");
157                 System.out.println("(7) - admin username of Fedora server");
158                 System.out
159                         .println("(8) - password for admin user of Fedora server\n");
160                 System.out
161                         .println("(9) - protocol to communicate with Fedora server (http or https)");
162                 System.out
163                 .println("(10) - optional, alternate context location of the Fedora server (default is fedora)");
164             }
