77      public ResultFrame(String frameTitle,
78                         String[] displayFields,
79                         int maxResults,
80                         FieldSearchQuery query) {
81          super(frameTitle, true, //resizable
82                true, //closable
83                true, //maximizable
84                true);//iconifiable
85  
86          // Make sure resultFields has pid, even though they may not
87          // want to display it. Also, signal that the pid should or
88          // should not be displayed.
89          boolean displayPid = false;
90          for (String element : displayFields) {
91              if (element.equals("pid")) {
92                  displayPid = true;
93              }
94          }
95          String[] resultFields;
96          if (displayPid) {
97              resultFields = displayFields;
98          } else {
99              resultFields = new String[displayFields.length + 1];
100             resultFields[0] = "pid";
101             for (int i = 1; i < displayFields.length + 1; i++) {
102                 resultFields[i] = displayFields[i - 1];
103             }
104         }
105         try {
106             if (m_finder == null) {
107                 m_finder = new AutoFinder(Administrator.APIA);
108             }
109             searchAndDisplay(m_finder.findObjects(TypeUtility
110                                                           .convertStringtoAOS(resultFields),
111                                                   maxResults,
112                                                   query),
113                              displayFields);
114         } catch (Exception e) {
115             e.printStackTrace();
116             System.out.println("ERROR: " + e.getClass().getName() + ":"
117                     + e.getMessage());
118         }
119     }

121     private void searchAndDisplay(FieldSearchResult fsr, String[] displayFields)
122             throws Exception {
123         // put the resulting data into a structure suitable for display
124         ResultList ofs = fsr.getResultList();
125         Object[][] data = null;
126         if (ofs != null && ofs.getObjectFields() != null) {
127             data = new Object[ofs.getObjectFields().size()][displayFields.length];
128             // while adding the pids to m_rowPids so they can be used later
129             m_rowPids = new String[ofs.getObjectFields().size()];
130             for (int i = 0; i < ofs.getObjectFields().size(); i++) {
131                 ObjectFields o = ofs.getObjectFields().get(i);
132                 m_rowPids[i] = o.getPid().getValue();
133                 for (int j = 0; j < displayFields.length; j++) {
134                     data[i][j] = getValue(o, displayFields[j]);
135                 }
136             }
137         }
138 
139         DefaultSortTableModel model =
140                 new DefaultSortTableModel(data, displayFields);
141         m_table = new JSortTable(model);
142         m_table.setPreferredScrollableViewportSize(new Dimension(400, 400));
143         m_table.setShowVerticalLines(false);
144         m_table.setCellSelectionEnabled(false);
145         m_table.setRowSelectionAllowed(true);
146         m_table.setUI(new ResultFrame.BrowserTableUI());
147 
148         JScrollPane browsePanel = new JScrollPane(m_table);
149         getContentPane().setLayout(new BorderLayout());
150         getContentPane().add(browsePanel, BorderLayout.CENTER);
151         if (fsr.getListSession() != null
152                 && fsr.getListSession().getValue().getToken() != null) {
153             m_moreButton = new JButton("More Results...");
154             m_moreButton
155                     .addActionListener(new MoreResultsListener(displayFields,
156                                                                fsr.getListSession()
157                                                                        .getValue()
158                                                                        .getToken(),
159                                                                this));
160             getContentPane().add(m_moreButton, BorderLayout.SOUTH);
161         }
162         ImageIcon zoomIcon =
163                 new ImageIcon(ClassLoader.getSystemResource("images/client/standard/general/Zoom16.gif"));
164         setFrameIcon(zoomIcon);
165         pack();
166         setSize(Administrator.getDesktop().getWidth() - 40, getSize().height);
167     }

175     public String getValue(ObjectFields o, String name) {
176         if (o == null || name == null) {
177             return null;
178         }
179         if (name.equals("pid")) {
180             return o.getPid() != null ? o.getPid().getValue() : "";
181         }
182         if (name.equals("label")) {
183             return o.getLabel() != null ? o.getLabel().getValue() : "";
184         }
185         if (name.equals("state")) {
186             return o.getState() != null ? o.getState().getValue() : "";
187         }
188         if (name.equals("ownerId")) {
189             return o.getOwnerId() != null ? o.getOwnerId().getValue() : "";
190         }
191         if (name.equals("cDate")) {
192             return o.getCDate() != null ? o.getCDate().getValue() : "";
193         }
194         if (name.equals("mDate")) {
195             return o.getMDate() != null ? o.getMDate().getValue() : "";
196         }
197         if (name.equals("dcmDate")) {
198             return o.getDcmDate() != null ? o.getDcmDate().getValue() : "";
199         }
200         if (name.equals("title")) {
201             return getList(o.getTitle());
202         }
203         if (name.equals("creator")) {
204             return getList(o.getCreator());
205         }
206         if (name.equals("subject")) {
207             return getList(o.getSubject());
208         }
209         if (name.equals("description")) {
210             return getList(o.getDescription());
211         }
212         if (name.equals("publisher")) {
213             return getList(o.getPublisher());
214         }
215         if (name.equals("contributor")) {
216             return getList(o.getContributor());
217         }
218         if (name.equals("date")) {
219             return getList(o.getDate());
220         }
221         if (name.equals("type")) {
222             return getList(o.getType());
223         }
224         if (name.equals("format")) {
225             return getList(o.getFormat());
226         }
227         if (name.equals("identifier")) {
228             return getList(o.getIdentifier());
229         }
230         if (name.equals("source")) {
231             return getList(o.getSource());
232         }
233         if (name.equals("language")) {
234             return getList(o.getLanguage());
235         }
236         if (name.equals("relation")) {
237             return getList(o.getRelation());
238         }
239         if (name.equals("coverage")) {
240             return getList(o.getCoverage());
241         }
242         if (name.equals("rights")) {
243             return getList(o.getRights());
244         }
245         return null;
246     }

248     public String getList(List<String> s) {
249         if (s == null) {
250             return "";
251         }
252         StringBuffer out = new StringBuffer();
253         for (int i = 0; i < s.size(); i++) {
254             if (i > 0) {
255                 out.append(", ");
256             }
257             out.append(s.get(i));
258         }
259         return out.toString();
260     }

295     public class BrowserTableUI
296             extends BasicTableUI {
297 
298         @Override
299         protected MouseInputListener createMouseInputListener() {
300             return new BasicTableUI.MouseInputHandler() {
301 
302                 @Override
303                 public void mouseClicked(MouseEvent e) {
304                     if (e.getClickCount() == 2) {
305                         int rowNum =
306                                 m_table.rowAtPoint(new Point(e.getX(), e.getY()));
307                         if (rowNum >= 0) {
308                             // launch object viewer to view object
309                             new ViewObject(m_rowPids[rowNum]).launch();
310                         }
311                     }
312                 }
313 
314                 @Override
315                 public void mousePressed(MouseEvent e) {
316                     if (SwingUtilities.isRightMouseButton(e)) {
317                         int rowNum =
318                                 m_table.rowAtPoint(new Point(e.getX(), e.getY()));
319                         if (rowNum >= 0) {
320                             int[] sRows = m_table.getSelectedRows();
321                             boolean clickedOnSelected = false;
322                             HashSet pids = new HashSet();
323                             for (int element : sRows) {
324                                 if (element == rowNum) {
325                                     clickedOnSelected = true;
326                                 }
327                                 pids.add(m_rowPids[element]);
328                             }
329                             if (!clickedOnSelected) {
330                                 pids = new HashSet();
331                                 m_table.clearSelection();
332                                 m_table.addRowSelectionInterval(rowNum, rowNum);
333                                 pids.add(m_rowPids[rowNum]);
334                             }
335                             if (pids.size() == 1) {
336                                 Iterator pidIter = pids.iterator();
337                                 new ResultFrame.SingleSelectionPopup((String) pidIter
338                                         .next()).show(e.getComponent(),
339                                                       e.getX(),
340                                                       e.getY());
341                             } else {
342                                 new ResultFrame.MultiSelectionPopup(pids)
343                                         .show(e.getComponent(),
344                                               e.getX(),
345                                               e.getY());
346                             }
347                         }
348                     } else {
349                         // not a right click
350                         super.mousePressed(e);
351                     }
352                 }
353             };
354         }
355     }
