28  /**
29   * A client for performing HTTP GET requests on a Fedora server (with
30   * authentication) or any other server (without authentication). Each kind of
31   * request can either request an InputStream or request that the Downloader
32   * write the content directly to a provided OutputStream.
33   *
34   * @author Chris Wilper
35   */
36  public class Downloader {

47      /**
48       * Construct a downloader for a certain repository as a certain user.
49       */
50      public Downloader(String host,
51                        int port,
52                        String context,
53                        String user,
54                        String pass)
55              throws IOException {
56          m_fedoraUrlStart =
57                  Administrator.getProtocol() + "://" + host + ":" + port + "/"
58                          + context + "/" + "get/";
59          m_authScope =
60                  new AuthScope(host, AuthScope.ANY_PORT, AuthScope.ANY_REALM);
61          m_creds = new UsernamePasswordCredentials(user, pass);
62      }

64      public void getDatastreamContent(String pid,
65                                       String dsID,
66                                       String asOfDateTime,
67                                       OutputStream out) throws IOException {
68          InputStream in = getDatastreamContent(pid, dsID, asOfDateTime);
69          StreamUtility.pipeStream(in, out, 4096);

72      public InputStream getDatastreamContent(String pid,
73                                              String dsID,
74                                              String asOfDateTime)
75              throws IOException {
76          StringBuffer buf = new StringBuffer();
77          buf.append(m_fedoraUrlStart);
78          buf.append(pid);
79          buf.append('/');
80          buf.append(dsID);
81          if (asOfDateTime != null) {
82              buf.append('/');
83              buf.append(asOfDateTime);
84          }
85          return get(buf.toString());
86      }

121     /**
122      * Get data via HTTP as an InputStream, following redirects, and supplying
123      * credentials if the host is the Fedora server.
124      */
125     public InputStream get(String url) throws IOException {
126         GetMethod get = null;
127         boolean ok = false;
128         try {
129             m_cManager.getParams().setConnectionTimeout(20000);
130             HttpClient client = new HttpClient(m_cManager);
131             client.getState().setCredentials(m_authScope, m_creds);
132             client.getParams().setAuthenticationPreemptive(true); // don't bother with challenges
133             int redirectCount = 0; // how many redirects did we follow
134             int resultCode = 300; // not really, but enter the loop that way
135             Dimension d = null;
136             while (resultCode > 299 && resultCode < 400 && redirectCount < 25) {
137                 get = new GetMethod(url);
138                 get.setDoAuthentication(true);
139                 get.setFollowRedirects(true);
140                 if (Administrator.INSTANCE != null) {
141                     d = Administrator.PROGRESS.getSize();
142                     // if they're using Administrator, tell them we're downloading...
143                     Administrator.PROGRESS.setString("Downloading " + url
144                             + " . . .");
145                     Administrator.PROGRESS.setValue(100);
146                     Administrator.PROGRESS.paintImmediately(0, 0, (int) d
147                             .getWidth() - 1, (int) d.getHeight() - 1);
148                 }
149                 resultCode = client.executeMethod(get);
150                 if (resultCode > 299 && resultCode < 400) {
151                     redirectCount++;
152                     url = get.getResponseHeader("Location").getValue();
153                 }
154             }
155             if (resultCode != 200) {
156                 System.err.println(get.getResponseBodyAsString());
157                 throw new IOException("Server returned error: " + resultCode
158                         + " " + HttpStatus.getStatusText(resultCode));
159             }
160             ok = true;
161             if (Administrator.INSTANCE != null) {
162                 // cache it to a file
163                 File tempFile =
164                         File.createTempFile("fedora-client-download-", null);
165                 tempFile.deleteOnExit();
166                 HashMap PARMS = new HashMap();
167                 PARMS.put("in", get.getResponseBodyAsStream());
168                 PARMS.put("out", new FileOutputStream(tempFile));
169                 // do the actual download in a safe thread
170                 SwingWorker worker = new SwingWorker(PARMS) {
171 
172                     @Override
173                     public Object construct() {
174                         try {
175                             StreamUtility.pipeStream((InputStream) parms
176                                                              .get("in"),
177                                                      (OutputStream) parms
178                                                              .get("out"),
179                                                      8192);
180                         } catch (Exception e) {
181                             thrownException = e;
182                         }
183                         return "";
184                     }
185                 };
186                 worker.start();
187                 // The following code will run in the (safe)
188                 // Swing event dispatcher thread.
189                 int ms = 200;
190                 while (!worker.done) {
191                     try {
192                         Administrator.PROGRESS.setValue(ms);
193                         Administrator.PROGRESS.paintImmediately(0, 0, (int) d
194                                 .getWidth() - 1, (int) d.getHeight() - 1);
195                         Thread.sleep(100);
196                         ms = ms + 100;
197                         if (ms >= 2000) {
198                             ms = 200;
199                         }
200                     } catch (InterruptedException ie) {
201                     }
202                 }
203                 if (worker.thrownException != null) {
204                     throw worker.thrownException;
205                 }
206                 Administrator.PROGRESS.setValue(2000);
207                 Administrator.PROGRESS.paintImmediately(0, 0, (int) d
208                         .getWidth() - 1, (int) d.getHeight() - 1);
209                 try {
210                     Thread.sleep(100);
211                 } catch (InterruptedException ie) {
212                 }
213                 return new FileInputStream(tempFile);
214             }
215             return get.getResponseBodyAsStream();
216         } catch (Exception e) {
217             throw new IOException(e.getMessage());
218         } finally {
219             if (get != null && !ok) {
220                 get.releaseConnection();
221             }
222             if (Administrator.INSTANCE != null) {
223                 Administrator.PROGRESS.setValue(0);
224                 Administrator.PROGRESS.setString("");
225             }
226         }
227 
228     }
