47  /**
48   * Parses a stream of batch modify directives and makes appropriate calls to the
49   * a Fedora server. The parsing is configured to parse directives in the file
50   * sequentially. Logs are written for each successful and failed directive that
51   * is processed. Recoverable(non-fatal) errors are written to the log file and
52   * processing continues. Catastrophic errors will cause parsing to halt and set
53   * the count of failed directives to -1 indicating that parsing was halted prior
54   * to the end of the file. In this case the logs will contain all directives
55   * processed up to the point of failure.
56   *
57   * @author Ross Wayland
58   */
59  public class BatchModifyParser
60          extends DefaultHandler
61          implements Constants {

119     /**
120      * <p>
121      * Constructor allows this class to initiate the parsing.
122      * </p>
123      *
124      * @param UPLOADER
125      *        - An instance of Uploader.
126      * @param APIM
127      *        - An instance of FedoraAPIM.
128      * @param APIA
129      *        - An instance of Fedora APIA.
130      * @param in
131      *        - An input stream containing the xml to be parsed.
132      * @param out
133      *        - A print stream used for writing log info.
134      */
135     public BatchModifyParser(Uploader UPLOADER,
136                              FedoraAPIMMTOM APIM,
137                              FedoraAPIAMTOM APIA,
138                              InputStream in,
139                              PrintStream out) {
140         BatchModifyParser.out = out;
141         BatchModifyParser.APIM = APIM;
142         BatchModifyParser.APIA = APIA;
143         BatchModifyParser.UPLOADER = UPLOADER;
144         XMLReader xmlReader = null;
145         boolean schemaValidate = true;
146 
147         if ("true".equals(System.getenv("OFFLINE"))
148                 || "true".equals(System.getProperty("offline"))
149                 || "false".equals(System.getProperty("online"))
150                 || "false".equals(System.getenv("ONLINE"))) {
151                 System.out.println("NOT validating");
152                 schemaValidate = false;
153         }
154 
155         // Configure the SAX parser.
156         try {
157             SAXParserFactory saxfactory = SAXParserFactory.newInstance();
158             SAXParser parser = saxfactory.newSAXParser();
159             xmlReader = parser.getXMLReader();
160             xmlReader.setContentHandler(this);
161             xmlReader
162                     .setFeature("http://xml.org/sax/features/namespaces", true);
163             xmlReader
164                     .setFeature("http://xml.org/sax/features/namespace-prefixes",
165                                 true);
166             xmlReader
167                     .setFeature("http://apache.org/xml/features/validation/schema",
168                                 schemaValidate);
169             xmlReader.setErrorHandler(new BatchModifyXMLErrorHandler());
170         } catch (Exception e) {
171             // An Exception indicates a fatal error and parsing was halted. Set
172             // failedCount to -1 to indicate to the calling class that parsing failed.
173             // Throwing an Exception would make class variables for succeededCount
174             // and failedCount unavailable.
175             logParserError(e, null);
176             failedCount = -1;
177         }
178 
179         // Parse the file.
180         try {
181             xmlReader.parse(new InputSource(in));
182         } catch (Exception e) {
183             // An Exception indicates a fatal error and parsing was halted. Set
184             // failedCount to -1 to indicate to the calling class that parsing failed.
185             // Throwing an Exception would make class variables for succeededCount
186             // and failedCount unavailable.
187             logParserError(e, null);
188             failedCount = -1;
189         }
190     }

218     @Override
219     public void skippedEntity(String name) throws SAXException {
220         StringBuffer sb = new StringBuffer();
221         sb.append('&');
222         sb.append(name);
223         sb.append(';');
224         char[] text = new char[sb.length()];
225         sb.getChars(0, sb.length(), text, 0);
226         characters(text, 0, text.length);
227     }

229     private void appendElementStart(String uri,
230                                     String localName,
231                                     String qName,
232                                     Attributes a,
233                                     StringBuffer out) {
234         out.append("<" + qName);
235         for (int i = 0; i < a.getLength(); i++) {
236             out.append(" " + a.getQName(i) + "=\""
237                     + StreamUtility.enc(a.getValue(i)) + "\"");
238         }
239         out.append(">");
240     }

242     @Override
243     public void characters(char ch[], int start, int length)
244             throws SAXException {
245         if (m_inXMLMetadata) {
246             // since this data is encoded straight back to xml,
247             // we need to make sure special characters &, <, >, ", and '
248             // are re-converted to the xml-acceptable equivalents.
249             StreamUtility.enc(ch, start, length, m_dsXMLBuffer);
250         }
251     }

640     @Override
641     public void endElement(String namespaceURI, String localName, String qName)
642             throws SAXException {
643 
644         if (m_inXMLMetadata) {
645             if (namespaceURI.equalsIgnoreCase(BATCH_MODIFY.uri)
646                     && localName.equals("xmlData")) {
647                 try {
648                     m_ds.xmlContent =
649                             m_dsXMLBuffer.toString().getBytes("UTF-8");
650                 } catch (UnsupportedEncodingException e) {
651                     // won't happen
652                 }
653                 m_inXMLMetadata = false;
654             } else {
655                 // finished an element in xmlData...append end tag.
656                 m_dsXMLBuffer.append("</");
657                 m_dsXMLBuffer.append(qName);
658                 m_dsXMLBuffer.append(">");
659             }
660         } else if (namespaceURI.equalsIgnoreCase(BATCH_MODIFY.uri)
661                 && localName.equalsIgnoreCase("addObject")) {
662             try {
663                 if (addObject) {
664                     StringBuffer xml = new StringBuffer();
665                     xml.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
666                     xml.append("<" + FOXML.DIGITAL_OBJECT.qName + " ");
667                     xml.append(FOXML.VERSION.localName + "=\"1.1\"");
668                     if (m_obj.pid != null && m_obj.pid.length() > 0) {
669                         xml.append(" " + FOXML.PID.localName + "=\"");
670                         xml.append(StreamUtility.enc(m_obj.pid) + "\"");
671                     }
672                     xml.append("\n");
673                     xml.append("    xmlns:" + FOXML.prefix + "=\"");
674                     xml.append(FOXML.uri + "\"\n");
675                     xml.append("    xmlns:" + XSI.prefix + "=\"");
676                     xml.append(XSI.uri + "\"\n");
677                     xml.append("    " + XSI.SCHEMA_LOCATION.qName + "=\"");
678                     xml.append(FOXML.uri + " " + FOXML1_1.xsdLocation);
679                     xml.append("\">\n");
680                     xml.append("  <" + FOXML.OBJECT_PROPERTIES.qName + ">\n");
681                     appendProperty(xml, MODEL.LABEL.uri, m_obj.label);
