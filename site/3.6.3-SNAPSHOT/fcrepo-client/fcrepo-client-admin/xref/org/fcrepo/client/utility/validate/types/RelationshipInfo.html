6   package org.fcrepo.client.utility.validate.types;
7   
8   /**
9    * Information needed to validate an object's relationships against its content
10   * model.
11   * 
12   * @author Jim Blake
13   */
14  public class RelationshipInfo {
15  
16      private final String predicate;
17  
18      private final String object;
19  
20      public RelationshipInfo(String predicate, String object) {
21          this.predicate = predicate;
22          this.object = object;
23      }
24  
25      public String getPredicate() {
26          return predicate;
27      }
28  
29      public String getObject() {
30          return object;
31      }
32  
33      public String getObjectPid() {
34          if (object != null && object.startsWith("info:fedora/")) {
35              return object.substring(12);
36          }
37          return null;
38      }
39  
40      @Override
41      public boolean equals(Object obj) {
42          if (this == obj) {
43              return true;
44          }
45          if (obj == null) {
46              return false;
47          }
48          if (!this.getClass().equals(obj.getClass())) {
49              return false;
50          }
51  
52          RelationshipInfo that = (RelationshipInfo) obj;
53          return equivalent(predicate, that.predicate)
54                  && equivalent(object, that.object);
55      }
56  
57      @Override
58      public int hashCode() {
59          return hashIt(predicate) ^ hashIt(object);
60      }
61  
62      @Override
63      public String toString() {
64          return "RelationshipInfo[predicate='" + predicate + "', object='"
65                  + object + "']";
66      }
67  
68      private boolean equivalent(Object obj1, Object obj2) {
69          return obj1 == null ? obj2 == null : obj1.equals(obj2);
70      }
71  
72      private int hashIt(Object obj) {
73          return obj == null ? 0 : obj.hashCode();
74      }
75  
76  }
