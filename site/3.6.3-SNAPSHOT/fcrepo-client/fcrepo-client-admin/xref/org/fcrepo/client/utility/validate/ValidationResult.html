6   package org.fcrepo.client.utility.validate;

8   import java.util.ArrayList;
9   import java.util.Collection;
10  import java.util.List;
11  
12  import org.fcrepo.client.utility.validate.types.ObjectInfo;
13  
14  
15  /**
16   * <p>
17   * The accumulated result of validating on a digital object. This result
18   * contains a reference to the object and may also contain (0 or more)
19   * "notations" that describe the results of the validation.
20   * </p>
21   * <p>
22   * Each notation has a {@link Level} associated with it, and if any of these
23   * levels are {@link Level#ERROR ERROR}, the validation is considered to have
24   * failed.
25   * </p>
26   * 
27   * @author Jim Blake
28   */
29  public class ValidationResult {
30  
31      public enum Level {
32          /** Information of interest - not a problem. */
33          INFO,
34          /** May be a problem - subject to interpretation. */
35          WARN,
36          /** Validation failed - object is not valid. */
37          ERROR
38      };
39  
40      private final ObjectInfo object;
41  
42      private final List<ValidationResultNotation> notes =
43              new ArrayList<ValidationResultNotation>();
44  
45      public ValidationResult(ObjectInfo object) {
46          if (object == null) {
47              throw new IllegalArgumentException("object may not be null.");
48          }
49          this.object = object;
50      }
51  
52      public void addNote(ValidationResultNotation note) {
53          notes.add(note);
54      }
55  
56      public ObjectInfo getObject() {
57          return object;
58      }
59  
60      public Collection<ValidationResultNotation> getNotes() {
61          return new ArrayList<ValidationResultNotation>(notes);
62      }
63  
64      /**
65       * What's the highest severity level of any of the notations on this result?
66       */
67      public Level getSeverityLevel() {
68          Level severity = Level.INFO;
69          for (ValidationResultNotation note : notes) {
70              Level noteLevel = note.getLevel();
71              if (noteLevel.compareTo(severity) > 0) {
72                  severity = noteLevel;
73              }
74          }
75          return severity;
76      }
77  
78      @Override
79      public boolean equals(Object obj) {
80          if (this == obj) {
81              return true;
82          }
83          if (obj == null) {
84              return false;
85          }
86          if (!this.getClass().equals(obj.getClass())) {
87              return false;
88          }
89          ValidationResult that = (ValidationResult) obj;
90          return object.equals(that.object) && notes.equals(that.notes);
91      }
92  
93      @Override
94      public int hashCode() {
95          return object.hashCode() ^ notes.hashCode();
96      }
97  
98      @Override
99      public String toString() {
100         return "pid='" + object.getPid() + "', " + " Notes=" + notes;
101     }
