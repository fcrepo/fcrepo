46      /**
47       * Validates the candidate URL. The result of the validation also depends on the
48       * control group of the datastream in question. Managed datastreams may be ingested
49       * using the file URI scheme, other datastreams may not.
50       *
51       * @param url
52       *            The URL to validate.
53       * @param controlGroup
54       *            The control group of the datastream the URL belongs to.
55       *
56       * @throws ValidationException
57       *             if the URL is malformed.
58       */
59      public static void validateURL(String url, String controlGroup)
60              throws ValidationException {
61          if (!(controlGroup.equalsIgnoreCase("M") || controlGroup.equalsIgnoreCase("E")) && url.startsWith("file:")) {
62              throw new ValidationException(
63                      "Malformed URL (file: not allowed for control group "
64                              + controlGroup + ") " + url);
65          }
66          try {
67              new URL(url);
68          } catch (MalformedURLException e) {
69              if (url.startsWith(DatastreamManagedContent.UPLOADED_SCHEME)) {
70                  return;
71              }
72              throw new ValidationException("Malformed URL: " + url, e);
73          }
74      }

95      /**
96       * Validates the latest version of all reserved datastreams in the given
97       * object.
98       */
99      public static void validateReservedDatastreams(DOReader reader)
100             throws ValidationException {
101         try {
102             for (Datastream ds: reader.GetDatastreams(null, null)) {
103                 if ("X".equals(ds.DSControlGrp) || "M".equals(ds.DSControlGrp)) {
104                     validateReservedDatastream(PID.getInstance(reader.GetObjectPID()),
105                                                ds.DatastreamID,
106                                                ds);
107                 }
108             }
109         } catch (ValidationException e) {
110             throw e;
111         } catch (ServerException e) {
112             throw new FaultException(e);
113         }
114     }

116     /**
117      * Validates the given datastream if it's a reserved datastream.
118      *
119      * The given stream is guaranteed to be closed when this method completes.
120      */
121     public static void validateReservedDatastream(PID pid,
122                                                   String dsId,
123                                                   Datastream ds)
124             throws ValidationException {
125 
126         // NB: only want to generate inputstream from .getContentStream() once
127         // we know that this is a datastream to validate
128         // to prevent reading of non-reserved datstream content when it is not needed
129         InputStream content = null;
130 
131         try {
132         if ("POLICY".equals(dsId)) {
133             content = ds.getContentStream();
134             validatePOLICY(content);
135         } else if ("FESLPOLICY".equals(dsId)) {
136             content = ds.getContentStream();
137             validateFESLPOLICY(content);
138         } else if ("RELS-EXT".equals(dsId) || "RELS-INT".equals(dsId)) {
139                 content = ds.getContentStream();
140             validateRELS(pid, dsId, content);
141             }
142         } catch (StreamIOException e) {
143             throw new ValidationException("Failed to get content stream for " + pid + "/" + dsId + ": " + e.getMessage(), e);
144         }
145 
146         if (content != null) {
147             try {
148                 content.close();
149             } catch (IOException e) {
150                 throw new ValidationException("Error closing content stream for " + pid + "/" + dsId + ": " + e.getMessage(), e);
151             }
152         }
153     }

155     private static void validatePOLICY(InputStream content)
156             throws ValidationException {
157         logger.debug("Validating POLICY datastream");
158         policyParser.copy().parse(content, true);
159         logger.debug("POLICY datastream is valid");
160     }
161 
162     private static void validateFESLPOLICY(InputStream content)
163     throws ValidationException {
164         // if FeSL is not enabled, this won't be set
165         if (feslPolicyParser != null) {
166             logger.debug("Validating FESLPOLICY datastream");
167             // FIXME: maintaining backwards compatibility; policy validation should really be enforced
168             feslPolicyParser.copy().parse(content, validateFeslPolicy);
169             logger.debug("FESLPOLICY datastream is valid");
170         }
171     }

181     private static void validateRELS(PID pid, String dsId, InputStream content)
