35      public Condition(String property, String operator, String value)
36              throws InvalidOperatorException, QueryParseException {
37          m_property = property;

45      /**
46       * Gets a List of Conditions from a string like: a=x b~'that\'s' c>='z'
47       * 
48       * @param query
49       *        The query string.
50       * @return The Conditions.
51       */
52      public static List<Condition> getConditions(String query)
53              throws QueryParseException {
54          StringBuffer prop = new StringBuffer();
55          Operator oper = null;
56          StringBuffer val = new StringBuffer();
57          ArrayList<Condition> ret = new ArrayList<Condition>();
58          boolean inProp = true;
59          boolean inValue = false;
60          boolean firstValueChar = false;
61          boolean valueStartsWithQuote = false;
62          for (int i = 0; i < query.length(); i++) {
63              char c = query.charAt(i);
64              if (inProp) {
65                  if (c == '=') {
66                      oper = Operator.EQUALS;
67                      inProp = false;
68                      inValue = true;
69                      firstValueChar = true;
70                  } else if (c == '~') {
71                      oper = Operator.CONTAINS;
72                      inProp = false;
73                      inValue = true;
74                      firstValueChar = true;
75                  } else if (c == '>') {
76                      if (i + 1 < query.length()) {
77                          char d = query.charAt(i + 1);
78                          if (d == '=') {
79                              i++;
80                              oper = Operator.GREATER_OR_EQUAL;
81                          } else {
82                              oper = Operator.GREATER_THAN;
83                          }
84                          inProp = false;
85                          inValue = true;
86                          firstValueChar = true;
87                      } else {
88                          throw new QueryParseException("Found <end-of-string> "
89                                  + "immediately following '>' operator, but "
90                                  + "expected a value.");
91                      }
92                  } else if (c == '<') {
93                      if (i + 1 < query.length()) {
94                          char d = query.charAt(i + 1);
95                          if (d == '=') {
96                              i++;
97                              oper = Operator.LESS_OR_EQUAL;
98                          } else {
99                              oper = Operator.LESS_THAN;
100                         }
101                         inProp = false;
102                         inValue = true;
103                         firstValueChar = true;
104                     } else {
105                         throw new QueryParseException("Found <end-of-string> "
106                                 + "immediately following '<' operator, but "
107                                 + "expected a value.");
108                     }
109                 } else if ((c >= 0x41 && c <= 0x5A) || (c >= 0x61 && c <= 0x7A)) {
110                     prop.append(c);
111                 } else {
112                     throw new QueryParseException("Found ' " + c
113                             + "' at character " + i
114                             + " but expected operator");
115                 }
116             } else if (inValue) {
117                 if (prop.toString().length() == 0) {
118                     throw new QueryParseException("Found "
119                             + "operator but expected a non-zero length "
120                             + "property.");
121                 }
122                 if (firstValueChar) {
123                     // allow ', and mark it if it's there, add one to i
124                     if (c == '\'') {
125                         i++;
126                         if (i >= query.length()) {
127                             throw new QueryParseException("Found <end-of-string> "
128                                     + "immediately following start quote, but "
129                                     + "expected a value.");
130                         }
131                         c = query.charAt(i);
132                         valueStartsWithQuote = true;
133                     }
134                     firstValueChar = false;
135                 }
136                 if (c == '\'') {
137                     if (!valueStartsWithQuote) {
138                         throw new QueryParseException("Found ' character in "
139                                 + "value at position " + i + ", but the value "
140                                 + "did not start with a string, so this can't "
141                                 + " be a value terminator.");
142                     }
143                     // end of value part
144                     // next must be space or empty... check
145                     i++;
146                     if (i < query.length()) {
147                         if (query.charAt(i) != ' ') {
148                             throw new QueryParseException("Found value-terminator "
149                                     + "' but it was not followed by <end-of-string> "
150                                     + "or <space>.");
151                         }
152                     }
153                     ret
154                             .add(new Condition(prop.toString(), oper, val
155                                     .toString()));
156                     prop = new StringBuffer();
157                     oper = null;
158                     val = new StringBuffer();
159                     inValue = false;
160                     inProp = true;
161                     valueStartsWithQuote = false;
162                 } else if (c == '\\') {
163                     i++;
164                     if (i >= query.length()) {
165                         throw new QueryParseException("Found character-escaping "
166                                 + "character as last item in string.");
167                     }
168                     val.append(query.charAt(i));
169                 } else if (c == ' ') {
170                     // end of value part... or inside string?
171                     if (valueStartsWithQuote) {
172                         // was inside string..ok
173                         val.append(c);
174                     } else {
175                         // end of value part...cuz not quotes
176                         ret.add(new Condition(prop.toString(), oper, val
177                                 .toString()));
178                         prop = new StringBuffer();
179                         oper = null;
180                         val = new StringBuffer();
181                         inValue = false;
182                         inProp = true;
183                     }
184                 } else if (c == '=') {
185                     throw new QueryParseException("Found <operator> at position "
186                             + i + ", but expected <value>");
187                 } else if (c == '~') {
188                     throw new QueryParseException("Found <operator> at position "
189                             + i + ", but expected <value>");
190                 } else if (c == '>') {
191                     throw new QueryParseException("Found <operator> at position "
192                             + i + ", but expected <value>");
193                 } else if (c == '<') {
194                     throw new QueryParseException("Found <operator> at position "
195                             + i + ", but expected <value>");
196                 } else {
197                     val.append(c);
198                 }
199             }
200         }
201         if (inProp) {
202             if (prop.toString().length() > 0) {
203                 throw new QueryParseException("String ended before operator "
204                         + "was found");
205             }
206         }
207         if (inValue) {
208             if (valueStartsWithQuote) {
209                 throw new QueryParseException("String ended before quoted value"
210                         + "'s ending quote.");
211             }
212             ret.add(new Condition(prop.toString(), oper, val.toString()));
213         }
214         return ret;
215     }

229     @Override
230     public boolean equals(Object o) {
231         if (o == this) {
232             return true;
233         }
234         if (o == null) {
235             return false;
236         }
237         if (!o.getClass().equals(this.getClass())) {
238             return false;

243                 && equivalent(m_value, that.m_value)
244                 && equivalent(m_operator, that.m_operator);
