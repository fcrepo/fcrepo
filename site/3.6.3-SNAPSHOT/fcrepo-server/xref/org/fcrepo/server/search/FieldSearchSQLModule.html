21  /**
22   * Module that wraps FieldSearchSQLImpl.
23   *
24   * @author Chris Wilper
25   */
26  public class FieldSearchSQLModule
27          extends Module
28          implements FieldSearch {
29  
30      private static final Logger logger =
31              LoggerFactory.getLogger(FieldSearchSQLModule.class);
32  
33      private FieldSearchSQLImpl m_wrappedFieldSearch;
34  
35      public FieldSearchSQLModule(Map params, Server server, String role)
36              throws ModuleInitializationException {
37          super(params, server, role);
38      }
39  
40      @Override
41      public void postInitModule() throws ModuleInitializationException {
42  
43          //
44          // get and validate maxResults
45          //
46          if (getParameter("maxResults") == null) {
47              throw new ModuleInitializationException("maxResults parameter must be specified.",
48                                                      getRole());
49          }
50          int maxResults = 0;
51          try {
52              maxResults = Integer.parseInt(getParameter("maxResults"));
53              if (maxResults < 1) {
54                  throw new NumberFormatException("");
55              }
56          } catch (NumberFormatException nfe) {
57              throw new ModuleInitializationException("maxResults must be a positive integer.",
58                                                      getRole());
59          }
60  
61          //
62          // get and validate maxSecondsPerSession
63          //
64          if (getParameter("maxSecondsPerSession") == null) {
65              throw new ModuleInitializationException("maxSecondsPerSession parameter must be specified.",
66                                                      getRole());
67          }
68          int maxSecondsPerSession = 0;
69          try {
70              maxSecondsPerSession =
71                      Integer.parseInt(getParameter("maxSecondsPerSession"));
72              if (maxSecondsPerSession < 1) {
73                  throw new NumberFormatException("");
74              }
75          } catch (NumberFormatException nfe) {
76              throw new ModuleInitializationException("maxSecondsPerSession must be a positive integer.",
77                                                      getRole());
78          }
79  
80          //
81          // get indexDCFields parameter (default to true if unspecified)
82          //
83          boolean indexDCFields = true;
84          String indexDCFieldsValue = getParameter("indexDCFields");
85          if (indexDCFieldsValue != null) {
86              String val = indexDCFieldsValue.trim().toLowerCase();
87              if (val.equals("false") || val.equals("no")) {
88                  indexDCFields = false;
89              } else if (!val.equals("true") && !val.equals("yes")) {
90                  throw new ModuleInitializationException("indexDCFields param "
91                          + "was not a boolean", getRole());
92              }
93          }
94  
95          //
96          // get connectionPool from ConnectionPoolManager
97          //
98          ConnectionPoolManager cpm =
99                  (ConnectionPoolManager) getServer()
100                         .getModule("org.fcrepo.server.storage.ConnectionPoolManager");
101         if (cpm == null) {
102             throw new ModuleInitializationException("ConnectionPoolManager module was required, but apparently has "
103                                                             + "not been loaded.",
104                                                     getRole());
105         }
106         String cPoolName = getParameter("connectionPool");
107         ConnectionPool cPool = null;
108         try {
109             if (cPoolName == null) {
110                 logger.debug("connectionPool unspecified; using default from "
111                         + "ConnectionPoolManager.");
112                 cPool = cpm.getPool();
113             } else {
114                 logger.debug("connectionPool specified: " + cPoolName);
115                 cPool = cpm.getPool(cPoolName);
116             }
117         } catch (ConnectionPoolNotFoundException cpnfe) {
118             throw new ModuleInitializationException("Could not find requested "
119                     + "connectionPool.", getRole());
120         }
121         //
122         // get the doManager
123         //
124         DOManager doManager =
125                 (DOManager) getServer()
126                         .getModule("org.fcrepo.server.storage.DOManager");
127         if (doManager == null) {
128             throw new ModuleInitializationException("DOManager module was required, but apparently has "
129                                                             + "not been loaded.",
130                                                     getRole());
131         }
132         //
133         // things look ok...get the wrapped instance
134         //
135         m_wrappedFieldSearch =
136                 new FieldSearchSQLImpl(cPool,
137                                        doManager,
138                                        maxResults,
139                                        maxSecondsPerSession,
140                                        indexDCFields);
141     }
142 
143     @Override

146                 "org.fcrepo.server.storage.DOManager"};
147     }
