30  /**
31   * <p>
32   * The abstract base for all JournalWriter classes.
33   * </p>
34   * <p>
35   * Each child class is responsible for providing an XMLEventWriter that will
36   * receive the JournalEntry tag. This class will format a JournalEntry object
37   * into XML and add it to the XMLEventWriter.
38   * </p>
39   * <p>
40   * Note that the writing of an entry is necessarily a three step process,
41   * consisting of
42   * <ol>
43   * <li>calling {@link #prepareToWriteJournalEntry()},</li>
44   * <li>invoking the management method,</li>
45   * <li>calling {@link #writeJournalEntry(CreatorJournalEntry)}</li>
46   * </ol>
47   * Several factors combine to require this sequence.
48   * <ul>
49   * <li>Each new journal file starts with a repository hash.</li>
50   * <li>Repository hashes are expensive to calculate (on the order of 10
51   * seconds). We don't want to calculate a hash unless it is needed for a new
52   * journal file.</li>
53   * <li>We cannot predict in advance which journal entry will require that a new
54   * journal file be opened. The preceding file may be closed asynchronously as a
55   * result of a period of inactivity.</li>
56   * <li>The repository hash must be calculated before the management method is
57   * invoked, so the receiver can confirm the file before making any changes to
58   * its own repository.</li>
59   * <li>The journal entry must be written after the management method is
60   * invoked. The management method may add values to the context object in the
61   * {@link CreatorJournalEntry}, and these values must be written to the
62   * journal.</li>
63   * </ul>
64   * </p>
65   *
66   * @author Jim Blake
67   */
68  public abstract class JournalWriter
69          extends AbstractXmlWriter {

157     protected void writeDocumentHeader(XMLEventWriter writer,

160             throws JournalException {
161         try {
162             putStartDocument(writer);
163             putStartTag(writer, QNAME_TAG_JOURNAL);
164             putAttribute(writer, QNAME_ATTR_REPOSITORY_HASH, repositoryHash);
165             putAttribute(writer, QNAME_ATTR_TIMESTAMP, JournalHelper

184     /**
185      * Format a JournalEntry object and write a JournalEntry tag to the journal.
186      */
187     protected void writeJournalEntry(CreatorJournalEntry journalEntry,
188                                      XMLEventWriter writer)
189             throws JournalException {
190         try {
191             writeJournaEntryStartTag(journalEntry, writer);
192 
193             new ContextXmlWriter().writeContext(journalEntry.getContext(),
194                                                 writer);
195 
196             writeArguments(journalEntry.getArgumentsMap(), writer);
197 
198             putEndTag(writer, QNAME_TAG_JOURNAL_ENTRY);
199             writer.flush();
200         } catch (XMLStreamException e) {
201             throw new JournalException(e);
202         }
203     }

205     private void writeJournaEntryStartTag(CreatorJournalEntry journalEntry,
206                                           XMLEventWriter writer)
207             throws XMLStreamException {
208         putStartTag(writer, QNAME_TAG_JOURNAL_ENTRY);
209         putAttribute(writer, QNAME_ATTR_METHOD, journalEntry.getMethodName());
210         putAttribute(writer, QNAME_ATTR_TIMESTAMP, JournalHelper
211                 .formatDate(journalEntry.getContext().now()));
212 
213         String[] clientIpArray =
214                 journalEntry
215                         .getContext()
216                         .getEnvironmentValues(Constants.HTTP_REQUEST.CLIENT_IP_ADDRESS.uri);
217         if (clientIpArray != null && clientIpArray.length > 0) {
218             putAttribute(writer, QNAME_ATTR_CLIENT_IP, clientIpArray[0]);
219         }
220 
221         String[] loginIdArray =
222                 journalEntry.getContext()
223                         .getSubjectValues(Constants.SUBJECT.LOGIN_ID.uri);
224         if (loginIdArray != null && loginIdArray.length > 0) {
225             putAttribute(writer, QNAME_ATTR_LOGIN_ID, loginIdArray[0]);
226         }
227     }

229     private void writeArguments(Map<String, Object> arguments,
230                                 XMLEventWriter writer)
231             throws XMLStreamException, JournalException {
232         for (String key : arguments.keySet()) {
233             Object value = arguments.get(key);
234             if (value == null) {
235                 writeNullArgument(key, writer);
236             } else if (value instanceof String) {
237                 writeStringArgument(key, (String) value, writer);
238             } else if (value instanceof String[]) {
239                 writeStringArrayArgument(key, (String[]) value, writer);
240             } else if (value instanceof Date) {
241                 writeDateArgument(key, (Date) value, writer);
242             } else if (value instanceof Integer) {
243                 writeIntegerArgument(key, (Integer) value, writer);
244             } else if (value instanceof Boolean) {
245                 writeBooleanArgument(key, (Boolean) value, writer);
246             } else if (value instanceof File) {
247                 writeFileArgument(key, (File) value, writer);
248             } else {
249                 throw new JournalException("Unknown argument type: name='"
250                         + key + "', type='" + value.getClass().getName() + "'");
251             }
252         }
253     }

323     /**
324      * An InputStream argument must be written as a Base64-encoded String. It is
325      * read from the temp file in segments. Each segment is encoded and written
326      * to the XML writer as a series of character events.
327      */
328     private void writeFileArgument(String key, File file, XMLEventWriter writer)
329             throws XMLStreamException, JournalException {
330         try {
331             putStartTag(writer, QNAME_TAG_ARGUMENT);
332             putAttribute(writer, QNAME_ATTR_NAME, key);
333             putAttribute(writer, QNAME_ATTR_TYPE, ARGUMENT_TYPE_STREAM);
334 
335             EncodingBase64InputStream encoder =
336                     new EncodingBase64InputStream(new BufferedInputStream(new FileInputStream(file)));
337             String encodedChunk;
338             while (null != (encodedChunk = encoder.read(1000))) {
339                 putCharacters(writer, encodedChunk);
340             }
341             encoder.close();
342             putEndTag(writer, QNAME_TAG_ARGUMENT);
343         } catch (IOException e) {
344             throw new JournalException("IO Exception on temp file", e);
345         }
346     }

348     /**
349      * This method must not be called before the server has completed
350      * initialization. That's the only way we can be confident that the
351      * DOManager is present, and ready to create the repository has that we will
352      * compare to.
353      */
354     private String getRepositoryHash() throws JournalException {
355         if (!server.hasInitialized()) {
356             throw new IllegalStateException("The repository hash is not available until "
357                     + "the server is fully initialized.");
358         }
359 
360         try {
361             return server.getRepositoryHash();
362         } catch (ServerException e) {
363             throw new JournalException(e);
364         }
