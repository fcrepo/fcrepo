36   * The abstract base for all JournalReader classes.
37   * <p>
38   * Each child class is responsible for providing an XMLEventReader that is
39   * positioned at the beginning of a JournalEntry tag. This class will read the
40   * entry and leave the XMLEventReader positioned after the corresponding closing

43   * @author Jim Blake
44   */

111     private void parseParameters() throws JournalException {
112         String ignore = parameters.get(PARAMETER_IGNORE_HASH);
113         if (ignore == null) {
114             ignoreHashErrors = false;
115         } else if (ignore.equals(VALUE_FALSE)) {
116             ignoreHashErrors = false;
117         } else if (ignore.equals(VALUE_TRUE)) {
118             ignoreHashErrors = true;
119         } else {
120             throw new JournalException("'" + PARAMETER_IGNORE_HASH
121                     + "' parameter must be '" + VALUE_FALSE + "'(default) or '"
122                     + VALUE_TRUE + "'");
123         }
124     }

142     /**
143      * Compare the repository hash from the journal file with the current hash
144      * obtained from the server. If they do not match, either throw an exception
145      * or simply log it, depending on the parameters. This method must not be
146      * called before the server has completed initialization. That's the only
147      * way we can be confident that the DOManager is present, and ready to
148      * create the repository has that we will compare to.
149      */
150     protected void checkRepositoryHash(String hash) throws JournalException {
151         if (!server.hasInitialized()) {
152             throw new IllegalStateException("The repository has is not available until "
153                     + "the server is fully initialized.");
154         }
155 
156         JournalException hashException = null;
157 
158         if (hash == null) {
159             hashException =
160                     new JournalException("'" + QNAME_TAG_JOURNAL
161                             + "' tag must have a '"
162                             + QNAME_ATTR_REPOSITORY_HASH + "' attribute.");
163         } else {
164             try {
165                 String currentHash = server.getRepositoryHash();
166                 if (hash.equals(currentHash)) {
167                     recoveryLog.log("Validated repository hash: '" + hash
168                             + "'.");
169                 } else {
170                     hashException =
171                             new JournalException("'"
172                                     + QNAME_ATTR_REPOSITORY_HASH
173                                     + "' attribute in '"
174                                     + QNAME_TAG_JOURNAL
175                                     + "' tag does not match current repository hash: '"
176                                     + hash + "' vs. '" + currentHash + "'.");
177                 }
178             } catch (ServerException e) {
179                 hashException = new JournalException(e);
180             }
181         }
182 
183         if (hashException != null) {
184             if (ignoreHashErrors) {
185                 recoveryLog.log("WARNING: " + hashException.getMessage());
186             } else {
187                 throw hashException;
188             }
189         }
190 
191     }

193     /**
194      * Read a JournalEntry from the journal, to produce a
195      * <code>ConsumerJournalEntry</code> instance. Concrete sub-classes should
196      * insure that the XMLEventReader is positioned at the beginning of a
197      * JournalEntry before calling this method.
198      */
199     protected ConsumerJournalEntry readJournalEntry(XMLEventReader reader)
200             throws JournalException, XMLStreamException {
201         StartElement startTag = getJournalEntryStartTag(reader);
202         String methodName =
203                 getRequiredAttributeValue(startTag, QNAME_ATTR_METHOD);
204 
205         JournalEntryContext context =
206                 new ContextXmlReader().readContext(reader);
207         ConsumerJournalEntry cje =
208                 new ConsumerJournalEntry(methodName, context);
209 
210         readArguments(reader, cje);
211 
212         return cje;
213     }

215     /**
216      * Get the next event and complain if it isn't a JournalEntry start tag.
217      */
218     private StartElement getJournalEntryStartTag(XMLEventReader reader)
219             throws XMLStreamException, JournalException {
220         XMLEvent event = reader.nextTag();
221         if (!isStartTagEvent(event, QNAME_TAG_JOURNAL_ENTRY)) {
222             throw getNotStartTagException(QNAME_TAG_JOURNAL_ENTRY, event);
223         }
224         return event.asStartElement();
225     }

227     /**
228      * Read arguments and add them to the event, until we hit the end tag for
229      * the event.
230      */
231     private void readArguments(XMLEventReader reader, ConsumerJournalEntry cje)
232             throws XMLStreamException, JournalException {
233         while (true) {
234             XMLEvent nextTag = reader.nextTag();
235             if (isStartTagEvent(nextTag, QNAME_TAG_ARGUMENT)) {
236                 readArgument(nextTag, reader, cje);
237             } else if (isEndTagEvent(nextTag, QNAME_TAG_JOURNAL_ENTRY)) {
238                 return;
239             } else {
240                 throw getNotNextMemberOrEndOfGroupException(QNAME_TAG_JOURNAL_ENTRY,
241                                                             QNAME_TAG_ARGUMENT,
242                                                             nextTag);
243             }
244         }
245     }

247     private void readArgument(XMLEvent nextTag,
248                               XMLEventReader reader,
249                               ConsumerJournalEntry journalEntry)
250             throws JournalException, XMLStreamException {
251         StartElement element = nextTag.asStartElement();
252 
253         String argName = getRequiredAttributeValue(element, QNAME_ATTR_NAME);
254         String argType = getRequiredAttributeValue(element, QNAME_ATTR_TYPE);
255 
256         if (ARGUMENT_TYPE_NULL.equals(argType)) {
257             readNullArgument(reader, journalEntry, argName);
258         } else if (ARGUMENT_TYPE_STRING.equals(argType)) {
259             readStringArgument(reader, journalEntry, argName);
260         } else if (ARGUMENT_TYPE_STRINGARRAY.equals(argType)) {
261             readStringArrayArgument(reader, journalEntry, argName);
262         } else if (ARGUMENT_TYPE_INTEGER.equals(argType)) {
263             readIntegerArgument(reader, journalEntry, argName);
264         } else if (ARGUMENT_TYPE_BOOLEAN.equals(argType)) {
265             readBooleanArgument(reader, journalEntry, argName);
266         } else if (ARGUMENT_TYPE_DATE.equals(argType)) {
267             readDateArgument(reader, journalEntry, argName);
268         } else if (ARGUMENT_TYPE_STREAM.equals(argType)) {
269             readStreamArgument(reader, journalEntry, argName);
270         } else {
271             throw new JournalException("Unknown argument type: name='"
272                     + argName + "', type='" + argType + "'");
273         }
274     }

276     private void readStringArgument(XMLEventReader reader,
277                                     ConsumerJournalEntry journalEntry,
278                                     String name) throws XMLStreamException,
279             JournalException {
280         String value =
281                 readCharactersUntilEndOfArgument(reader,
282                                                  QNAME_TAG_ARGUMENT,
283                                                  journalEntry.getMethodName(),
284                                                  name,
285                                                  ARGUMENT_TYPE_STRING);
286         journalEntry.addArgument(name, value);
287     }

289     private void readStringArrayArgument(XMLEventReader reader,
290                                          ConsumerJournalEntry journalEntry,
291                                          String name)
292             throws XMLStreamException, JournalException {
293         List<String> values = new ArrayList<String>();
294         while (true) {
295             XMLEvent event = reader.nextTag();
296             if (isStartTagEvent(event, QNAME_TAG_ARRAYELEMENT)) {
297                 values
298                         .add(readCharactersUntilEndOfArgument(reader,
299                                                               QNAME_TAG_ARRAYELEMENT,
300                                                               journalEntry
301                                                                       .getMethodName(),
302                                                               name,
303                                                               ARGUMENT_TYPE_STRINGARRAY));
304             } else if (isEndTagEvent(event, QNAME_TAG_ARGUMENT)) {
305                 break;
306             } else {
307                 throw getUnexpectedEventInArgumentException(name,
308                                                             ARGUMENT_TYPE_STRINGARRAY,
309                                                             journalEntry
310                                                                     .getMethodName(),
311                                                             event);
312             }
313         }
314         Object[] valuesArray = values.toArray(new String[values.size()]);
315         journalEntry.addArgument(name, valuesArray);
316     }

333         XMLEvent endTag = reader.nextEvent();

338                                                         ARGUMENT_TYPE_INTEGER,
339                                                         journalEntry

400     /**
401      * An InputStream argument appears as a Base64-encoded String. It must be
402      * decoded and written to a temp file, so it can be presented to the
403      * management method as an InputStream again.
404      */
405     private void readStreamArgument(XMLEventReader reader,
406                                     ConsumerJournalEntry journalEntry,
407                                     String name) throws XMLStreamException,
408             JournalException {
409         try {
410             File tempFile = JournalHelper.createTempFile();
411             DecodingBase64OutputStream decoder =
412                     new DecodingBase64OutputStream(new FileOutputStream(tempFile));
413 
414             while (true) {
415                 XMLEvent event = reader.nextEvent();
416                 if (event.isCharacters()) {
417                     decoder.write(event.asCharacters().getData());
418                 } else if (isEndTagEvent(event, QNAME_TAG_ARGUMENT)) {
419                     break;
420                 } else {
421                     throw getUnexpectedEventInArgumentException(name,
422                                                                 ARGUMENT_TYPE_STREAM,
423                                                                 journalEntry
424                                                                         .getMethodName(),
425                                                                 event);
426                 }
427             }
428             decoder.close();
429             journalEntry.addArgument(name, tempFile);
430         } catch (IOException e) {
431             throw new JournalException("failed to write stream argument to temp file",
432                                        e);
433         }
434     }

450     /**
451      * Loop through a series of character events, accumulating the data into a
452      * String. The character events should be terminated by an EndTagEvent with
453      * the expected tag name.
454      */
455     private String readCharactersUntilEndOfArgument(XMLEventReader reader,
456                                                     QName tagName,
457                                                     String methodName,
458                                                     String argumentName,
459                                                     String argumentType)
460             throws XMLStreamException, JournalException {
461         StringBuffer stringValue = new StringBuffer();
462         while (true) {
463             XMLEvent event = reader.nextEvent();
464             if (event.isCharacters()) {
465                 stringValue.append(event.asCharacters().getData());
466             } else if (isEndTagEvent(event, tagName)) {
467                 break;
468             } else {
469                 throw getUnexpectedEventInArgumentException(argumentName,
470                                                             argumentType,
471                                                             methodName,
472                                                             event);
473             }
474         }
475         return stringValue.toString();
476     }

478     /**
479      * If we encounter an unexpected event when reading the a method argument,
480      * create an exception with all of the pertinent information.
481      */
482     private JournalException getUnexpectedEventInArgumentException(String name,
483                                                                    String argumentType,
484                                                                    String methodName,
485                                                                    XMLEvent event) {
486         return new JournalException("Unexpected event while processing '"
487                 + name + "' argument (type = '" + argumentType + "') for '"
488                 + methodName + "' method call: event is '" + event + "'");
489     }
