43  /**
44   * Implements the "describeRepository" functionality of the Fedora Access LITE
45   * (API-A-LITE) interface using a java servlet front end. The syntax defined by
46   * API-A-LITE has for getting a description of the repository has the following
47   * binding:
48   * <ol>
49   * <li>describeRepository URL syntax:
50   * protocol://hostname:port/fedora/describe[?xml=BOOLEAN] This syntax requests
51   * information about the repository. The xml parameter determines the type of
52   * output returned. If the parameter is omitted or has a value of "false", a
53   * MIME-typed stream consisting of an html table is returned providing a
54   * browser-savvy means of viewing the object profile. If the value specified is
55   * "true", then a MIME-typed stream consisting of XML is returned.</li>
56   * <ul>
57   * <li>protocol - either http or https.</li>
58   * <li>hostname - required hostname of the Fedora server.</li>
59   * <li>port - required port number on which the Fedora server is running.</li>
60   * <li>fedora - required name of the Fedora access service.</li>
61   * <li>describe - required verb of the Fedora service.</li>
62   * <li>xml - an optional parameter indicating the requested output format. A
63   * value of "true" indicates a return type of text/xml; the absence of the xml
64   * parameter or a value of "false" indicates format is to be text/html.</li>
65   * </ul>
66   *
67   * @author Ross Wayland
68   */
69  public class DescribeRepositoryServlet

86      /**
87       * <p>
88       * Process Fedora Access Request. Parse and validate the servlet input
89       * parameters and then execute the specified request.
90       * </p>
91       *
92       * @param request
93       *        The servlet request.
94       * @param response
95       *        servlet The servlet response.
96       * @throws ServletException
97       *         If an error occurs that effects the servlet's basic operation.
98       * @throws IOException
99       *         If an error occurrs with an input or output operation.
100      */
101     @Override
102     public void doGet(HttpServletRequest request, HttpServletResponse response)
103             throws ServletException, IOException {
104         boolean xml = false;
105 
106         logger.debug("Got request: " + request.getRequestURL().toString() + "?"
107                 + request.getQueryString());
108 
109         // Check for xml parameter.
110         for (Enumeration<?> e = request.getParameterNames(); e.hasMoreElements();) {
111             String name = URLDecoder.decode((String) e.nextElement(), "UTF-8");
112             if (name.equalsIgnoreCase("xml")) {
113                 xml = new Boolean(request.getParameter(name)).booleanValue();
114             }
115         }
116         Context context =
117                 ReadOnlyContext.getContext(Constants.HTTP_REQUEST.REST.uri,
118                                            request);
119         try {
120             describeRepository(context, xml, response);
121         } catch (AuthzException ae) {
122             throw RootException.getServletException(ae,
123                                                     request,
124                                                     ACTION_LABEL,
125                                                     new String[0]);
126         } catch (Throwable th) {
127             throw new InternalError500Exception("",
128                                                 th,
129                                                 request,
130                                                 ACTION_LABEL,
131                                                 "",
132                                                 new String[0]);
133         }
134 
135     }

137     public void describeRepository(Context context,
138                                    boolean xml,
139                                    HttpServletResponse response)
140             throws ServerException {
141 
142         OutputStreamWriter out = null;
143         RepositoryInfo repositoryInfo = null;
144         PipedWriter pw = null;
145         PipedReader pr = null;
146 
147         try {
148             pw = new PipedWriter();
149             pr = new PipedReader(pw);
150             repositoryInfo = m_access.describeRepository(context);
151             if (repositoryInfo != null) {
152                 // Repository info obtained.
153                 // Serialize the RepositoryInfo object into XML
154                 new ReposInfoSerializerThread(context, repositoryInfo, pw)
155                         .start();
156                 if (xml) {
157                     // Return results as raw XML
158                     response.setContentType(CONTENT_TYPE_XML);
159 
160                     // Insures stream read from PipedReader correctly translates
161                     // utf-8
162                     // encoded characters to OutputStreamWriter.
163                     out =
164                             new OutputStreamWriter(response.getOutputStream(),
165                                                    "UTF-8");
166                     int bufSize = 4096;
167                     char[] buf = new char[bufSize];
168                     int len = 0;
169                     while ((len = pr.read(buf, 0, bufSize)) != -1) {
170                         out.write(buf, 0, len);
171                     }
172                     out.flush();
173                 } else {
174                     // Transform results into an html table
175                     response.setContentType(CONTENT_TYPE_HTML);
176                     out =
177                             new OutputStreamWriter(response.getOutputStream(),
178                                                    "UTF-8");
179                     File xslFile =
180                             new File(m_server.getHomeDir(),
181                                      "access/viewRepositoryInfo.xslt");
182                     TransformerFactory factory =
183                             XmlTransformUtility.getTransformerFactory();
184                     Templates template =
185                             factory.newTemplates(new StreamSource(xslFile));
186                     Transformer transformer = template.newTransformer();
187                     transformer.setParameter("fedora", context
188                                              .getEnvironmentValue(FEDORA_APP_CONTEXT_NAME));
189                     transformer.transform(new StreamSource(pr),
190                                           new StreamResult(out));
191                 }
192                 out.flush();
193 
194             } else {
195                 // Describe request returned nothing.
196                 logger.error("No repository info returned");
197             }
198         } catch (AuthzException ae) {
199             throw ae;
200         } catch (Throwable th) {
201             String msg = "Error describing repository";
202             logger.error(msg, th);
203             throw new GeneralException(msg, th);
204         } finally {
205             try {
206                 if (pr != null) {
207                     pr.close();
208                 }
209                 if (out != null) {
210                     out.close();
211                 }
212             } catch (Throwable th) {
213                 String message =
214                         "[DescribeRepositoryServlet] An error has occured. "
215                                 + " The error was a \" "
216                                 + th.getClass().getName() + " \". Reason: "
217                                 + th.getMessage();
218                 throw new StreamIOException(message);
219             }
220         }
221     }

223     /**
224      * <p>
225      * A Thread to serialize an ObjectProfile object into XML.
226      * </p>
227      */
228     public class ReposInfoSerializerThread
229             extends Thread {
230 
231         private PipedWriter pw = null;
232 
233         private RepositoryInfo repositoryInfo = null;
234 
235         /**
236          * <p>
237          * Constructor for ReposInfoSerializerThread.
238          * </p>
239          *
240          * @param repositoryInfo
241          *        A repository info data structure.
242          * @param pw
243          *        A PipedWriter to which the serialization info is written.
244          */
245         public ReposInfoSerializerThread(Context context,
246                                          RepositoryInfo repositoryInfo,
247                                          PipedWriter pw) {
248             this.pw = pw;
249             this.repositoryInfo = repositoryInfo;
250             if (Constants.HTTP_REQUEST.SECURE.uri.equals(context
251                     .getEnvironmentValue(Constants.HTTP_REQUEST.SECURITY.uri))) {
252             } else if (Constants.HTTP_REQUEST.INSECURE.uri.equals(context
253                     .getEnvironmentValue(Constants.HTTP_REQUEST.SECURITY.uri))) {
254             }
255         }
256 
257         /**
258          * <p>
259          * This method executes the thread.
260          * </p>
261          */
262         @Override
263         public void run() {
264             if (pw != null) {
265                 try {
266                     pw.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
267                     pw.write("<fedoraRepository"
268                             + " xmlns=\"" + REPO_DESC1_0.namespace.uri + "\""
269                             + " xmlns:xsd=\"" + XML_XSD.uri + "\""
270                             + " xmlns:xsi=\"" + XSI.uri + "\""
271                             + " xsi:schemaLocation=\"" + ACCESS.uri + " "
272                             + REPO_DESC1_0.xsdLocation + "\">");
273 
274                     // REPOSITORY INFO FIELDS SERIALIZATION
275                     pw.write("<repositoryName>" + repositoryInfo.repositoryName
276                             + "</repositoryName>");
277                     pw.write("<repositoryBaseURL>"
278                             + repositoryInfo.repositoryBaseURL
279                             + "</repositoryBaseURL>");
280                     pw.write("<repositoryVersion>"
281                             + repositoryInfo.repositoryVersion
282                             + "</repositoryVersion>");
283                     pw.write("<repositoryPID>");
284                     pw.write("    <PID-namespaceIdentifier>"
285                             + repositoryInfo.repositoryPIDNamespace
286                             + "</PID-namespaceIdentifier>");
287                     pw.write("    <PID-delimiter>" + ":" + "</PID-delimiter>");
288                     pw.write("    <PID-sample>" + repositoryInfo.samplePID
289                             + "</PID-sample>");
290                     String[] retainPIDs = repositoryInfo.retainPIDs;
291                     for (String element : retainPIDs) {
292                         pw.write("    <retainPID>" + element + "</retainPID>");
293                     }
294                     pw.write("</repositoryPID>");
295                     pw.write("<repositoryOAI-identifier>");
296                     pw.write("    <OAI-namespaceIdentifier>"
297                             + repositoryInfo.OAINamespace
298                             + "</OAI-namespaceIdentifier>");
299                     pw.write("    <OAI-delimiter>" + ":" + "</OAI-delimiter>");
300                     pw.write("    <OAI-sample>"
301                             + repositoryInfo.sampleOAIIdentifer
302                             + "</OAI-sample>");
303                     pw.write("</repositoryOAI-identifier>");
304                     pw.write("<sampleSearch-URL>"
305                             + repositoryInfo.sampleSearchURL
306                             + "</sampleSearch-URL>");
307                     pw.write("<sampleAccess-URL>"
308                             + repositoryInfo.sampleAccessURL
309                             + "</sampleAccess-URL>");
310                     pw.write("<sampleOAI-URL>" + repositoryInfo.sampleOAIURL
311                             + "</sampleOAI-URL>");
312                     String[] emails = repositoryInfo.adminEmailList;
313                     for (String element : emails) {
314                         pw.write("<adminEmail>" + element + "</adminEmail>");
315                     }
316                     pw.write("</fedoraRepository>");
317                     pw.flush();
318                     pw.close();
319                 } catch (IOException ioe) {
320                     logger.error("WriteThread IOException", ioe);
321                 } finally {
322                     try {
323                         if (pw != null) {
324                             pw.close();
325                         }
326                     } catch (IOException ioe) {
327                         logger.error("WriteThread IOException", ioe);
328                     }
329                 }
330             }
331         }
332     }
