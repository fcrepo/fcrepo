53  /**
54   * Implements listMethods method of Fedora Access LITE (API-A-LITE) interface
55   * using a java servlet front end.
56   * <ol>
57   * <li>ListMethods URL syntax:
58   * <p>
59   * protocol://hostname:port/fedora/listMethods/PID[/dateTime][?xml=BOOLEAN]
60   * </p>
61   * <p>
62   * This syntax requests a list of methods for the specified digital object. The
63   * xml parameter determines the type of output returned. If the parameter is
64   * omitted or has a value of "false", a MIME-typed stream consisting of an html
65   * table is returned providing a browser-savvy means of viewing the object
66   * profile. If the value specified is "true", then a MIME-typed stream
67   * consisting of XML is returned.
68   * </p>
69   * </li>
70   * <ul>
71   * <li>protocol - either http or https.</li>
72   * <li>hostname - required hostname of the Fedora server.</li>
73   * <li>port - required port number on which the Fedora server is running.</li>
74   * <li>fedora - required name of the Fedora access service.</li>
75   * <li>get - required verb of the Fedora service.</li>
76   * <li>PID - required persistent identifier of the digital object.</li>
77   * <li>dateTime - optional dateTime value indicating dissemination of a version
78   * of the digital object at the specified point in time. (NOT currently
79   * implemented.)
80   * <li>xml - an optional parameter indicating the requested output format. A
81   * value of "true" indicates a return type of text/xml; the absence of the xml
82   * parameter or a value of "false" indicates format is to be text/html.</li>
83   * </ul>
84   * </ol>
85   *
86   * @author Ross Wayland
87   * @version $Id$
88   */

133     @Override
134     public void doGet(HttpServletRequest request, HttpServletResponse response)
135             throws ServletException, IOException {
136         String PID = null;
137         Date asOfDateTime = null;
138         Date versDateTime = null;
139         boolean xml = false;
140         requestURI =
141                 request.getRequestURL().toString() + "?"
142                         + request.getQueryString();
143 
144         // Parse servlet URL.
145         String[] URIArray = request.getRequestURL().toString().split("/");
146         if (URIArray.length == 6 || URIArray.length == 7) {
147             // Request is either unversioned or versioned listMethods request
148             try {
149                 PID = Server.getPID(URIArray[5]).toString(); // normalize PID
150             } catch (Throwable th) {
151                 logger.error("Bad pid syntax in request", th);
152                 throw new BadRequest400Exception(request,
153                                                  ACTION_LABEL,
154                                                  "",
155                                                  new String[0]);
156             }
157             if (URIArray.length == 7) {
158                 // Request is a versioned listMethods request
159                 try {
160                     versDateTime = DateUtility.parseDateStrict(URIArray[6]);
161                 } catch(ParseException e) {
162                     logger.error("Bad date format in request");
163                     throw new BadRequest400Exception(request,
164                                                      ACTION_LABEL,
165                                                      "",
166                                                      new String[0]);
167                 }
168                 asOfDateTime = versDateTime;
169             }
170             logger.debug("Listing methods (PID=" + PID + ", asOfDate="
171                     + versDateTime + ")");
172         } else {
173             logger.error("Bad syntax (expected 6 or 7 parts) in request");
174             throw new BadRequest400Exception(request,
175                                              ACTION_LABEL,
176                                              "",
177                                              new String[0]);
178         }
179 
180         if (request.getParameter("xml") != null) {
181             xml = new Boolean(request.getParameter("xml")).booleanValue();
182         }
183 
184         try {
185             Context context =
186                     ReadOnlyContext.getContext(HTTP_REQUEST.REST.uri, request);
187             listMethods(context, PID, asOfDateTime, xml, request, response);
188             logger.debug("Finished listing methods");
189         } catch (ObjectNotFoundException e) {
190             logger.error("Object not found for request: " + requestURI
191                     + " (actionLabel=" + ACTION_LABEL + ")", e);
192             throw new NotFound404Exception(request,
193                                            ACTION_LABEL,
194                                            "",
195                                            new String[0]);
196         } catch (DisseminationException e) {
197             logger.error("Error Listing Methods: " + requestURI + " (actionLabel="
198                     + ACTION_LABEL + ")", e);
199             throw new NotFound404Exception("Error Listing Methods",
200                                            e,
201                                            request,
202                                            ACTION_LABEL,
203                                            "",
204                                            new String[0]);
205         } catch (ObjectNotInLowlevelStorageException e) {
206             logger.error("Object not found for request: " + requestURI
207                     + " (actionLabel=" + ACTION_LABEL + ")", e);
208             throw new NotFound404Exception(request,
209                                            ACTION_LABEL,
210                                            "",
211                                            new String[0]);
212         } catch (AuthzException ae) {
213             logger.error("Authorization error listing methods", ae);
214             throw RootException.getServletException(ae,
215                                                     request,
216                                                     ACTION_LABEL,
217                                                     new String[0]);
218         } catch (Throwable th) {
219             logger.error("Error listing methods", th);
220             throw new InternalError500Exception("Error listing methods",
221                                                 th,
222                                                 request,
223                                                 ACTION_LABEL,
224                                                 "",
225                                                 new String[0]);
226         }

229     public void listMethods(Context context,
230                             String PID,
231                             Date asOfDateTime,
232                             boolean xml,
233                             HttpServletRequest request,
234                             HttpServletResponse response)
235             throws ServerException {
236 
237         OutputStreamWriter out = null;
238         Date versDateTime = asOfDateTime;
239         ObjectMethodsDef[] methodDefs = null;
240         PipedWriter pw = null;
241         PipedReader pr = null;
242 
243         try {
244             pw = new PipedWriter();
245             pr = new PipedReader(pw);
246             methodDefs = m_access.listMethods(context, PID, asOfDateTime);
247 
248             // Object Profile found.
249             // Serialize the ObjectProfile object into XML
250             new ObjectMethodsDefSerializerThread(context,
251                                                  PID,
252                                                  methodDefs,
253                                                  versDateTime,
254                                                  pw).start();
255             if (xml) {
256                 // Return results as raw XML
257                 response.setContentType(CONTENT_TYPE_XML);
258 
259                 // Insures stream read from PipedReader correctly translates
260                 // utf-8
261                 // encoded characters to OutputStreamWriter.
262                 out =
263                         new OutputStreamWriter(response.getOutputStream(),
264                                                "UTF-8");
265                 int bufSize = 4096;
266                 char[] buf = new char[bufSize];
267                 int len = 0;
268                 while ((len = pr.read(buf, 0, bufSize)) != -1) {
269                     out.write(buf, 0, len);
270                 }
271                 out.flush();
272             } else {
273                 // Transform results into an html table
274                 response.setContentType(CONTENT_TYPE_HTML);
275                 out =
276                         new OutputStreamWriter(response.getOutputStream(),
277                                                "UTF-8");
278                 File xslFile =
279                         new File(m_server.getHomeDir(),
280                                  "access/listMethods.xslt");
281                 TransformerFactory factory =
282                         XmlTransformUtility.getTransformerFactory();
283                 Templates template =
284                         factory.newTemplates(new StreamSource(xslFile));
285                 Transformer transformer = template.newTransformer();
286                 transformer.setParameter("fedora", context
287                         .getEnvironmentValue(FEDORA_APP_CONTEXT_NAME));
288                 transformer.transform(new StreamSource(pr),
289                                       new StreamResult(out));
290             }
291             out.flush();
292         } catch (ServerException e) {
293             throw e;
294         } catch (Throwable th) {
295             String message = "Error listing methods";
296             logger.error(message, th);
297             throw new GeneralException(message, th);
298         } finally {
299             try {
300                 if (pr != null) {
301                     pr.close();
302                 }
303                 if (out != null) {
304                     out.close();
305                 }
306             } catch (Throwable th) {
307                 String message =
308                         "[ListMethodsServlet] An error has occured. "
309                                 + " The error was a \" "
310                                 + th.getClass().getName() + " \". Reason: "
311                                 + th.getMessage();
312                 throw new StreamIOException(message);
313             }
314         }
315     }

317     /**
318      * <p>
319      * A Thread to serialize an ObjectMethodDef object into XML.
320      * </p>
321      */
322     public class ObjectMethodsDefSerializerThread
323             extends Thread {
324 
325         private PipedWriter pw = null;
326 
327         private String PID = null;
328 
329         private ObjectMethodsDef[] methodDefs = null;
330 
331         private Date versDateTime = null;
332 
333         private String fedoraServerProtocol = null;
334 
335         private String fedoraServerPort = null;
336 
337         private String fedoraAppServerContext = null;
338 
339         /**
340          * <p>
341          * Constructor for ProfileSerializeThread.
342          * </p>
343          *
344          * @param PID
345          *        The persistent identifier of the specified digital object.
346          * @param methodDefs
347          *        An array of ObjectMethodsDefs.
348          * @param versDateTime
349          *        The version datetime stamp of the request.
350          * @param pw
351          *        A PipedWriter to which the serialization info is written.
352          */
353         public ObjectMethodsDefSerializerThread(Context context,
354                                                 String PID,
355                                                 ObjectMethodsDef[] methodDefs,
356                                                 Date versDateTime,
357                                                 PipedWriter pw) {
358             this.pw = pw;
359             this.PID = PID;
360             this.methodDefs = methodDefs;
361             this.versDateTime = versDateTime;
362             fedoraServerPort =
363                     context.getEnvironmentValue(HTTP_REQUEST.SERVER_PORT.uri);
364 
365             fedoraAppServerContext =
366                     context
367                             .getEnvironmentValue(Constants.FEDORA_APP_CONTEXT_NAME);
368             if (HTTP_REQUEST.SECURE.uri.equals(context
369                     .getEnvironmentValue(HTTP_REQUEST.SECURITY.uri))) {
370                 fedoraServerProtocol = HTTPS;
371             } else if (HTTP_REQUEST.INSECURE.uri.equals(context
372                     .getEnvironmentValue(HTTP_REQUEST.SECURITY.uri))) {
373                 fedoraServerProtocol = HTTP;
374             }
375         }
376 
377         /**
378          * <p>
379          * This method executes the thread.
380          * </p>
381          */
382         @Override
383         public void run() {
384             if (pw != null) {
385                 try {
386                     pw.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
387                     pw.write("<objectMethods");
388                     pw.write(" pid=\"" + PID + "\"");
389                     if (versDateTime != null) {
390                         pw.write(" asOfDateTime=\"");
391                         pw.write(DateUtility.convertDateToString(versDateTime));
392                         pw.write("\"");
393                     }
394                     pw.write(" baseURL=\""
395                             + StreamUtility.enc(fedoraServerProtocol) + "://"
396                             + StreamUtility.enc(m_fedoraServerHost) + ":"
397                             + StreamUtility.enc(fedoraServerPort) + "/"
398                             + fedoraAppServerContext + "/\"");
399                     pw.write(" xmlns=\"" + OBJ_METHODS1_0.namespace.uri + "\" ");
400                     pw.write(" xmlns:xsi=\"" + XSI.uri + "\" ");
401                     pw.write(" xsi:schemaLocation=\"" + OBJ_METHODS1_0.namespace.uri);
402                     pw.write(" " + OBJ_METHODS1_0.xsdLocation + "\">");
403 
404                     // ObjectMethodsDef SERIALIZATION
405                     String nextSdef = "null";
406                     String currentSdef = "";
407                     for (int i = 0; i < methodDefs.length; i++) {
408                         currentSdef = methodDefs[i].sDefPID;
409                         if (!currentSdef.equalsIgnoreCase(nextSdef)) {
410                             if (i != 0) {
411                                 pw.write("</sDef>");
412                             }
413                             pw.write("<sDef pid=\""
414                                     + StreamUtility.enc(methodDefs[i].sDefPID)
415                                     + "\" >");
416                         }
417                         pw.write("<method name=\""
418                                 + StreamUtility.enc(methodDefs[i].methodName)
419                                 + "\" >");
420                         MethodParmDef[] methodParms =
421                                 methodDefs[i].methodParmDefs;
422                         for (MethodParmDef element : methodParms) {
423                             pw.write("<methodParm parmName=\""
424                                     + StreamUtility.enc(element.parmName)
425                                     + "\" parmDefaultValue=\""
426                                     + StreamUtility
427                                             .enc(element.parmDefaultValue)
428                                     + "\" parmRequired=\""
429                                     + element.parmRequired + "\" parmLabel=\""
430                                     + StreamUtility.enc(element.parmLabel)
431                                     + "\" >");
432                             if (element.parmDomainValues.length > 0) {
433                                 pw.write("<methodParmDomain>");
434                                 for (String element2 : element.parmDomainValues) {
435                                     pw.write("<methodParmValue>"
436                                             + StreamUtility.enc(element2)
437                                             + "</methodParmValue>");
438                                 }
439                                 pw.write("</methodParmDomain>");
440                             }
441                             pw.write("</methodParm>");
442                         }
443 
444                         pw.write("</method>");
445                         nextSdef = currentSdef;
446                     }
447                     pw.write("</sDef>");
448                     pw.write("</objectMethods>");
449 
450                     pw.flush();
451                     pw.close();
452                 } catch (IOException ioe) {
453                     logger.error("WriteThread error", ioe);
454                 } finally {
455                     try {
456                         if (pw != null) {
457                             pw.close();
458                         }
459                     } catch (IOException ioe) {
460                         logger.error("WriteThread error", ioe);
461                     }
462                 }
463             }
464         }
465     }
