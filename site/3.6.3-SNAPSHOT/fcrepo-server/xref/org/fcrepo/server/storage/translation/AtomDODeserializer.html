130     /**
131      * {@inheritDoc}
132      */
133     public void deserialize(InputStream in,
134                             DigitalObject obj,
135                             String encoding,
136                             int transContext) throws ObjectIntegrityException,
137             StreamIOException, UnsupportedEncodingException {
138         if (m_format.equals(ATOM_ZIP1_1)) {
139             try {
140                 m_tempDir = FileUtils.createTempDir("atomzip", null);
141                 m_zin = new ZipInputStream(new BufferedInputStream(in));
142                 ZipEntry entry;
143                 while ((entry = m_zin.getNextEntry()) != null) {
144                     FileUtils.copy(m_zin, new FileOutputStream(new File(m_tempDir, entry.getName())));
145                 }
146                 in = new FileInputStream(new File(m_tempDir, "atommanifest.xml"));
147             } catch (FileNotFoundException e) {
148                 throw new StreamIOException(e.getMessage(), e);
149             } catch (IOException e) {
150                 throw new StreamIOException(e.getMessage(), e);
151             }
152         }
153 
154         Parser parser = abdera.getParser();
155         Document<Feed> feedDoc = parser.parse(in);
156         m_feed = feedDoc.getRoot();
157         m_xpath = abdera.getXPath();
158 
159         m_obj = obj;
160         m_encoding = encoding;
161         m_transContext = transContext;
162         addObjectProperties();
163         addDatastreams();
164 
165         DOTranslationUtility.normalizeDatastreams(m_obj,
166                                                   m_transContext,
167                                                   m_encoding);
168         FileUtils.delete(m_tempDir);
169     }

179      * Set the Fedora Object properties from the Feed metadata.
180      *
181      * @throws ObjectIntegrityException
182      */
183     private void addObjectProperties() throws ObjectIntegrityException {
184         PID pid;
185         try {
186             pid = new PID(m_feed.getId().toString());
187         } catch (MalformedPIDException e) {
188             throw new ObjectIntegrityException(e.getMessage(), e);
189         }
190 
191         String label = m_feed.getTitle();
192         String state =
193                 m_xpath.valueOf("/a:feed/a:category[@scheme='"
194                         + MODEL.STATE.uri + "']/@term", m_feed);
195         String createDate =
196                 m_xpath.valueOf("/a:feed/a:category[@scheme='"
197                         + MODEL.CREATED_DATE.uri + "']/@term", m_feed);
198 
199         m_obj.setPid(pid.toString());
200 
201         try {
202             m_obj.setState(DOTranslationUtility.readStateAttribute(state));
203         } catch (ParseException e) {
204             throw new ObjectIntegrityException("Could not read object state", e);
205         }
206 
207         m_obj.setLabel(label);
208         m_obj.setOwnerId(getOwnerId());
209         m_obj.setCreateDate(DateUtility.convertStringToDate(createDate));
210         m_obj.setLastModDate(m_feed.getUpdated());
211 
212         setExtProps();
213     }

215     private void addDatastreams() throws UnsupportedEncodingException,
216             StreamIOException, ObjectIntegrityException {
217         m_feed.sortEntries(new UpdatedIdComparator(true));
218         List<Entry> entries = m_feed.getEntries();
219         for (Entry entry : entries) {
220             if (ThreadHelper.getInReplyTo(entry) != null) {
221                 addDatastreamVersion(entry);
222             }
223         }
224     }

226     private void addDatastreamVersion(Entry entry)
227             throws UnsupportedEncodingException, StreamIOException,
228             ObjectIntegrityException {
229         IRI ref = ThreadHelper.getInReplyTo(entry).getRef();
230         Entry parent = m_feed.getEntry(ref.toString());
231 
232         Datastream ds;
233         String controlGroup = getDSControlGroup(parent);
234         if (controlGroup.equals("X")) {
235             ds = addInlineDatastreamVersion(entry);
236         } else if (controlGroup.equals("M")) {
237             ds = addManagedDatastreamVersion(entry);
238         } else {
239             ds = addExternalReferencedDatastreamVersion(entry);
240         }
241         m_obj.addDatastreamVersion(ds, true);
242     }

244     private Datastream addInlineDatastreamVersion(Entry entry)
245             throws ObjectIntegrityException, StreamIOException {
246         DatastreamXMLMetadata ds = new DatastreamXMLMetadata();
247         setDSCommonProperties(ds, entry);
248         String dsId = ds.DatastreamID;
249         String dsvId = ds.DSVersionID;
250         ds.DSLocation = m_obj.getPid() + "+" + dsId + "+" + dsvId;
251 
252         if (ds.DSVersionID.equals("AUDIT.0")) {
253             addAuditDatastream(entry);
254         } else {
255             try {
256                 if (m_format.equals(ATOM_ZIP1_1)) {
257                     ByteArrayOutputStream bout = new ByteArrayOutputStream();
258                     FileUtils.copy(new FileInputStream(getContentSrcAsFile(entry.getContentSrc())),
259                             bout);
260                     ds.xmlContent = bout.toByteArray();
261 
262                 } else {
263                     ds.xmlContent = entry.getContent().getBytes(m_encoding); //IOUtils.toByteArray(entry.getContentStream());
264                 }
265             } catch (UnsupportedEncodingException e) {
266                 throw new StreamIOException(e.getMessage(), e);
267             } catch (FileNotFoundException e) {
268                 throw new ObjectIntegrityException(e.getMessage(), e);
269             }
270         }
271 
272         if (ds.xmlContent != null) {
273             ds.DSSize = ds.xmlContent.length;
274         }
275 
276         MimeType mimeType = entry.getContentMimeType();
277         if (mimeType == null) {
278             ds.DSMIME = "text/xml";
279         } else {
280             ds.DSMIME = mimeType.toString();
281         }
282         return ds;
283     }

301     private Datastream addManagedDatastreamVersion(Entry entry)
302             throws StreamIOException, ObjectIntegrityException {
303         Datastream ds = new DatastreamManagedContent();
304         setDSCommonProperties(ds, entry);
305         ds.DSLocationType = Datastream.DS_LOCATION_TYPE_INTERNAL;
306 
307         ds.DSMIME = getDSMimeType(entry);
308 
309         // Managed Content can take any of the following forms:
310         // 1) inline text (plaintext, html, xml)
311         // 2) inline Base64
312         // 3) referenced content
313         IRI contentLocation = entry.getContentSrc();
314         if (contentLocation != null) {
315             // URL FORMAT VALIDATION for dsLocation:
316             // For Managed Content the URL is only checked when we are parsing a
317             // a NEW ingest file because the URL is replaced with an internal identifier
318             // once the repository has sucked in the content for storage.
319 
320         	// AtomZIP files can have a simple filename (nb, not file: )for the content location, so don't validate that
321             if (m_obj.isNew() && !m_format.equals(ATOM_ZIP1_1)) {
322                 ValidationUtility
323                         .validateURL(contentLocation.toString(),ds.DSControlGrp);
324             }
325 
326             if (m_format.equals(ATOM_ZIP1_1)) {
327                 if (!contentLocation.isAbsolute() && !contentLocation.isPathAbsolute()) {
328                     File f = getContentSrcAsFile(contentLocation);
329                     contentLocation = new IRI(DatastreamManagedContent.TEMP_SCHEME +
330                                               f.getAbsolutePath());
331                 }
332             }
333 
334             ds.DSLocation = contentLocation.toString();
335             ds.DSLocation =
336                     (DOTranslationUtility.normalizeDSLocationURLs(m_obj
337                             .getPid(), ds, m_transContext)).DSLocation;
338             return ds;
339         }
340 
341         try {
342             File temp = File.createTempFile("binary-datastream", null);
343             OutputStream out = new FileOutputStream(temp);
344             if (MimeTypeHelper.isText(ds.DSMIME)
345                     || MimeTypeHelper.isXml(ds.DSMIME)) {
346                 IOUtils.copy(new StringReader(entry.getContent()),
347                              out,
348                              m_encoding);
349             } else {
350                 IOUtils.copy(entry.getContentStream(), out);
351             }
352             ds.DSLocation = DatastreamManagedContent.TEMP_SCHEME + temp.getAbsolutePath();
353         } catch (IOException e) {
354             throw new StreamIOException(e.getMessage(), e);
355         }
356 
357         return ds;
358     }

360     private void addAuditDatastream(Entry entry)
361             throws ObjectIntegrityException, StreamIOException {
362         try {
363             Reader auditTrail;
364             if (m_format.equals(ATOM_ZIP1_1)) {
365                 File f = getContentSrcAsFile(entry.getContentSrc());
366                 auditTrail = new InputStreamReader(new FileInputStream(f), m_encoding);
367             } else {
368                 auditTrail = new StringReader(entry.getContent());
369             }
370             m_obj.getAuditRecords().addAll(DOTranslationUtility
371                     .getAuditRecords(auditTrail));
372             auditTrail.close();
373         } catch (XMLStreamException e) {
374             throw new ObjectIntegrityException(e.getMessage(), e);
375         } catch (IOException e) {
376             throw new StreamIOException(e.getMessage(), e);
377         }
378     }

390      * Parses the id to determine a datastreamId.
391      *
392      * @param id
393      * @return
394      */
395     private String getDatastreamId(Entry entry) {
396         String entryId = entry.getId().toString();
397         // matches info:fedora/pid/dsid/timestamp
398         Pattern pattern =
399                 Pattern.compile("^" + Constants.FEDORA.uri + ".+?/([^/]+)/?.*");
400         Matcher matcher = pattern.matcher(entryId);
401 
402         if (matcher.find()) {
403             return matcher.group(1);
404         } else {
405             return m_obj.newDatastreamID();
406         }
407     }

409     private String getDatastreamVersionId(Entry entry) {
410         String dsId = getDatastreamId(entry);
411         String dsvId = entry.getTitle();
412         // e.g. Match DS1.0 but not DS1
413         if (dsvId.matches("^" + dsId + ".*\\.[\\w]")) {
414             return dsvId;
415         } else {
416             if (!m_obj.datastreams(dsId).iterator().hasNext()) {
417                 return dsId + ".0";
418             } else {
419                 return m_obj.newDatastreamID(dsId);
420             }
421         }

424     private String getDSControlGroup(Entry entry)
425             throws ObjectIntegrityException {
426         List<Category> controlGroups =
427                 entry.getCategories(MODEL.CONTROL_GROUP.uri);
428 
429         // Try to infer the control group if not provided
430         if (controlGroups.isEmpty() || controlGroups.size() > 1) {
431             if (entry.getContentType() != null) {
432                 if (entry.getContentType().equals(Content.Type.XML)) {
433                     return "X";
434                 } else {
435                     // only XML can be inline
436                     return "M";
437                 }
438             }
439 
440             if (entry.getContentSrc() != null) {
441                 return "M";
442             }
443             // TODO other cases
444             // link alts, link enclosures
445 
446             else {
447                 throw new ObjectIntegrityException("No control group provided by "
448                         + m_obj.getPid());
449             }
450         } else {
451             return controlGroups.get(0).getTerm();
452         }
453     }

509     private String getDSMimeType(Entry entry) {
510         String dsMimeType = "application/unknown";
511         MimeType mimeType = entry.getContentMimeType();
512         if (mimeType == null) {
513             Content.Type type = entry.getContentType();
514             if (type != null) {
515                 if (type == Content.Type.HTML) {
516                     dsMimeType = "text/html";
517                 } else if (type == Content.Type.TEXT) {
518                     dsMimeType = "text/plain";
519                 } else if (type == Content.Type.XHTML) {
520                     dsMimeType = "application/xhtml+xml";
521                 } else if (type == Content.Type.XML) {
522                     dsMimeType = "text/xml";
523                 }
524             }
525         } else {
526             dsMimeType = mimeType.toString();
527         }
528         return dsMimeType;
529     }

560     private void setDatastreamVersionProperties(Datastream ds, Entry entry)
561             throws ValidationException {
562         ds.DatastreamAltIDs = getDSAltIds(entry);
563         ds.DSCreateDT = entry.getUpdated();
564         ds.DSFormatURI = getDSFormatURI(entry);
565         ds.DSLabel = getDSLabel(entry);
566         ds.DSVersionID = getDatastreamVersionId(entry);
567         ds.DSChecksumType = getDSChecksumType(entry);
568         String checksum = getDSChecksum(entry);
569         if (m_obj.isNew()) {
570             if (logger.isDebugEnabled()) {
571                 logger.debug("New Object: checking supplied checksum");
572             }
573             if (checksum != null && !checksum.equals("")
574                     && !checksum.equals(Datastream.CHECKSUM_NONE)) {
575                 String tmpChecksum = ds.getChecksum();
576                 if (logger.isDebugEnabled()) {
577                     logger.debug("checksum = " + tmpChecksum);
578                 }
579                 if (!checksum.equals(tmpChecksum)) {
580                     throw new ValidationException("Checksum Mismatch: "
581                             + tmpChecksum);
582                 }
583             }
584             ds.DSChecksumType = ds.getChecksumType();
585         } else {
586             ds.DSChecksum = checksum;
587         }
588     }

598     /**
599      * Returns the an Entry's contentSrc as a File relative to {@link #m_tempDir}.
600      *
601      * @param contentSrc
602      * @return the contentSrc as a File relative to m_tempDir.
603      * @throws ObjectIntegrityException
604      */
605     protected File getContentSrcAsFile(IRI contentSrc) throws ObjectIntegrityException {
606         if (contentSrc.isAbsolute() || contentSrc.isPathAbsolute()) {
607             throw new ObjectIntegrityException("contentSrc must not be absolute");
608         }
609         try {
610             // Normalize the IRI to resolve percent-encoding and
611             // backtracking (e.g. "../")
612             NormalizedURI nUri = new NormalizedURI(m_tempDir.toURI().toString() + contentSrc.toString());
613             nUri.normalize();
614 
615             File f = new File(nUri.toURI());
616             if (f.getParentFile().equals(m_tempDir)) {
617                 return f;
618             } else {
619                 throw new ObjectIntegrityException(contentSrc.toString()
620                                                    + " is not a valid path.");
621             }
622         } catch (URISyntaxException e) {
623             throw new ObjectIntegrityException(e.getMessage(), e);
624         }
625     }

627     private static class UpdatedIdComparator
628             implements Comparator<Entry> {
629 
630         private boolean ascending = true;
631 
632         UpdatedIdComparator(boolean ascending) {
633             this.ascending = ascending;
634         }
635 
636         public int compare(Entry o1, Entry o2) {
637             Date d1 = o1.getUpdated();
638             Date d2 = o2.getUpdated();
639             String id1 = o1.getId().toString();
640             String id2 = o2.getId().toString();
641 
642             int r = d1.compareTo(d2);
643             if (d1.equals(d2)) {
644                 r = id1.compareTo(id2);
645             }
646             return (ascending) ? r : -r;
647         }
