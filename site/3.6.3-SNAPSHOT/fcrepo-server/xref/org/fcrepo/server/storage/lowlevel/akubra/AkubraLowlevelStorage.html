62      /**
63       * Creates an instance using the given blob stores.
64       * <p>
65       * The blob stores <b>MUST</b>:
66       * <ul>
67       *   <li> be <i>non-transactional</i></li>
68       *   <li> be able to accept <code>info:fedora/</code> URIs as blob ids.
69       * </ul>
70       * <p>
71       * The blob stores <b>MAY</b>:
72       * <ul>
73       *   <li> support atomic overwrites natively. If not,
74       *        <code>forceSafe..Overwrites</code> MUST be given as
75       *        <code>true</code> and the blob store MUST support
76       *        {@link org.akubraproject.core.Blob#renameTo}
77       *   </li>
78       * </ul>
79       *
80       * @param objectStore the store for serialized objects.
81       * @param datastreamStore the store for datastream content.
82       * @param forceSafeObjectOverwrites if true, replaceObject calls will
83       *        be done in a way that ensures the old content is not deleted
84       *        until the new content is safely written. If the objectStore
85       *        already does this, this should be given as false.
86       * @param forceSafeDatastreamOverwrites same as above, but for
87       *        replaceDatastream calls.
88       */

90                                   BlobStore datastreamStore,
91                                   boolean forceSafeObjectOverwrites,
92                                   boolean forceSafeDatastreamOverwrites) {
93          this.objectStore = objectStore;
94          this.datastreamStore = datastreamStore;
95          this.forceSafeObjectOverwrites = forceSafeObjectOverwrites;
96          this.forceSafeDatastreamOverwrites = forceSafeDatastreamOverwrites;

209     private static long add(BlobStore store,
210                             String key,
211                             InputStream content, Map<String, String> hints)
212             throws ObjectAlreadyInLowlevelStorageException {
213         BlobStoreConnection connection = null;
214         try {
215             URI blobId = getBlobId(key);
216             connection = getConnection(store, hints);
217             Blob blob = getBlob(connection, blobId, hints);
218             OutputStream out = openOutputStream(blob, -1, false);
219             copy(content, out);
220             try {
221                 return blob.getSize();
222             } catch (MissingBlobException e) { // should never happen
223                 throw new RuntimeException("Missing blob after adding the blob: " +e.getMessage(), e);
224             } catch (IOException e) { // should also never happen
225                 throw new RuntimeException("Error reading blob size: " +e.getMessage(), e);
226             }
227         } catch (DuplicateBlobException e) {
228             throw new ObjectAlreadyInLowlevelStorageException(key, e);
229         } finally {
230             closeConnection(connection);
231         }
232     }

245                                String key)
246             throws ObjectNotInLowlevelStorageException {
247         BlobStoreConnection connection = null;
248         try {
249             URI blobId = getBlobId(key);
250             connection = getConnection(store, null);
251             Blob blob = getBlob(connection, blobId, null);
252             if (exists(blob)) {

262     private static long replace(BlobStore store,
263                                 String key,
264                                 InputStream content,
265                                 boolean forceSafeOverwrite,
266                                 Map<String, String> hints)
267             throws LowlevelStorageException {
268         BlobStoreConnection connection = null;
269         try {
270             URI blobId = getBlobId(key);
271             connection = getConnection(store, hints);
272             Blob blob = getBlob(connection, blobId, null);
273             if (exists(blob)) {
274                 if (forceSafeOverwrite) {
275                     safeOverwrite(blob, content);
276                 } else {
277                     // leave it to the store impl to ensure atomicity
278                     OutputStream out = openOutputStream(blob, -1, true);
279                     copy(content, out);
280                 }
281             } else {
282                 throw new ObjectNotInLowlevelStorageException("Object not found in low-level storage: " + key);
283             }
284             try {
285                 return blob.getSize();
286             } catch (MissingBlobException e) { // should never happen
287                 throw new RuntimeException("Missing blob after replcaing the blob: " +e.getMessage(), e);
288             } catch (IOException e) { // should also never happen
289                 throw new RuntimeException("Error reading blob size: " +e.getMessage(), e);
290             }
291         } catch (DuplicateBlobException wontHappen) {
292             throw new FaultException(wontHappen);
293         } finally {
294             closeConnection(connection);
295         }
296     }

313     /**
314      * Overwrites the content of the given blob in a way that guarantees the
315      * original content is not destroyed until the replacement is successfully
316      * put in its place.
317      */
318     private static void safeOverwrite(Blob origBlob, InputStream content) {
319         BlobStoreConnection connection = origBlob.getConnection();
320         String origId = origBlob.getId().toString();
321 
322         // write new content to origId/new
323         Blob newBlob = null;
324         try {
325             newBlob = connection.getBlob(new URI(origId + "/new"), null);
326             copy(content, newBlob.openOutputStream(-1, false));
327         } catch (Throwable th) {
328             // any error or exception here is an unrecoverable fault
329             throw new FaultException(th);
330         }
331 
332         // At this point, we have origId (with old content) and origId/new
333 
334         // rename origId to origId/old
335         Blob oldBlob = null;
336         try {
337             oldBlob = rename(origBlob, origId + "/old");
338         } finally {
339             if (oldBlob == null) {
340                 // rename failed; attempt recovery before throwing the fault
341                 try {
342                     delete(newBlob);
343                 } catch (Throwable th) {
344                     logger.error("Failed to delete " + newBlob.getId() + " while"
345                               + " recovering from rename failure during safe"
346                               + " overwrite", th);
347                 }
348             }
349         }
350 
351         // At this point, we have origId/old and origId/new
352 
353         // rename origId/new to origId
354         boolean successful = false;
355         try {
356             rename(newBlob, origId);
357             successful = true;
358         } finally {
359             if (!successful) {
360                 // rename failed; attempt recovery before throwing the fault
361                 try {
362                     rename(oldBlob, origId);
363                 } catch (Throwable th) {
364                     logger.error("Failed to rename " + oldBlob.getId() + " to "
365                               + origId + " while recovering from rename"
366                               + " failure during safe overwrite", th);
367                 }
368                 try {
369                     newBlob.delete();
370                 } catch (Throwable th) {
371                     logger.error("Failed to delete " + newBlob.getId()
372                               + " while recovering from rename"
373                               + " failure during safe overwrite", th);
374                 }
375             }
376         }
377 
378         // At this point, we have origId (with new content) and origId/old
379 
380         // remove origId/old; we don't need it anymore
381         try {
382             delete(oldBlob);
383         } catch (Throwable th) {
384             logger.error("Failed to delete " + oldBlob.getId()
385                     + " while cleaning up after committed"
386                     + " safe overwrite", th);
387         }
388     }

400     private static InputStream retrieve(BlobStore store,
401                                         String key)
402             throws ObjectNotInLowlevelStorageException {
403         BlobStoreConnection connection = null;
404         InputStream content = null;
405         boolean successful = false;
406         try {
407             URI blobId = getBlobId(key);
408             connection = getConnection(store, null);
409             Blob blob = getBlob(connection, blobId, null);
410             content = openInputStream(blob);
411             successful = true;
412             return new ConnectionClosingInputStream(connection, content);
413         } catch (MissingBlobException e) {
414             throw new ObjectNotInLowlevelStorageException("Object not found in low-level storage: " + key);
415         } finally {
416             if (!successful) {
417                 IOUtils.closeQuietly(content);
418                 closeConnection(connection);
419             }
420         }
421     }

546     /**
547      * Converts a token to a token-as-blobId.
548      * <p>
549      * Object tokens are simply prepended with <code>info:fedora/</code>,
550      * whereas datastream tokens are additionally converted such that
551      * <code>ns:id+dsId+dsVersionId</code> becomes
552      * <code>info:fedora/ns:id/dsId/dsVersionId</code>, with the dsId
553      * and dsVersionId segments URI-percent-encoded with UTF-8 character
554      * encoding.
555      *
556      * @param token the token to convert.
557      * @return the blob id.
558      * @throws IllegalArgumentException if the token is not a well-formed
559      *         pid or datastream token.
560      */
561     private static URI getBlobId(String token) {
562         try {
563             int i = token.indexOf('+');
564             if (i == -1) {
565                 return new URI(new PID(token).toURI());
566             } else {
567                 String[] dsParts = token.substring(i + 1).split("\\+");
568                 if (dsParts.length != 2) {
569                     throw new IllegalArgumentException(
570                             "Malformed datastream token: " + token);
571                 }
572                 return new URI(Constants.FEDORA.uri
573                              + token.substring(0, i) + "/"
574                              + uriEncode(dsParts[0]) + "/"
575                              + uriEncode(dsParts[1]));
576             }
577         } catch (MalformedPIDException e) {
578             throw new IllegalArgumentException(
579                     "Malformed object token: " + token, e);
580         } catch (URISyntaxException e) {
581             throw new IllegalArgumentException(
582                     "Malformed object or datastream token: " + token, e);
583         }
584     }

586     /**
587      * Converts a token-as-blobId back to a token.
588      *
589      * @param blobId the blobId to convert.
590      * @return the resulting object or datastream token.
591      */
592     private static String getToken(URI blobId) {
593         String[] parts = blobId.getSchemeSpecificPart().split("/");
594         if (parts.length == 2) {
595             return parts[1];
596         } else if (parts.length == 4) {
597             return parts[1] + "+"  + uriDecode(parts[2]) + "+"
598                     + uriDecode(parts[3]);
599         } else {
600             throw new IllegalArgumentException("Malformed token-as-blobId: "
601                     + blobId);
602         }
603     }

622      * Closes the stream and connection automatically when closed or finalized.

624     static class ConnectionClosingInputStream extends FilterInputStream {
625 
626         private final BlobStoreConnection connection;
627 
628         public ConnectionClosingInputStream(BlobStoreConnection connection,
629                                             InputStream wrapped) {
630             super(wrapped);
631             this.connection = connection;
632         }
633 
634         @Override
635         public void close() {
636             if (!connection.isClosed()) {
637                 try {
638                     super.close();
639                 } catch (IOException e) {
640                     throw new FaultException("System error closing stream", e);
641                 } finally {
642                     connection.close();
643                 }
644             }
645         }
646 
647         @Override
648         protected void finalize() {
649             close();
650         }

654     /**
655      * Converts a blob id iterator to a key iterator and closes the
656      * connection automatically when exhausted or finalized.
657      */
658     static class ConnectionClosingKeyIterator implements Iterator<String> {
659 
660         private final BlobStoreConnection connection;
661         private final Iterator<URI> blobIds;
662 
663         public ConnectionClosingKeyIterator(BlobStoreConnection connection,
664                                             Iterator<URI> blobIds) {
665             this.connection = connection;
666             this.blobIds = blobIds;
667         }
668 
669         @Override
670         public boolean hasNext() {
671             if (!blobIds.hasNext()) {
672                 connection.close();
673                 return false;
674             }
675             return true;
676         }
677 
678         @Override
679         public String next() {
680             return getToken(blobIds.next());
681         }
682 
683         @Override
684         public void remove() {
685             throw new UnsupportedOperationException();
686         }
687 
688         @Override
689         protected void finalize() {
690             connection.close();
691         }
