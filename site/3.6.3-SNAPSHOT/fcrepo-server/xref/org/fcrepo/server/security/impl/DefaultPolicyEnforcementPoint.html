39              LoggerFactory.getLogger(DefaultPolicyEnforcementPoint.class);

41      private static final String ROLE = org.fcrepo.server.security.PolicyEnforcementPoint.class.getName();

43      private static final String ENFORCE_MODE_CONFIG_KEY = "ENFORCE-MODE";
44  
45      static final String ENFORCE_MODE_ENFORCE_POLICIES = "enforce-policies";
46  
47      static final String ENFORCE_MODE_PERMIT_ALL_REQUESTS =
48              "permit-all-requests";

50      static final String ENFORCE_MODE_DENY_ALL_REQUESTS = "deny-all-requests";

52      private final ContextRegistry m_registry;
53  
54      private String m_enforceMode = ENFORCE_MODE_ENFORCE_POLICIES;

56      public DefaultPolicyEnforcementPoint(PDPConfig pdpConfig, ContextRegistry registry, ModuleConfiguration authzConfiguration)
57              throws ModuleInitializationException {
58  
59          super(pdpConfig);
60  
61          m_registry = registry;
62          Map<String,String> moduleParameters = authzConfiguration.getParameters();
63          if (moduleParameters.containsKey(ENFORCE_MODE_CONFIG_KEY)) {
64              m_enforceMode = moduleParameters.get(ENFORCE_MODE_CONFIG_KEY);
65              if (ENFORCE_MODE_ENFORCE_POLICIES.equals(m_enforceMode)) {
66              } else if (ENFORCE_MODE_PERMIT_ALL_REQUESTS.equals(m_enforceMode)) {
67              } else if (ENFORCE_MODE_DENY_ALL_REQUESTS.equals(m_enforceMode)) {
68              } else {
69                  throw new ModuleInitializationException("invalid enforceMode from config \"" + m_enforceMode + "\"", ROLE);
70              }
71          }
72      }

75      private int n = 0;
76  
77      private synchronized int next() {
78          return n++;
79      }

81      /* (non-Javadoc)
82       * @see org.fcrepo.server.security.PolicyEnforcementPoint#enforce(java.lang.String, java.lang.String, java.lang.String, java.lang.String, java.lang.String, org.fcrepo.server.Context)
83       */
84      @Override
85      public final void enforce(String subjectId,
86                                String action,
87                                String api,
88                                String pid,
89                                String namespace,
90                                Context context) throws AuthzException {
91  
92          long enforceStartTime = System.currentTimeMillis();
93          try {
94              synchronized (this) {
95                  //wait, if pdp update is in progress
96              }
97              if (ENFORCE_MODE_PERMIT_ALL_REQUESTS.equals(m_enforceMode)) {
98                  logger.debug("permitting request because enforceMode==ENFORCE_MODE_PERMIT_ALL_REQUESTS");
99              } else if (ENFORCE_MODE_DENY_ALL_REQUESTS.equals(m_enforceMode)) {
100                 logger.debug("denying request because enforceMode==ENFORCE_MODE_DENY_ALL_REQUESTS");
101                 throw new AuthzDeniedException("all requests are currently denied");
102             } else if (!ENFORCE_MODE_ENFORCE_POLICIES.equals(m_enforceMode)) {
103                 logger.debug("denying request because enforceMode is invalid");
104                 throw new AuthzOperationalException("invalid enforceMode from config \"" + m_enforceMode + "\"");
105             } else {
106                 ResponseCtx response = null;
107                 String contextIndex = null;
108                 try {
109                     contextIndex = (new Integer(next())).toString();
110                     logger.debug("context index set={}", contextIndex);
111                     Set<Subject> subjects = wrapSubjects(subjectId);
112                     Set<Attribute> actions = wrapActions(action, api, contextIndex);
113                     Set<Attribute> resources = wrapResources(pid, namespace);
114 
115                     RequestCtx request =
116                             new RequestCtx(subjects,
117                                            resources,
118                                            actions,
119                                            Collections.EMPTY_SET);
120                     Iterator<Attribute> tempit = actions.iterator();
121                     while (tempit.hasNext()) {
122                         Attribute tempobj = tempit.next();
123                         logger.debug("request action has {}={}", tempobj.getId(), tempobj.getValue().toString());
124                     }
125                     m_registry.registerContext(contextIndex, context);
126                     long st = System.currentTimeMillis();
127                     try {
128                         response = m_pdp.evaluate(request);
129                     } finally {
130                         long dur = System.currentTimeMillis() - st;
131                         logger.debug("Policy evaluation took {}ms.", dur);
132                     }
133 
134                     logger.debug("in pep, after evaluate() called");
135                 } catch (Throwable t) {
136                     logger.error("Error evaluating policy", t);
137                     throw new AuthzOperationalException("");
138                 } finally {
139                     m_registry.unregisterContext(contextIndex);
140                 }
141                 logger.debug("in pep, before denyBiasedAuthz() called");
142                 if (!denyBiasedAuthz(response.getResults())) {
143                     throw new AuthzDeniedException("");
144                 }
145             }
146             if (context.getNoOp()) {
147                 throw new AuthzPermittedException("noOp");
148             }
149         } finally {
150             long dur = System.currentTimeMillis() - enforceStartTime;
151             logger.debug("Policy enforcement took {}ms.", dur);
152         }
153     }
