5   package org.fcrepo.server.errors;

7   import java.text.MessageFormat;
8   
9   import java.util.Locale;
10  import java.util.ResourceBundle;
11  
12  import org.fcrepo.server.Server;

15  /**
16   * The superclass for all Fedora server exceptions.
17   * 
18   * <p>This class encourages the use of resource bundles as message (and detail
19   * text) sources so that localization may easily be implemented.
20   *
21   * <p>Methods that return text will attempt to find the message by code in the
22   * ResourceBundle, but will fall back to simply returning the code if no such
23   * message can be found in the ResourceBundle, or no such ResourceBundle exists
24   * or the bundleName given in the constructor is null.
25   *
26   * <p>This enables developers to temporarily construct exceptions with something
27   * like:
28   * <pre>
29   *     throw new MyException(null, "myMessageId", null, null, null);
30   * </pre>
31   * 
32   * <p>Exceptions of this type have the benefit that they can be easily 
33   * converted to informative, localized SOAP Fault envelopes.
34   * 
35   * @author Chris Wilper
36   */
37  public abstract class ServerException
38          extends Exception {
39  
40      /** The bundle in which the message, identified by m_code, resides. */
41      private final String m_bundleName;
42  
43      /** The identifier for the message in the bundle. */
44      private final String m_code;
45  
46      /** The message in the default locale, if it's already been determined. */
47      private String m_defaultMessage;
48  
49      /** Replacements for placeholders in the message, starting at {0}. */
50      private final String[] m_values;
51  
52      /** Identifiers for messages that provide detail on the error */
53      private final String[] m_details;
54  
55      /** Whether the error was internal to the server, returned by wasServer() */
56      private boolean m_wasServer;
57  
58      /** An empty string array */
59      private static String[] s_emptyStringArray = new String[] {};
60  
61      /**
62       * Constructs a new ServerException.
63       * 
64       * @param bundleName
65       *        The bundle in which the message resides.
66       * @param code
67       *        The identifier for the message in the bundle, aka the key.
68       * @param values
69       *        Replacements for placeholders in the message, where placeholders
70       *        are of the form {num} where num starts at 0, indicating the 0th
71       *        (1st) item in this array.
72       * @param details
73       *        Identifiers for messages which provide detail on the error. This
74       *        may empty or null.
75       * @param cause
76       *        The underlying exception if known, null meaning unknown or none.
77       */
78      public ServerException(String bundleName,
79                             String code,
80                             String[] values,
81                             String[] details,
82                             Throwable cause) {
83          super(code, cause);
84          m_bundleName = bundleName;
85          m_code = code;
86          m_values = values;
87          m_details = details;
88          m_wasServer = false;
89      }
90  
91      /**
92       * Gets the identifier for the message.
93       * 
94       * @return The code, which is also the key in the <code>MessageBundle</code>
95       *         for this exception.
96       */
97      public String getCode() {
98          return m_code;
99      }
100 
101     /**
102      * Tells whether the error occurred because of an unexpected error in the
103      * server, likely requiring action on the part of the server administrator.
104      * <p>
105      * </p>
106      * If it's not an error in the server, it means that the client made a
107      * mistake, which is the more likely case.
108      */
109     public boolean wasServer() {
110         return m_wasServer;
111     }
112 
113     /**
114      * Sets the value for the "wasServer" flag for this exception.
115      */
116     public void setWasServer() {
117         m_wasServer = true;
118     }
119 
120     /**
121      * Gets the message, preferring the <code>Server</code> locale.
122      * 
123      * @return The message, with {num}-indexed placeholders populated, if
124      *         needed.
125      */
126     @Override
127     public String getMessage() {
128         if (m_defaultMessage == null) {
129             m_defaultMessage =
130                     getLocalizedOrCode(m_bundleName,
131                                        Server.getLocale(),
132                                        m_code,
133                                        m_values);
134         }
135         return m_defaultMessage;
136     }
137 
138     /**
139      * Gets the message, preferring the provided locale.
140      * 
141      * <p>When a message in the desired locale is not found, the locale 
142      * selection logic described by <a
143      * href="http://java.sun.com/j2se/1.4/docs/api/java/util/ResourceBundle.html">the
144      * java.util.ResourceBundle</a> class javadoc is used.
145      * 
146      * @param locale
147      *        The preferred locale.
148      * @return The message, with {num}-indexed placeholders populated, if
149      *         needed.
150      */
151     public String getMessage(Locale locale) {
152         return getLocalizedOrCode(m_bundleName, locale, m_code, m_values);
153     }
154 
155     /**
156      * Gets any detail messages, preferring the <code>Server</code> locale.
157      * 
158      * @return The detail messages, with {num}-indexed placeholders populated,
159      *         if needed.
160      */
161     public String[] getDetails() {
162         return getDetails(Server.getLocale());
163     }
164 
165     /**
166      * Gets any detail messages, preferring the provided locale.
167      * 
168      * @return The detail messages, with {num}-indexed placeholders populated,
169      *         if needed.
170      */
171     public String[] getDetails(Locale locale) {
172         if (m_details == null || m_details.length == 0) {
173             return s_emptyStringArray;
174         }
175         String[] ret = new String[m_details.length];
176         for (int i = 0; i < m_details.length; i++) {
177             ret[i] = getLocalizedOrCode(m_bundleName, locale, m_code, null);
178         }
179         return ret;
180     }
181 
182     /**
183      * Gets the message from the resource bundle, formatting it if needed, and
184      * on failure, just returns the code.
185      * 
186      * @param bundleName
187      *        The ResourceBundle where the message is defined.
188      * @param locale
189      *        The preferred locale.
190      * @param code
191      *        The message key.
192      * @param values
193      *        The replacement values, assumed empty if null.
194      * @return The detail messages, with {num}-indexed placeholders populated,
195      *         if needed.
196      */
197     private static String getLocalizedOrCode(String bundleName,
198                                              Locale locale,
199                                              String code,
200                                              String[] values) {
201         ResourceBundle bundle = null;
202         if (bundleName != null) {
203             bundle = ResourceBundle.getBundle(bundleName, locale);
204         }
205         if (bundle == null) {
206             return code;
207         }
208         String locMessage = bundle.getString(code);
209         if (locMessage == null) {
210             return code;
211         }
212         if (values == null) {
213             return locMessage;
214         }
215         return MessageFormat.format(locMessage, (Object[]) values);
216     }
