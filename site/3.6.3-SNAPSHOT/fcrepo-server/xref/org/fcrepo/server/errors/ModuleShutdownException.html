5   package org.fcrepo.server.errors;

7   /**
8    * Signals that an error occurred during a module's shutdown.
9    * 
10   * @author Chris Wilper
11   */
12  public class ModuleShutdownException
13          extends ShutdownException {
14  
15      private static final long serialVersionUID = 1L;
16  
17      /** The role of the module in which the error occurred */
18      private final String m_role;
19  
20      /**
21       * Creates a ModuleShutdownException.
22       * 
23       * @param message
24       *        An informative message explaining what happened and (possibly) how
25       *        to fix it.
26       * @param role
27       *        The role of the module.
28       */
29      public ModuleShutdownException(String message, String role) {
30          super(message);
31          m_role = role;
32      }
33  
34      public ModuleShutdownException(String message, String role, Throwable cause) {
35          super(null, message, null, null, cause);
36          m_role = role;
37      }
38  
39      /**
40       * Gets the role of the module in which the error occurred.
41       * 
42       * @return The role.
43       */
44      public String getRole() {
45          return m_role;
46      }
47  }
