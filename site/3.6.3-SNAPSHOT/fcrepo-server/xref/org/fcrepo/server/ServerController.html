59      @Override
60      public void doGet(HttpServletRequest request, HttpServletResponse response)
61              throws ServletException, IOException {
62          String actionLabel = "server control";
63          String action = request.getParameter("action");
64  
65          if (action == null) {
66              throw new BadRequest400Exception(request,
67                                               actionLabel,
68                                               "no action",
69                                               new String[0]);
70          }
71  
72          if (action.equals("status")) {
73              statusAction(request, response);
74          } else if (action.equals("reloadPolicies")) {
75              reloadPoliciesAction(request, response);
76          } else if (action.equals("modifyDatastreamControlGroup")) {
77              modifyDatastreamControlGroupAction(request, response);
78          } else {
79              throw new BadRequest400Exception(request, actionLabel, "bad action:  "
80                                               + action, new String[0]);
81          }
82      }

84      private void statusAction(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
85          String actionLabel = "getting server status";
86              Context context =
87                      ReadOnlyContext.getContext(Constants.HTTP_REQUEST.REST.uri,
88                                                 request);
89              if (m_server == null) {
90                  throw new InternalError500Exception(request,
91                                                      actionLabel,
92                                                      "server not available",
93                                                      new String[0]);
94              }
95              try {
96                  m_server.status(context);
97              } catch (AuthzOperationalException aoe) {
98                  throw new Forbidden403Exception(request,
99                                                  actionLabel,
100                                                 "authorization failed",
101                                                 new String[0]);
102             } catch (AuthzDeniedException ade) {
103                 throw new Forbidden403Exception(request,
104                                                 actionLabel,
105                                                 "authorization denied",
106                                                 new String[0]);
107             } catch (AuthzPermittedException ape) {
108                 throw new Continue100Exception(request,
109                                                actionLabel,
110                                                "authorization permitted",
111                                                new String[0]);
112             } catch (Throwable t) {
113                 throw new InternalError500Exception(request,
114                                                     actionLabel,
115                                                     "error performing action2",
116                                                     new String[0]);
117             }
118             throw new Ok200Exception(request,
119                                      actionLabel,
120                                      "server running",
121                                      new String[0]);
122 
123         }

176     private boolean getParameterAsBoolean(HttpServletRequest request, String name, boolean defaultValue) {
177 
178         String parameter = request.getParameter(name);
179         boolean res;
180 
181         if (parameter == null || parameter.equals("")) {
182             res = defaultValue;
183         } else {
184             if (parameter.toLowerCase().equals("true") || parameter.toLowerCase().equals("yes")) {
185                 res = true;
186             } else if (parameter.toLowerCase().equals("false") || parameter.toLowerCase().equals("no")) {
187                 res = false;
188             } else {
189                 throw new IllegalArgumentException("Invalid value " + parameter + " supplied for " + name + ".  Please use true or false");
190     }
191         }
192 
193 
194 
195         return res;
196     }

199     // FIXME: see FCREPO-765 - this should be migrated to an admin API (possibly with other methods from this class)
200     private void modifyDatastreamControlGroupAction(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
201         String actionLabel = "modifying datastream control group";
202 
203         Context context =
204                 ReadOnlyContext.getContext(Constants.HTTP_REQUEST.REST.uri,
205                                            request);
206         if (m_server == null) {
207             throw new InternalError500Exception(request,
208                                                 actionLabel,
209                                                 "server not available",
210                                                 new String[0]);
211         }
212         // FIXME: see FCREPO-765 Admin methods are currently in DefaultManagement and carried through to ManagementModule
213         ManagementModule apimDefault = (ManagementModule) m_server.getModule("org.fcrepo.server.management.Management");
214 
215         // FIXME: see FCREPO-765. tidy up output writing
216 
217         // get parameters
218         String pid = request.getParameter("pid");
219         String dsID = request.getParameter("dsID");
220         String controlGroup = request.getParameter("controlGroup");
221         boolean addXMLHeader = getParameterAsBoolean(request, "addXMLHeader", false);
222         boolean reformat = getParameterAsBoolean(request, "reformat", false);
223         boolean setMIMETypeCharset = getParameterAsBoolean(request, "setMIMETypeCharset", false);
224 
225         // get datastream list (single ds id is a list of one)
226         String[] datastreams = dsID.split(",");
227 
228         // get iterable for pid looping
229         boolean singlePID;
230         Iterable<String> pids = null;
231         if (pid.startsWith(PROTOCOL_FILE)) {
232             File pidFile = new File(pid.substring(PROTOCOL_FILE.length()));
233             pids = new PIDStreamIterableWrapper(new FileInputStream(pidFile));
234             singlePID = false;
235 
236         } else { // pid list
237             String[] pidList = pid.split(",");
238             pids = new ArrayList<String>(Arrays.asList(pidList));
239             singlePID = (pidList.length == 1);
240         }
241 
242         try {
243 
244             response.setStatus(HttpServletResponse.SC_OK);
245             response.setCharacterEncoding("UTF-8");
246 
247             PrintWriter pw = response.getWriter();
248 
249             // if doing a single pid/datastream, simple xml output
250             if (singlePID && datastreams.length == 1) {
251                 response.setContentType("text/xml; charset=UTF-8");
252                 Date[] versions = apimDefault.modifyDatastreamControlGroup(context, pid, dsID, controlGroup, addXMLHeader, reformat, setMIMETypeCharset);
253                 pw.write("<versions>\n");
254                     for (Date version : versions) {
255                         pw.write("<version>" + version.toString() + "</version>\n");
256                     }
257                 pw.write("</versions>\n");
258 
259             } else { // logging style output
260                 response.setContentType("text/plain; charset=UTF-8");
261                 ServerUtilitySerializer ser = new ServerUtilitySerializer(pw);
262                 for (String curpid : pids) {
263                     ser.startObject(curpid);
264                     for (String curdsID : datastreams) {
265                         ser.startDatastream(curdsID);
266                         Date[] versions;
267                         try {
268                             versions = apimDefault.modifyDatastreamControlGroup(context, curpid, curdsID, controlGroup, addXMLHeader, reformat, setMIMETypeCharset);
269                         } catch (DatastreamNotFoundException e) {
270                             versions = null;
271                         }
272                         ser.writeVersions(versions);
273                         ser.endDatastream();
274                     }
275                     ser.endObject();
276                 }
277 
278                 ser.finish();
279             }
280             pw.flush();
281 
282 
283         } catch (ObjectNotFoundException e) {
284             logger.error("Object not found: " + pid + " - " + e.getMessage());
285             throw new NotFound404Exception(request,
286                                            actionLabel,
287                                            e.getMessage(),
288                                            new String[0]);
289         } catch (DatastreamNotFoundException e) {
290             logger.error("Datastream not found: " + pid + "/" + dsID + " - " + e.getMessage());
291             throw new NotFound404Exception(request,
292                                            actionLabel,
293                                            e.getMessage(),
294                                            new String[0]);
295 
296 
297         } catch (GeneralException e) {
298             logger.error(e.getMessage());
299             throw new InternalError500Exception(request,
300                                                 actionLabel,
301                                                 e.getMessage(),
302                                                 new String[0]);
303         } catch (AuthzOperationalException aoe) {
304             throw new Forbidden403Exception(request,
305                                             actionLabel,
306                                             "authorization failed",
307                                             new String[0]);
308         } catch (AuthzDeniedException ade) {
309             throw new Forbidden403Exception(request,
310                                             actionLabel,
311                                             "authorization denied",
312                                             new String[0]);
313         } catch (AuthzPermittedException ape) {
314             throw new Continue100Exception(request,
315                                            actionLabel,
316                                            "authorization permitted",
317                                            new String[0]);
318 
319         } catch (ServerException e) {
320             logger.error(e.getMessage(),e);
321             throw new InternalError500Exception(request,
322                                                 actionLabel,
323                                                 "Unexpected error: " + e.getMessage(),
324                                                 new String[0]);
325         }

335     @Override
336     public void destroy() {
337 
338         if (m_server != null) {
339             try {
340                 m_status.append(ServerState.STOPPING,
341                                "Shutting down Fedora Server and modules");
342                 m_server.shutdown(null);
343                 m_status.append(ServerState.STOPPED, "Shutdown Successful");
344             } catch (Throwable th) {
345                 try {
346                     m_status.appendError(ServerState.STOPPED_WITH_ERR, th);
347                 } catch (Exception e) {
348                 }
349             }
350             m_server = null;
351         }
352 
353         super.destroy();
354     }
