62      @Override
63      protected void registerBeanDefinitions()
64              throws ServerInitializationException{
65          ServerConfiguration sc = getConfig();
66          ModuleConfiguration rim = sc.getModuleConfiguration(ResourceIndex.class.getName());
67          if (rim != null){
68              String ds = rim.getParameter("datastore");
69              if (ds != null){
70                  DatastoreConfiguration dsConfig = sc.getDatastoreConfiguration(ds);
71                  if (dsConfig != null){
72                      try{
73                          String name = TriplestoreConnector.class.getName();
74                          GenericBeanDefinition beanDefinition = Server.getTriplestoreConnectorBeanDefinition(dsConfig);
75                          beanDefinition.setAttribute("name", name);
76                          beanDefinition.setAttribute("id", name);
77                          registerBeanDefinition(name, beanDefinition);
78                      }
79                      catch (IOException e){
80                          throw new ServerInitializationException(e.toString(),e);
81                      }
82                  }
83              }
84          }
85          for (String rebuilder:REBUILDERS){
86              try{
87                  ScannedGenericBeanDefinition beanDefinition = Server.getScannedBeanDefinition(rebuilder);
88                  beanDefinition.setAutowireCandidate(true);
89                  beanDefinition.setAutowireMode(AbstractBeanDefinition.AUTOWIRE_BY_NAME);
90                  beanDefinition.setAttribute("id", rebuilder);
91                  beanDefinition.setAttribute("name", rebuilder);
92                  beanDefinition.setInitMethodName("init");
93                  beanDefinition.setScope(BeanDefinition.SCOPE_SINGLETON);
94                  beanDefinition.setDependencyCheck(AbstractBeanDefinition.DEPENDENCY_CHECK_NONE);
95                  registerBeanDefinition(rebuilder, beanDefinition);
96              }
97              catch (IOException e){
98                  throw new ServerInitializationException(e.toString(),e);
99              }
100         }
101         File fedoraHome = new File(Constants.FEDORA_HOME);
102         File springWeb = new File(fedoraHome, "server/config/spring/web");
103         String policyIndexName = System.getProperty("FEDORA_POLICY_INDEX_CONFIG");
104         File policyIndex;
105         if (policyIndexName == null) {
106             policyIndex = new File(springWeb,"config-policy-index.xml");
107         } else {
108             policyIndex = new File(policyIndexName);
109             if (!policyIndex.getAbsolutePath().equals(policyIndexName)){
110                 policyIndex = new File(springWeb,policyIndexName);
111             }
112         }
113         if (policyIndex.exists()){
114             XmlBeanDefinitionReader beanReader = new XmlBeanDefinitionReader(this);
115             FileSystemResource beanConfig = new FileSystemResource(policyIndex);
116             beanReader.loadBeanDefinitions(beanConfig);
117         }
118     }

122         if ("org.fcrepo.server.security.Authorization".equals(moduleRole)
123                 || "org.fcrepo.server.access.Access".equals(moduleRole)
124                 || "org.fcrepo.server.access.DynamicAccess".equals(moduleRole)
125                 || "org.fcrepo.oai.OAIProvider".equals(moduleRole)
126                 || "org.fcrepo.oai.OAIProvider".equals(moduleRole)
127                 || "org.fcrepo.server.management.Management".equals(moduleRole)
128                 || "org.fcrepo.server.resourceIndex.ResourceIndex".equals(moduleRole)) {
129             return true;
130         }
131         return false;

154     public final static Server getRebuildInstance(File homeDir)
155             throws ServerInitializationException, ModuleInitializationException {
156         // instantiate a new special purpose server for rebuilding
157         // SQL databases given the class provided in the root element
158         // in the config file and return it
159 
160             String className = "org.fcrepo.server.utilities.rebuild.RebuildServer";
161             try {
162                 Server inst = new RebuildServer(homeDir);
163                 s_instances.put(homeDir, inst);
164                 inst.init();
165                 return inst;
166             } catch (IllegalArgumentException iae) {
167                 // improbable
168                 throw new ServerInitializationException(MessageFormat
169                         .format(INIT_SERVER_SEVERE_BADARGS,
170                                 new Object[] {className}));
