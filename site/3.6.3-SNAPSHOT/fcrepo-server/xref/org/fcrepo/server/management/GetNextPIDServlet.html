46  /**
47   * Implements the "getNextPID" functionality of the Fedora Management LITE
48   * (API-M-LITE) interface using a java servlet front end. The syntax defined by
49   * API-M-LITE for getting a list of the next available PIDs has the following
50   * binding:
51   * <ol>
52   * <li>getNextPID URL syntax:
53   * protocol://hostname:port/fedora/management/getNextPID[?numPIDs=NUMPIDS&namespace=NAMESPACE&xml=BOOLEAN]
54   * This syntax requests a list of next available PIDS. The parameter numPIDs
55   * determines the number of requested PIDS to generate. If omitted, numPIDs
56   * defaults to 1. The namespace parameter determines the namespace to be used in
57   * generating the PIDs. If omitted, namespace defaults to the namespace defined
58   * in the fedora.fcfg configuration file for the parameter pidNamespace. The xml
59   * parameter determines the type of output returned. If the parameter is omitted
60   * or has a value of "false", a MIME-typed stream consisting of an html table is
61   * returned providing a browser-savvy means of viewing the object profile. If
62   * the value specified is "true", then a MIME-typed stream consisting of XML is
63   * returned.</li>
64   * <ul>
65   * <li>protocol - either http or https.</li>
66   * <li>hostname - required hostname of the Fedora server.</li>
67   * <li>port - required port number on which the Fedora server is running.</li>
68   * <li>fedora - required name of the Fedora access service.</li>
69   * <li>describe - required verb of the Fedora service.</li>
70   * <li>numPIDs - an optional parameter indicating the number of PIDs to be
71   * generated. If omitted, it defaults to 1.</li>
72   * <li>namespace - an optional parameter indicating the namesapce to be used in
73   * generating the PIDs. If omitted, it defaults to the namespace defined in the
74   * <code>fedora.fcfg</code> configuration file for the parameter pidNamespace.</li>
75   * <li>xml - an optional parameter indicating the requested output format. A
76   * value of "true" indicates a return type of text/xml; the absence of the xml
77   * parameter or a value of "false" indicates format is to be text/html.</li>
78   * </ul>
79   *
80   * @author Ross Wayland
81   */

99      /**
100      * <p>
101      * Process the Fedora API-M-LITE request to generate a list of next
102      * available PIDs. Parse and validate the servlet input parameters and then
103      * execute the specified request.
104      * </p>
105      *
106      * @param request
107      *        The servlet request.
108      * @param response
109      *        servlet The servlet response.
110      * @throws ServletException
111      *         If an error occurs that effects the servlet's basic operation.
112      * @throws IOException
113      *         If an error occurrs with an input or output operation.
114      */
115     @Override
116     public void doGet(HttpServletRequest request, HttpServletResponse response)
117             throws ServletException, IOException {
118         boolean xml = false;
119         int numPIDs = 1;
120         String namespace = null;
121 
122         Context context =
123                 ReadOnlyContext.getContext(HTTP_REQUEST.REST.uri, request);
124 
125         // Get optional supplied parameters.
126         for (Enumeration<?> e = request.getParameterNames(); e.hasMoreElements();) {
127             String name = URLDecoder.decode((String) e.nextElement(), "UTF-8");
128             if (name.equalsIgnoreCase("xml")) {
129                 xml = new Boolean(request.getParameter(name)).booleanValue();
130             }
131             if (name.equalsIgnoreCase("numPIDs")) {
132                 numPIDs =
133                         new Integer(URLDecoder.decode(request
134                                 .getParameter(name), "UTF-8")).intValue();
135             }
136             if (name.equalsIgnoreCase("namespace")) {
137                 namespace =
138                         URLDecoder.decode(request.getParameter(name), "UTF-8");
139             }
140         }
141         try {
142             getNextPID(context, numPIDs, namespace, xml, response);
143         } catch (AuthzException ae) {
144             throw RootException.getServletException(ae,
145                                                     request,
146                                                     ACTION_LABEL,
147                                                     new String[0]);
148         } catch (Throwable th) {
149             final String msg = "Unexpected error getting next PID";
150             logger.error(msg, th);
151             throw new InternalError500Exception(msg,
152                                                 th,
153                                                 request,
154                                                 ACTION_LABEL,
155                                                 "Internal Error",
156                                                 new String[0]);
157         }
158     }

160     /**
161      * <p>
162      * Get the requested list of next Available PIDs by invoking the approriate
163      * method from the Management subsystem.
164      * </p>
165      *
166      * @param context
167      *        The context of this request.
168      * @param numPIDs
169      *        The number of PIDs requested.
170      * @param namespace
171      *        The namespace of the requested PIDs.
172      * @param xml
173      *        Boolean that determines format of response; true indicates
174      *        response format is xml; false indicates response format is html.
175      * @param response
176      *        The servlet response.
177      * @throws ServerException
178      *         If an error occurred while accessing the Fedora Management
179      *         subsystem.
180      */
181     public void getNextPID(Context context,
182                            int numPIDs,
183                            String namespace,
184                            boolean xml,
185                            HttpServletResponse response) throws ServerException {
186 
187         OutputStreamWriter out = null;
188         PipedWriter pw = null;
189         PipedReader pr = null;
190 
191         try {
192             pw = new PipedWriter();
193             pr = new PipedReader(pw);
194             String[] pidList =
195                     m_management.getNextPID(context, numPIDs, namespace);
196             if (pidList.length > 0) {
197                 // Repository info obtained.
198                 // Serialize the RepositoryInfo object into XML
199                 new GetNextPIDSerializerThread(context, pidList, pw).start();
200                 if (xml) {
201                     // Return results as raw XML
202                     response.setContentType(CONTENT_TYPE_XML);
203 
204                     // Insures stream read from PipedReader correctly translates
205                     // utf-8
206                     // encoded characters to OutputStreamWriter.
207                     out =
208                             new OutputStreamWriter(response.getOutputStream(),
209                                                    "UTF-8");
210                     int bufSize = 4096;
211                     char[] buf = new char[bufSize];
212                     int len = 0;
213                     while ((len = pr.read(buf, 0, bufSize)) != -1) {
214                         out.write(buf, 0, len);
215                     }
216                     out.flush();
217                 } else {
218                     // Transform results into an html table
219                     response.setContentType(CONTENT_TYPE_HTML);
220                     out =
221                             new OutputStreamWriter(response.getOutputStream(),
222                                                    "UTF-8");
223                     File xslFile =
224                             new File(m_server.getHomeDir(),
225                                      "management/getNextPIDInfo.xslt");
226                     TransformerFactory factory =
227                             XmlTransformUtility.getTransformerFactory();
228                     Templates template =
229                             factory.newTemplates(new StreamSource(xslFile));
230                     Transformer transformer = template.newTransformer();
231                     transformer.transform(new StreamSource(pr),
232                                           new StreamResult(out));
233                 }
234                 out.flush();
235 
236             } else {
237                 // GetNextPID request returned no PIDs.
238                 String message = "[GetNextPIDServlet] No PIDs returned.";
239                 logger.error(message);
240             }
241         } catch (ServerException e) {
242             throw e;
243         } catch (Throwable th) {
244             throw new GeneralException("Error while getting next PID", th);
245         } finally {
246             try {
247                 if (pr != null) {
248                     pr.close();
249                 }
250                 if (out != null) {
251                     out.close();
252                 }
253             } catch (Throwable th) {
254                 String message =
255                         "[GetNextPIDServlet] An error has occured. "
256                                 + " The error was a \" "
257                                 + th.getClass().getName() + " \". Reason: "
258                                 + th.getMessage();
259                 throw new StreamIOException(message);
260             }
261         }
262     }

264     /**
265      * <p>
266      * A Thread to serialize an array of PIDs into XML.
267      * </p>
268      */
269     public class GetNextPIDSerializerThread
270             extends Thread {
271 
272         private PipedWriter pw = null;
273 
274         private String[] pidList = null;
275 
276         /**
277          * <p>
278          * Constructor for GetNextPIDSerializerThread.
279          * </p>
280          *
281          * @param pidList
282          *        An array of the requested next available PIDs.
283          * @param pw
284          *        A PipedWriter to which the serialization info is written.
285          */
286         public GetNextPIDSerializerThread(Context context,
287                                           String[] pidList,
288                                           PipedWriter pw) {
289             this.pw = pw;
290             this.pidList = pidList;
291             if (HTTP_REQUEST.SECURE.uri.equals(context
292                     .getEnvironmentValue(HTTP_REQUEST.SECURITY.uri))) {
293             } else if (HTTP_REQUEST.INSECURE.uri.equals(context
294                     .getEnvironmentValue(HTTP_REQUEST.SECURITY.uri))) {
295             }
296         }
297 
298         /**
299          * <p>
300          * This method executes the thread.
301          * </p>
302          */
303         @Override
304         public void run() {
305             if (pw != null) {
306                 try {
307                     pw.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
308                     pw.write("<pidList "
309                             + " xmlns=\"" + Constants.PID_LIST1_0.namespace.uri + "\" "
310                             + " xmlns:xsd=\"http://www.w3.org/2001/XMLSchema\""
311                             + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\""
312                             + " xsi:schemaLocation=\"" + Constants.PID_LIST1_0.namespace.uri + " "
313                             + Constants.PID_LIST1_0.xsdLocation + "\">");
314 
315                     // PID array serialization
316                     for (String element : pidList) {
317                         pw.write("  <pid>" + element + "</pid>\n");
318                     }
319                     pw.write("</pidList>\n");
320                     pw.flush();
321                     pw.close();
322                 } catch (IOException ioe) {
323                     logger.error("WriteThread error", ioe);
324                 } finally {
325                     try {
326                         if (pw != null) {
327                             pw.close();
328                         }
329                     } catch (IOException ioe) {
330                         logger.warn("WriteThread error", ioe);
331                     }
332                 }
333             }
334         }
335     }
