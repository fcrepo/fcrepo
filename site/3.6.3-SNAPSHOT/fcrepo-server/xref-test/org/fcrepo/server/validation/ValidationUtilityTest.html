228     private static void validatePolicy(PolicyParser parser, String policy)
229             throws IOException, SAXException, ValidationException {
230         ValidationUtility.setPolicyParser(parser);
231         // need a datastream to validate
232         DatastreamXMLMetadata dsxml = new DatastreamXMLMetadata();
233         dsxml.xmlContent = policy.getBytes();
234 
235         ValidationUtility.validateReservedDatastream(PID.getInstance(TEST_PID),
236                                                      "POLICY",
237                                                      dsxml);
238     }

240     private static void validateRels(String dsId, String rels)
241             throws ValidationException {

243         DatastreamXMLMetadata dsxml = new DatastreamXMLMetadata();
244         dsxml.xmlContent = rels.getBytes();

251     private static void validateReserved(PolicyParser parser, String[] dsData)
252             throws ValidationException {
253         ValidationUtility.setPolicyParser(parser);
254         ValidationUtility.validateReservedDatastreams(getDOReader(dsData));
255     }

257     private static DOReader getDOReader(String[] dsData) {
258         MockRepositoryReader repo = new MockRepositoryReader();
259         try {
260             DigitalObject obj = new BasicDigitalObject();
261             obj.setPid(TEST_PID);
262             for (int i = 0; i < dsData.length; i+=2) {
263                 ObjectBuilder.addXDatastream(obj, dsData[i], dsData[i+1]);
264             }
265             repo.putObject(obj);
266             DOReader reader = repo.getReader(false, ReadOnlyContext.EMPTY, TEST_PID);
267             return reader;
268         } catch (Exception wontHappen) {
269             throw new FaultException(wontHappen);
270         }
271     }
