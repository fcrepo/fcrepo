394     /**
395      * <p>
396      * Test a Journaled method.
397      * </p>
398      * <p>
399      * Call the selected method on the JournalCreator. Then tell the
400      * JournalConsumer to process the journal. Compare the calls received by
401      * both Management delegates to the call we expected them to see.
402      * </p>
403      * <p>
404      * Calling the method directly on the JournalConsumer should produce an
405      * exception.
406      * </p>
407      *
408      * @throws ModuleInitializationException
409      * @throws ModuleShutdownException
410      */
411     private void testJournaledMethod(String methodName, Object... arguments)
412             throws ServerException {
413         buildExpectedCall(methodName, arguments);
414 
415         setupLeader();
416         executeManagmentMethod(creator, methodName, arguments);
417         closeLeader();
418 
419         setupFollower();
420         letFollowerCatchUp();
421 
422         assertExpectedCall("leading", leadingDelegate);
423         assertExpectedCall("following", followingDelegate);
424 
425         try {
426             executeManagmentMethod(consumer, methodName, arguments);
427             fail("expected an InvalidStateException");
428         } catch (InvalidStateException e) {
429             // That's the one we expected.
430         }
431     }

433     /**
434      * <p>
435      * Test a non-Journaled method.
436      * </p>
437      * <p>
438      * Call the selected method on the JournalCreator. Check that the Journal is
439      * empty. Compare the call received by the Management delegates to the call
440      * we expected it to see.
441      * </p>
442      * <p>
443      * Call the selected method on the JournalConsumer. Again, compare the call
444      * to the one that we expected.
445      * </p>
446      */
447     private void testNonJournaledMethod(String methodName, Object... arguments)
448             throws ServerException {
449         buildExpectedCall(methodName, arguments);
450 
451         setupLeader();
452         executeManagmentMethod(creator, methodName, arguments);
453         closeLeader();
454 
455         assertEmptyJournal();
456         assertExpectedCall("leading", leadingDelegate);
457 
458         setupFollower();
459         letFollowerCatchUp();
460         executeManagmentMethod(consumer, methodName, arguments);
461         assertExpectedCall("following", followingDelegate);
462     }

465      * <p>
466      * The expected call includes the expected context, as well as the supplied
467      * arguments.
468      * </p>
469      * <p>
470      * Note that "getTempStream" is the only Management call that doesn't use

474     private void buildExpectedCall(String methodName, Object[] arguments) {
475         loadExpectedContext();

478         }
479         expectedCall = new Call(methodName, arguments);

504     /**
505      * <p>
506      * Note that "getTempStream" is the only Management call that doesn't use
507      * "context" as its first argument.
508      * </p>
509      */
510     private void executeManagmentMethod(Management management,
511                                         String methodName,
512                                         Object[] arguments)
513             throws ServerException {
514         Class<?>[] argTypes = new Class[arguments.length];
515         for (int i = 0; i < arguments.length; i++) {
516             argTypes[i] = getArgType(arguments[i]);
517         }
518 
519         try {
520             Method method =
521                     Management.class.getDeclaredMethod(methodName, argTypes);
522             method.invoke(management, arguments);
523         } catch (SecurityException e) {
524             e.printStackTrace();
525             fail("Failed to invoke the method: " + e);
526         } catch (IllegalArgumentException e) {
527             e.printStackTrace();
528             fail("Failed to invoke the method: " + e);
529         } catch (NoSuchMethodException e) {
530             e.printStackTrace();
531             fail("Failed to invoke the method: " + e);
532         } catch (IllegalAccessException e) {
533             e.printStackTrace();
534             fail("Failed to invoke the method: " + e);
535         } catch (InvocationTargetException e) {
536             Throwable cause = e.getCause();
537             if (cause instanceof ServerException) {
538                 throw (ServerException) cause;
539             } else {
540                 e.printStackTrace();
541                 fail("Failed to invoke the method: " + e);
542             }
543         }
544     }

546     /**
547      * Figure the class of each argument to the call, assuming that we always
548      * use <code>boolean</code> or <code>int</code> rather than
549      * {@link Boolean} or {@link Integer}. Also, use {@link Context} and
550      * {@link InputStream} rather than one of their subclasses.
551      */
552     private Class<?> getArgType(Object argument) {
553         if (argument == null) {
554             fail("Can't run unit test with null arguments.");
555         }
556         Class<?> argType = argument.getClass();
557         if (argType.equals(Integer.class)) {
558             argType = Integer.TYPE;
559         }
560         if (argType.equals(Boolean.class)) {
561             argType = Boolean.TYPE;
562         }
563         if (Context.class.isAssignableFrom(argType)) {
564             argType = Context.class;
565         }
566         if (InputStream.class.isAssignableFrom(argType)) {
567             argType = InputStream.class;
568         }
569         return argType;
570     }

590     private void letFollowerCatchUp() throws ModuleShutdownException {
591         // Start the follower.
592         consumer.setManagementDelegate(followingDelegate);
593         // Wait for it to catch up.
594         waitForConsumerThread();
595         // Shut it down.
596         consumer.shutdown();
597     }

599     /**
600      * Compare the expected call with the call that the delegate actually
601      * recorded.
602      */
603     private void assertExpectedCall(String label,
604                                     MockManagementDelegate delegate) {
605         if (delegate.getCallCount() != 1) {
606             fail("Wrong number of " + label + " calls: expected 1, actual "
607                  + delegate.getCallCount() + ". Calls are as follows:\n"
608                  + delegate.getCalls());
609         }
610         assertEquals(label + " calls", expectedCall, delegate.getCalls().get(0));
611     }
