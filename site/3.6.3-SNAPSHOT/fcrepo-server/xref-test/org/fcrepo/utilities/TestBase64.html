78      private static final boolean sameBytes(byte[] a, byte[] b) {
79          if (a.length != b.length) return false;
80          for (int i = 0; i < a.length; i++) {
81              if (a[i] != b[i]) {
82                  return false;
83              }
84          }
85          return true;
86      }
87      
88      private static final void assertSameBytes(byte [] a, byte [] b) {
89          if (a.length != b.length) fail("Bytes not equal\nE: " + inspectBytes(a) + "\nA: " + inspectBytes(b));
90          for (int i = 0; i < a.length; i++) {
91              if (a[i] != b[i]) {
92                  fail("Bytes not equal at position " + Integer.toString(i) + "\nE: " + inspectBytes(a) + "\nA: " + inspectBytes(b));
93              }
94          }
95      }
96      
97      private static String inspectBytes(byte [] a) {
98          StringBuilder builder = new StringBuilder(a.length * 3);
99          for (byte b: a){
100             builder.append(Integer.toHexString(b));
101             builder.append(' ');
102         }
103         return builder.toString();
104     }
