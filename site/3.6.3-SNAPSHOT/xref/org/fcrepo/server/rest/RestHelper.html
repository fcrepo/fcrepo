25      /**
26       * Return the first argument that is not null. This is used to specify
27       * default parameters
28       *
29       * @param args
30       * @return the first argument that is not null.
31       */
32      static <T> T firstNotNull(T... args) {
33          for (T arg : args) {
34              if (arg != null) {
35                  return arg;
36              }
37          }
38  
39          return null;
40      }

42      /**
43       * Return the submitted content type, either through a "format" parameter,
44       * the HTTP content type header. If the content type is application/xml",
45       * return "text/xml".
46       *
47       * @param request
48       * @return content type
49       */
50      static String getConsumedContentType(HttpServletRequest request) {
51          String mimeType = firstNotNull(request.getParameter("format"), request
52                  .getContentType());
53  
54          return ((mimeType != null) ? mimeType.replace("application/xml",
55                  BaseRestResource.XML) : null);
56      }

75      /**
76       * Return the content type represented by the given string.  If the string does not have a "/", assume
77       * it's "text/{format}".  For example, if format is "xml", the returned media type will be "text/xml".
78       *
79       * @param format
80       * @return the content type represented by the given string
81       */
82      public static MediaType getContentType(String format) {
83          if (format == null) {
84              format = "text/html";
85          } else if (format.indexOf('/') <= 0) {
86              format = "text/" + format;
87          }
88  
89          return new MediaTypeHeaderProvider().fromString(format);
90      }
