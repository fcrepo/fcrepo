72      @GET
73      @Path("/")
74      @Produces( { HTML, XML })
75      public Response searchObjects(
76              @QueryParam(RestParam.TERMS)
77              String terms,
78              @QueryParam(RestParam.QUERY)
79              String query,
80              @QueryParam(RestParam.MAX_RESULTS)
81              @DefaultValue("25")
82              int maxResults,
83              @QueryParam(RestParam.SESSION_TOKEN)
84              String sessionToken,
85              @QueryParam(RestParam.RESULT_FORMAT)
86              @DefaultValue(HTML)
87              String format,
88              @QueryParam(RestParam.FLASH)
89              @DefaultValue("false")
90              boolean flash) {
91  
92          try {
93              Context context = getContext();
94              String[] wantedFields = getWantedFields(m_servletRequest);
95              MediaType mime = RestHelper.getContentType(format);
96  
97              FieldSearchResult result = null;
98  
99              if (wantedFields.length > 0 || sessionToken != null) {
100                 if (sessionToken != null) {
101                     result = m_access.resumeFindObjects(context, sessionToken);
102                 } else {
103                     if ((terms != null) && (terms.length() != 0)) {
104                         result = m_access.findObjects(context, wantedFields, maxResults, new FieldSearchQuery(terms));
105                     } else {
106                         result = m_access.findObjects(context, wantedFields, maxResults, new FieldSearchQuery(Condition.getConditions(query)));
107                     }
108                 }
109             }
110 
111             String output;
112             if (TEXT_HTML.isCompatible(mime)) {
113                 output = getSerializer(context).searchResultToHtml(query, terms, SEARCHABLE_FIELDS, wantedFields, maxResults, result);
114             } else {
115                 output = getSerializer(context).searchResultToXml(result);
116 
117             }
118 
119             return Response.ok(output, mime).build();
120         } catch (Exception ex) {
121             return handleException(ex, flash);
122         }
123     }

125     /**
126      * Implements the "getNextPID" functionality of the Fedora Management LITE
127      * (API-M-LITE) interface using a java servlet front end. The syntax defined
128      * by API-M-LITE for getting a list of the next available PIDs has the
129      * following binding:
130      * <ol>
131      * <li>getNextPID URL syntax:
132      * protocol://hostname:port/fedora/objects/nextPID[?numPIDs=NUMPIDS&namespace=NAMESPACE&format=html,xml]
133      * This syntax requests a list of next available PIDS. The parameter numPIDs
134      * determines the number of requested PIDS to generate. If omitted, numPIDs
135      * defaults to 1. The namespace parameter determines the namespace to be
136      * used in generating the PIDs. If omitted, namespace defaults to the
137      * namespace defined in the fedora.fcfg configuration file for the parameter
138      * pidNamespace. The xml parameter determines the type of output returned.
139      * If the parameter is omitted or has a value of "false", a MIME-typed
140      * stream consisting of an html table is returned providing a browser-savvy
141      * means of viewing the object profile. If the value specified is "true",
142      * then a MIME-typed stream consisting of XML is returned.</li>
143      */
144     @Path("/nextPID")
145     @POST
146     public Response getNextPID(
147             @QueryParam("numPIDs")
148             @DefaultValue("1")
149             int numPIDS,
150             @QueryParam(RestParam.NAMESPACE)
151             String namespace,
152             @QueryParam("format")
153             @DefaultValue(HTML)
154             String format,
155             @QueryParam(RestParam.FLASH)
156             @DefaultValue("false")
157             boolean flash) throws Exception {
158 
159         try {
160             Context context = getContext();
161             String[] pidList = m_management.getNextPID(context, numPIDS, namespace);
162             MediaType mime = RestHelper.getContentType(format);
163 
164             if (pidList.length > 0) {
165                 String output = getSerializer(context).pidsToXml(pidList);
166 
167                 if (TEXT_HTML.isCompatible(mime)) {
168                     CharArrayWriter writer = new CharArrayWriter();
169                     transform(output, "management/getNextPIDInfo.xslt", writer);
170                     output = writer.toString();
171                 }
172 
173                 return Response.ok(output, mime).build();
174             } else {
175                 return Response.noContent().build();
176             }
177         } catch (Exception ex) {
178             return handleException(ex, flash);
179         }
180     }

423     /**
424      * Create/Update a new digital object. If no xml given in the body, will
425      * create an empty object.
426      * <p/>
427      * POST /objects/{pid} ? label logMessage format encoding namespace ownerId state
428      */
429     @POST
430     @Path(VALID_PID_PART)
431     @Consumes({XML, FORM, ZIP})
432     public Response createObject(
433             @javax.ws.rs.core.Context
434             HttpHeaders headers,
435             @PathParam(RestParam.PID)
436             String pid,
437             @QueryParam(RestParam.LABEL)
438             String label,
439             @QueryParam(RestParam.LOG_MESSAGE)
440             String logMessage,
441             @QueryParam(RestParam.FORMAT)
442             @DefaultValue(FOXML1_1)
443             String format,
444             @QueryParam(RestParam.ENCODING)
445             @DefaultValue(DEFAULT_ENC)
446             String encoding,
447             @QueryParam(RestParam.NAMESPACE)
448             String namespace,
449             @QueryParam(RestParam.OWNER_ID)
450             String ownerID,
451             @QueryParam(RestParam.STATE)
452             @DefaultValue("A")
453             String state,
454             @QueryParam(RestParam.IGNORE_MIME)
455             @DefaultValue("false")
456             boolean ignoreMime,
457             @QueryParam(RestParam.FLASH)
458             @DefaultValue("false")
459             boolean flash) {
460         try {
461             Context context = getContext();
462 
463             InputStream is = null;
464 
465             // Determine if content is provided
466             RestUtil restUtil = new RestUtil();
467             RequestContent content =
468                     restUtil.getRequestContent(m_servletRequest, headers);
469             if (content != null && content.getContentStream() != null) {
470                 if (ignoreMime) {
471                     is = content.getContentStream();
472                 } else {
473                     // Make sure content is XML or ZIP
474                     String contentMime = content.getMimeType();
475                     if (contentMime != null) {
476                         MediaType t = MediaType.valueOf(contentMime);
477                         if (TEXT_XML.isCompatible(t) || APP_ZIP.isCompatible(t)) {
478                             is = content.getContentStream();
479                         }
480                     }
481                 }
482             }
483 
484             // If no content is provided, use a FOXML template
485             if (is == null) {
486                 if (pid == null || pid.equals("new")) {
487                     pid = m_management.getNextPID(context, 1, namespace)[0];
488                 }
489 
490                 if (ownerID == null || "".equals(ownerID.trim())) {
491                     ownerID = context.getSubjectValue(Constants.SUBJECT.LOGIN_ID.uri);
492                 }
493                 is = new ByteArrayInputStream(getFOXMLTemplate(pid, label, ownerID, encoding).getBytes());
494             } else {
495 
496                 if (namespace != null && !namespace.equals("")) {
497                     logger.warn("The namespace parameter is only applicable whene object " +
498                                 "content is not provided, thus the namespace provided '" +
499                                 namespace + "' has been ignored.");
500                 }
501             }
502 
503             pid = m_management.ingest(context, is, logMessage, format, encoding, pid);
504 
505             URI createdLocation = m_uriInfo.getRequestUri().resolve(URLEncoder.encode(pid, DEFAULT_ENC));
506             return Response.created(createdLocation).entity(pid).build();
507         } catch (Exception ex) {
508             return handleException(ex, flash);
509         }
510     }

561     private static String getFOXMLTemplate(
562             String pid,
563             String label,
564             String ownerId,
565             String encoding) {
566         StringBuilder xml = new StringBuilder();
567         xml.append("<?xml version=\"1.0\" encoding=\"" + encoding + "\"?>\n");
568         xml.append("<foxml:digitalObject VERSION=\"1.1\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"\n");
569         xml.append("    xmlns:foxml=\"info:fedora/fedora-system:def/foxml#\"\n");
570         xml.append(
571                 "           xsi:schemaLocation=\"" + Constants.FOXML.uri + " " + Constants.FOXML1_1.xsdLocation + "\"");
572         if (pid != null && pid.length() > 0) {
573             xml.append("\n           PID=\"" + StreamUtility.enc(pid) + "\">\n");
574         } else {
575             xml.append(">\n");
576         }
577         xml.append("  <foxml:objectProperties>\n");
578         xml.append("    <foxml:property NAME=\"info:fedora/fedora-system:def/model#state\" VALUE=\"A\"/>\n");
579         xml.append("    <foxml:property NAME=\"info:fedora/fedora-system:def/model#label\" VALUE=\""
580                    + StreamUtility.enc(label) + "\"/>\n");
581         xml.append("    <foxml:property NAME=\"info:fedora/fedora-system:def/model#ownerId\" VALUE=\""
582                    + ownerId + "\"/>\n");
583         xml.append("  </foxml:objectProperties>\n");
584         xml.append("</foxml:digitalObject>");
585 
586         return xml.toString();
587     }
