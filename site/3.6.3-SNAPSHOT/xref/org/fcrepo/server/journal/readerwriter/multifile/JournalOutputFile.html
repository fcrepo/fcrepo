28  /**
29   * Encapsulate the information that goes with the creation of a Journal file.
30   * <p>
31   * <b>CAUTION:</b> This file includes an asynchronous timer thread that can
32   * close the file. The {@link #isOpen()} and {@link #close()} methods are
33   * synchronized against the {@link JournalWriter#SYNCHRONIZER} to guard against
34   * problems. Any other operations on the file or on its
35   * <code>XMLEventWriter</code> should also be synchronized against the
36   * {@link JournalWriter#SYNCHRONIZER}.
37   * 
38   * @author Jim Blake
39   */
40  class JournalOutputFile
41          implements MultiFileJournalConstants {
42  
43      /** A "dummy" file instance that is already closed. */
44      public static final JournalOutputFile DUMMY_FILE = new JournalOutputFile();
45  
46      /** The parent JournalWriter, used for the document header and trailer. */
47      private final MultiFileJournalWriter parent;
48  
49      /** The name of the file after it is closed. */
50      private final File file;
51  
52      /** The name of the file while it is being written. */
53      private final File tempFile;
54  
55      private final FileWriter fileWriter;
56  
57      private final XMLEventWriter xmlWriter;
58  
59      /** If the file is larger than this (in bytes), close it. */
60      private final long sizeLimit;
61  
62      /** The timer that monitors the age of this file. */
63      private final Timer timer;
64  
65      /** Is this file still open? */
66      private boolean open = true;
67  
68      /**
69       * This private constructor creates a "dummy" file that is closed to start
70       * with.
71       */
72      private JournalOutputFile() {
73          sizeLimit = 0;
74          file = null;
75          tempFile = null;
76          fileWriter = null;
77          xmlWriter = null;
78          parent = null;
79          timer = null;
80          open = false;
81      }

83      /**
84       * Open the file, initialize all of the fields, write the document header,
85       * and set the timer.
86       */
87      JournalOutputFile(MultiFileJournalWriter parent,
88                        String filenamePrefix,
89                        File journalDirectory,
90                        long sizeLimit,
91                        long ageLimit)
92              throws JournalException {
93          try {
94              this.parent = parent;
95              this.sizeLimit = sizeLimit;
96              file = createFilename(filenamePrefix, journalDirectory);
97              tempFile = createTempFilename(file, journalDirectory);
98              fileWriter = createTempFile(tempFile);
99              xmlWriter = createXmlEventWriter(fileWriter);
100             this.parent.getDocumentHeader(xmlWriter);
101             timer = createTimer(ageLimit);
102         } catch (IOException e) {
103             throw new JournalException(e);
104         } catch (XMLStreamException e) {
105             throw new JournalException(e);
106         }
107     }
108 
109     /**
110      * Create a <code>File</code> object with the name of this Journal file.
111      * Check to be sure that no such file already exists.
112      */
113     private File createFilename(String filenamePrefix, File journalDirectory)
114             throws JournalException {
115         String filename =
116                 JournalHelper.createTimestampedFilename(filenamePrefix,
117                                                         new Date());
118         File theFile = new File(journalDirectory, filename);
119 
120         if (theFile.exists()) {
121             throw new JournalException("File '" + theFile.getPath()
122                     + "' already exists.");
123         }
124 
125         return theFile;
126     }
127 
128     /**
129      * The "temporary" filename is the permanent name preceded by an underscore.
130      */
131     private File createTempFilename(File permanentFile, File journalDirectory) {
132         String tempFilename = "_" + permanentFile.getName();
133         File file2 = new File(journalDirectory, tempFilename);
134         return file2;
135     }
136 
137     /**
138      * Create and open the temporary file.
139      */
140     private FileWriter createTempFile(File tempfile) throws IOException,
141             JournalException {
142         boolean created = tempfile.createNewFile();
143         if (!created) {
144             throw new JournalException("Unable to create file '"
145                     + tempfile.getPath() + "'.");
146         }
147         return new FileWriter(tempfile);
148     }
149 
150     /**
151      * Create an XMLEventWriter for this file. Make it a pretty, indenting
152      * writer.
153      */
154     private XMLEventWriter createXmlEventWriter(FileWriter fileWriter)
155             throws FactoryConfigurationError, XMLStreamException {
156         XMLOutputFactory factory = XMLOutputFactory.newInstance();
157         return new IndentingXMLEventWriter(factory
158                 .createXMLEventWriter(fileWriter));
159     }
160 
161     /**
162      * Create the timer, and schedule a task that will let us know when the file
163      * is too old to continue. If the age limit is 0 or negative, we treat it as
164      * "no limit".
165      */
166     private Timer createTimer(long ageLimit) {
167         Timer fileTimer = new Timer();
168 
169         // if the age limit is 0 or negative, treat it as "no limit".
170         if (ageLimit >= 0) {
171             fileTimer.schedule(new CloseFileTimerTask(), ageLimit);
172         }
173 
174         return fileTimer;
175     }
176 
177     /**
178      * Get an XMLEventWriter that we can write the JournalEvents to. NOTE: any
179      * operations against this writer should be synchronized on the
180      * {@link JournalWriter#SYNCHRONIZER}.
181      */
182     public XMLEventWriter getXmlWriter() {
183         return xmlWriter;
184     }
185 
186     /**
187      * Check the size limit and see whether the file is big enough to close. We
188      * could also check the age limit here, but we trust the timer to handle
189      * that.
190      */
191     public void closeIfAppropriate() throws JournalException {
192         synchronized (JournalWriter.SYNCHRONIZER) {
193             if (!open) {
194                 return;
195             }
196 
197             // if the size limit is 0 or negative, treat it as "no limit".
198             long currentSize = tempFile.length();
199             if (sizeLimit > 0 && currentSize > sizeLimit) {
200                 close();
201             }
202         }
203     }
204 
205     /**
206      * Is this file available for writing?
207      */
208     public boolean isOpen() {
209         synchronized (JournalWriter.SYNCHRONIZER) {
210             return open;
211         }
212     }
213 
214     /**
215      * Write the document trailer, clean up everything and rename the file. Set
216      * the flag saying we are closed.
217      */
218     public void close() throws JournalException {
219         synchronized (JournalWriter.SYNCHRONIZER) {
220             if (!open) {
221                 return;
222             }
223 
224             try {
225                 parent.getDocumentTrailer(xmlWriter);
226                 xmlWriter.close();
227                 fileWriter.close();
228                 timer.cancel();
229 
230                 /*
231                  * java.io.File.renameTo() has a known bug when working across
232                  * file-systems, see:
233                  * http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=4073756 So
234                  * instead of this call: tempFile.renameTo(file); We use the
235                  * following line, and check for exception...
236                  */
237                 try {
238                     FileMovingUtil.move(tempFile, file);
239                 } catch (IOException e) {
240                     throw new JournalException("Failed to rename file from '"
241                             + tempFile.getPath() + "' to '" + file.getPath()
242                             + "'", e);
243                 }
244 
245                 open = false;
246             } catch (XMLStreamException e) {
247                 throw new JournalException(e);
248             } catch (IOException e) {
249                 throw new JournalException(e);
250             }
251         }
252     }
253 
254     /**
255      * When the timer goes off, close the file.
256      */
257     private final class CloseFileTimerTask
258             extends TimerTask {
259 
260         @Override
261         public void run() {
262             try {
263                 close();
264             } catch (JournalException e) {
265                 /*
266                  * What to do with this exception? If we print it, where is the
267                  * console? If we throw it, where will it be recorded?
268                  */
269                 e.printStackTrace();
270                 throw new IllegalStateException(e);
271             }
272         }
273     }
