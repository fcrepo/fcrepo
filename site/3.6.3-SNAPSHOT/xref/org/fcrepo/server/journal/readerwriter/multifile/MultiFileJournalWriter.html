23  /**
24   * An implementation of JournalWriter that writes a series of Journal files to a
25   * specified directory. New files are begun when the current file becomes too
26   * large or too old.
27   * 
28   * @author Jim Blake
29   */
30  public class MultiFileJournalWriter
31          extends JournalWriter
32          implements MultiFileJournalConstants {
33  
34      /** the directory that will hold the journal files. */
35      private final File journalDirectory;
36  
37      /** journal file names will start with this string. */
38      private final String filenamePrefix;
39  
40      /** number of bytes before we start a new file - 0 means no limit */
41      private final long sizeLimit;
42  
43      /** number of milliseconds before we start a new file - 0 means no limit */
44      private final long ageLimit;
45  
46      /** the current journal file - start with a dummy that is already closed. */
47      private JournalOutputFile currentJournal = JournalOutputFile.DUMMY_FILE;
48  
49      private boolean open = true;
50  
51      /**
52       * Parse the parameters to find out how we are operating.
53       */
54      public MultiFileJournalWriter(Map<String, String> parameters,
55                                    String role,
56                                    ServerInterface server)
57              throws JournalException {
58          super(parameters, role, server);
59          journalDirectory =
60                  ParameterHelper
61                          .parseParametersForWritableDirectory(parameters,
62                                                               PARAMETER_JOURNAL_DIRECTORY);
63          filenamePrefix =
64                  ParameterHelper.parseParametersForFilenamePrefix(parameters);
65          sizeLimit = ParameterHelper.parseParametersForSizeLimit(parameters);
66          ageLimit = ParameterHelper.parseParametersForAgeLimit(parameters);
67  
68          checkForPotentialFilenameConflict();
69      }
70  
71      /**
72       * Look at the list of files in the current directory, and make sure that
73       * any new files we create won't conflict with them.
74       */
75      private void checkForPotentialFilenameConflict() throws JournalException {
76          File[] journalFiles =
77                  MultiFileJournalHelper
78                          .getSortedArrayOfJournalFiles(journalDirectory,
79                                                        filenamePrefix);
80          if (journalFiles.length == 0) {
81              return;
82          }
83  
84          String newestFilename = journalFiles[journalFiles.length - 1].getName();
85          String potentialFilename =
86                  JournalHelper.createTimestampedFilename(filenamePrefix,
87                                                          new Date());
88          if (newestFilename.compareTo(potentialFilename) > 0) {
89              throw new JournalException("The name of one or more existing files in the journal "
90                      + "directory (e.g. '"
91                      + newestFilename
92                      + "') may conflict with new Journal "
93                      + "files. Has the system clock changed?");
94          }
95      }
96  
97      /**
98       * Before writing an entry, check to see whether we need to close the
99       * current file and/or open a new one.
100      */
101     @Override
102     public void prepareToWriteJournalEntry() throws JournalException {
103         if (open) {
104             currentJournal.closeIfAppropriate();
105 
106             if (!currentJournal.isOpen()) {
107                 currentJournal =
108                         new JournalOutputFile(this,
109                                               filenamePrefix,
110                                               journalDirectory,
111                                               sizeLimit,
112                                               ageLimit);
113             }
114         }
115     }
116 
117     /**
118      * We've prepared for the entry, so just write it, but remember to
119      * synchronize on the file, so we don't get an asynchronous close while
120      * we're writing. After writing the entry, flush the file.
121      */
122     @Override
123     public void writeJournalEntry(CreatorJournalEntry journalEntry)
124             throws JournalException {
125         if (open) {
126             try {
127                 synchronized (JournalWriter.SYNCHRONIZER) {
128                     XMLEventWriter xmlWriter = currentJournal.getXmlWriter();
129                     super.writeJournalEntry(journalEntry, xmlWriter);
130                     xmlWriter.flush();
131                     currentJournal.closeIfAppropriate();
132                 }
133             } catch (XMLStreamException e) {
134                 throw new JournalException(e);
135             }
136         }
137     }
138 
139     /**
140      * Close the current journal file.
141      */
142     @Override
143     public void shutdown() throws JournalException {
144         if (open) {
145             currentJournal.close();
146             open = false;
147         }
148     }
149 
150     /**
151      * A convenience method so the JournalOutputFile can request its own header.
152      */
153     void getDocumentHeader(XMLEventWriter xmlWriter) throws JournalException {
154         super.writeDocumentHeader(xmlWriter);
155     }
156 
157     /**
158      * A convenience method so the JournalOutputFile can request its own
159      * trailer.
160      */
161     void getDocumentTrailer(XMLEventWriter xmlWriter) throws JournalException {
162         super.writeDocumentTrailer(xmlWriter);
163     }
164 
165     /**
166      * Create an informative message for debugging purposes.
167      */
168     @Override
169     public String toString() {
170         return super.toString() + ", journalDirectory='" + journalDirectory
171                 + "', filenamePrefix='" + filenamePrefix + "', sizeLimit="
172                 + sizeLimit + "(bytes), ageLimit=" + ageLimit + "(msec)";
173     }
