6   package org.fcrepo.server.journal.readerwriter.multicast;
7   
8   import java.io.File;
9   import java.io.FileWriter;
10  import java.io.IOException;
11  import java.io.Writer;
12  
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  /**
17   * <p>
18   * <b>Title:</b> TransportOutputFile.java
19   * </p>
20   * <p>
21   * <b>Description:</b> A "renaming" journal file, for use by Transports.
22   * </p>
23   * <p>
24   * When this file is created, a prefix is put on the filename to show that
25   * output is in progress. When the file is closed, it is renamed to remove the
26   * prefix.
27   * </p>
28   *
29   * @author jblake
30   * @version $Id: TransportOutputFile.java,v 1.3 2007/06/01 17:21:31 jblake Exp $
31   */
32  public class TransportOutputFile {
33  
34      private static final Logger logger =
35              LoggerFactory.getLogger(TransportOutputFile.class);
36  
37      private enum State {
38          READY, OPEN, CLOSED
39      };
40  
41      private final File file;
42  
43      private final File tempFile;
44  
45      private State state = State.READY;
46  
47      private FileWriter fileWriter;
48  
49      /**
50       * Sture the filename, and the "in process" filename.
51       *
52       * @throws IOException
53       *         if either file already exists.
54       */
55      public TransportOutputFile(File directory, String name)
56              throws IOException {
57          logger.debug("creating TransportOutputFile: '" + directory + "', '" + name
58                  + "'");
59  
60          file = new File(directory, name);
61          if (file.exists()) {
62              throw new IOException("File " + file + " already exists.");
63          }
64  
65          tempFile = new File(directory, ("_" + name));
66          if (tempFile.exists()) {
67              throw new IOException("File " + tempFile + " already exists.");
68          }
69      }
70  
71      /**
72       * Create the file with its "in progress" filename.
73       *
74       * @return a Writer on the new file.
75       */
76      public Writer open() throws IOException {
77          switch (state) {
78              case OPEN:
79                  throw new IllegalStateException("File " + tempFile
80                          + " is already open.");
81              case CLOSED:
82                  throw new IllegalStateException("File " + tempFile
83                          + " has been closed already.");
84              default: // READY
85                  state = State.OPEN;
86                  tempFile.createNewFile();
87                  fileWriter = new FileWriter(tempFile);
88                  return fileWriter;
89          }
90      }
91  
92      /**
93       * Close the writer and rename the file.
94       *
95       * @throws IOException
96       */
97      public void close() throws IOException {
98          switch (state) {
99              case READY:
100                 throw new IllegalStateException("File " + tempFile
101                         + " hasn't been opened yet.");
102             case CLOSED:
103                 throw new IllegalStateException("File " + tempFile
104                         + " has been closed already.");
105             default: // OPEN
106                 fileWriter.close();
107                 tempFile.renameTo(file);
108                 state = State.CLOSED;
109         }
110     }
111 
112     /**
113      * Did somebody ask who we are?
114      */
115     public String getName() {
116         return file.getName();
117     }
118 }
