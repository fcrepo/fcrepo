5   package org.fcrepo.server.journal.xmlhelpers;

7   import java.util.Iterator;
8   
9   import javax.xml.stream.XMLEventWriter;
10  import javax.xml.stream.XMLStreamException;
11  
12  import org.fcrepo.server.Context;
13  import org.fcrepo.server.MultiValueMap;
14  import org.fcrepo.server.journal.entry.JournalEntryContext;
15  import org.fcrepo.server.journal.helpers.JournalHelper;
16  import org.fcrepo.server.journal.helpers.PasswordCipher;

18  
19  /**
20   * Write an entire Context object to the Journal file.
21   * 
22   * @author Jim Blake
23   */
24  public class ContextXmlWriter
25          extends AbstractXmlWriter {
26  
27      public void writeContext(JournalEntryContext context, XMLEventWriter writer)
28              throws XMLStreamException {
29          putStartTag(writer, QNAME_TAG_CONTEXT);
30  
31          writeContextPassword(context, writer);
32          writeContextNoOp(context, writer);
33          writeContextNow(context, writer);
34  
35          writeMultiMap(writer, CONTEXT_MAPNAME_ENVIRONMENT, context
36                  .getEnvironmentAttributes());
37          writeMultiMap(writer, CONTEXT_MAPNAME_SUBJECT, context
38                  .getSubjectAttributes());
39          writeMultiMap(writer, CONTEXT_MAPNAME_ACTION, context
40                  .getActionAttributes());
41          writeMultiMap(writer, CONTEXT_MAPNAME_RESOURCE, context
42                  .getResourceAttributes());
43          writeMultiMap(writer, CONTEXT_MAPNAME_RECOVERY, context
44                  .getRecoveryAttributes());
45  
46          putEndTag(writer, QNAME_TAG_CONTEXT);
47      }
48  
49      private void writeContextPassword(Context context, XMLEventWriter writer)
50              throws XMLStreamException {
51          String password = context.getPassword();
52          if (password == null) {
53              password = "";
54          }
55  
56          putStartTag(writer, QNAME_TAG_PASSWORD);
57          putAttribute(writer, QNAME_ATTR_PASSWORD_TYPE, PASSWORD_CIPHER_TYPE);
58          putCharacters(writer, encipherPassword(context, password));
59          putEndTag(writer, QNAME_TAG_PASSWORD);
60      }
61  
62      private String encipherPassword(Context context, String password) {
63          String key = JournalHelper.formatDate(context.now());
64          return PasswordCipher.encipher(key, password);
65      }
66  
67      private void writeContextNoOp(Context context, XMLEventWriter writer)
68              throws XMLStreamException {
69          putStartTag(writer, QNAME_TAG_NOOP);
70          putCharacters(writer, String.valueOf(context.getNoOp()));
71          putEndTag(writer, QNAME_TAG_NOOP);
72      }
73  
74      private void writeContextNow(Context context, XMLEventWriter writer)
75              throws XMLStreamException {
76          putStartTag(writer, QNAME_TAG_NOW);
77          putCharacters(writer, JournalHelper.formatDate(context.now()));
78          putEndTag(writer, QNAME_TAG_NOW);
79      }
80  
81      private void writeMultiMap(XMLEventWriter writer,
82                                 String mapName,
83                                 MultiValueMap map) throws XMLStreamException {
84          putStartTag(writer, QNAME_TAG_MULTI_VALUE_MAP);
85          putAttribute(writer, QNAME_ATTR_NAME, mapName);
86          for (Iterator attributes = map.names(); attributes.hasNext();) {
87              String attribute = (String) attributes.next();
88              putStartTag(writer, QNAME_TAG_MULTI_VALUE_MAP_KEY);
89              putAttribute(writer, QNAME_ATTR_NAME, attribute);
90              String[] values = map.getStringArray(attribute);
91              for (String element : values) {
92                  putStartTag(writer, QNAME_TAG_MULTI_VALUE_MAP_VALUE);
93                  putCharacters(writer, element);
94                  putEndTag(writer, QNAME_TAG_MULTI_VALUE_MAP_VALUE);
95              }
96              putEndTag(writer, QNAME_TAG_MULTI_VALUE_MAP_KEY);
97          }
98          putEndTag(writer, QNAME_TAG_MULTI_VALUE_MAP);
99      }
