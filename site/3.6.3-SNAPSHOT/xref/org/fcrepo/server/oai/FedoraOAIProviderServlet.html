22  /**
23   * FedoraOAIProviderServlet.
24   *
25   * @author Chris Wilper
26   */
27  public class FedoraOAIProviderServlet
28          extends OAIProviderServlet {
29  
30      private static final long serialVersionUID = 1L;
31  
32      Server m_server;
33  
34      OAIResponder m_responder;
35  
36      @Override
37      public OAIResponder getResponder() throws RepositoryException {
38          if (m_responder == null) {
39              try {
40                  OAIProvider provider = m_server
41                          .getBean("org.fcrepo.oai.OAIProvider", OAIProvider.class);
42                  Authorization authz = m_server
43                          .getBean("org.fcrepo.server.security.Authorization", Authorization.class);
44                  m_responder = new OAIResponder(provider, authz);
45              } catch (Exception e) {
46                  throw new RepositoryException(e.getClass().getName() + ": "
47                          + e.getMessage());
48              }
49          }
50          return m_responder;
51      }
52  
53      @Override
54      public void init(ServletConfig config) throws ServletException {
55          super.init(config);
56  
57          try {
58              WebApplicationContext appContext = WebApplicationContextUtils.getRequiredWebApplicationContext(config.getServletContext());
59              m_server = (Server)appContext.getBean("org.fcrepo.server.Server");
60              if (m_server == null) failStartup("Could not retrieve org.fcrepo.server.Server bean",null);
61          } catch (Throwable th) {
62              String msg = "Fedora startup failed";
63              failStartup(msg, th);
64          }
65      }
66  
67      /**
68       * Prints a "FEDORA STARTUP ERROR" to STDERR along with the stacktrace of
69       * the Throwable (if given) and finally, throws a ServletException.
70       */
71      private void failStartup(String message, Throwable th)
72              throws ServletException {
73          System.err.println("\n**************************");
74          System.err.println("** FEDORA STARTUP ERROR **");
75          System.err.println("**************************\n");
76          System.err.println(message);
77          if (th == null) {
78              System.err.println();
79              throw new ServletException(message);
80          } else {
81              th.printStackTrace();
82              System.err.println();
83              throw new ServletException(message, th);
84          }
85      }
