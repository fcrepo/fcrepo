60      @Override
61      public void doGet(HttpServletRequest request, HttpServletResponse response)
62              throws ServletException, IOException {
63          String userId = request.getRemoteUser();
64          if (userId == null) {
65              userId = "anonymous";
66          }
67  
68          Map<String, Set<String>> subjectAttributes =
69                  getSubjectAttributes(request);
70  
71          Document doc = documentBuilder.newDocument();
72          doc.setXmlVersion("1.0");
73  
74          Element root = doc.createElement("user");
75          root.setAttribute("id", userId);
76  
77          doc.appendChild(root);
78  
79          for (String attr : subjectAttributes.keySet()) {
80              Element attribute = doc.createElement("attribute");
81              attribute.setAttribute("name", attr);
82              root.appendChild(attribute);
83  
84              for (String value : subjectAttributes.get(attr)) {
85                  Element v = doc.createElement("value");
86                  v.appendChild(doc.createTextNode(value));
87                  attribute.appendChild(v);
88              }
89          }
90  
91          byte[] output = null;
92          try {
93              output = DataUtils.format(doc).getBytes();
94          } catch (Exception e) {
95              logger.error("Error obtaining user information: " + e.getMessage(), e);
96          }
97  
98          response.setContentType("text/xml");
99          response.setContentLength(output.length);
100         OutputStream out = response.getOutputStream();
101         out.write(output);
102         out.flush();
103         out.close();
104     }
105 
106     @SuppressWarnings("unchecked")
107     private Map<String, Set<String>> getSubjectAttributes(HttpServletRequest request) {
108         Map<String, Set<String>> subjectAttr = null;
109         subjectAttr =
110                 (Map<String, Set<String>>) request
111                         .getAttribute(FEDORA_ATTRS_KEY);
112 
113         if (subjectAttr == null) {
114             subjectAttr = new HashMap<String, Set<String>>();
115         }
116 
117         return subjectAttr;
118     }
