46  /**
47   * Handles the get operations.
48   *
49   * @author nishen@melcoe.mq.edu.au
50   */
51  public class GetFilter
52          extends AbstractFilter {
53  
54      private static final Logger logger =
55              LoggerFactory.getLogger(GetFilter.class);
56  
57      /**
58       * Default constructor.
59       *
60       * @throws PEPException
61       */
62      public GetFilter()
63              throws PEPException {
64          super();
65      }
66  
67      /*
68       * (non-Javadoc)
69       * @see
70       * org.fcrepo.server.security.xacml.pep.rest.filters.RESTFilter#handleRequest(javax.servlet
71       * .http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
72       */
73      @Override
74      @SuppressWarnings("deprecation")
75      public RequestCtx handleRequest(HttpServletRequest request,
76                                      HttpServletResponse response)
77              throws IOException, ServletException {
78          if (request.getPathInfo() == null) {
79              logger.error("Bad request: " + request.getRequestURI());
80              throw new ServletException("Bad request: "
81                      + request.getRequestURI());
82          }
83  
84          String[] parts = request.getPathInfo().split("/");
85          if (parts.length < 2) {
86              logger.warn("Not enough path components on the URI.");
87              return null;
88              // throw new ServletException("Not enough path components on the
89              // URI.");
90          }
91  
92          RequestCtx req = null;
93  
94          String pid = null;
95          String dsID = null;
96          String dissID = null;
97          String methodName = null;
98          String dateTime = null;
99  
100         if (logger.isDebugEnabled()) {
101             for (String p : parts) {
102                 logger.debug("Parts: {}", p);
103             }
104         }
105 
106         Map<URI, AttributeValue> actions = new HashMap<URI, AttributeValue>();
107         Map<URI, AttributeValue> resAttr;
108 
109         String logAction = null;
110 
111         // Starting assumption is that we are doing a GetObjectProfile
112         pid = parts[1];
113         actions.put(Constants.ACTION.ID.getURI(),
114                     Constants.ACTION.GET_OBJECT_PROFILE
115                             .getStringAttribute());
116         logAction =
117                 Constants.ACTION.GET_OBJECT_PROFILE.uri;
118 
119         if (parts.length > 2) {
120             if (isDate(parts[2])) {
121                 dateTime = parts[2];
122             } else if (isDatastream(parts[2])) {
123                 dsID = parts[2];
124                 actions.clear();
125                 actions
126                         .put(Constants.ACTION.ID.getURI(),
127                              Constants.ACTION.GET_DATASTREAM_DISSEMINATION
128                                      .getStringAttribute());
129                 actions.put(Constants.ACTION.ID.getURI(),
130                             Constants.ACTION.GET_DATASTREAM
131                                     .getStringAttribute());
132                 logAction =
133                         Constants.ACTION.GET_DATASTREAM_DISSEMINATION.getURI()
134                                 .toASCIIString();
135             } else {
136                 dissID = parts[2];
137                 actions.clear();
138                 actions
139                         .put(Constants.ACTION.ID.getURI(),
140                              Constants.ACTION.GET_DISSEMINATION
141                                      .getStringAttribute());
142                 logAction =
143                         Constants.ACTION.GET_DISSEMINATION.uri;
144             }
145         }
146 
147         if (parts.length > 3) {
148             if (isDate(parts[3])) {
149                 dateTime = parts[3];
150             } else {
151                 methodName = parts[3];
152             }
153         }
154 
155         if (parts.length > 4) {
156             if (isDate(parts[4])) {
157                 dateTime = parts[4];
158             }
159         }
160 
161         try {
162             resAttr = ResourceAttributes.getResources(pid);
163             
164             if (dsID != null && !"".equals(dsID)) {
165                 resAttr.put(Constants.DATASTREAM.ID.getURI(),
166                             new StringAttribute(dsID));
167             }
168             if (dissID != null && !"".equals(dissID)) {
169                 resAttr.put(Constants.DISSEMINATOR.ID.getURI(),
170                             new StringAttribute(dissID));
171             }
172             if (methodName != null && !"".equals(methodName)) {
173                 resAttr.put(Constants.DISSEMINATOR.METHOD.getURI(),
174                             new StringAttribute(methodName));
175             }
176             if (dateTime != null && !"".equals(dateTime)) {
177                 resAttr.put(Constants.DATASTREAM.AS_OF_DATETIME.getURI(),
178                             DateTimeAttribute.getInstance(dateTime));
179             }
180 
181             actions.put(Constants.ACTION.API.getURI(),
182                         Constants.ACTION.APIA.getStringAttribute());
183 
184             req =
185                     getContextHandler().buildRequest(getSubjects(request),
186                                                      actions,
187                                                      resAttr,
188                                                      getEnvironment(request));
189 
190             LogUtil.statLog(request.getRemoteUser(), logAction, pid, dsID);
191         } catch (Exception e) {
192             logger.error(e.getMessage(), e);
193             throw new ServletException(e);
194         }
195 
196         return req;
197     }
