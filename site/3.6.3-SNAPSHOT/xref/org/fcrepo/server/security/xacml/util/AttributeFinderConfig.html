14  /**
15   * Configuration for an attribute finder.
16   *
17   * Encapsulates a set of designators (corresponding to XACML policy targets), each
18   * of which encapsulates a set of XACML attribute IDs and configurations
19   *
20   * @author Stephen Bayliss
21   * @version $Id$
22   */
23  public class AttributeFinderConfig {
24  
25      private final Map<Integer, Designator> designators;
26  
27      public AttributeFinderConfig() {
28          designators = new HashMap<Integer, Designator>();
29      }
30      /**
31       * Get a Designator based on the Sun XACML designator ID
32       * @param designator
33       * @return
34       */
35      public Designator get(int designator) {
36          return designators.get(new Integer(designator));
37      }
38      /**
39       * Get a Designator based on the XACML target name (in lower case)
40       * @param designatorName
41       * @return
42       */
43      public Designator get(String designatorName) {
44          return designators.get(getTarget(designatorName));
45      }
46      /**
47       * Gets the designator IDs that have been configured
48       * @return
49       */
50      public Set<Integer> getDesignatorIds() {
51          return designators.keySet();
52      }
53  
54      /**
55       * Add or update a Designator.  The added/updated designator will have no attributes.
56       * Returns the added/updated designator
57       *
58       * @param designatorName
59       * @return
60       * @throws AttributeFinderException
61       */
62      public Designator put(String designatorName) throws AttributeFinderException {
63          int target = getTarget(designatorName);
64          if (target != -1) {
65              Designator des = designators.get(target);
66              if (des == null) {
67                  des = new Designator();
68              }
69              designators.put(new Integer(target), des);
70              return des;
71          } else {
72              throw new AttributeFinderException("Invalid attribute designator name " + designatorName);
73          }
74      }
75  
76      /**
77       * Converts a XACML target name (in lower case) to the Sun XACML designator identifier
78       * @param targetName
79       * @return
80       */
81      private static Integer getTarget(String targetName) {
82          if (targetName.equals("resource")) {
83              return AttributeDesignator.RESOURCE_TARGET;
84          } else if (targetName.equals("subject")) {
85              return AttributeDesignator.SUBJECT_TARGET;
86          } else if (targetName.equals("environment")) {
87              return AttributeDesignator.ENVIRONMENT_TARGET;
88          } else if (targetName.equals("action")) {
89              return AttributeDesignator.ACTION_TARGET;
90          } else {
91              return -1;
92          }
93      }
