5   package org.fcrepo.server.security.xacml.pdp.decorator;

7   import java.io.InputStream;
8   
9   import org.fcrepo.server.Context;
10  import org.fcrepo.server.errors.ServerException;
11  import org.fcrepo.server.security.xacml.pdp.data.FedoraPolicyStore;
12  import org.fcrepo.server.storage.DOManager;
13  import org.fcrepo.server.storage.DOReader;
14  import org.fcrepo.server.storage.types.Datastream;
15  /**
16   * Wrapper for a FeSL XACML policy stored as a Fedora digital object.
17   *
18   * Contains method for determining whether or not this is a digital object that
19   * represents an active FeSL XACML policy.
20   *
21   * Only objects that have a policy datastream which is active within an active object
22   * are deemed to be active policies.
23   *
24   * @author Stephen Bayliss
25   * @version $Id$
26   */
27  
28  
29  
30  
31  public class PolicyObject {
32  
33      private Datastream m_datastream = null;
34  
35      private InputStream m_dsContent = null;
36  
37      public static final String POLICY_DATASTREAM = FedoraPolicyStore.FESL_POLICY_DATASTREAM;
38  
39      private Boolean m_hasPolicyDatastream = null;
40  
41      private String m_pid = null;
42  
43      private String m_dsState = null;
44  
45      private String m_objectState = null;
46  
47      private DOManager m_DOManager = null;
48      private DOReader m_reader = null;
49      private Context m_context = null;
50  
51      @SuppressWarnings("unused")
52      private PolicyObject() {}
53  
54      /**
55       * Create a new PolicyObject, initialising the object based on the supplied parameters.
56       * Parameters are the same parameters that are generally supplied to API-M methods.
57       * Parameters therefore represent "already-known" information about the policy object
58       * (after the API-M method succeeded).
59       * @param manager
60       * @param context
61       * @param pid
62       * @param objectState
63       * @param dsID
64       * @param dsState
65       */
66      public PolicyObject(DOManager manager, Context context, String pid, String objectState, String dsID, String dsState) {
67  
68          m_DOManager = manager;
69          m_context = context;
70          m_pid = pid;
71          if (objectState != null)
72              m_objectState = objectState;
73          if (dsID != null) {
74              if (dsID.equals(POLICY_DATASTREAM)) {
75                  m_hasPolicyDatastream = true;
76              }
77          }
78          if (dsState != null)
79              m_dsState = dsState;
80  
81      }
82  
83  
84  
85      /**
86       * determines if this object represents an active FeSL policy
87       *
88       * @return
89       * @throws ServerException
90       */
91      public boolean isPolicyActive() throws ServerException {
92  
93          return (hasPolicyDatastream() && isDatastreamActive() && isObjectActive());
94      }
95  
96      /**
97       * determines if this digital object is active
98       * @return
99       * @throws ServerException
100      */
101     public boolean isObjectActive() throws ServerException {
102         if (m_objectState == null)
103             m_objectState = getReader().GetObjectState();
104 
105         return m_objectState.equals("A");
106 
107     }
108 
109     /**
110      * determines if the policy datastream in this object is active
111      * @return
112      * @throws ServerException
113      */
114     public boolean isDatastreamActive() throws ServerException {
115         if (m_dsState == null && getDatastream() != null)
116             m_dsState = m_datastream.DSState;
117 
118         return m_dsState != null && m_dsState.equals("A");
119     }
120 
121     /**
122      * determines if this object contains a policy datastream
123      * @return
124      * @throws ServerException
125      */
126     public boolean hasPolicyDatastream() throws ServerException {
127 
128         if (m_hasPolicyDatastream == null) {
129             m_hasPolicyDatastream = false;
130             Datastream[] allDs = getReader().GetDatastreams(null, null);
131             for (int i = 0; i < allDs.length; i++) {
132                 if (allDs[i].DatastreamID.equals(POLICY_DATASTREAM)) {
133                     m_hasPolicyDatastream = true;
134                     // also have the datastream now
135                     m_datastream = allDs[i];
136                     break;
137                 }
138             }
139         }
140         return m_hasPolicyDatastream;
141     }
142 
143     private DOReader getReader() throws ServerException {
144         if (m_reader == null) {
145             m_reader = m_DOManager.getReader(false, m_context, m_pid);
146         }
147         return m_reader;
148     }
149 
150     /**
151      * get the policy datastream content
152      * @return
153      * @throws ServerException
154      */
155     public InputStream getDsContent() throws ServerException {
156         if (m_dsContent == null && getDatastream() != null) {
157             m_dsContent = getDatastream().getContentStream();
158 
159             // remember we found the policy datastream
160             m_hasPolicyDatastream = true;
161         }
162         return m_dsContent;
163     }
164 
165     private Datastream getDatastream() throws ServerException {
166         if (m_datastream == null) {
167             m_datastream = getReader().GetDatastream(POLICY_DATASTREAM, null);
168 
169         }
170         // if we got the policy datastream, we know it has one
171         m_hasPolicyDatastream = true;
172         return m_datastream;
173     }
