111     @Override
112     public RequestCtx handleRequest(SOAPMessageContext context)
113             throws OperationHandlerException {
114         if (logger.isDebugEnabled()) {
115             logger.debug("ListDatastreamsHandler/handleRequest!");
116         }
117 
118         RequestCtx req = null;
119         Object oMap = null;
120 
121         String pid = null;
122         String asOfDateTime = null;
123 
124         try {
125             oMap = getSOAPRequestObjects(context);
126             logger.debug("Retrieved SOAP Request Objects");
127         } catch (SoapFault af) {
128             logger.error("Error obtaining SOAP Request Objects", af);
129             throw new OperationHandlerException("Error obtaining SOAP Request Objects",
130                                                 af);
131         }
132 
133         try {
134             pid = (String) callGetter("getPid",oMap);
135             asOfDateTime = (String) callGetter("getAsOfDateTime", oMap);
136         } catch (Exception e) {
137             logger.error("Error obtaining parameters", e);
138             throw new OperationHandlerException("Error obtaining parameters.",
139                                                 e);
140         }
141 
142         logger.debug("Extracted SOAP Request Objects");
143 
144         Map<URI, AttributeValue> actions = new HashMap<URI, AttributeValue>();
145         Map<URI, AttributeValue> resAttr;
146 
147         try {
148             resAttr = ResourceAttributes.getResources(pid);
149             if (asOfDateTime != null && !"".equals(asOfDateTime)) {
150                 resAttr.put(Constants.DATASTREAM.AS_OF_DATETIME.getURI(),
151                             DateTimeAttribute.getInstance(asOfDateTime));
152             }
153 
154             actions.put(Constants.ACTION.ID.getURI(),
155                         Constants.ACTION.LIST_DATASTREAMS
156                                 .getStringAttribute());
157             actions.put(Constants.ACTION.API.getURI(),
158                         Constants.ACTION.APIA
159                                 .getStringAttribute());
160 
161             req =
162                     getContextHandler().buildRequest(getSubjects(context),
163                                                      actions,
164                                                      resAttr,
165                                                      getEnvironment(context));
166 
167             LogUtil.statLog(getUser(context),
168                             Constants.ACTION.LIST_DATASTREAMS.uri,
169                             pid,
170                             null);
171         } catch (Exception e) {
172             logger.error(e.getMessage(), e);
173             throw new OperationHandlerException(e.getMessage(), e);
174         }
175 
176         return req;
177     }

179     public List<DatastreamDef> filter(SOAPMessageContext context,
180                                   List<DatastreamDef> dsDefs,
181                                   String pid) throws OperationHandlerException,
182             PEPException {
183         List<String> requests = new ArrayList<String>();
184         Map<String, DatastreamDef> objects =
185                 new HashMap<String, DatastreamDef>();
186 
187         for (DatastreamDef dsDef : dsDefs) {
188             if (logger.isDebugEnabled()) {
189                 logger.debug("Checking: " + dsDef.getID());
190             }
191 
192             objects.put(dsDef.getID(), dsDef);
193 
194             Map<URI, AttributeValue> actions =
195                     new HashMap<URI, AttributeValue>();
196             Map<URI, AttributeValue> resAttr;
197 
198             try {
199                 actions.put(Constants.ACTION.ID.getURI(),
200                             Constants.ACTION.GET_DATASTREAM
201                                     .getStringAttribute());
202 
203                 resAttr = ResourceAttributes.getResources(pid);
204                 resAttr.put(Constants.DATASTREAM.ID.getURI(),
205                             new StringAttribute(dsDef.getID()));
206 
207                 RequestCtx req =
208                         getContextHandler()
209                                 .buildRequest(getSubjects(context),
210                                               actions,
211                                               resAttr,
212                                               getEnvironment(context));
213 
214                 requests.add(m_contextUtil.makeRequestCtx(req));
215             } catch (Exception e) {
216                 logger.error(e.getMessage(), e);
217                 throw new OperationHandlerException(e.getMessage(), e);
218             }
219         }
220 
221         String response =
222                 getContextHandler().evaluateBatch(requests
223                         .toArray(new String[requests.size()]));
224         ResponseCtx resCtx;
225         try {
226             resCtx = m_contextUtil.makeResponseCtx(response);
227         } catch (MelcoeXacmlException e) {
228             throw new PEPException(e);
229         }
230 
231         @SuppressWarnings("unchecked")
232         Set<Result> results = resCtx.getResults();
233 
234         List<DatastreamDef> resultObjects = new ArrayList<DatastreamDef>();
235         for (Result r : results) {
236             if (r.getResource() == null || "".equals(r.getResource())) {
237                 logger.warn("This resource has no resource identifier in the xacml response results!");
238             } else if (logger.isDebugEnabled()) {
239                 logger.debug("Checking: " + r.getResource());
240             }
241 
242             String[] ridComponents = r.getResource().split("\\/");
243             String rid = ridComponents[ridComponents.length - 1];
244 
245             if (r.getStatus().getCode().contains(Status.STATUS_OK)
246                     && r.getDecision() == Result.DECISION_PERMIT) {
247                 DatastreamDef tmp = objects.get(rid);
248                 if (tmp != null) {
249                     resultObjects.add(tmp);
250                     if (logger.isDebugEnabled()) {
251                         logger.debug("Adding: " + r.getResource() + "[" + rid
252                                 + "]");
253                     }
254                 } else {
255                     logger.warn("Not adding this object as no object found for this key: "
256                                     + r.getResource() + "[" + rid + "]");
257                 }
258             }
259         }
260 
261         return resultObjects;
262     }
