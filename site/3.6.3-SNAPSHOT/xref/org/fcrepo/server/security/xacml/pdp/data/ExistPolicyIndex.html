51   * A PolicyIndex based on an XPath XML database.

53   * This implementation only tested on eXist, but as only generic xmldb API methods have

114     @Override
115     public String addPolicy(String name, String document) throws PolicyIndexException {
116         String Id = nameToId(name);
117         XMLResource res;
118         try {
119             writeLock.lock();
120             // check it doesn't already exist
121             res = (XMLResource) m_collection.getResource(Id);
122             if (res != null ) {
123                 throw new PolicyIndexException("Tried to add an already-existing resource " + name);
124             }
125 
126             // create the new resource
127             res = (XMLResource) m_collection.createResource(Id, XMLResource.RESOURCE_TYPE);
128 
129             // set the resource content
130             res.setContentAsDOM(createDocument(document));
131 
132             // add it
133             m_collection.storeResource(res);
134         } catch (XMLDBException e) {
135             log.error("Error adding resource " + name + " " + e.getMessage(), e);
136             throw new PolicyIndexException("Error adding resource " + name + " " + e.getMessage(), e);
137         } finally {
138             writeLock.unlock();
139         }
140         return name;
141     }

190     @Override
191     public Map<String, AbstractPolicy> getPolicies(EvaluationCtx eval, PolicyFinder policyFinder)
192     throws PolicyIndexException {
193 
194         Map<String, Set<AttributeBean>> attributeMap;
195         try {
196             // get evaluation context attributes to query on
197             attributeMap = getAttributeMap(eval);
198         } catch (URISyntaxException e) {
199             log.error("Error getting attribute map " + e.getMessage(), e);
200             throw new PolicyIndexException("Error getting attribute map " + e.getMessage(), e);
201         }
202 
203         Map<String, AbstractPolicy> documents = new HashMap<String, AbstractPolicy>();
204 
205         // generate the xpath query and variables
206         String query = getXpath(attributeMap);
207         Map<String, String> variables = getXpathVariables(attributeMap);
208 
209         try {
210             readLock.lock();
211             // do the query
212             ResourceSet resources = doQuery(query, variables);
213 
214             try {
215                 // get each result
216                 ResourceIterator ri = resources.getIterator();
217                 while (ri.hasMoreResources()) {
218                     XMLResource res = (XMLResource)ri.nextResource();
219                     // get the result's document name (the query result just contains content as selected by the xpath query)
220                     String id = res.getDocumentId();
221                     log.trace("Query matched document: " + IdToName(id));
222                     Document policyDoc = m_policyReader.readPolicy(((String)m_collection.getResource(id).getContent()).getBytes("UTF-8"));
223                     documents.put(IdToName(id), handleDocument(policyDoc, policyFinder));
224                 }
225             } catch (XMLDBException e) {
226                 log.error("Error retrieving query results " + e.getMessage(), e);
227                 throw new PolicyIndexException("Error retrieving query results " + e.getMessage(), e);
228             } catch (ParsingException e) {
229                 log.error("Error retrieving query results " + e.getMessage(), e);
230                 throw new PolicyIndexException("Error retrieving query results " + e.getMessage(), e);
231             } catch (UnsupportedEncodingException e) {
232                 // Should never happen
233                 throw new RuntimeException("Unsupported encoding " + e.getMessage(), e);
234             }
235         } finally {
236             readLock.unlock();
237         }
238 
239         return documents;

305         OutputFormat format = new OutputFormat();
306         format.setEncoding("UTF-8");
307         format.setIndenting(true);
308         format.setIndent(2);
309         format.setOmitXMLDeclaration(false);
310 
311         ByteArrayOutputStream out = new ByteArrayOutputStream();
312         Writer output = new OutputStreamWriter(out);
313 
314         XMLSerializer serializer = new XMLSerializer(output, format);
315         try {

317         } catch (IOException e) {
318             throw new PolicyIndexException("Failed to serialise node " + e.getMessage(), e);
319         }
320         return out.toByteArray();

347     protected  static String[] sortDescending(String[] s) {
348         Arrays.sort(s,
349                     new Comparator<String>() {
350             @Override

352                 if (o1.length() < o2.length())

363     protected ResourceSet doQuery(String query, Map<String, String> variables) throws PolicyIndexException {
364         try {
365             XPathQueryService queryService = (XPathQueryService) m_collection.getService("XPathQueryService", "1.0");
366 
367             // set namespaces
368             for (String prefix : PolicyIndexBase.namespaces.keySet()) {
369                 queryService.setNamespace(prefix, PolicyIndexBase.namespaces.get(prefix));
370             }
371 
372             // note: eXist extensions support "declareVariable", but the base API does not
373             // so we substitute the variables here to avoid dependency on the eXist implementation (and libraries)
374 
375             // do in descending order as some variable names could be substrings of others
376             String[] varNames = sortDescending((variables.keySet().toArray(new String[0])));
377 
378             for (String name : varNames) {
379                 // nb, treat as strings (variables currently does not specify type)
380                 query = query.replace("$" + name, "\"" + variables.get(name) + "\"");
381             }
382             log.trace("XPath query with variables substituted:\n" + query);
383 
384 
385             /* eXist impl version, not used, to remove dependency on eXist API
386             // set the xpath variables if supplied
387             if (variables != null) {
388                 for (String name : variables.keySet()) {
389                     String value = variables.get(name);
390                     // query service needs to be eXist extension to xmldb
391                     queryService.declareVariable(name, value);
392                 }
393             }
394              */
395             long start = System.nanoTime();
396             ResourceSet results = queryService.query(query);
397             log.debug("XPath query time: " + (System.nanoTime() - start) + "ns");
398 
399             return results;
400 
401         } catch (XMLDBException e) {
402             log.error("Error running query " + e.getMessage(), e);
403             throw new PolicyIndexException("Error running query " + e.getMessage(), e);
404         }
405     }

407     /**
408      * Create a collection given a full path to the collection.  The collection path must include
409      * the root collection.  Intermediate collections in the path are created if they do not
410      * already exist.
411      *
412      * @param collectionPath
413      * @param rootCollection
414      * @return
415      * @throws PolicyIndexException
416      */
417     protected Collection createCollectionPath(String collectionPath, Collection rootCollection) throws PolicyIndexException {
418         try {
419             if (rootCollection.getParentCollection()  != null) {
420                 throw new PolicyIndexException("Collection supplied is not a root collection");
421             }
422             String rootCollectionName = rootCollection.getName();
423             if (!collectionPath.startsWith(rootCollectionName)) {
424                 throw new PolicyIndexException("Collection path " + collectionPath + " does not start from root collection - " + rootCollectionName );
425             }
426 
427             // strip root collection from path, obtain each individual collection name in the path
428             String pathToCreate = collectionPath.substring(rootCollectionName.length());
429             String[] collections = pathToCreate.split("/");
430 
431             // iterate each and create as necessary
432             Collection nextCollection = rootCollection;
433             for (String collectionName : collections ) {
434                 Collection childCollection = nextCollection.getChildCollection(collectionName);
435                 if (childCollection != null) {
436                     // child exists
437                     childCollection = nextCollection.getChildCollection(collectionName);
438                 } else {
439                     // does not exist, create it
440                     CollectionManagementService mgtService = (CollectionManagementService) nextCollection.getService("CollectionManagementService", "1.0");
441                     childCollection = mgtService.createCollection(collectionName);
442                     log.debug("Created collection " + collectionName);
443                 }
444                 if (nextCollection.isOpen()) {
445                     nextCollection.close();
446                 }
447                 nextCollection = childCollection;
448             }
449             return nextCollection;
450         } catch (XMLDBException e) {
451             log.error("Error creating collections from path " + e.getMessage(), e);
452             throw new PolicyIndexException("Error creating collections from path " + e.getMessage(), e);
453         }
454     }

507     protected void initCollection() throws PolicyIndexException {
508         try {
509             // try to get FeSL policy collection
510             m_collection = DatabaseManager.getCollection(m_databaseURI + m_collectionPath, m_user, m_password);
511 
512             // if it doesn't exist, create it and create the index document if this doesn't already exist
513             if (m_collection == null) {
514                 // get root collection
515                 Collection rootCol = DatabaseManager.getCollection(m_databaseURI + ROOT_COLLECTION_PATH ,m_user, m_password);
516                 CollectionManagementService mgtService = (CollectionManagementService) rootCol.getService("CollectionManagementService", "1.0");
517 
518                 // first create the index for the collection
519                 Collection indexCollection = DatabaseManager.getCollection(m_databaseURI + m_indexCollectionPath, m_user, m_password);
520                 if (indexCollection == null) {
521                     log.debug("creating index collection");
522                     indexCollection = createCollectionPath(m_indexCollectionPath, rootCol);
523                 }
524                 if (rootCol.isOpen())
525                     rootCol.close();
526                 XMLResource ixd = (XMLResource) indexCollection.getResource(INDEX_DOCUMENT_NAME);
527                 // get an already-existing index config document; or create if it doesn't exist
528                 if (ixd == null ) {
529                     ixd = (XMLResource) indexCollection.createResource(INDEX_DOCUMENT_NAME, XMLResource.RESOURCE_TYPE);
530                 }
531 
532                 // Note: although eXist allows full path-based indexes, recommendation is to use
533                 // qname-based indexes for efficiency, therefore the following defines indexing
534                 // on resource attribute IDs and values throughout rather than following the
535                 // (dbxml-style) index definition.
536                 // Thus the index configuration in the config file is not followed.
537 
538                 String configDoc =
539                     "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" +
540                     "<collection xmlns=\"http://exist-db.org/collection-config/1.0\">" +
541                     "<index xmlns:p=\"urn:oasis:names:tc:xacml:2.0:policy:schema:os\">" +
542                     "<create qname=\"p:AttributeValue\" type=\"xs:string\"/>" +
543                     "<create qname=\"@AttributeId\" type=\"xs:string\"/>" +
544                     "</index>" +
545                     "</collection>";
546 
547                 Document config = createDocument(configDoc);
548                 log.debug("Storing index document");
549                 ixd.setContentAsDOM(config);
550                 indexCollection.storeResource(ixd);
551                 indexCollection.close();
552 
553                 // create the collection
554                 log.debug("Creating policy collection");
555                 m_collection = mgtService.createCollection(m_collectionName);
556             }
557 
558         } catch (XMLDBException e) {
559             throw new PolicyIndexException("Error getting/creating policy collection " + e.getMessage(), e);
560         }
561     }
