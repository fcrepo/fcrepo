125     /*
126      * synchronize so that each access gets the same item instance (protect
127      * against overlapping calls) note that expiration logic of cache element
128      * changes the element's state -- elements are never removed from cache or
129      * replaced
130      */
131     private final synchronized CacheElement getCacheElement(String userid /*
132                                                                              * ,
133                                                                              * String
134                                                                              * password
135                                                                              */) {
136         String m = getCacheAbbrev() + " getCacheElement() ";
137         CacheElement cacheElement = null;
138         String keytemp = getKey(userid/* ,password,CACHE_KEY_SEPARATOR */);
139         Integer key = new Integer(keytemp.hashCode());
140         logger.debug(m + "keytemp==" + keytemp);
141         logger.debug(m + "key==" + key);
142         if (cache.containsKey(key)) {
143             logger.debug(m + "cache already has element");
144         } else {
145             logger.debug(m + "cache does not have element; create and put");
146             CacheElement itemtemp =
147                     new CacheElement(userid, getCacheId(), getCacheAbbrev());
148             cache.put(key, itemtemp);
149         }
150         cacheElement = (CacheElement) cache.get(key);
151         if (cacheElement == null) {
152             logger.error(m + "cache does not contain element");
153         } else {
154             logger.debug(m + "element retrieved from cache successfully");
155         }
156         return cacheElement;
157     }
