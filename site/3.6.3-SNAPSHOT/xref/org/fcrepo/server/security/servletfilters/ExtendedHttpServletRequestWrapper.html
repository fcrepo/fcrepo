243     public Set getAttributeValues(String key) throws AuthzOperationalException {
244         Set accumulatedValues4Key = null;
245         Map map = null;
246         if (isUserSponsored()) {
247             map = sponsoredAttributes;
248         } else {
249             map = authenticatedAttributes;
250         }
251         for (Iterator iterator = map.values().iterator(); iterator.hasNext();) {
252             Map attributesFromOneAuthority = (Map) iterator.next();
253             if (attributesFromOneAuthority.containsKey(key)) {
254                 Set someValues4Key = (Set) attributesFromOneAuthority.get(key);
255                 if (someValues4Key != null && !someValues4Key.isEmpty()) {
256                     if (accumulatedValues4Key == null) {
257                         accumulatedValues4Key = new HashSet();
258                     }
259                     accumulatedValues4Key.addAll(someValues4Key);
260                 }
261             }
262         }
263         if (accumulatedValues4Key == null) {
264             accumulatedValues4Key = IMMUTABLE_NULL_SET;
265         }
266         return accumulatedValues4Key;

282     private void putIntoMap(Map map, String key, Object value) throws Exception {
283         if (wrapperWriteLocked) {
284             throw new Exception();
285         }
286         if (!isAuthenticated()) {
287             throw new Exception("can't collect user roles/attributes/groups until after authentication");
288         }
289         if (map == null || key == null || value == null) {
290             throw new Exception("null parm, map==" + map + ", key==" + key
291                     + ", value==" + value);
292         }
293         if (map.containsKey(key)) {
294             throw new Exception("map already contains key==" + key);
295         }
296         logger.debug("mapping " + key + " => " + value + " in " + map);
297         map.put(key, value);
298     }

340     private final String[] parseUsernamePassword(String header)
341             throws Exception {
342         String here = "parseUsernamePassword():";
343         String[] usernamePassword = null;
344 
345         String msg = here + "header intact";
346         if (header == null || "".equals(header)) {
347             String exceptionMsg = msg + FAILED;
348             logger.error(exceptionMsg + ", header==" + header);
349             throw new Exception(exceptionMsg);
350         }
351         logger.debug(msg + SUCCEEDED);
352 
353         String authschemeUsernamepassword[] = header.split("\\s+");
354 
355         msg = here + "header split";
356         if (authschemeUsernamepassword.length != 2) {
357             String exceptionMsg = msg + FAILED;
358             logger.error(exceptionMsg + ", header==" + header);
359             throw new Exception(exceptionMsg);
360         }
361         logger.debug(msg + SUCCEEDED);
362 
363         msg = here + "auth scheme";
364         String authscheme = authschemeUsernamepassword[0];
365         if (authscheme == null && !BASIC.equalsIgnoreCase(authscheme)) {
366             String exceptionMsg = msg + FAILED;
367             logger.error(exceptionMsg + ", authscheme==" + authscheme);
368             throw new Exception(exceptionMsg);
369         }
370         logger.debug(msg + SUCCEEDED);
371 
372         msg = here + "digest non-null";
373         String usernamepassword = authschemeUsernamepassword[1];
374         if (usernamepassword == null || "".equals(usernamepassword)) {
375             String exceptionMsg = msg + FAILED;
376             logger.error(exceptionMsg + ", usernamepassword==" + usernamepassword);
377             throw new Exception(exceptionMsg);
378         }
379         logger.debug(msg + SUCCEEDED + ", usernamepassword==" + usernamepassword);
380 
381         byte[] encoded = usernamepassword.getBytes();
382         msg = here + "digest base64-encoded";
383         if (!Base64.isArrayByteBase64(encoded)) {
384             String exceptionMsg = msg + FAILED;
385             logger.error(exceptionMsg + ", encoded==" + encoded);
386             throw new Exception(exceptionMsg);
387         }
388         if (logger.isDebugEnabled()) {
389             logger.debug(msg + SUCCEEDED + ", encoded==" + encoded);
390         }
391 
392         byte[] decodedAsByteArray = Base64.decodeBase64(encoded);
393         logger.debug(here + "got decoded bytes" + SUCCEEDED
394                 + ", decodedAsByteArray==" + decodedAsByteArray);
395 
396         String decoded = new String(decodedAsByteArray); //decodedAsByteArray.toString();
397         logger.debug(here + "got decoded string" + SUCCEEDED + ", decoded=="
398                 + decoded);
399 
400         msg = here + "digest decoded";
401         if (decoded == null || "".equals(decoded)) {
402             String exceptionMsg = msg + FAILED;
403             logger.error(exceptionMsg + ", digest decoded==" + decoded);
404             throw new Exception(exceptionMsg);
405         }
406         logger.debug(msg + SUCCEEDED);
407 
408         String DELIMITER = ":";
409         if (decoded == null) {
410             logger.error("decoded user/password is null . . . returning 0-length strings");
411             usernamePassword = new String[2];
412             usernamePassword[0] = "";
413             usernamePassword[1] = "";
414         } else if (decoded.indexOf(DELIMITER) < 0) {
415             String exceptionMsg = "decoded user/password lacks delimiter";
416             logger.error(exceptionMsg + " . . . throwing exception");
417             throw new Exception(exceptionMsg);
418         } else if (decoded.startsWith(DELIMITER)) {
419             logger.error("decoded user/password is lacks user . . . returning 0-length strings");
420             usernamePassword = new String[2];
421             usernamePassword[0] = "";
422             usernamePassword[1] = "";
423         } else if (decoded.endsWith(DELIMITER)) { // no password, e.g., user == "guest"
424             usernamePassword = new String[2];
425             usernamePassword[0] = decoded.substring(0, decoded.length() - 1);
426             usernamePassword[1] = "";
427         } else { // usual, expected case
428             usernamePassword = decoded.split(DELIMITER);
429         }
430 
431         msg = here + "user/password split";
432         if (usernamePassword.length != 2) {
433             String exceptionMsg = msg + FAILED;
434             logger.error(exceptionMsg + ", digest decoded==" + decoded);
435             throw new Exception(exceptionMsg);
436         }
437         logger.debug(msg + SUCCEEDED);
438 
439         return usernamePassword;
440     }

469     public final String getUser() throws Exception {
470         if (username == null) {
471             logger.debug("username==null, so will grok now");
472             String authorizationHeader = getAuthorizationHeader();
473             logger.debug("authorizationHeader==" + authorizationHeader);
474             if (authorizationHeader != null && !"".equals(authorizationHeader)) {
475                 logger.debug("authorizationHeader is intact");
476                 String[] usernamePassword =
477                         parseUsernamePassword(authorizationHeader);
478                 logger.debug("usernamePassword[] length=="
479                         + usernamePassword.length);
480                 username = usernamePassword[0];
481                 logger.debug("username (usernamePassword[0])==" + username);
482                 if (super.getRemoteUser() == null) {
483                     logger.debug("had none before");
484                 } else if (super.getRemoteUser() == username
485                         || super.getRemoteUser().equals(username)) {
486                     logger.debug("got same now");
487                 } else {
488                     throw new Exception("somebody got it wrong");
489                 }
490             }
491         }
492         logger.debug("return user==" + username);
493         return username;
494     }

496     public final String getPassword() throws Exception {
497         if (password == null) {
498             String authorizationHeader = getAuthorizationHeader();
499             if (authorizationHeader != null && !"".equals(authorizationHeader)) {
500                 String[] usernamePassword =
501                         parseUsernamePassword(authorizationHeader);
502                 password = usernamePassword[1];
503             }
504         }
505         logger.debug("return password==" + password);
506         return password;
