100     @Override
101     public boolean login() throws LoginException {
102         if (debug) {
103             logger.debug(this.getClass().getName() + " login called.");
104         }
105 
106         if (Constants.FEDORA_HOME == null || "".equals(Constants.FEDORA_HOME.trim())) {
107             logger.error("FEDORA_HOME constant is not set");
108             return false;
109         }
110 
111         // The only 2 callback types that are supported.
112         Callback[] callbacks = new Callback[2];
113         callbacks[0] = new NameCallback("username");
114         callbacks[1] = new PasswordCallback("password", false);
115 
116         String password = null;
117         try {
118             // sets the username and password from the callback handler
119             handler.handle(callbacks);
120             username = ((NameCallback) callbacks[0]).getName();
121             char[] passwordCharArray =
122                     ((PasswordCallback) callbacks[1]).getPassword();
123             password = new String(passwordCharArray);
124         } catch (IOException ioe) {
125             ioe.printStackTrace();
126             throw new LoginException("IOException occured: " + ioe.getMessage());
127         } catch (UnsupportedCallbackException ucbe) {
128             ucbe.printStackTrace();
129             throw new LoginException("UnsupportedCallbackException encountered: "
130                     + ucbe.getMessage());
131         }
132 
133         successLogin = authenticate(username, password);
134 
135         return successLogin;
136     }

138     @Override
139     public boolean commit() throws LoginException {
140         if (!successLogin) {
141             return false;
142         }
143 
144         try {
145             subject.getPrincipals().add(principal);
146             subject.getPublicCredentials().add(attributes);
147         } catch (Exception e) {
148             logger.error(e.getMessage(), e);
149             return false;
150         }
151 
152         return true;
153     }

187     private boolean authenticate(String username, String password) {
188         if (!userFile.exists()) {
189             logger.error("XmlUsersFile not found: {}", userFile.getAbsolutePath());
190             return false;
191         }
192 
193         try {
194             Document userDoc = getUserDocument();
195             // go through each user
196             NodeList userList = userDoc.getElementsByTagName("user");
197             for (int x = 0; x < userList.getLength(); x++) {
198                 Element user = (Element) userList.item(x);
199                 String a_username = user.getAttribute("name");
200                 String a_password = user.getAttribute("password");
201                 if (!a_username.equals(username)
202                         || !a_password.equals(password)) {
203                     continue;
204                 }
205 
206                 principal = new UserPrincipal(username);
207 
208                 // for a matched user, go through each attribute
209                 NodeList attributeList = user.getElementsByTagName("attribute");
210                 for (int y = 0; y < attributeList.getLength(); y++) {
211                     Element attribute = (Element) attributeList.item(y);
212                     String name = attribute.getAttribute("name");
213 
214                     // go through each value
215                     NodeList valueList =
216                             attribute.getElementsByTagName("value");
217                     for (int z = 0; z < valueList.getLength(); z++) {
218                         Element value = (Element) valueList.item(z);
219                         String v = value.getFirstChild().getNodeValue();
220 
221                         Set<String> values = attributes.get(name);
222                         if (values == null) {
223                             values = new HashSet<String>();
224                             attributes.put(name, values);
225                         }
226                         values.add(v);
227                     }
228                 }
229 
230                 return true;
231             }
232         } catch (Exception e) {
233             logger.error(e.getMessage());
234         }
235         return false;
236     }

238     private static File getUsersFile() {
239         if (Constants.FEDORA_HOME == null || "".equals(Constants.FEDORA_HOME)) {
240             logger.error("FEDORA_HOME constant is not set");
241         } else {
242             if (logger.isDebugEnabled()) {
243                 logger.debug("using FEDORA_HOME: " + Constants.FEDORA_HOME);
244             }
245         }
246 
247         return new File(Constants.FEDORA_HOME + "/server/config/fedora-users.xml");
248     }

250     /**
251      * Quick and dirty caching.
252      * @return DOM of a parsed user file
253      * @throws Exception
254      */
255     private static Document getUserDocument() throws Exception {
256         if (userDoc == null || userFileParsed != userFile.lastModified() || userFileBytesLoaded != userFile.length()) {
257             synchronized(XmlUsersFileModule.class) {
258                 if (userDoc == null || userFileParsed != userFile.lastModified() || userFileBytesLoaded != userFile.length()) {
259                     userDoc = DataUtils.getDocumentFromFile(userFile);
260                     userFileParsed = userFile.lastModified();
261                     userFileBytesLoaded = userFile.length();
262                 }
263             }
264         }
265         return userDoc;
266     }
