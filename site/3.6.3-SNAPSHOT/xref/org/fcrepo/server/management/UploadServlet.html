33  /**
34   * Accepts and HTTP Multipart POST of a file from an authorized user, and if
35   * successful, returns a status of "201 Created" and a text/plain response with
36   * a single line containing an opaque identifier that can be used to later
37   * submit to the appropriate API-M method.
38   * <p>
39   * If it fails it will return a non-201 status code with a text/plain
40   * explanation. The submitted file must be named "file".
41   *
42   * @author Chris Wilper
43   */
44  public class UploadServlet
45          extends SpringManagementServlet {
46  
47      private static final Logger logger =
48              LoggerFactory.getLogger(UploadServlet.class);
49  
50      private static final long serialVersionUID = 1L;
51  
52      /**
53       * The servlet entry point. http://host:port/fedora/management/upload
54       */
55      @Override
56      public void doPost(HttpServletRequest request, HttpServletResponse response)
57              throws ServletException, IOException {
58          Context context =
59                  ReadOnlyContext.getContext(Constants.HTTP_REQUEST.REST.uri,
60                                             request);
61          try {
62              // Create a new file upload handler
63              ServletFileUpload upload = new ServletFileUpload();
64  
65              // Parse the request, looking for "file"
66              InputStream in = null;
67              FileItemIterator iter = upload.getItemIterator(request);
68              while (in == null && iter.hasNext()) {
69                  FileItemStream item = iter.next();
70                  logger.info("Got next item: isFormField=" + item.isFormField() + " fieldName=" + item.getFieldName());
71                  if (!item.isFormField() && item.getFieldName().equals("file")) {
72                      in = item.openStream();
73                  }
74              }
75              if (in == null) {
76                  sendResponse(HttpServletResponse.SC_BAD_REQUEST,
77                               "No data sent.",
78                               response);
79              } else {
80                  sendResponse(HttpServletResponse.SC_CREATED,
81                               m_management.putTempStream(context, in),
82                               response);
83              }
84          } catch (AuthzException ae) {
85              throw RootException.getServletException(ae,
86                                                      request,
87                                                      "Upload",
88                                                      new String[0]);
89          } catch (Exception e) {
90              e.printStackTrace();
91              sendResponse(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e
92                      .getClass().getName()
93                      + ": " + e.getMessage(), response);
94          }
95      }
96  
97      @Override
98      public void doGet(HttpServletRequest request, HttpServletResponse response)
99              throws IOException {
100         sendResponse(HttpServletResponse.SC_OK,
101                      "Client must use HTTP Multipart POST",
102                      response);
103     }
104 
105     public void sendResponse(int status,
106                              String message,
107                              HttpServletResponse response) {
108         try {
109             if (status == HttpServletResponse.SC_CREATED) {
110                 logger.info("Successful upload, id=" + message);
111             } else {
112                 logger.error("Failed upload: " + message);
113             }
114             response.setStatus(status);
115             response.setContentType("text/plain");
116             PrintWriter w = response.getWriter();
117             w.println(message);
118         } catch (Exception e) {
119             logger.error("Unable to send response", e);
120         }
121     }
