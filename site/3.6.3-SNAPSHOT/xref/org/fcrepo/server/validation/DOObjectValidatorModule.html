19  /**
20   * A module for validating digital objects.  Instead of validating the FOXML
21   * (as per DOValidatorModule), the digital object itself is validated, wrapped
22   * in a DOReader.
23   * 
24   * If validators have been configured (see Spring config doobjectvalidator.xml), then
25   * any object creation or modification will result in the object being validated before
26   * it is committed to storage.  Any validation errors will result in the modification 
27   * being aborted.
28   * 
29   * @author stephen.bayliss
30   *
31   */
32  public class DOObjectValidatorModule extends Module implements
33  		DOObjectValidator {
34  
35  	private static final Logger logger = LoggerFactory
36  			.getLogger(DOValidatorModule.class);
37  	
38  	// holds the set of validators to validate against
39      private final Map<String, DOObjectValidator> m_validators =
40              new HashMap<String, DOObjectValidator>();
41      
42      private boolean m_enabled = false;
43  
44  	public DOObjectValidatorModule(Map<String, String> moduleParameters,
45  			Server server, String role) throws ModuleInitializationException {
46  		super(moduleParameters, server, role);
47  		
48  	}
49      @Override
50      public void postInitModule() throws ModuleInitializationException {
51          try {
52          } catch (Exception e) {
53              throw new ModuleInitializationException(e.getMessage(),
54                                                      "org.fcrepo.server.validation.DOObjectValidatorModule");
55          }
56      }	
57  
58      @Override
59      public void validate(Context context, DOReader reader) throws ServerException {
60  		if (m_enabled) {
61  			// validate against all of the configured validators
62  			for (DOObjectValidator validator : m_validators.values()) {
63  				try {
64  					validator.validate(context, reader);
65  				} catch (ObjectValidityException e) {
66  					logger.error("Object validation error " + reader.GetObjectPID() + ": " + e.getMessage());
67  					throw e;
68  				}
69  			}
70  		}
71  	}
72  	
73  	// spring config of the validators
74  	public void setValidators(Map<String,? extends DOObjectValidator> validators){
75  		logger.info("Adding " + validators.size() + " object validators");
76          m_validators.putAll(validators);
77          if (m_validators.size() > 0) {
78          	m_enabled = true;
79          }
80      }
81  	
82  
83  }
