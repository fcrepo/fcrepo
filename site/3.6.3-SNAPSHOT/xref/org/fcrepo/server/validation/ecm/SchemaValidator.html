55      void validate(Context context, DsTypeModel typeModel, Datastream objectDatastream, Validation validation,
56                    DOReader contentmodelReader, Date asOfDateTime, ExternalContentManager m_exExternalContentManager) throws ServerException {
57  
58          List<Extension> extensions = typeModel.getExtension();
59          List<List<String>> schemaStreamsToProblemsMap = new ArrayList<List<String>>();
60          for (Extension extension : extensions) {
61              String name = extension.getName();
62  
63  
64              Element reference = null;
65              Source source = null;
66  
67              if (!"SCHEMA".equals(name)) {//ignore non schema extensions
68                  continue;
69              }
70  
71              List<Element> contents = extension.getAny();
72              for (Element content : contents) { //find the reference
73                  String tagname = content.getTagName();
74                  if (tagname.equals("reference")) {
75                      reference = content;
76                      break;
77                  }
78              }
79  
80              if (reference == null){ //if no reference
81                  boolean found = false;
82                  for (Element content : contents) {
83                      if ( content.getNodeType() == Element.ELEMENT_NODE){
84                          source = new DOMSource(content);//parse the inline schema
85                          found = true;
86                          break;
87                      }
88                  }
89                  if (!found){//empty tag
90                      List<String> validationProblems = validation.getDatastreamProblems(objectDatastream.DatastreamID);
91                      validationProblems.add(Errors.schemaNotFound(contentmodelReader.GetObjectPID()));
92                      validation.setValid(false);
93                  }
94              }
95              else {
96                  String type = reference.getAttribute("type");
97                  String value = reference.getAttribute("value");
98  
99                  if ("datastream".equalsIgnoreCase(type)) {
100                     Datastream schemaDS = contentmodelReader.GetDatastream(value, asOfDateTime);
101                     if (schemaDS == null) {//No schema datastream, ignore and continue
102                         continue;
103                     }
104                     InputStream schemaStream;
105                     schemaStream = schemaDS.getContentStream();
106                     source = new StreamSource(schemaStream);
107 
108                 } else if ("url".equalsIgnoreCase(type)){
109                     InputStream schemaStream;
110                     ContentManagerParams params = new ContentManagerParams(value);
111                     MIMETypedStream externalContent = m_exExternalContentManager.getExternalContent(params);
112                     schemaStream = externalContent.getStream();
113                     source = new StreamSource(schemaStream);
114                 } else { //reference used, but type not recognized
115                     List<String> validationProblems = validation.getDatastreamProblems(objectDatastream.DatastreamID);
116                     validationProblems.add(Errors.schemaNotFound(contentmodelReader.GetObjectPID()));
117                     validation.setValid(false);
118                     continue;
119                 }
120 
121             }
122 
123             LSResourceResolver resourceResolver
124                     = new ResourceResolver(context, doMgr, contentmodelReader, asOfDateTime);
125 
126 
127             Schema schema;
128             try {
129                 schema = parseAsSchema(source, resourceResolver);
130             } catch (SAXException e) {
131                 List<String> validationProblems = validation.getDatastreamProblems(objectDatastream.DatastreamID);
132                 validationProblems.add(Errors.schemaCannotParse(contentmodelReader.GetObjectPID(),objectDatastream.DatastreamID,e));
133                 validation.setValid(false);
134                 continue;
135             }
136 
137             List<String> problems = checkSchema( objectDatastream.getContentStream(),
138                                                  schema,
139                                                  contentmodelReader.GetObjectPID(),
140                                                  objectDatastream.DatastreamID);
141             schemaStreamsToProblemsMap.add(problems);
142             if (problems.isEmpty()) {//if multiple SCHEMAS have been defined, only one is needed to be compliant. If one schema
143                 //produce no errors, do not bother validating against the others.
144                 break;
145 
146             }
147         }
148 
149 
150         boolean foundProblem = false;
151         for (List<String> problems : schemaStreamsToProblemsMap) {
152             if (!problems.isEmpty()){
153                 foundProblem = true;
154                 break;
155             }
156         }
157         if (foundProblem) {
158             validation.setValid(false);
159             List<String> validationProblems = validation.getDatastreamProblems(objectDatastream.DatastreamID);
160             for (List<String> problems : schemaStreamsToProblemsMap) {
161                 validationProblems.addAll(problems);
162             }
163         }
164     }

167     private Schema parseAsSchema(Source input, LSResourceResolver resolver) throws SAXException {
168         SchemaFactory schemaFactory = SchemaFactory
169                 .newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
170         Schema schema;
171         schemaFactory.setResourceResolver(resolver);
172 
173         schema = schemaFactory.newSchema(input);
174         return schema;
175     }

177     public List<String> checkSchema(InputStream objectStream, Schema schema,
178                                     String contentModel,
179                                     String datastreamID) {
180 
181         List<String> problems = new ArrayList<String>();
182         ErrorHandler errorhandler = new ReportingErrorHandler(problems, contentModel, datastreamID);
183 
184         try {
185             Validator validator = schema.newValidator();
186             validator.setErrorHandler(errorhandler);
187             validator.validate(new StreamSource(objectStream));
188         } catch (SAXException e) {
189             problems.add(Errors.invalidContentInDatastream(datastreamID,contentModel,e));
190         } catch (IOException e) {
191             problems.add(Errors.unableToReadDatastream(datastreamID,e));
192         }
193         return problems;
194     }

213         @Override
214         public LSInput resolveResource(String type, String namespaceURI, String publicId, String systemId,
215                                        String baseURI) {
216             //The key here is SystemID.
217             if (systemId == null) {
218                 return null;
219             }
220             if (systemId.startsWith("$THIS$/")) {//other datastream in this object
221                 String[] tokens = systemId.split("/");
222                 //String is of the format $THIS$/DSID
223                 try {
224                     final Datastream schemastream = contentmodelReader.GetDatastream(tokens[1], asOfDateTime);
225                     LSInput input = new MyLSInput(schemastream);
226                     input.setBaseURI(baseURI);
227                     input.setPublicId(publicId);
228                     input.setBaseURI(baseURI);
229                     return input;
230                 } catch (ServerException e) {
231                     return null;
232                 }
233             } else {
234                 return null;
235             }
236         }
