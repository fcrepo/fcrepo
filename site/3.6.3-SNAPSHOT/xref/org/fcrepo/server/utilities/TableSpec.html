19  /**
20   * A holder of table specification information that helps in producing 
21   * RDBMS-generic DDL "CREATE TABLE" commands.
22   * 
23   * <p>An application constructs a TableSpec without regard to the underlying
24   * database kind, and then the TableSpec is converted to a DB-specific CREATE
25   * TABLE DDL command by a DDLConverter before the command is issued via JDBC.
26   * 
27   * @author Chris Wilper
28   */

39      /**
40       * Constructs a TableSpec given a name, a set of ColumnSpecs, and the name

42       * 
43       * @param name
44       *        The table name.
45       * @param columnSpecs
46       *        ColumnSpec objects describing columns in the table.
47       * @param primaryColumnName
48       *        The column that is the primary key for the table.
49       * @throws InconsistentTableSpecException

90      /**
91       * Gets a TableSpec for each table element in the stream, where the stream
92       * contains a valid XML document containing one or more table elements,
93       * wrapped in the root element.
94       * 
95       * <p>
96       * Input is of the form:
97       * 
98       * <pre>
99       * &lt;database&gt;
100      *   &lt;table name=&quot;<i>tableName</i>&quot; primaryKey=&quot;<i>primaryColumnName</i>&quot; type=&quot;<i>tableType</i>&quot;&gt;
101      *     &lt;column name=&quot;<i>columnName</i>&quot;
102      *                type=&quot;<i>typeSpec</i>&quot;
103      *                autoIncrement=&quot;<i>isAutoIncremented</i>&quot;
104      *                index=&quot;<i>indexName</i>&quot;
105      *                notNull=&quot;<i>isNotNull</i>&quot;
106      *                unique=&quot;<i>isUnique</i>&quot;
107      *                default=&quot;<i>defaultValue</i>&quot;
108      *                foreignKey=&quot;<i>foreignTableName.columnName onDeleteAction</i>&quot;/&gt;
109      *   &lt;/table&gt;
110      * &lt;/database&gt;
111      * </pre>
112      * 
113      * About the attributes:
114      * <ul>
115      * <li> <b>tableName</b> - The desired name of the table.
116      * <li> <b>primaryColumnName</b> - Identifies column that is the primary
117      * key for the table. A column that is a primary key must be notNull, and
118      * can't be a foreign key.
119      * <li> <b>type</b> - The table type, which is RDBMS-specific. See
120      * TableSpec(String, Set, String, String) for detail.
121      * <li> <b>columnName</b> - The name of the column.
122      * <li> <b>typeSpec</b> - The value type of the column. For instance,
123      * varchar(255). This is not checked for validity. See <a
124      * href="ColumnSpec.html">ColumnSpec javadoc</a> for detail.
125      * <li> <b>isAutoIncremented</b> - (true|false) Whether values in the
126      * column should be automatically generated by the database upon insert.
127      * This requires that the type be some numeric variant, and is
128      * RDBMS-specific. NUMERIC will generally work.
129      * <li> <b>indexName</b> - Specifies that an index should be created on
130      * this column and provides the column name.
131      * <li> <b>isNotNull</b> - (true|false) Whether input should be limited to
132      * actual values.
133      * <li> <b>isUnique</b> - (true|false) Whether input should be limited such
134      * that all values in the column are unique.
135      * <li> <b>default</b> - The value to be used when inserts don't specify a
136      * value for the column. This cannot be specified with autoIncrement true.
137      * <li> <b>foreignTableName.column</b> - Specifies that this is a foreign
138      * key column and identifies the (primary key) column in the database
139      * containing the rows that values in this column refer to. This cannot be
140      * specified with autoIncrement true.
141      * <li> <b>onDeleteAction</b> - Optionally specifies a "CASCADE" or "SET
142      * NULL" action to be taken on matching rows in this table when a row from
143      * the parent (foreign) table is deleted. If "CASCADE", matching rows in
144      * this table are automatically deleted. If "SET NULL", this column's value
145      * will be set to NULL for all matching rows. This value is not checked for
146      * validity.
147      * </ul>
148      * 
149      * @param in
150      *        The xml-encoded table specs.
151      * @return TableSpec objects.
152      * @throws InconsistentTableSpecException
153      *         if inconsistencies are detected in table specifications.
154      * @throws IOException
155      *         if an IO error occurs.
156      */
157     public static List<TableSpec> getTableSpecs(InputStream in)
158             throws InconsistentTableSpecException, IOException {
159         try {
160             SAXParserFactory spf = SAXParserFactory.newInstance();
161             spf.setNamespaceAware(true);
162             SAXParser parser = spf.newSAXParser();
163             TableSpecDeserializer tsd = new TableSpecDeserializer();
164             parser.parse(in, tsd);
165             tsd.assertTableSpecsConsistent();
166             return tsd.getTableSpecs();
167         } catch (InconsistentTableSpecException itse) {
168             throw itse;
169         } catch (Exception e) {
170             throw new IOException("Error parsing XML: " + e.getMessage());
171         }
172     }
