39          copy(ds);

43      /**
44       * Gets the external content manager which is used for the retrieval of
45       * content.
46       * 
47       * @return an instance of <code>ExternalContentManager</code> 
48       * @throws Exception is thrown in case the server is not able to find the module.
49       */
50      private ExternalContentManager getExternalContentManager()
51              throws Exception {
52          if (s_ecm == null) {
53              Server server;
54              try {
55                  server = Server.getInstance(new File(Constants.FEDORA_HOME),
56                          false);
57                  s_ecm = (ExternalContentManager) server
58                          .getModule("org.fcrepo.server.storage.ExternalContentManager");
59              } catch (InitializationException e) {
60                  throw new Exception(
61                          "Unable to get ExternalContentManager Module: "
62                                  + e.getMessage(), e);
63              }
64          }
65          return s_ecm;
66      }

68      /**
69       * Gets an InputStream to the content of this externally-referenced
70       * datastream.
71       * 
72       * <p>The DSLocation of this datastream must be non-null before invoking 
73       * this method.
74       * 
75       * <p>If successful, the DSMIME type is automatically set based on the web
76       * server's response header. If the web server doesn't send a valid
77       * Content-type: header, as a last resort, the content-type is guessed by
78       * using a map of common extensions to mime-types.
79       * 
80       * <p>If the content-length header is present in the response, DSSize will 
81       * be set accordingly.
82       *
83       * @see org.fcrepo.server.storage.types.Datastream#getContentStream()
84       */
85      @Override
86      public InputStream getContentStream(Context context) throws StreamIOException {
87          try {
88              ContentManagerParams params = new ContentManagerParams(DSLocation);
89              if (context != null ) {
90                  params.setContext(context);
91              }
92              MIMETypedStream stream = getExternalContentManager()
93                      .getExternalContent(params);
94              DSSize = getContentLength(stream);
95              return stream.getStream();
96          } catch (Exception ex) {
97              throw new StreamIOException("Error getting content stream", ex);
98          }
99      }

101     /**
102      * Returns the length of the content of this stream.
103      * @param stream the MIMETypedStream 
104      * @return length the length of the content
105      */
106     public long getContentLength(MIMETypedStream stream) {
107         long length = 0;
108         if (stream.header != null) {
109             for (int i = 0; i < stream.header.length; i++) {
110                 if (stream.header[i].name != null
111                         && !stream.header[i].name.equalsIgnoreCase("")
112                         && stream.header[i].name.equalsIgnoreCase("content-length")) {
113                     length = Long.parseLong(stream.header[i].value);
114                     break;
115                 }
116             }
117         }
118         return length;
119     }
