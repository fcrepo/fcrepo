34      private static final String ANY_MIME_TYPE = "*/*";
35  
36      /**
37       * In human readable string, describe which mime types are allowed.
38       */
39      public String describeAllowedMimeTypes() {
40          StringBuffer out = new StringBuffer();
41          if (bindingMIMETypes == null || bindingMIMETypes.length == 0) {
42              return ANY_MIME_TYPE;
43          }
44          for (int i = 0; i < bindingMIMETypes.length; i++) {
45              String allowed = bindingMIMETypes[i];
46              if (allowed == null) {
47                  return ANY_MIME_TYPE;
48              }
49              if (allowed.equals("*/*")) {
50                  return ANY_MIME_TYPE;
51              }
52              if (allowed.equals("*")) {
53                  return ANY_MIME_TYPE;
54              }
55              if (i > 0) {
56                  if (i < bindingMIMETypes.length - 1) {
57                      out.append(", ");
58                  } else {
59                      if (i > 1) {
60                          out.append(",");
61                      }
62                      out.append(" or ");
63                  }
64              }
65              out.append(allowed);
66          }
67          return out.toString();
68      }
