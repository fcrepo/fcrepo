18  /**
19   * Wrapper class for a datastream that has XML metadata.
20   *
21   * Allows managed content and inline xml datastreams to be treated generically
22   * as datastreams having XML metadata content, eg for DC, RELS-EXT and RELS-INT which
23   * may be stored either inline or as managed content.
24   *
25   * @author Stephen Bayliss
26   * @version $Id$
27   */

29  public class XMLDatastreamProcessor {
30  
31      private static final Logger logger =
32          LoggerFactory.getLogger(XMLDatastreamProcessor.class);

35      protected Datastream m_ds; // the wrapped datastream

37      private enum DS_TYPE {
38          INLINE_XML,
39          MANAGED
40      }

42      // when creating new datastreams via this class, these determine the
43      // type (M or X) of datastream to create.  See fedora.fcfg for config
44      private static String DC_DEFAULT_CONTROLGROUP;
45      private static String RELS_DEFAULT_CONTROLGROUP;
46  
47      private static boolean initialized = false;
48  
49      protected DS_TYPE m_dsType;
50  
51      @SuppressWarnings("unused")
52      private XMLDatastreamProcessor() {}

54      /**
55       * Construct a new XML datastream processor and the associated datastream
56       * using the datastream ID to determine the type (M or X) of datastream to
57       * construct (see fedora.fcfg for configuring the default types)
58       *
59       * @param dsId
60       */
61      public XMLDatastreamProcessor(String dsId) {
62          init();
63  
64          String controlGroup = null;
65          if (dsId.equals("DC")) {
66              controlGroup = DC_DEFAULT_CONTROLGROUP;
67          } else if (dsId.equals("RELS-EXT") || dsId.equals("RELS-INT")) {
68              controlGroup = RELS_DEFAULT_CONTROLGROUP;
69          } else {
70              // coding error if trying to handle other types of datastream
71              throw new RuntimeException("XML Datastream Processor only handles DC and RELS datastreams.  Datastream ID supplied was " + dsId);
72          }
73  
74          if (controlGroup.equals("X")) {
75              m_ds = new DatastreamXMLMetadata();
76              m_dsType = DS_TYPE.INLINE_XML;
77          } else if (controlGroup.equals("M")) {
78              m_ds = new DatastreamManagedContent();
79              m_dsType = DS_TYPE.MANAGED;
80          }
81          m_ds.DSControlGrp = controlGroup;
82          m_ds.DatastreamID = dsId;
83      }
84  
85      /**
86       * Construct an XML Datastream Processor by wrapping an existing datastream
87       *
88       * @param ds - datastream to wrap
89       */
90      public XMLDatastreamProcessor(Datastream ds) {
91  
92          if (ds instanceof DatastreamXMLMetadata) {
93              m_ds = ds;
94              m_dsType = DS_TYPE.INLINE_XML;
95          } else if (ds instanceof DatastreamManagedContent) {
96              m_ds = ds;
97              m_dsType = DS_TYPE.MANAGED;
98          } else {
99              // unhandled type is a coding error
100             throw new RuntimeException("XML datastreams must be of type Managed or Inline,"
101                     + " but type was " + ds.getClass().getName());
102         }
103     }

105     private static void init() {
106         if (!initialized) {
107             Server server;
108             // get default types of datastream (M or X) to be used for reserved datastreams
109             try {
110                 server = Server.getInstance(new File(Constants.FEDORA_HOME),
111                                             false);
112                 Module module = server.getModule("org.fcrepo.server.storage.DOManager");
113                 DC_DEFAULT_CONTROLGROUP = module.getParameter("defaultDCControlGroup");
114                 RELS_DEFAULT_CONTROLGROUP = module.getParameter("defaultRELSControlGroup");
115             } catch (ServerInitializationException e) {
116                 logger.error("Unable to get server", e);
117             } catch (ModuleInitializationException e) {
118                 logger.error("Unable to get DOManager module", e);
119             }
120             if (DC_DEFAULT_CONTROLGROUP == null) {
121                 logger.error("Unable to determine default controlgroup for DC datastreams, using X");
122                 DC_DEFAULT_CONTROLGROUP = "X";
123             }
124             if (RELS_DEFAULT_CONTROLGROUP == null) {
125                 logger.error("Unable to determine default controlgroup for RELS datastreams, using X");
126                 RELS_DEFAULT_CONTROLGROUP = "X";
127             }
128             initialized = true;
129         }
130     }

132     /**
133      * Return a new XML Datastream processor wrapping a new datastream.  The type (M or X) of the
134      * new datastream will be the same as the existing one.  Use to generate new versions of existing
135      * datastreams and wrap them in an XML datastream processor.
136      *
137      * @throws ServerInitializationException
138      * @throws ModuleInitializationException
139      */
140     public XMLDatastreamProcessor newVersion() throws ServerInitializationException, ModuleInitializationException {
141 
142         // create new datastream (version) based on existing datastream control group
143         Datastream ds;
144         if (m_dsType == DS_TYPE.INLINE_XML) {
145             ds = new DatastreamXMLMetadata();
146             ds.DSControlGrp = "X";
147         } else if (m_dsType == DS_TYPE.MANAGED) {
148             ds = new DatastreamManagedContent();
149             ds.DSControlGrp = "M";
150         } else {
151             // unhandled type is a coding error
152             throw new RuntimeException("XML datastreams must be of type Managed or Inline");
153         }
154         return new XMLDatastreamProcessor(ds);
155 
156     }

166     public byte[] getXMLContent(Context ctx) {
167         // could use getContentStream generically instead?
168         if (m_dsType == DS_TYPE.INLINE_XML)
169             return ((DatastreamXMLMetadata)m_ds).xmlContent;
170         else if (m_dsType == DS_TYPE.MANAGED)
171             try {
172                 if (ctx == null) {
173                     return IOUtils.toByteArray(m_ds.getContentStream());
174                 } else {
175                     return IOUtils.toByteArray(m_ds.getContentStream(ctx));
176                 }
177             } catch (IOException e) {
178                 throw new RuntimeException("Unable to read managed stream contents", e);
179             } catch (StreamIOException e) {
180                 throw new RuntimeException("Unable to read managed stream contents", e);
181             }
182             else
183                 throw new RuntimeException("XML datastreams must be of type Managed or Inline");
184 
185     }

195     /**
196      * Update the XML content of the datastream wrapped by this class
197      * @param xmlContent
198      */
199     public void setXMLContent(byte[] xmlContent) {
200         if (m_dsType == DS_TYPE.INLINE_XML) {
201             ((DatastreamXMLMetadata)m_ds).xmlContent = xmlContent;
202             ((DatastreamXMLMetadata)m_ds).DSSize = (xmlContent != null) ? xmlContent.length : -1;
203         } else if (m_dsType == DS_TYPE.MANAGED) {
204             ByteArrayInputStream bais = new ByteArrayInputStream(xmlContent);
205             MIMETypedStream s = new MIMETypedStream("text/xml", bais, null,xmlContent.length);
206             try {
207                 ((DatastreamManagedContent)m_ds).putContentStream(s);
208             } catch (StreamIOException e) {
209                 throw new RuntimeException("Unable to update managed datastream contents", e);
210             }
211         } else
212             // coding error if trying to use other datastream type
213             throw new RuntimeException("XML datastreams must be of type Managed or Inline");
214     }
