29  /**
30   * A class for parsing the special XML format in Fedora for a Method Map. A
31   * DSInputSpec exists within a Service Deployment (sDep) and a Service Definition Object (sDef). The Method Map defines abstract methods
32   * definitions. In a sDef these are the "behavior contract." In a sDep, these
33   * are abstract definitions that are then implemented by the service represented
34   * by the sDep.
35   * 
36   * @author Sandy Payette
37   */

79      /**
80       * Constructor allows this class to initiate the parsing
81       */
82      public MmapParser(String parentPID, InputStream in)
83              throws RepositoryConfigurationException, ObjectIntegrityException {
84          XMLReader xmlReader = null;
85          try {
86              SAXParserFactory saxfactory = SAXParserFactory.newInstance();
87              saxfactory.setValidating(false);
88              SAXParser parser = saxfactory.newSAXParser();
89              xmlReader = parser.getXMLReader();
90              xmlReader.setContentHandler(this);
91              xmlReader.setFeature("http://xml.org/sax/features/namespaces",
92                                   false);
93              xmlReader
94                      .setFeature("http://xml.org/sax/features/namespace-prefixes",
95                                  false);
96          } catch (Exception e) {
97              throw new RepositoryConfigurationException("Internal SAX error while "
98                      + "preparing for Method Map datastream parsing: "
99                      + e.getMessage());
100         }
101         try {
102             xmlReader.parse(new InputSource(in));
103         } catch (Exception e) {
104             throw new ObjectIntegrityException("Error parsing Method Map datastream"
105                     + e.getClass().getName() + ": " + e.getMessage());
106         }
107     }

130     @Override
131     public void skippedEntity(String name) throws SAXException {
132         StringBuffer sb = new StringBuffer();
133         sb.append('&');
134         sb.append(name);
135         sb.append(';');
136         char[] text = new char[sb.length()];
137         sb.getChars(0, sb.length(), text, 0);
138         characters(text, 0, text.length);

146     @Override
147     public void startElement(String namespaceURI,
148                              String localName,
149                              String qName,
150                              Attributes attrs) throws SAXException {
151         if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
152                 && localName.equalsIgnoreCase("MethodMap")) {
153             methodMap = new Mmap();
154             methodMap.mmapName = attrs.getValue("name");
155             tmp_methods = new Vector();
156         } else if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
157                 && localName.equalsIgnoreCase("Method")) {
158             inMethod = true;
159             methodMapMethod = new MmapMethodDef();
160             methodMapMethod.methodName = attrs.getValue("operationName");
161             methodMapMethod.methodLabel = "fix me";
162             methodMapMethod.wsdlOperationName = attrs.getValue("operationName");
163             methodMapMethod.wsdlMessageName = attrs.getValue("wsdlMsgName");
164             methodMapMethod.wsdlOutputMessageName =
165                     attrs.getValue("wsdlMsgOutput");
166             tmp_parms = new Vector();
167             wsdlMsgPartToParmDefTbl = new Hashtable();
168         } else if (inMethod) {
169             if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
170                     && localName.equalsIgnoreCase("DatastreamInputParm")) {
171                 methodMapParm = new MmapMethodParmDef();
172                 methodMapParm.wsdlMessagePartName = attrs.getValue("parmName");
173                 methodMapParm.parmName = attrs.getValue("parmName");
174                 methodMapParm.parmLabel = "fix me";
175                 methodMapParm.parmPassBy = attrs.getValue("passBy");
176                 methodMapParm.parmType = MethodParmDef.DATASTREAM_INPUT;
177                 if (attrs.getValue("required") == null) {
178                     methodMapParm.parmRequired = true;
179                 } else {
180                     methodMapParm.parmRequired =
181                             new Boolean(attrs.getValue("required"))
182                                     .booleanValue();
183                 }
184                 methodMapParm.parmDefaultValue = null;
185                 methodMapParm.parmDomainValues = new String[0];
186             } else if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
187                     && localName.equalsIgnoreCase("DefaultInputParm")) {
188                 methodMapParm = new MmapMethodParmDef();
189                 methodMapParm.wsdlMessagePartName = attrs.getValue("parmName");
190                 methodMapParm.parmName = attrs.getValue("parmName");
191                 methodMapParm.parmLabel = "fix me";
192                 methodMapParm.parmPassBy = MethodParmDef.PASS_BY_VALUE;
193                 methodMapParm.parmType = MethodParmDef.DEFAULT_INPUT;
194                 if (attrs.getValue("required") == null) {
195                     methodMapParm.parmRequired = true;
196                 } else {
197                     methodMapParm.parmRequired =
198                             new Boolean(attrs.getValue("required"))
199                                     .booleanValue();
200                 }
201                 methodMapParm.parmDefaultValue = attrs.getValue("defaultValue");
202                 methodMapParm.parmDomainValues = new String[0];
203             } else if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
204                     && localName.equalsIgnoreCase("UserInputParm")) {
205                 inUserInputParm = true;
206                 methodMapParm = new MmapMethodParmDef();
207                 methodMapParm.wsdlMessagePartName = attrs.getValue("parmName");
208                 methodMapParm.parmName = attrs.getValue("parmName");
209                 methodMapParm.parmLabel = "fix me";
210                 methodMapParm.parmPassBy = MethodParmDef.PASS_BY_VALUE;
211                 methodMapParm.parmType = MethodParmDef.USER_INPUT;
212                 if (attrs.getValue("required") == null) {
213                     methodMapParm.parmRequired = true;
214                 } else {
215                     methodMapParm.parmRequired =
216                             new Boolean(attrs.getValue("required"))
217                                     .booleanValue();
218                 }
219                 methodMapParm.parmDefaultValue = attrs.getValue("defaultValue");
220             } else if (inUserInputParm) {
221                 if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
222                         && localName.equalsIgnoreCase("ValidParmValues")) {
223                     tmp_enum = new Vector();
224                 } else if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
225                         && localName.equalsIgnoreCase("ValidParm")) {
226                     tmp_enum.add(attrs.getValue("value"));
227                 }
228             }
229         }

232     @Override
233     public void endElement(String namespaceURI, String localName, String qName)
234             throws SAXException {
235         if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
236                 && localName.equalsIgnoreCase("MethodMap")) {
237             methodMap.mmapMethods =
238                     (MmapMethodDef[]) tmp_methods.toArray(new MmapMethodDef[0]);
239             methodMap.wsdlOperationToMethodDef = wsdlOperationToMethodDefTbl;
240             tmp_methods = null;
241         } else if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
242                 && localName.equalsIgnoreCase("Method")) {
243             methodMapMethod.methodParms =
244                     (MethodParmDef[]) tmp_parms.toArray(new MethodParmDef[0]);
245             methodMapMethod.wsdlMsgParts =
246                     (MmapMethodParmDef[]) tmp_parms
247                             .toArray(new MmapMethodParmDef[0]);
248             methodMapMethod.wsdlMsgPartToParmDefTbl = wsdlMsgPartToParmDefTbl;
249             tmp_methods.add(methodMapMethod);
250             wsdlOperationToMethodDefTbl.put(methodMapMethod.methodName,
251                                             methodMapMethod);
252             wsdlMsgPartToParmDefTbl = null;
253             methodMapMethod = null;
254             tmp_parms = null;
255             inMethod = false;
256         } else if (inMethod) {
257             if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
258                     && localName.equalsIgnoreCase("DatastreamInputParm")) {
259                 tmp_parms.add(methodMapParm);
260                 wsdlMsgPartToParmDefTbl.put(methodMapParm.wsdlMessagePartName,
261                                             methodMapParm);
262                 methodMapParm = null;
263             } else if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
264                     && localName.equalsIgnoreCase("DefaultInputParm")) {
265                 tmp_parms.add(methodMapParm);
266                 wsdlMsgPartToParmDefTbl.put(methodMapParm.wsdlMessagePartName,
267                                             methodMapParm);
268                 methodMapParm = null;
269             } else if (namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
270                     && localName.equalsIgnoreCase("UserInputParm")) {
271                 tmp_parms.add(methodMapParm);
272                 wsdlMsgPartToParmDefTbl.put(methodMapParm.wsdlMessagePartName,
273                                             methodMapParm);
274                 methodMapParm = null;
275                 inUserInputParm = false;
276             } else if (inUserInputParm
277                     && namespaceURI.equalsIgnoreCase(METHOD_MAP.uri)
278                     && localName.equalsIgnoreCase("ValidParmValues")) {
279                 methodMapParm.parmDomainValues =
280                         (String[]) tmp_enum.toArray(new String[0]);
281                 tmp_enum = null;
282             }
283         }
284     }
