29  /**
30   * Parses WSDL document into a Service object. The Service object encapsulates
31   * all information about the service: the service name, the PortType (with
32   * abstract operation definitions), and one or more Ports (with operation
33   * binding information). This parser will not yet handle every pattern we may
34   * encounter in WSDL, therefore, it throws exceptions for WSDL elements that it
35   * does not yet support. The goal is to evolve the WSDLParser to be able to
36   * handle any kind of WSDL, and to have Fedora be able to play nicely with any
37   * kind of WSDL.
38   * 
39   * @author Sandy Payette
40   */
41  class WSDLParser
42          extends DefaultHandler
43          implements Constants {

140     public WSDLParser(InputStream in)
141             throws RepositoryConfigurationException, ObjectIntegrityException {
142         XMLReader xmlReader = null;
143         try {
144             SAXParserFactory saxfactory = SAXParserFactory.newInstance();
145             saxfactory.setValidating(false);
146             SAXParser parser = saxfactory.newSAXParser();
147             xmlReader = parser.getXMLReader();
148             xmlReader.setContentHandler(this);
149             xmlReader.setFeature("http://xml.org/sax/features/namespaces",
150                                  false);
151             xmlReader
152                     .setFeature("http://xml.org/sax/features/namespace-prefixes",
153                                 false);
154         } catch (Exception e) {
155             throw new RepositoryConfigurationException("Internal SAX error while "
156                     + "preparing for WSDL datastream parsing: "
157                     + e.getMessage());
158         }
159         try {
160             xmlReader.parse(new InputSource(in));
161         } catch (Exception e) {
162             throw new ObjectIntegrityException("Error parsing WSDL datastream"
163                     + e.getClass().getName() + ": " + e.getMessage());
164         }
165     }

171     public void startDocument() throws SAXException {
172         nsPrefixMap = new HashMap<String, String>();
173         wsdlTypeTbl = new Hashtable<String, SimpleType>();
174         wsdlMessageTbl = new Hashtable<String, Message>();
175         wsdlPortBindingTbl = new Hashtable<String, String>();
176         wsdlBindingTbl = new Hashtable<String, Binding>();
177         wsdlAbstrOperTbl = new Hashtable<String, AbstractOperation>();

186         wsdlAbstrOperTbl = null;

194     public void skippedEntity(String name) throws SAXException {
195         StringBuffer sb = new StringBuffer();
196         sb.append('&');
197         sb.append(name);
198         sb.append(';');
199         char[] text = new char[sb.length()];
200         sb.getChars(0, sb.length(), text, 0);
201         characters(text, 0, text.length);
202     }

208     public void startElement(String namespaceURI,
209                              String localName,
210                              String qName,
211                              Attributes attrs) throws SAXException {
212         // Look for things in the WSDL that this parser is not ready to support yet!
213         // Exception will be thrown and the parsing will stop if an unsupported
214         // pattern is found.
215         checkForUnsupportedPattern(namespaceURI, localName, qName, attrs);
216 
217         // Gather up all XML schema type definitions for the service.
218         if (namespaceURI.equalsIgnoreCase(WSDL.uri)
219                 && localName.equalsIgnoreCase("types")) {
220             inWSDLTypes = true;
221         } else if (inWSDLTypes) {
222             if (namespaceURI.equalsIgnoreCase(XML_XSD.uri)
223                     && localName.equalsIgnoreCase("simpleType")) {
224                 inSimpleType = true;
225                 wsdlSimpleType = new SimpleType();
226                 wsdlSimpleType.typeName = attrs.getValue("name");
227             } else if (inSimpleType) {
228                 if (namespaceURI.equalsIgnoreCase(XML_XSD.uri)
229                         && localName.equalsIgnoreCase("restriction")) {
230                     inRestriction = true;
231                     tmp_enum = new Vector<String>();
232                     wsdlSimpleType.baseTypeName = attrs.getValue("base");
233                     String nsprefix = null;
234                     StringTokenizer st =
235                             new StringTokenizer(wsdlSimpleType.baseTypeName,
236                                                 ":");
237                     if (st.hasMoreTokens()) {
238                         nsprefix = st.nextToken();
239                         wsdlSimpleType.baseTypeLocalName = st.nextToken();
240                     } else {
241                         wsdlSimpleType.baseTypeLocalName =
242                                 wsdlSimpleType.baseTypeName;
243                     }
244                     // FIXIT!! Test whether null nsprefix makes this bag.
245                     if (nsprefix.equalsIgnoreCase((String) nsPrefixMap
246                             .get(XML_XSD.uri))) {
247                         wsdlSimpleType.baseTypeNamespaceURI = XML_XSD.uri;
248                     } else {
249                         throw new SAXException("WSDLParser: base for simpleType cannot be other than XSD type"
250                                 + wsdlSimpleType.baseTypeName);
251                     }
252                 } else if (inRestriction
253                         && namespaceURI.equalsIgnoreCase(XML_XSD.uri)
254                         && localName.equalsIgnoreCase("enumeration")) {
255                     tmp_enum.add(attrs.getValue("value"));
256                 }
257             }
258         }
259 
260         // Second, parse WSDL message definitions that will be used in defining abstract operations.
261 
262         else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
263                 && localName.equalsIgnoreCase("message")) {
264             inMessage = true;
265             wsdlMessage = new Message();
266             wsdlMessage.messageName = attrs.getValue("name");
267             tmp_parts = new Vector<Part>();
268         } else if (inMessage && namespaceURI.equalsIgnoreCase(WSDL.uri)
269                 && localName.equalsIgnoreCase("part")) {
270             wsdlMessagePart = new Part();
271             wsdlMessagePart.partName = attrs.getValue("name");
272             wsdlMessagePart.partTypeName = attrs.getValue("type");
273         }
274 
275         // Third, process the WSDL portType for the abstract operations
276 
277         else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
278                 && localName.equalsIgnoreCase("portType")) {
279             inPortType = true;
280             wsdlPortType = new PortType();
281             wsdlPortType.portTypeName = attrs.getValue("name");
282             tmp_operations = new Vector<AbstractOperation>();
283         } else if (inPortType) {
284             if (namespaceURI.equalsIgnoreCase(WSDL.uri)
285                     && localName.equalsIgnoreCase("operation")) {
286                 inAbstractOperation = true;
287                 wsdlOperation = new AbstractOperation();
288                 wsdlOperation.operationName = attrs.getValue("name");
289             } else if (inAbstractOperation) {
290                 // FIXIT!! According to the WSDL XML schema, the input and output
291                 // elements can have an optional "name" attribute in addition to the
292                 // "message" attribute. I am ignoring the "name" attribute and just picking
293                 // up the message.  Consider if there are reasons to parse the "name" attribute.
294                 if (namespaceURI.equalsIgnoreCase(WSDL.uri)
295                         && localName.equalsIgnoreCase("input")) {
296                     wsdlOperation.inputMessage = new Message();
297                     wsdlOperation.inputMessage.messageName =
298                             attrs.getValue("message");
299                     StringTokenizer st =
300                             new StringTokenizer(wsdlOperation.inputMessage.messageName,
301                                                 ":");
302                     if (st.hasMoreTokens()) {
303                         st.nextToken();
304                         wsdlOperation.inputMessage.messageName = st.nextToken();
305                     }
306                 } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
307                         && localName.equalsIgnoreCase("output")) {
308                     wsdlOperation.outputMessage = new Message();
309                     wsdlOperation.outputMessage.messageName =
310                             attrs.getValue("message");
311                     StringTokenizer st =
312                             new StringTokenizer(wsdlOperation.outputMessage.messageName,
313                                                 ":");
314                     if (st.hasMoreTokens()) {
315                         st.nextToken();
316                         wsdlOperation.outputMessage.messageName =
317                                 st.nextToken();
318                     }
319                 }
320                 // FIXIT!! Parse the fault element.  Also note that according to the
321                 // WSDL XML schema, the "name" attribute on the fault element is NOT optional
322                 // like it is with input and output elements.
323             }
324         }
325 
326         // Get the Service and Port information
327         else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
328                 && localName.equalsIgnoreCase("service")) {
329             inService = true;
330             wsdlService = new Service();
331             wsdlService.serviceName = attrs.getValue("name");
332             tmp_ports = new Vector<Port>();
333         } else if (inService) {
334             if (namespaceURI.equalsIgnoreCase(WSDL.uri)
335                     && localName.equalsIgnoreCase("port")) {
336                 wsdlPort = new Port();
337                 wsdlPort.portName = attrs.getValue("name");
338                 tmp_portBindingName = attrs.getValue("binding");
339                 String nsprefix = null;
340                 tmp_portBindingLocalName = null;
341                 StringTokenizer st =
342                         new StringTokenizer(tmp_portBindingName, ":");
343                 if (st.hasMoreTokens()) {
344                     nsprefix = st.nextToken();
345                     tmp_portBindingLocalName = st.nextToken();
346                 } else {
347                     tmp_portBindingLocalName = tmp_portBindingName;
348                 }
349             } else if (namespaceURI.equalsIgnoreCase(WSDL_HTTP.uri)
350                     && localName.equalsIgnoreCase("address")) {
351                 wsdlPort.portBaseURL = attrs.getValue("location");
352             } else if (namespaceURI.equalsIgnoreCase(SOAP.uri)
353                     && localName.equalsIgnoreCase("address")) {
354                 wsdlPort.portBaseURL = attrs.getValue("location");
355             }
356         }
357 
358         // Get the Operation Bindings
359 
360         else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
361                 && localName.equalsIgnoreCase("binding")) {
362             inBinding = true;
363             tmp_bindingName = attrs.getValue("name");
364             tmp_bindingPortTypeName = attrs.getValue("type");
365             tmp_bindingPortTypeLocalName = null;
366             StringTokenizer st =
367                     new StringTokenizer(tmp_bindingPortTypeName, ":");
368             if (st.hasMoreTokens()) {
369                 String nsprefix = st.nextToken();
370                 tmp_bindingPortTypeLocalName = st.nextToken();
371             } else {
372                 tmp_bindingPortTypeLocalName = tmp_bindingPortTypeName;
373             }
374         } else if (inBinding) {
375             if (namespaceURI.equalsIgnoreCase(WSDL_HTTP.uri)
376                     && localName.equalsIgnoreCase("binding")) {
377                 wsdlBinding = new HTTPBinding();
378                 wsdlBinding.bindingName = tmp_bindingName;
379                 wsdlBinding.portTypeLocalName = tmp_bindingPortTypeLocalName;
380                 ((HTTPBinding) wsdlBinding).bindingVerb =
381                         attrs.getValue("verb");
382                 tmp_bindOperations = new Vector<AbstractOperation>();
383             } else if (namespaceURI.equalsIgnoreCase(SOAP.uri)
384                     && localName.equalsIgnoreCase("binding")) {
385                 wsdlBinding = new SOAPBinding();
386                 wsdlBinding.bindingName = tmp_bindingName;
387                 wsdlBinding.portTypeLocalName = tmp_bindingPortTypeLocalName;
388                 ((SOAPBinding) wsdlBinding).bindingStyle =
389                         attrs.getValue("style");
390                 ((SOAPBinding) wsdlBinding).bindingTransport =
391                         attrs.getValue("transport");
392                 tmp_bindOperations = new Vector<AbstractOperation>();
393             } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
394                     && localName.equalsIgnoreCase("operation")) {
395                 tmp_operationName = attrs.getValue("name");
396             } else if (namespaceURI.equalsIgnoreCase(WSDL_HTTP.uri)
397                     && localName.equalsIgnoreCase("operation")) {
398                 wsdlOperation = new HTTPOperation();
399                 wsdlOperation.operationName = tmp_operationName;
400                 ((HTTPOperation) wsdlOperation).operationLocation =
401                         attrs.getValue("location");
402                 isHTTPOperation = true;
403             } else if (namespaceURI.equalsIgnoreCase(SOAP.uri)
404                     && localName.equalsIgnoreCase("operation")) {
405                 wsdlOperation = new SOAPOperation();
406                 wsdlOperation.operationName = tmp_operationName;
407                 ((SOAPOperation) wsdlOperation).soapAction =
408                         attrs.getValue("soapAction");
409                 ((SOAPOperation) wsdlOperation).soapActionStyle =
410                         attrs.getValue("style");
411                 isSOAPOperation = true;
412             } else if (isHTTPOperation) {
413                 if (namespaceURI.equalsIgnoreCase(WSDL.uri)
414                         && localName.equalsIgnoreCase("input")) {
415                     inInput = true;
416                     wsdlHTTPOpInOut = new HTTPOperationInOut();
417                     tmp_MIMEContent = new Vector<MIMEContent>();
418                 } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
419                         && localName.equalsIgnoreCase("output")) {
420                     inOutput = true;
421                     wsdlHTTPOpInOut = new HTTPOperationInOut();
422                     tmp_MIMEContent = new Vector<MIMEContent>();
423                 } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
424                         && localName.equalsIgnoreCase("fault")) {
425                     // FIXIT!! do something!!
426                 } else if (inInput) {
427                     if (namespaceURI.equalsIgnoreCase(WSDL_HTTP.uri)
428                             && localName.equalsIgnoreCase("urlReplacement")) {
429                         wsdlHTTPOpInOut.ioBindingType =
430                                 HTTPOperationInOut.URL_REPLACE_BINDING_TYPE;
431                         wsdlHTTPOpInOut.ioMIMEContent = null;
432                     } else if (namespaceURI.equalsIgnoreCase(WSDL_MIME.uri)) {
433                         wsdlHTTPOpInOut.ioBindingType =
434                                 HTTPOperationInOut.MIME_BINDING_TYPE;
435                         tmp_MIMEContent.add(parseMIMEBinding(namespaceURI,
436                                                              localName,
437                                                              qName,
438                                                              attrs));
439                     } else {
440                         // FIXIT!!  Probably unnecessary, but...
441                         throw new SAXException("WSDLParser: Found an input extension element on HTTP operation "
442                                 + "that Fedora does not yet support: " + qName);
443                     }
444                 } else if (inOutput) {
445                     if (namespaceURI.equalsIgnoreCase(WSDL_MIME.uri)) {
446                         wsdlHTTPOpInOut.ioBindingType =
447                                 HTTPOperationInOut.MIME_BINDING_TYPE;
448                         tmp_MIMEContent.add(parseMIMEBinding(namespaceURI,
449                                                              localName,
450                                                              qName,
451                                                              attrs));
452                     } else {
453                         // FIXIT!!  Probably unnecessary, but...
454                         throw new SAXException("WSDLParser: Found an output extension element on HTTP operation "
455                                 + "that Fedora does not yet support: " + qName);
456                     }
457                 }
458             } else if (isSOAPOperation) {
459                 // FIXIT!! implement something
460             }
461         }
462     }

464     public void endElement(String namespaceURI, String localName, String qName)
465             throws SAXException {
466         // Type post-processing
467         if (inWSDLTypes) {
468             if (inSimpleType) {
469                 if (inRestriction) {
470                     if (namespaceURI.equalsIgnoreCase(XML_XSD.uri)
471                             && localName.equalsIgnoreCase("restriction")) {
472                         wsdlSimpleType.enumerationOfValues = tmp_enum;
473                         tmp_enum = null;
474                         inRestriction = false;
475                     }
476                 } else if (namespaceURI.equalsIgnoreCase(XML_XSD.uri)
477                         && localName.equalsIgnoreCase("simpleType")) {
478                     wsdlTypeTbl.put(wsdlSimpleType.typeName, wsdlSimpleType);
479                     wsdlSimpleType = null;
480                     inSimpleType = false;
481                 }
482             } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
483                     && localName.equalsIgnoreCase("types")) {
484                 inWSDLTypes = false;
485             }
486         }
487         // Message post-processing
488         else if (inMessage) {
489             if (namespaceURI.equalsIgnoreCase(WSDL.uri)
490                     && localName.equalsIgnoreCase("part")) {
491                 tmp_parts.add(wsdlMessagePart);
492                 wsdlMessagePart = null;
493             }
494 
495             else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
496                     && localName.equalsIgnoreCase("message")) {
497                 wsdlMessage.messageParts =
498                         (Part[]) tmp_parts.toArray(new Part[0]);
499                 wsdlMessageTbl.put(wsdlMessage.messageName, wsdlMessage);
500                 wsdlMessage = null;
501                 tmp_parts = null;
502                 inMessage = false;
503             }
504         }
505         // PortType post-processing
506         else if (inPortType) {
507             if (inAbstractOperation) {
508                 if (namespaceURI.equalsIgnoreCase(WSDL.uri)
509                         && localName.equalsIgnoreCase("operation")) {
510                     tmp_operations.add(wsdlOperation);
511                     wsdlOperation = null;
512                     inAbstractOperation = false;
513                 }
514             } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
515                     && localName.equalsIgnoreCase("portType")) {
516                 wsdlPortType.operations =
517                         (AbstractOperation[]) tmp_operations
518                                 .toArray(new AbstractOperation[0]);
519                 tmp_operations = null;
520                 inPortType = false;
521             }
522         }
523         // Service post-processing
524         else if (inService) {
525             if (namespaceURI.equalsIgnoreCase(WSDL.uri)
526                     && localName.equalsIgnoreCase("port")) {
527                 tmp_ports.add(wsdlPort);
528                 wsdlPortBindingTbl.put(wsdlPort.portName,
529                                        tmp_portBindingLocalName);
530                 wsdlPort = null;
531                 tmp_portBindingLocalName = null;
532                 tmp_portBindingName = null;
533             } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
534                     && localName.equalsIgnoreCase("service")) {
535                 wsdlService.ports = (Port[]) tmp_ports.toArray(new Port[0]);
536                 tmp_ports = null;
537                 inService = false;
538             }
539         }
540         // Binding post-processing
541         else if (inBinding) {
542             if (isHTTPOperation) {
543                 if (namespaceURI.equalsIgnoreCase(WSDL.uri)
544                         && localName.equalsIgnoreCase("input")) {
545                     wsdlHTTPOpInOut.ioMIMEContent =
546                             (MIMEContent[]) tmp_MIMEContent
547                                     .toArray(new MIMEContent[0]);
548                     ((HTTPOperation) wsdlOperation).inputBinding =
549                             wsdlHTTPOpInOut;
550                     tmp_MIMEContent = null;
551                     wsdlHTTPOpInOut = null;
552                     inInput = false;
553                 } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
554                         && localName.equalsIgnoreCase("output")) {
555                     wsdlHTTPOpInOut.ioMIMEContent =
556                             (MIMEContent[]) tmp_MIMEContent
557                                     .toArray(new MIMEContent[0]);
558                     ((HTTPOperation) wsdlOperation).outputBinding =
559                             wsdlHTTPOpInOut;
560                     tmp_MIMEContent = null;
561                     wsdlHTTPOpInOut = null;
562                     inOutput = false;
563                 } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
564                         && localName.equalsIgnoreCase("operation")) {
565                     tmp_bindOperations.add(wsdlOperation);
566                     tmp_operationName = null;
567                     isHTTPOperation = false;
568                     wsdlOperation = null;
569                 }
570             } else if (isSOAPOperation) {
571                 // FIXIT!! implement something!!!
572             } else if (namespaceURI.equalsIgnoreCase(WSDL.uri)
573                     && localName.equalsIgnoreCase("binding")) {
574                 if (wsdlBinding
575                         .getClass()
576                         .getName()
577                         .equalsIgnoreCase("org.fcrepo.server.storage.service.HTTPBinding")) {
578                     ((HTTPBinding) wsdlBinding).operations =
579                             (HTTPOperation[]) tmp_bindOperations
580                                     .toArray(new HTTPOperation[0]);
581                     tmp_bindOperations = null;
582                 } else if (wsdlBinding
583                         .getClass()
584                         .getName()
585                         .equalsIgnoreCase("org.fcrepo.server.storage.service.SOAPBinding")) {
586                     // FIXIT!! Finish up class definition for SOAPOperations.
587                     ((SOAPBinding) wsdlBinding).operations =
588                             (SOAPOperation[]) tmp_bindOperations
589                                     .toArray(new SOAPOperation[0]);
590                     tmp_bindOperations = null;
591                 }
592                 wsdlBindingTbl.put(wsdlBinding.bindingName, wsdlBinding);
593                 wsdlBinding = null;
594                 tmp_bindingName = null;
595                 tmp_bindingPortTypeName = null;
596                 tmp_bindingPortTypeLocalName = null;
597                 inBinding = false;
598             }
599         }
600     }

602     private MIMEContent parseMIMEBinding(String namespaceURI,
603                                          String localName,
604                                          String qName,
605                                          Attributes attrs) throws SAXException {
606         MIMEContent content = new MIMEContent();
607         content.elementType = localName;
608         content.messagePartName = attrs.getValue("part");
609 
610         if (namespaceURI.equalsIgnoreCase(WSDL_MIME.uri)
611                 && localName.equalsIgnoreCase("content")) {
612             content.mimeType = attrs.getValue("type");
613         } else if (namespaceURI.equalsIgnoreCase(WSDL_MIME.uri)
614                 && localName.equalsIgnoreCase("mimeXml")) {
615             content.mimeType = "text/xml";
616         } else {
617             throw new SAXException("WSDLParser: Detected a MIME Binding that Fedora does not yet support:"
618                     + qName);
619         }
620         return content;
621     }

623     private void doServiceJoins() throws SAXException {
624         doPortTypeToMessageJoin();
625         doPortToMessageJoin();
626     }

628     private void doPortTypeToMessageJoin() throws SAXException {
629         // For each Abstract Operation, get the input and output messages.
630         // Each Abstract Operation has input and output Message objects, but the
631         // Message objects only have the message name filled in. The full message
632         // information will be added via a lookup to the wsdlMessageTbl which was
633         // created when the WSDL message definitions were parsed.
634         // But, the messages also need some augmentation.  We must lookup the
635         // schema type information for the message parts.  This is done via the
636         // doMessageToTypeJoin operation.
637         AbstractOperation[] absOps = wsdlPortType.operations;
638         for (int i = 0; i < absOps.length; i++) {
639             // FIXIT!! What if either an input message or an output message does not exist on an abstract operation??
640             absOps[i].inputMessage =
641                     doMessageToTypeJoin((Message) wsdlMessageTbl
642                             .get(absOps[i].inputMessage.messageName));
643             wsdlAbstrOperTbl.put(absOps[i].operationName, absOps[i]);
644             absOps[i].outputMessage =
645                     doMessageToTypeJoin((Message) wsdlMessageTbl

686     /**
687      * For each Message, iterate through Message Parts. For each MessagePart,
688      * get partTypeName and break into ns/localName. Lookup to wsdlTypeTbl to
689      * get type info for parts.
690      */
691     private Message doMessageToTypeJoin(Message msg) throws SAXException {
692         for (int i = 0; i < msg.messageParts.length; i++) {
693             Type schemaType = null;
694             String nsprefix = null;
695             String partTypeLocalName = null;
696             StringTokenizer st =
697                     new StringTokenizer(msg.messageParts[i].partTypeName, ":");
698             if (st.hasMoreTokens()) {
699                 nsprefix = st.nextToken();
700                 partTypeLocalName = st.nextToken();
701             } else {
702                 partTypeLocalName = msg.messageParts[i].partTypeName;
703             }
704             // FIXIT!! Test whether null nsprefix makes this bag.
705             if (nsprefix
706                     .equalsIgnoreCase((String) nsPrefixMap.get(XML_XSD.uri))) {
707                 msg.messageParts[i].partBaseTypeNamespaceURI = XML_XSD.uri;
708                 msg.messageParts[i].partBaseTypeLocalName = partTypeLocalName;
709             }
710             // If the part is not an XSD base type, then we assume it is a type defined
711             // in the schema definition section of THIS WSDL document (i.e., either
712             // a simpleType or a complexType).  NOTE that complexType not currently supported
713             // by Fedora.  Also, note that if the part type is not found it the type table
714             // (wsdlTypeTbl was created from parsing the schema types section of the WSDL),
715             // then an exception will be thrown.  This means that Fedora only supports WSDL
716             // whose message parts are either XSD base types or simpleTypes defined in the WSDL document.
717             else if ((schemaType = (Type) wsdlTypeTbl.get(partTypeLocalName)) != null) {
718                 if (schemaType
719                         .getClass()
720                         .getName()
721                         .equalsIgnoreCase("org.fcrepo.server.storage.service.SimpleType")) {
722                     msg.messageParts[i].partBaseTypeNamespaceURI =
723                             schemaType.baseTypeNamespaceURI;
724                     msg.messageParts[i].partBaseTypeLocalName =
725                             schemaType.baseTypeLocalName;
726                     msg.messageParts[i].enumerationOfValues =
727                             ((SimpleType) schemaType).enumerationOfValues;
728                     // FIXIT!! Do we want to put more information in the Part object??
729                 }
730             } else {
731                 throw new SAXException("WSDLParser: message part type must be XSD type or defined in WSDL as simpleType"
732                         + msg.messageParts[i].partTypeName);
733             }
734         }
735         return msg;
736     }
