171      * {@inheritDoc}
172      */
173     @Override
174     public InputStream Export(String format, String exportContext)
175             throws ObjectIntegrityException, StreamIOException,
176                    UnsupportedTranslationException, ServerException {
177         ByteArrayOutputStream bytes = new ByteArrayOutputStream();
178         int transContext;
179         // first, set the translation context...
180         logger.debug("Export context: " + exportContext);
181 
182         if (exportContext == null || exportContext.equals("")
183             || exportContext.equalsIgnoreCase("default")) {
184             // null and default is set to PUBLIC translation
185             transContext = DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC;
186         } else if (exportContext.equalsIgnoreCase("public")) {
187             transContext = DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC;
188         } else if (exportContext.equalsIgnoreCase("migrate")) {
189             transContext = DOTranslationUtility.SERIALIZE_EXPORT_MIGRATE;
190         } else if (exportContext.equalsIgnoreCase("archive")) {
191             transContext = DOTranslationUtility.SERIALIZE_EXPORT_ARCHIVE;
192         } else {
193             throw new UnsupportedTranslationException("Export context "
194                                                       + exportContext + " is not valid.");
195         }
196         // now serialize for export in the proper XML format...
197         if (format == null || format.equals("")
198             || format.equalsIgnoreCase("default")) {
199             logger.debug("Export in default format: " + m_exportFormat);
200             m_translator.serialize(m_obj,
201                                    bytes,
202                                    m_exportFormat,
203                                    "UTF-8",
204                                    transContext);
205         } else {
206             logger.debug("Export in format: " + format);
207             m_translator.serialize(m_obj, bytes, format, "UTF-8", transContext);
208         }
209 
210         return new ByteArrayInputStream(bytes.toByteArray());
211     }

270     @Override
271     public String[] ListDatastreamIDs(String state) {
272         Iterator<String> iter = m_obj.datastreamIdIterator();
273         ArrayList<String> al = new ArrayList<String>();
274         while (iter.hasNext()) {
275             String dsId = iter.next();
276             if (state == null) {
277                 al.add(dsId);
278             } else {
279                 // below should never return null -- already know id exists,
280                 // and am asking for any the latest existing one.
281                 Datastream ds = GetDatastream(dsId, null);
282                 if (ds.DSState.equals(state)) {
283                     al.add(dsId);
284                 }
285             }
286         }
287         iter = al.iterator();
288         String[] out = new String[al.size()];
289         int i = 0;
290         while (iter.hasNext()) {
291             out[i] = iter.next();
292             i++;
293         }
294         return out;
295     }

310     /**
311      * {@inheritDoc}
312      */
313     @Override
314     public Datastream GetDatastream(String datastreamID, Date versDateTime) {
315         // get the one with the closest creation date
316         // without going over
317         Datastream result = null;
318         long bestTimeDifference = Long.MAX_VALUE;
319         long latestCreateTime = -1;
320         long vTime = -1;
321         if (versDateTime != null) {
322             vTime = versDateTime.getTime();
323         }
324         for (Datastream ds : m_obj.datastreams(datastreamID)) {
325             if (versDateTime == null) {
326                 if (ds.DSCreateDT == null || ds.DSCreateDT.getTime() > latestCreateTime || result == null) {
327                     result = ds;
328                     if (ds.DSCreateDT != null) latestCreateTime = ds.DSCreateDT.getTime();
329                 }
330             } else {
331                 //TODO If none of the versions have a create date, what should behavior be?
332                 long diff = (ds.DSCreateDT == null)? vTime : vTime - ds.DSCreateDT.getTime();
333                 if (diff >= 0) {
334                     if (diff < bestTimeDifference) {
335                         bestTimeDifference = diff;
336                         result = ds;
337                     }
338                 }
339             }
340         }
341         return result;
342     }

396      * {@inheritDoc}
397      */
398     private MethodDef[] listMethods(String sDefPID,
399                                     ServiceDefinitionReader sDefReader,
400                                     Date versDateTime)
401             throws MethodNotFoundException, ServerException {
402         if (sDefPID.equalsIgnoreCase("fedora-system:1")
403             || sDefPID.equalsIgnoreCase("fedora-system:3")) {
404             throw new MethodNotFoundException("[getObjectMethods] The object, "
405                                               + m_obj.getPid()
406                                               + ", will not report on dynamic method definitions "
407                                               + "at this time (fedora-system:1 and fedora-system:3.");
408         }
409 
410         if (sDefReader == null) {
411             return null;
412         }
413         MethodDef[] methods = sDefReader.getAbstractMethods(versDateTime);
414         // Filter out parms that are internal to the mechanism and not part
415         // of the abstract method definition. We just want user parms.
416         for (int i = 0; i < methods.length; i++) {
417             methods[i].methodParms = filterParms(methods[i]);
418         }
419         return methods;
420     }
421 
422     /**
423      * Filters out mechanism-specific parms (system default parms and datastream
424      * input parms) so that what is returned is only method parms that reflect
425      * abstract method definitions. Abstract method definitions only expose
426      * user-supplied parms.
427      *
428      * @param method
429      * @return
430      */
431     private MethodParmDef[] filterParms(MethodDef method) {
432         ArrayList<MethodParmDef> filteredParms = new ArrayList<MethodParmDef>();
433         for (MethodParmDef element : method.methodParms) {
434             if (element.parmType.equalsIgnoreCase(MethodParmDef.USER_INPUT)) {
435                 filteredParms.add(element);
436             }
437         }
438         return filteredParms.toArray(new MethodParmDef[0]);
439     }
