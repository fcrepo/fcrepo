39      @Override
40      public void postInitModule() throws ModuleInitializationException {
41          try {
42              m_llstore = new DefaultLowlevelStorage(getModuleParameters());
43          } catch (LowlevelStorageException e) {
44              throw new ModuleInitializationException(e.getMessage(), getRole());
45          }
46      }

48      protected Map<String, Object> getModuleParameters() throws ModuleInitializationException {
49          // Parameter validation
50          String objectStoreBase =
51                  getModuleParameter(DefaultLowlevelStorage.OBJECT_STORE_BASE,
52                                     true);
53          String datastreamStoreBase =
54                  getModuleParameter(DefaultLowlevelStorage.DATASTREAM_STORE_BASE,
55                                     true);
56          // FIXME object/datastream store location sanity checks (e.g. no overlapping)
57  
58          String filesystem =
59                  getModuleParameter(DefaultLowlevelStorage.FILESYSTEM, false);
60          String pathAlgorithm =
61                  getModuleParameter(DefaultLowlevelStorage.PATH_ALGORITHM, false);
62          String pathRegistry =
63                  getModuleParameter(DefaultLowlevelStorage.PATH_REGISTRY, false);
64  
65          // parameter required by DBPathRegistry
66          String backslashIsEscape;
67          String param =
68                  getModuleParameter("backslash_is_escape", false).toLowerCase();
69          if (param.equals("true") || param.equals("false")) {
70              backslashIsEscape = param;
71          } else {
72              throw new ModuleInitializationException("backslash_is_escape parameter must be either true or false",
73                                                      getRole());
74          }
75  
76          // get connectionPool from ConnectionPoolManager
77          ConnectionPoolManager cpm =
78                  (ConnectionPoolManager) getServer()
79                          .getModule("org.fcrepo.server.storage.ConnectionPoolManager");
80          if (cpm == null) {
81              throw new ModuleInitializationException("ConnectionPoolManager module was required, but apparently has "
82                                                              + "not been loaded.",
83                                                      getRole());
84          }
85  
86          ConnectionPool cPool;
87          try {
88              cPool = cpm.getPool();
89          } catch (ConnectionPoolNotFoundException e1) {
90              throw new ModuleInitializationException("Could not find requested "
91                      + "connectionPool.", getRole());
92          }
93  
94          Map<String, Object> configuration = new HashMap<String, Object>();
95          configuration.put(DefaultLowlevelStorage.FILESYSTEM, filesystem);
96          configuration.put(DefaultLowlevelStorage.PATH_ALGORITHM, pathAlgorithm);
97          configuration.put(DefaultLowlevelStorage.PATH_REGISTRY, pathRegistry);
98          configuration.put(DefaultLowlevelStorage.OBJECT_STORE_BASE,
99                            objectStoreBase);
100         configuration.put(DefaultLowlevelStorage.DATASTREAM_STORE_BASE,
101                           datastreamStoreBase);
102         configuration.put("connectionPool", cPool);
103         configuration.put("backslashIsEscape", backslashIsEscape);
104 
105         return configuration;
106     }

108     protected String getModuleParameter(String parameterName,
109                                         boolean parameterAsAbsolutePath)
110             throws ModuleInitializationException {
111         String parameterValue =
112                 getParameter(parameterName, parameterAsAbsolutePath);
113 
114         if (parameterValue == null) {
115             throw new ModuleInitializationException(parameterName
116                     + " parameter must be specified", getRole());
117         }
118         return parameterValue;
119     }
