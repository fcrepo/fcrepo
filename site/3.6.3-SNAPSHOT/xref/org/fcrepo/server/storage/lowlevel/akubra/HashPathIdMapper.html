17  /**
18   * Provides a hash-based <code>file:</code> mapping for any URI.
19   * <p>
20   * The path component of each internal URI is derived from an MD5 hash of
21   * the external URI. The filename component is a reversible encoding of the
22   * external URI that is safe to use as a filename on modern filesystems.
23   * <p>
24   * <h2>Hash Path Patterns</h2>
25   * The pattern given at construction time determines how the path component
26   * of each internal URI will be composed. Within the pattern, the # character
27   * is a stand-in for a hexadecimal [0-f] digit from the MD5 hash of the
28   * external id.
29   * <p>
30   * Patterns:
31   * <ul>
32   *   <li> must consist only of # and / characters.</li>
33   *   <li> must contain between 1 and 32 # characters.</li>
34   *   <li> must not begin or end with the / character.</li>
35   *   <li> must not contain consecutive / characters.</li>
36   * </ul>
37   * <p>
38   * Example patterns:
39   * <ul>
40   *   <li> Good: #</li>
41   *   <li> Good: ##/#</li>
42   *   <li> Good: ##/##/##</li>
43   *   <li> Bad: a</li>
44   *   <li> Bad: ##/</li>
45   *   <li> Bad: ##//##</li>
46   * </ul>
47   * <p>
48   * <h2>Filesystem-Safe Encoding</h2>
49   * The last part of the internal URI is a "filesystem-safe" encoding of the
50   * external URI. All characters will be UTF-8 percent-encoded ("URI escaped")
51   * except for the following: <code>a-z A-Z 0-9 = ( ) [ ] -</code>
52   * In addition, <code>.</code> (period) will be escaped as <code>%2E</code> when
53   * it occurs as the last character of the URI.
54   * <p>
55   * <h2>Example Mappings</h2>
56   * With pattern <em>#/#</em>:
57   * <ul>
58   *   <li> <code>urn:example1</code> becomes <code>file:0/8/urn%3Aexample1</code></li>
59   *   <li> <code>http://tinyurl.com/cxzzf</code> becomes <code>file:6/2/http%3A%2F%2Ftinyurl.com%2Fcxzzf</code></li>
60   * </ul>
61   * With pattern <em>##/##</em>:
62   * <ul>
63   *   <li> <code>urn:example1</code> becomes <code>file:08/86/urn%3Aexample1</code></li>
64   *   <li> <code>http://tinyurl.com/cxzzf</code> becomes <code>file:62/ca/http%3A%2F%2Ftinyurl.com%2Fcxzzf</code></li>
65   * </ul>
66   *
67   * @author Chris Wilper
68   */
69  public class HashPathIdMapper
70          implements IdMapper {
71  
72      private static final String internalScheme = "file";
73  
74      private final String pattern;
75  
76      /**
77       * Creates an instance that will use the given pattern.
78       *
79       * @param pathPattern the pattern to use, possibly <code>null</code> or "".
80       * @throws IllegalArgumentException if the pattern is invalid.
81       */
82      public HashPathIdMapper(String pattern) {
83          this.pattern = validatePattern(pattern);
84      }

86      //@Override
87      public URI getExternalId(URI internalId) throws NullPointerException {
88          String fullPath = internalId.toString().substring(
89                  internalScheme.length() + 1);
90          int i = fullPath.lastIndexOf('/');
91          String encodedURI;
92          if (i == -1)
93              encodedURI = fullPath;
94          else
95              encodedURI = fullPath.substring(i + 1);
96          return URI.create(decode(encodedURI));
97      }
98  
99      //@Override
100     public URI getInternalId(URI externalId) throws NullPointerException {
101         if (externalId == null) {
102             throw new NullPointerException();
103         }
104         String uri = externalId.toString();
105         return URI.create(internalScheme + ":" + getPath(uri) + encode(uri));
106     }
107 
108     //@Override
109     public String getInternalPrefix(String externalPrefix)
110             throws NullPointerException {
111         if (externalPrefix == null) {
112             throw new NullPointerException();
113         }
114         // we can only do this if pattern is ""
115         if (pattern.length() == 0) {
116             return internalScheme + ":" + encode(externalPrefix);
117         } else {
118             return null;
119         }
120     }

122     // gets the path based on the hash of the uri, or "" if the pattern is empty
123     private String getPath(String uri) {
124         if (pattern.length() == 0) {
125             return "";
126         }
127         StringBuilder builder = new StringBuilder();
128         String hash = getHash(uri);
129         int hashPos = 0;
130         for (int i = 0; i < pattern.length(); i++) {
131             char c = pattern.charAt(i);
132             if (c == '#') {
133                 builder.append(hash.charAt(hashPos++));
134             } else {
135                 builder.append(c);
136             }
137         }
138         builder.append('/');
139         return builder.toString();
140     }
141 
142     // computes the md5 and returns a 32-char lowercase hex string
143     private static String getHash(String uri) {
144         return MD5Utility.getBase16Hash(uri);
145     }
146 
147     private static String encode(String uri) {
148         // encode char-by-char because we only want to borrow
149         // URLEncoder.encode's behavior for some characters
150         StringBuilder out = new StringBuilder();
151         for (int i = 0; i < uri.length(); i++) {
152             char c = uri.charAt(i);
153             if (c >= 'a' && c <= 'z') {
154                 out.append(c);
155             } else if (c >= '0' && c <= '9') {
156                 out.append(c);
157             } else if (c >= 'A' && c <= 'Z') {
158                 out.append(c);
159             } else if (c == '-' || c == '=' || c == '(' || c == ')'
160                     || c == '[' || c == ']' || c == ';') {
161                 out.append(c);
162             } else if (c == ':') {
163                 out.append("%3A");
164             } else if (c == ' ') {
165                 out.append("%20");
166             } else if (c == '+') {
167                 out.append("%2B");
168             } else if (c == '_') {
169                 out.append("%5F");
170             } else if (c == '*') {
171                 out.append("%2A");
172             } else if (c == '.') {
173                 if (i == uri.length() - 1) {
174                     out.append("%2E");
175                 } else {
176                     out.append(".");
177                 }
178             } else {
179                 try {
180                     out.append(URLEncoder.encode("" + c, "UTF-8"));
181                 } catch (UnsupportedEncodingException wontHappen) {
182                     throw new RuntimeException(wontHappen);
183                 }
184             }
185         }
186         return out.toString();
187     }

189     private static String decode(String encodedURI) {
190         if (encodedURI.endsWith("%2E")) {
191             encodedURI = encodedURI.substring(0, encodedURI.length() - 3) + ".";
192         }
193         try {
194             return URLDecoder.decode(encodedURI, "UTF-8");
195         } catch (UnsupportedEncodingException wontHappen) {
196             throw new RuntimeException(wontHappen);
197         }
198     }
199 
200     private static String validatePattern(String pattern) {
201         if (pattern == null) {
202             return "";
203         }
204         int count = 0;
205         boolean prevWasSlash = false;
206         for (int i = 0; i < pattern.length(); i++) {
207             char c = pattern.charAt(i);
208             if (c == '#') {
209                 count++;
210                 prevWasSlash = false;
211             } else if (c == '/') {
212                 if (i == 0 || i == pattern.length() - 1) {
213                     throw new IllegalArgumentException("Pattern must not begin"
214                             + " or end with '/'");
215                 } else if (prevWasSlash) {
216                     throw new IllegalArgumentException("Pattern must not"
217                             + " contain consecutive '/' characters");
218                 } else {
219                     prevWasSlash = true;
220                 }
221             } else {
222                 throw new IllegalArgumentException("Illegal character in"
223                         + " pattern: " + c);
224             }
225         }
226         if (count > 32) {
227             throw new IllegalArgumentException("Pattern must not contain more"
228                     + " than 32 '#' characters");
229         }
230         return pattern;
231     }
