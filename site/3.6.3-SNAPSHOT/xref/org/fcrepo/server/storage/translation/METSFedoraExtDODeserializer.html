229     /**
230      * Creates a deserializer that reads the given Fedora METS Extension format.
231      *
232      * @param format
233      *        the version-specific Fedora METS Extension format.
234      * @throws IllegalArgumentException
235      *         if format is not a known Fedora METS Extension format.
236      */
237     public METSFedoraExtDODeserializer(XMLFormat format) {
238         if (format.equals(METS_EXT1_0)) {
239             m_xlink = OLD_XLINK;
240         } else if (format.equals(METS_EXT1_1)) {
241             m_xlink = XLINK;
242         } else {
243             throw new IllegalArgumentException("Not a METSFedoraExt format: "
244                     + format.uri);
245         }
246         m_format = format;
247     }

260     /**
261      * {@inheritDoc}
262      */
263     public void deserialize(InputStream in,
264                             DigitalObject obj,
265                             String encoding,
266                             int transContext) throws ObjectIntegrityException,
267             StreamIOException, UnsupportedEncodingException {
268         logger.debug("Deserializing " + m_format.uri + " for transContext: "
269                 + transContext);
270 
271         // initialize sax for this parse
272         try {
273             m_parser = spf.newSAXParser();
274         } catch (Exception e) {
275             throw new RuntimeException("Error initializing SAX parser", e);
276         }
277 
278         m_obj = obj;
279         m_obj.setOwnerId("");
280         m_obj.setLabel("");
281         m_characterEncoding = encoding;
282         m_transContext = transContext;
283         initialize();
284         try {
285             m_parser.parse(in, this);
286         } catch (IOException ioe) {
287             throw new StreamIOException("Low-level stream IO problem occurred "
288                     + "while SAX parsing this object.");
289         } catch (SAXException se) {
290             throw new ObjectIntegrityException("METS stream was bad : "
291                     + se.getMessage());
292         }
293         if (!m_rootElementFound) {
294             throw new ObjectIntegrityException("METS root element not found");
295         }
296 
297         // POST-PROCESSING...
298         // convert audit records to contain component ids
299         convertAudits();
300         // preserve ADMID and DMDID relationships in a RELS-INT
301         // datastream, if one does not already exist.
302         createRelsInt();
303 
304         DOTranslationUtility.normalizeDatastreams(m_obj,
305                                                   m_transContext,
306                                                   m_characterEncoding);
307 
308         if (m_format.equals(METS_EXT1_0)) {
309             // DISSEMINATORS... put disseminators in the instantiated digital
310             // object
311             Iterator<Disseminator> dissemIter = m_dissems.values().iterator();
312             while (dissemIter.hasNext()) {
313                 Disseminator diss = dissemIter.next();
314                 m_obj.disseminators(diss.dissID).add(diss);
315             }
316         }
317 
318     }

324     /**
325      * {@inheritDoc}
326      */
327     @Override
328     public void startPrefixMapping(String prefix, String uri) {
329         // Keep the prefix map up-to-date throughout the entire parse,
330         // and maintain a list of newly mapped prefixes on a per-element basis.
331         m_prefixMap.put(prefix, uri);
332         if (m_inXMLMetadata) {
333             m_localPrefixMap.put(prefix, uri);
334             m_prefixList.add(prefix);
335         }
336     }
337 
338     /**
339      * {@inheritDoc}
340      */
341     @Override
342     public void endPrefixMapping(String prefix) {
343         m_prefixMap.remove(prefix);
344         if (m_inXMLMetadata) {
345             m_localPrefixMap.remove(prefix);
346         }
347     }

349     /**
350      * {@inheritDoc}
351      */
352     @Override
353     public void startElement(String uri,
354                              String localName,
355                              String qName,
356                              Attributes a) throws SAXException {
357         if (uri.equals(METS.uri) && !m_inXMLMetadata) {
358             // a new mets element is starting
359             if (localName.equals("mets")) {
360                 m_rootElementFound = true;
361                 m_obj.setPid(grab(a, METS.uri, "OBJID"));
362                 m_obj.setLabel(grab(a, METS.uri, "LABEL"));
363                 if (m_format.equals(METS_EXT1_0)) {
364                     // In METS_EXT 1.0, the PROFILE attribute mapped to an
365                     // object property, fedora-model:contentModel.  This will be
366                     // retained as an extended property in the DigitalObject.
367                     m_obj.setExtProperty(MODEL.CONTENT_MODEL.uri,
368                                          grab(a, METS.uri, "PROFILE"));
369                     // Similarly, the TYPE attribute mapped to rdf:type, and
370                     // will also be retained as an external property.
371                     m_obj.setExtProperty(RDF.TYPE.uri,
372                                          grab(a, METS.uri, "TYPE"));
373                 }
374             } else if (localName.equals("metsHdr")) {
375                 m_obj.setCreateDate(DateUtility
376                         .convertStringToDate(grab(a, METS.uri, "CREATEDATE")));
377                 m_obj.setLastModDate(DateUtility
378                         .convertStringToDate(grab(a, METS.uri, "LASTMODDATE")));
379                 try {
380                     m_obj.setState(DOTranslationUtility
381                         .readStateAttribute(grab(a, METS.uri, "RECORDSTATUS")));
382                 } catch (ParseException e) {
383                     throw new SAXException("Could not read object state", e);
384                 }
385             } else if (localName.equals("agent")) {
386                 m_agentRole = grab(a, METS.uri, "ROLE");
387             } else if (localName.equals("name")
388                     && m_agentRole.equals("IPOWNER")) {
389                 m_readingContent = true;
390                 m_elementContent = new StringBuffer();
391             } else if (localName.equals("amdSec")) {
392                 m_dsId = grab(a, METS.uri, "ID");
393                 m_dsState = grab(a, METS.uri, "STATUS");
394                 String dsVersionable = grab(a, METS.uri, "VERSIONABLE");
395                 if (dsVersionable != null && !dsVersionable.equals("")) {
396                     m_dsVersionable =
397                             new Boolean(grab(a, METS.uri, "VERSIONABLE"))
398                                     .booleanValue();
399                 } else {
400                     m_dsVersionable = true;
401                 }
402             } else if (localName.equals("dmdSecFedora")) {
403                 m_dsId = grab(a, METS.uri, "ID");
404                 m_dsState = grab(a, METS.uri, "STATUS");
405                 String dsVersionable = grab(a, METS.uri, "VERSIONABLE");
406                 if (dsVersionable != null && !dsVersionable.equals("")) {
407                     m_dsVersionable =
408                             new Boolean(grab(a, METS.uri, "VERSIONABLE"))
409                                     .booleanValue();
410                 } else {
411                     m_dsVersionable = true;
412                 }
413             } else if (localName.equals("techMD") || localName.equals("descMD")
414                     || localName.equals("sourceMD")
415                     || localName.equals("rightsMD")
416                     || localName.equals("digiprovMD")) {
417                 m_dsVersId = grab(a, METS.uri, "ID");
418                 if (localName.equals("techMD")) {
419                     m_dsMDClass = DatastreamXMLMetadata.TECHNICAL;
420                 }
421                 if (localName.equals("sourceMD")) {
422                     m_dsMDClass = DatastreamXMLMetadata.SOURCE;
423                 }
424                 if (localName.equals("rightsMD")) {
425                     m_dsMDClass = DatastreamXMLMetadata.RIGHTS;
426                 }
427                 if (localName.equals("digiprovMD")) {
428                     m_dsMDClass = DatastreamXMLMetadata.DIGIPROV;
429                 }
430                 if (localName.equals("descMD")) {
431                     m_dsMDClass = DatastreamXMLMetadata.DESCRIPTIVE;
432                 }
433                 String dateString = grab(a, METS.uri, "CREATED");
434                 if (dateString != null && !dateString.equals("")) {
435                     m_dsCreateDate =
436                             DateUtility.convertStringToDate(dateString);
437                 }
438             } else if (localName.equals("mdWrap")) {
439                 m_dsInfoType = grab(a, METS.uri, "MDTYPE");
440                 m_dsOtherInfoType = grab(a, METS.uri, "OTHERMDTYPE");
441                 m_dsLabel = grab(a, METS.uri, "LABEL");
442                 m_dsMimeType = grab(a, METS.uri, "MIMETYPE");
443                 m_dsFormatURI = grab(a, METS.uri, "FORMAT_URI");
444                 String altIDs = grab(a, METS.uri, "ALT_IDS");
445                 if (altIDs.length() == 0) {
446                     m_dsAltIDs = new String[0];
447                 } else {
448                     m_dsAltIDs = altIDs.split(" ");
449                 }
450                 m_dsChecksum = grab(a, METS.uri, "CHECKSUM");
451                 m_dsChecksumType = grab(a, METS.uri, "CHECKSUMTYPE");
452             } else if (localName.equals("xmlData")) {
453                 m_dsXMLBuffer = new StringBuffer();
454                 m_xmlDataLevel = 0;
455                 m_inXMLMetadata = true;
456             } else if (localName.equals("fileGrp")) {
457                 m_dsId = grab(a, METS.uri, "ID");
458                 String dsVersionable = grab(a, METS.uri, "VERSIONABLE");
459                 if (dsVersionable != null && !dsVersionable.equals("")) {
460                     m_dsVersionable =
461                             new Boolean(grab(a, METS.uri, "VERSIONABLE"))
462                                     .booleanValue();
463                 } else {
464                     m_dsVersionable = true;
465                 }
466                 // reset the values for the next file
467                 m_dsVersId = "";
468                 m_dsCreateDate = null;
469                 m_dsMimeType = "";
470                 m_dsControlGrp = "";
471                 m_dsFormatURI = "";
472                 m_dsAltIDs = new String[0];
473                 m_dsState = grab(a, METS.uri, "STATUS");
474                 m_dsSize = -1;
475                 m_dsChecksum = "";
476                 m_dsChecksumType = "";
477             } else if (localName.equals("file")) {
478                 m_dsVersId = grab(a, METS.uri, "ID");
479                 String dateString = grab(a, METS.uri, "CREATED");
480                 if (dateString != null && !dateString.equals("")) {
481                     m_dsCreateDate =
482                             DateUtility.convertStringToDate(dateString);
483                 }
484                 m_dsMimeType = grab(a, METS.uri, "MIMETYPE");
485                 m_dsControlGrp = grab(a, METS.uri, "OWNERID");
486                 String ADMID = grab(a, METS.uri, "ADMID");
487                 if (ADMID != null && !"".equals(ADMID)) {
488                     ArrayList<String> al = new ArrayList<String>();
489                     if (ADMID.indexOf(" ") != -1) {
490                         String[] admIds = ADMID.split(" ");
491                         for (String element : admIds) {
492                             al.add(element);
493                         }
494                     } else {
495                         al.add(ADMID);
496                     }
497                     m_dsADMIDs.put(m_dsVersId, al);
498                 }
499                 String DMDID = grab(a, METS.uri, "DMDID");
500                 if (DMDID != null && !"".equals(DMDID)) {
501                     ArrayList<String> al = new ArrayList<String>();
502                     if (DMDID.indexOf(" ") != -1) {
503                         String[] dmdIds = DMDID.split(" ");
504                         for (String element : dmdIds) {
505                             al.add(element);
506                         }
507                     } else {
508                         al.add(DMDID);
509                     }
510                     m_dsDMDIDs.put(m_dsVersId, al);
511                 }
512                 String sizeString = grab(a, METS.uri, "SIZE");
513                 if (sizeString != null && !sizeString.equals("")) {
514                     try {
515                         m_dsSize = Long.parseLong(sizeString);
516                     } catch (NumberFormatException nfe) {
517                         throw new SAXException("If specified, a datastream's "
518                                 + "SIZE attribute must be an xsd:long.");
519                     }
520                 }
521                 String formatURI = grab(a, METS.uri, "FORMAT_URI");
522                 if (formatURI != null && !formatURI.equals("")) {
523                     m_dsFormatURI = formatURI;
524                 }
525                 String altIDs = grab(a, METS.uri, "ALT_IDS");
526                 if (altIDs.length() == 0) {
527                     m_dsAltIDs = new String[0];
528                 } else {
529                     m_dsAltIDs = altIDs.split(" ");
530                 }
531                 m_dsChecksum = grab(a, METS.uri, "CHECKSUM");
532                 m_dsChecksumType = grab(a, METS.uri, "CHECKSUMTYPE");
533                 // inside a "file" element, it's either going to be
534                 // FLocat (a reference) or FContent (inline)
535             } else if (localName.equals("FLocat")) {
536                 m_dsLabel = grab(a, m_xlink.uri, "title");
537                 String dsLocation = grab(a, m_xlink.uri, "href");
538                 if (dsLocation == null || dsLocation.equals("")) {
539                     throw new SAXException("xlink:href must be specified in FLocat element");
540                 }
541 
542                 if (m_dsControlGrp.equalsIgnoreCase("E")
543                         || m_dsControlGrp.equalsIgnoreCase("R")) {
544 
545                     // URL FORMAT VALIDATION for dsLocation:
546                     // make sure we have a properly formed URL (must have protocol)
547                     try {
548                         ValidationUtility.validateURL(dsLocation, m_dsControlGrp);
549                     } catch (ValidationException ve) {
550                         throw new SAXException(ve.getMessage());
551                     }
552                     // system will set dsLocationType for E and R datastreams...
553                     m_dsLocationType = Datastream.DS_LOCATION_TYPE_URL;
554                     m_dsInfoType = "DATA";
555                     m_dsLocation = dsLocation;
556                     instantiateDatastream(new DatastreamReferencedContent());
557                 } else if (m_dsControlGrp.equalsIgnoreCase("M")) {
558                     // URL FORMAT VALIDATION for dsLocation:
559                     // For Managed Content the URL is only checked when we are parsing a
560                     // a NEW ingest file because the URL is replaced with an internal identifier
561                     // once the repository has sucked in the content for storage.
562                     if (m_obj.isNew()) {
563                         try {
564                             ValidationUtility.validateURL(dsLocation, m_dsControlGrp);
565                             m_dsLocationType = Datastream.DS_LOCATION_TYPE_URL;
566                         } catch (ValidationException ve) {
567                             throw new SAXException(ve.getMessage());
568                         }
569                     } else {
570                         m_dsLocationType = Datastream.DS_LOCATION_TYPE_INTERNAL;
571                     }
572                     m_dsInfoType = "DATA";
573                     m_dsLocation = dsLocation;
574                     instantiateDatastream(new DatastreamManagedContent());
575                 }
576             } else if (localName.equals("FContent")) {
577                 // In METS_EXT, the FContent element contains base64-encoded
578                 // data.
579                 m_readingContent = true;
580                 m_elementContent = new StringBuffer();
581                 if (m_dsControlGrp.equalsIgnoreCase("M")) {
582                     m_readingBinaryContent = true;
583                     m_binaryContentTempFile = null;
584                     try {
585                         m_binaryContentTempFile =
586                                 File.createTempFile("binary-datastream", null);
587                     } catch (IOException ioe) {
588                         throw new SAXException(new StreamIOException("Unable to create temporary file for binary content"));
589                     }
590                 }
591 
592             } else if (m_format.equals(METS_EXT1_0)) {
593                 startDisseminators(localName, a);
594             }
595         } else {
596             if (m_inXMLMetadata) {
597                 // must be in xmlData... just output it, remembering the number
598                 // of METS:xmlData elements we see
599                 appendElementStart(uri, localName, qName, a, m_dsXMLBuffer);
600 
601                 // METS INSIDE METS! we have an inline XML datastream
602                 // that is itself METS.  We do not want to parse this!
603                 if (uri.equals(METS.uri) && localName.equals("xmlData")) {
604                     m_xmlDataLevel++;
605                 }
606                 // remember this stuff... (we don't have to look at level
607                 // because the audit schema doesn't allow for xml elements inside
608                 // these, so they're never set incorrectly)
609                 // signaling that we're interested in sending char data to
610                 // the m_auditBuffer by making it non-null, and getting
611                 // ready to accept data by allocating a new StringBuffer
612                 if (m_dsId.equals("FEDORA-AUDITTRAIL")
613                         || m_dsId.equals("AUDIT")) {
614                     if (localName.equals("record")) {
615                         m_auditId = grab(a, uri, "ID");
616                     } else if (localName.equals("process")) {
617                         m_auditProcessType = grab(a, uri, "type");
618                     } else if (localName.equals("action")
619                             || localName.equals("componentID")
620                             || localName.equals("responsibility")
621                             || localName.equals("date")
622                             || localName.equals("justification")) {
623                         m_auditBuffer = new StringBuffer();
624                     }
625                 }
626             } else {
627                 // ignore all else
628             }
629         }
630     }

632     /**
633      * {@inheritDoc}
634      */
635     @Override
636     public void characters(char[] ch, int start, int length) {
637         if (m_inXMLMetadata) {
638             if (m_auditBuffer != null) {
639                 m_auditBuffer.append(ch, start, length);
640             } else {
641                 // since this data is encoded straight back to xml,
642                 // we need to make sure special characters &, <, >, ", and '
643                 // are re-converted to the xml-acceptable equivalents.
644                 StreamUtility.enc(ch, start, length, m_dsXMLBuffer);
645             }
646         } else if (m_readingContent) {
647             // read normal element content into a string buffer
648             if (m_elementContent != null) {
649                 m_elementContent.append(ch, start, length);
650             }
651         }
652     }

654     /**
655      * {@inheritDoc}
656      */
657     @Override
658     public void endElement(String uri, String localName, String qName)
659             throws SAXException {
660         // first, deal with the situation when we are processing a block of inline XML
661         if (m_inXMLMetadata) {
662             if (uri.equals(METS.uri) && localName.equals("xmlData")
663                     && m_xmlDataLevel == 0) {
664                 // finished all xml metadata for this datastream
665                 if (m_dsId.equals("FEDORA-AUDITTRAIL")
666                         || m_dsId.equals("AUDIT")) {
667                     // we've been looking at an audit trail... set audit record
668                     AuditRecord a = new AuditRecord();
669                     // In METS each audit record is in its own <digiprovMD>
670                     // element within an <amdSec>.  So, pick up the XML ID
671                     // of the <digiprovMD> element for the audit record id.
672                     // This amdSec is treated like a datastream, and each
673                     // digiprovMD is a version, so id was parsed into dsVersId.
674                     a.id = m_auditId; //m_dsVersId;
675                     a.processType = m_auditProcessType;
676                     a.action = m_auditAction;
677                     a.componentID = m_auditComponentID;
678                     a.responsibility = m_auditResponsibility;
679                     a.date = DateUtility.convertStringToDate(m_auditDate);
680                     a.justification = m_auditJustification;
681                     m_obj.getAuditRecords().add(a);
682                     m_inXMLMetadata = false; // other stuff is re-initted upon
683                     // startElement for next xml metadata
684                     // element
685                 } else {
686                     // Create the right kind of datastream and add to the object
687                     DatastreamXMLMetadata ds = new DatastreamXMLMetadata();
688                     instantiateXMLDatastream(ds);
689                     m_inXMLMetadata = false;
690                     m_localPrefixMap.clear();
691                 }
692             } else {
693                 // finished an element within inline xml metadata
694                 m_dsXMLBuffer.append("</" + qName + ">");
695                 // make sure we know when to pay attention to METS again
696                 if (uri.equals(METS.uri) && localName.equals("xmlData")) {
697                     m_xmlDataLevel--;
698                 }
699                 if (m_dsId.equals("FEDORA-AUDITTRAIL")
700                         || m_dsId.equals("AUDIT")) {
701                     if (localName.equals("action")) {
702                         m_auditAction = m_auditBuffer.toString();
703                         m_auditBuffer = null;
704                     } else if (localName.equals("componentID")) {
705                         m_auditComponentID = m_auditBuffer.toString();
706                         m_auditBuffer = null;
707                     } else if (localName.equals("responsibility")) {
708                         m_auditResponsibility = m_auditBuffer.toString();
709                         m_auditBuffer = null;
710                     } else if (localName.equals("date")) {
711                         m_auditDate = m_auditBuffer.toString();
712                         m_auditBuffer = null;
713                     } else if (localName.equals("justification")) {
714                         m_auditJustification = m_auditBuffer.toString();
715                         m_auditBuffer = null;
716                     }
717                 }
718             }
719             // ALL OTHER ELEMENT CASES: we are NOT processing a block of inline XML metadata
720         } else {
721             if (m_readingBinaryContent) {
722                 // In the version of METS Fedora uses, FContent assumes base64-encoded content
723                 if (uri.equals(METS.uri) && localName.equals("FContent")) {
724                     if (m_binaryContentTempFile != null) {
725                         try {
726                             FileOutputStream os =
727                                     new FileOutputStream(m_binaryContentTempFile);
728                             // remove all spaces and newlines, this might not be necessary.
729                             String elementStr =
730                                     m_elementContent.toString()
731                                             .replaceAll("\\s", "");
732                             byte elementBytes[] = Base64.decode(elementStr);
733                             os.write(elementBytes);
734                             os.close();
735                             m_dsLocationType = Datastream.DS_LOCATION_TYPE_INTERNAL;
736                             m_dsLocation =
737                                 DatastreamManagedContent.TEMP_SCHEME
738                                             + m_binaryContentTempFile
739                                                     .getAbsolutePath();
740                             instantiateDatastream(new DatastreamManagedContent());
741                         } catch (FileNotFoundException fnfe) {
742                             throw new SAXException(new StreamIOException("Unable to open temporary file created for binary content"));
743                         } catch (IOException fnfe) {
744                             throw new SAXException(new StreamIOException("Error writing to temporary file created for binary content"));
745                         }
746                     }
747                 }
748                 m_binaryContentTempFile = null;
749                 m_readingBinaryContent = false;
750                 m_elementContent = null;
751                 // all other cases...
752             } else {
753                 if (m_readingContent) {
754                     // elements for which we were reading regular content
755                     if (uri.equals(METS.uri) && localName.equals("name")
756                             && m_agentRole.equals("IPOWNER")) {
757                         m_obj.setOwnerId(m_elementContent.toString());
758                     } else if (uri.equals(METS.uri)
759                             && localName.equals("agent")) {
760                         m_agentRole = null;
761                     }
762                     m_readingContent = false;
763                     m_elementContent = null;
764                 } else {
765                     // no other processing requirements at this time
766                 }
767             }
768         }
769     }

775     private void startDisseminators(String localName, Attributes a)
776             throws SAXException {
777         if (localName.equals("structMap")) {
778             // this is a component of a disseminator.  here we assume the rest
779             // of the disseminator's information will be seen later, so we
780             // construct a new Disseminator object to hold the structMap...
781             // and later, the other info
782             //
783             // Building up a global map of Disseminators, m_dissems,
784             // keyed by bindingmap ID.
785             //
786             if (grab(a, METS.uri, "TYPE").equals("fedora:dsBindingMap")) {
787                 String bmId = grab(a, METS.uri, "ID");
788                 if (bmId == null || bmId.equals("")) {
789                     throw new SAXException("structMap with TYPE "
790                             + "fedora:dsBindingMap must specify a non-empty "
791                             + "ID attribute.");
792                 } else {
793                     Disseminator diss = new Disseminator();
794                     diss.dsBindMapID = bmId;
795                     m_dissems.put(bmId, diss);
796                     m_diss = diss;
797                     m_diss.dsBindMap = new DSBindingMap();
798                     m_diss.dsBindMap.dsBindMapID = bmId;
799                     m_indiv = false; // flag we're not looking at inner part yet
800                 }
801             } else {
802                 throw new SAXException("StructMap must have TYPE fedora:dsBindingMap");
803             }
804         } else if (localName.equals("div")) {
805             if (m_indiv) {
806                 // inner part of structmap
807                 DSBinding binding = new DSBinding();
808                 if (m_diss.dsBindMap.dsBindings == null) {
809                     // none yet.. create array of size one
810                     DSBinding[] bindings = new DSBinding[1];
811                     m_diss.dsBindMap.dsBindings = bindings;
812                     m_diss.dsBindMap.dsBindings[0] = binding;
813                 } else {
814                     // need to expand the array size by one,
815                     // and do an array copy.
816                     int curSize = m_diss.dsBindMap.dsBindings.length;
817                     DSBinding[] oldArray = m_diss.dsBindMap.dsBindings;
818                     DSBinding[] newArray = new DSBinding[curSize + 1];
819                     for (int i = 0; i < curSize; i++) {
820                         newArray[i] = oldArray[i];
821                     }
822                     newArray[curSize] = binding;
823                     m_diss.dsBindMap.dsBindings = newArray;
824                 }
825                 // now populate 'binding' values...we'll have
826                 // everything at this point except datastreamID...
827                 // that comes as a child: <fptr FILEID="DS2"/>
828                 binding.bindKeyName = grab(a, METS.uri, "TYPE");
829                 binding.bindLabel = grab(a, METS.uri, "LABEL");
830                 binding.seqNo = grab(a, METS.uri, "ORDER");
831             } else {
832                 m_indiv = true;
833                 // first (outer div) part of structmap
834                 m_diss.dsBindMap.dsBindMechanismPID = grab(a, METS.uri, "TYPE");
835                 m_diss.dsBindMap.dsBindMapLabel = grab(a, METS.uri, "LABEL");
836             }
837         } else if (localName.equals("fptr")) {
838             // assume we're inside the inner div... that's the
839             // only place the fptr element is valid.
840             DSBinding binding =
841                     m_diss.dsBindMap.dsBindings[m_diss.dsBindMap.dsBindings.length - 1];
842             binding.datastreamID = grab(a, METS.uri, "FILEID");
843         } else if (localName.equals("behaviorSec")) {
844             // looks like we're in a disseminator... it should be in the
845             // hash by now because we've already gone through structmaps
846             // ...keyed by structmap id... remember the id (group id)
847             // so we can put it in when parsing serviceBinding
848             m_dissemId = grab(a, METS.uri, "ID");
849             m_dissemState = grab(a, METS.uri, "STATUS");
850         } else if (localName.equals("serviceBinding")) {
851             // remember the structId so we can grab the right dissem
852             // when parsing children
853             m_structId = grab(a, METS.uri, "STRUCTID");
854             // grab the disseminator associated with the provided structId
855             Disseminator dissem = m_dissems.get(m_structId);
856             // plug known items in..
857             dissem.dissID = m_dissemId;
858             dissem.dissState = m_dissemState;
859             // then grab the new stuff for the dissem for this element, and
860             // put it in.
861             dissem.dissVersionID = grab(a, METS.uri, "ID");
862             dissem.bDefID = grab(a, METS.uri, "BTYPE");
863             dissem.dissCreateDT =
864                     DateUtility
865                             .convertStringToDate(grab(a, METS.uri, "CREATED"));
866             dissem.dissLabel = grab(a, METS.uri, "LABEL");
867         } else if (localName.equals("interfaceMD")) {
868             Disseminator dissem = m_dissems.get(m_structId);
869         } else if (localName.equals("serviceBindMD")) {
870             Disseminator dissem = m_dissems.get(m_structId);
871             dissem.sDepID = grab(a, m_xlink.uri, "href");
872         }
873     }

875     private void appendElementStart(String uri,
876                                     String localName,
877                                     String qName,
878                                     Attributes a,
879                                     StringBuffer out) {
880         out.append("<" + qName);
881         // add the current qName's namespace to m_localPrefixMap
882         // and m_prefixList if it's not already in m_localPrefixMap
883         // This ensures that all namespaces used in inline XML are declared within,
884         // since it's supposed to be a standalone chunk.
885         String[] parts = qName.split(":");
886         if (parts.length == 2) {
887             String nsuri = m_localPrefixMap.get(parts[0]);
888             if (nsuri == null) {
889                 m_localPrefixMap.put(parts[0], parts[1]);
890                 m_prefixList.add(parts[0]);
891             }
892         }
893         // do we have any newly-mapped namespaces?
894         while (m_prefixList.size() > 0) {
895             String prefix = m_prefixList.remove(0);
896             out.append(" xmlns");
897             if (prefix.length() > 0) {
898                 out.append(":");
899             }
900             out.append(prefix + "=\""
901                     + StreamUtility.enc(m_prefixMap.get(prefix))
902                     + "\"");
903         }
904         for (int i = 0; i < a.getLength(); i++) {
905             out.append(" " + a.getQName(i) + "=\""
906                     + StreamUtility.enc(a.getValue(i)) + "\"");
907         }
908         out.append(">");
909     }

1014     /**
1015      * convertAudits: In Fedora 2.0 and beyond, we want self-standing audit
1016      * records. Make sure audit records are converted to new format that
1017      * contains a componentID to show what component in the object the audit
1018      * record is about.
1019      */
1020     private void convertAudits() {
1021         // Only do this if ADMID values were found in the object.
1022         if (m_dsADMIDs.size() > 0) {
1023             // Look at datastreams to see if there are audit records for them.
1024             // NOTE:  we do not look at disseminators because in pre-2.0
1025             // the disseminators did not point to their audit records as
1026             // did the datastreams.
1027             Iterator<String> dsIdIter = m_obj.datastreamIdIterator();
1028             while (dsIdIter.hasNext()) {
1029                 for (Datastream ds : m_obj.datastreams(dsIdIter.next())) {
1030                     // ADMID processing...
1031                     // get list of ADMIDs that go with a datastream version
1032                     List<String> admIdList = m_dsADMIDs.get(ds.DSVersionID);
1033                     List<String> cleanAdmIdList = new ArrayList<String>();
1034                     if (admIdList != null) {
1035                         Iterator<String> admIdIter = admIdList.iterator();
1036                         while (admIdIter.hasNext()) {
1037                             String admId = admIdIter.next();
1038                             // Detect ADMIDs that reference audit records
1039                             // vs. regular admin metadata. Drop audits from
1040                             // the list. We know we have an audit if the ADMID
1041                             // is not a regular datatream in the object.
1042                             Iterator<Datastream> matchedDatastreams =
1043                                     m_obj.datastreams(admId).iterator();
1044                             if (matchedDatastreams.hasNext()) {
1045 
1046                                 // Keep track of audit metadata correlated with the
1047                                 // datastream version it's about (for later use).
1048                                 m_AuditIdToComponentId.put(admId,
1049                                                            ds.DSVersionID);
1050                             } else {
1051                                 // Keep track of non-audit metadata in a new list.
1052                                 cleanAdmIdList.add(admId);
1053                             }
1054                         }
1055                     }
1056                     if (cleanAdmIdList.size() <= 0) {
1057                         // we keep track of admin metadata references
1058                         // for each datastream, but we exclude the audit
1059                         // records from this list.  If there are no
1060                         // non-audit metadata references, remove the
1061                         // datastream entry from the master hashmap.
1062                         m_dsADMIDs.remove(ds.DSVersionID);
1063                     } else {
1064                         // otherwise, update the master hashmap with the
1065                         // clean list of non-audit metadata
1066                         m_dsADMIDs.put(ds.DSVersionID, cleanAdmIdList);
1067                     }
1068                 }
1069             }
1070             // Now, put component ids on audit records.  Pre-Fedora 2.0
1071             // datastream versions pointed to their audit records.
1072             Iterator<AuditRecord> iter = m_obj.getAuditRecords().iterator();
1073             while (iter.hasNext()) {
1074                 AuditRecord au = iter.next();
1075                 if (au.componentID == null || au.componentID.equals("")) {
1076                     // Before Fedora 2.0 audit records were associated with
1077                     // datastream version ids.  From now on, the datastream id
1078                     // will be posted as the component id in the audit record,
1079                     // and associations to particular datastream versions can
1080                     // be derived via the datastream version dates and the audit
1081                     // record dates.
1082                     String dsVersId = m_AuditIdToComponentId.get(au.id);
1083                     if (dsVersId != null && !dsVersId.equals("")) {
1084                         au.componentID =
1085                                 dsVersId.substring(0, dsVersId.indexOf("."));
1086                     }
1087 
1088                 }
1089             }
1090         }
1091     }

1093     /**
1094      * addRelsInt: Build an RDF relationship datastream to preserve DMDID and
1095      * ADMID references in the digital object when METS is converted to FOXML
1096      * (or other formats in the future). If there is no pre-existing RELS-INT,
1097      * look for DMDID and ADMID attributes to create new RELS-INT datastream.
1098      */
1099     private void createRelsInt() {
1100 
1101         // create a new RELS-INT datastream only if one does not already exist.
1102         Iterator<Datastream> metsrels =
1103                 m_obj.datastreams("RELS-INT").iterator();
1104         if (metsrels.hasNext()) {
1105             m_relsBuffer = new StringBuffer();
1106             appendRDFStart(m_relsBuffer);
1107             Iterator<String> dsIds = m_obj.datastreamIdIterator();
1108             while (dsIds.hasNext()) {
1109                 // initialize hash sets to keep a list of
1110                 // unique DMDIDs or ADMIDs at the datatream id level.
1111                 HashSet<String> uniqueDMDIDs = new HashSet<String>();
1112                 HashSet<String> uniqueADMIDs = new HashSet<String>();
1113                 // get list of datastream *versions*
1114                 for (Datastream dsVersion : m_obj.datastreams(dsIds
1115                         .next())) {
1116                     // DMDID processing...
1117                     List<String> dmdIdList =
1118                             m_dsDMDIDs.get(dsVersion.DSVersionID);
1119                     if (dmdIdList != null) {
1120                         hasRels = true;
1121                         Iterator<String> dmdIdIter = dmdIdList.iterator();
1122                         while (dmdIdIter.hasNext()) {
1123                             String dmdId = dmdIdIter.next();
1124                             // APPEND TO RDF: record the DMDID relationship.
1125                             // Relationships will now be recorded at the
1126                             // datastream level, not the datastream version level.
1127                             // So, is the relationship existed on more than one
1128                             // datastream version, only write it once to the RDF.
1129                             if (!uniqueDMDIDs.contains(dmdId)) {
1130                                 appendRDFRel(m_relsBuffer,
1131                                              m_obj.getPid(),
1132                                              dsVersion.DatastreamID,
1133                                              "hasDescMetadata",
1134                                              dmdId);
1135                             }
1136                             uniqueDMDIDs.add(dmdId);
1137                         }
1138                     }
1139                     // ADMID processing (already cleansed of audit refs)...
1140                     List<String> cleanAdmIdList =
1141                             m_dsADMIDs.get(dsVersion.DSVersionID);
1142                     if (cleanAdmIdList != null) {
1143                         hasRels = true;
1144                         Iterator<String> admIdIter = cleanAdmIdList.iterator();
1145                         while (admIdIter.hasNext()) {
1146                             String admId = admIdIter.next();
1147                             // APPEND TO RDF: record the ADMID relationship.
1148                             // Relationships will now be recorded at the
1149                             // datastream level, not the datastream version level.
1150                             // So, is the relationship existed on more than one
1151                             // datastream version, only write it once to the RDF.
1152                             if (!uniqueADMIDs.contains(admId)) {
1153                                 appendRDFRel(m_relsBuffer,
1154                                              m_obj.getPid(),
1155                                              dsVersion.DatastreamID,
1156                                              "hasAdminMetadata",
1157                                              admId);
1158                             }
1159                             uniqueADMIDs.add(admId);
1160                         }
1161                     }
1162                 }
1163             }
1164             // APPEND RDF: finish up and add RDF as a system-generated datastream
1165             if (hasRels) {
1166                 appendRDFEnd(m_relsBuffer);
1167                 setRDFAsDatastream(m_relsBuffer);
1168             } else {
1169                 m_relsBuffer = null;
1170             }
1171         }
1172     }
