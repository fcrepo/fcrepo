105     /**
106      * Creates a JMS manager using jndi properties to start a connection
107      * to a JMS provider. If jndi properties are null an attempt will
108      * be made to create a connection based on a <code>java:comp/env</code>
109      * context
110      *
111      * A connection must have a clientId in order to create durable
112      * subscriptions. This clientId can either be set administratively
113      * on the Connection object created in the JNDI store or by providing
114      * a non-null value for the clientId parameter of this method.
115      *
116      * @param jndiProps
117      * @param clientId
118      * @throws MessagingException
119      */
120     public JMSManager(Properties jndiProps, String clientId)
121             throws MessagingException {
122 
123         this.jndiProps = jndiProps;
124 
125         // Check properties
126         if (jndiProps == null) {
127             // Attempt a lookup to see if properties are provided by the container
128             String providerUrl = null;
129             try {
130                 providerUrl = (String) jndiLookup(Context.PROVIDER_URL);
131             } catch (MessagingException me) {
132                 providerUrl = null;
133             }
134             if(providerUrl == null) {
135                 throw new MessagingException("JMS connection properties must either be "
136                         + "provided by the container or by a non-null properties file "
137                         + "containing, at minimum, a value for "
138                         + Context.INITIAL_CONTEXT_FACTORY + " and "
139                         + Context.PROVIDER_URL + ".");
140             }
141         } else {
142             String icf = jndiProps.getProperty(Context.INITIAL_CONTEXT_FACTORY);
143             if (icf == null) {
144                 throw new MessagingException("A value for "
145                         + Context.INITIAL_CONTEXT_FACTORY
146                         + " must be included in the JNDI properties.");
147             }
148 
149             String pUrl = jndiProps.getProperty(Context.PROVIDER_URL);
150             if (pUrl == null) {
151                 throw new MessagingException("A value for "
152                         + Context.PROVIDER_URL
153                         + "must be included in the JNDI properties.");
154             }
155         }
156 
157         connectToJMS(clientId);
158     }

173     /**
174      * Creates a Destination if the Destination has not already been created.
175      *
176      * @param name - the name of the destination to create
177      * @param type - the destination type (topic or queue)
178      * @param fTransacted - determines whether the session will maintain transactions
179      * @param ackMode - determines the session acknowledgment mode
180      * @throws MessagingException
181      */
182     public Destination createDestination(String name,
183                                   DestinationType type,
184                                   boolean fTransacted,
185                                   int ackMode) throws MessagingException {
186         // If the destination already exists, just return it
187         JMSDestination jmsDest = jmsDestinations.get(name);
188         if (jmsDest != null) {
189             return jmsDest.destination;
190         }
191 
192         // Create the new destination and store it
193         Session session;
194         try {
195             session = connection.createSession(fTransacted, ackMode);
196         } catch (JMSException e) {
197             throw new MessagingException(e.getMessage(), e);
198         }
199 
200         // Look up the destination otherwise create it
201         Destination destination = null;
202         try {
203             destination = (Destination) jndiLookup(name);
204         } catch (MessagingException me) {
205             logger.debug("JNDI lookup for destination " + name + " failed. "
206                     + "Destination must be created.");
207             destination = null;
208         }
209         if (destination == null) {
210             // Create a topic or queue as specified
211             try {
212                 if (type.equals(DestinationType.Queue)) {
213                     logger.debug("setupDestination() - creating Queue" + name);
214                     destination = session.createQueue(name);
215                 } else {
216                     logger.debug("setupDestination() - creating Topic " + name);
217                     destination = session.createTopic(name);
218                 }
219             } catch (JMSException e) {
220                 throw new MessagingException(e.getMessage(), e);
221             }
222         }
223 
224         jmsDest = new JMSDestination(destination, session, null, null);
225 
226         jmsDestinations.put(name, jmsDest);
227 
228         return destination;
229     }

289             Session s =
290                     connection.createSession(false, Session.AUTO_ACKNOWLEDGE);

292             Message msg = c.receive();
293             s.close();
294             return msg;

296             throw new MessagingException(e.getMessage(), e);
297         }

431     /**
432      * This is an asynchronous and durable listen. The caller provides a JMS
433      * callback interface reference, and any messages received for the given
434      * destination are provided through the onMessage() callback method. If the
435      * listener becomes unavailable the JMS provider will store messages
436      * received on the given topic until the listener reestablishes a connection
437      * and will then deliver those messages.
438      * Messages will be filtered based on the provided message selector.
439      *
440      * @param topic
441      *        the topic on which to listen
442      * @param messageSelector
443      *        selection criteria for filtering messages
444      * @param callback
445      *        the listener to call when a message is received
446      * @param subscriptionName
447      *        the name of the subscription
448      * @return The subscription name which can be used to stop, unsubscribe,
449      *         and reconnect to this topic
450      * @throws MessagingException
451      */
452     public String listenDurable(Topic topic,
453                                 String messageSelector,
454                                 MessageListener callback,
455                                 String subscriptionName)
456             throws MessagingException {
457         try {
458             String clientId = connection.getClientID();
459             if (clientId == null) {
460                 throw new MessagingException("A non-null client ID must be provided upon "
461                         + "creation of a JMSManager in order to create a JMS connection "
462                         + "capable of creating durable subscriptions.");
463             }
464             if (subscriptionName == null) {
465                 subscriptionName = topic.getTopicName();
466             }
467 
468             Session s =
469                     connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
470             MessageConsumer c =
471                     s.createDurableSubscriber(topic, subscriptionName, messageSelector, false);
472             c.setMessageListener(callback);
473             durableSubscriptions.put(subscriptionName, c);
474         } catch (JMSException e) {
475             throw new MessagingException(e.getMessage(), e);
476         }
477 
478         if (logger.isDebugEnabled()) {
479             logger.debug("listen() - Asynchronous durable listen on topic " + topic);
480         }
481 
482         return subscriptionName;
483     }

556     /**
557      * Stops producers and consumers on a given destination.
558      * This has no effect on durable subscriptions.
559      *
560      * @param destName
561      * @throws MessagingException
562      */
563     public void stop(String destName) throws MessagingException {
564         try {
565             // Look for an existing destination for the given destination
566             //
567             JMSDestination jmsDest =
568                     jmsDestinations.get(destName);
569             if (jmsDest != null) {
570                 // Close out all JMS related state
571                 //
572                 if (jmsDest.producer != null) {
573                     jmsDest.producer.close();
574                     logger.debug("Closed producer for " + destName);
575                 }
576                 if (jmsDest.consumer != null) {
577                     jmsDest.consumer.close();
578                     logger.debug("Closed consumer for " + destName);
579                 }
580                 if (jmsDest.session != null) {
581                     jmsDest.session.close();
582                     logger.debug("Closed session for " + destName);
583                 }
584 
585                 jmsDest.destination = null;
586                 jmsDest.session = null;
587                 jmsDest.producer = null;
588                 jmsDest.consumer = null;
589 
590                 // Remove the JMS client entry
591                 //
592                 jmsDestinations.remove(destName);
593                 jmsDest = null;
594             }
595         } catch (JMSException e) {
596             throw new MessagingException(e.getMessage(), e);
597         }
598     }

600     /**
601      * Stops a durable message consumer. Note that this is not
602      * the same as unsubscribing. When a durable message consumer is
603      * restarted all messages received since it was stopped will be
604      * delivered.
605      *
606      * @param subscriptionName - the name of the subscription
607      * @throws MessagingException
608      */
609     public void stopDurable(String subscriptionName) throws MessagingException {
610         try {
611             MessageConsumer durableSubscriber =
612                     durableSubscriptions.get(subscriptionName);
613             if (durableSubscriber != null) {
614                 durableSubscriber.close();
615             }
616         } catch (JMSException jmse) {
617             throw new MessagingException("Exception encountered attempting to "
618                     + "stop durable subscription with name: "
619                     + subscriptionName + ". Exception message: "
620                     + jmse.getMessage(), jmse);
621         }
622     }

624     /**
625      * Removes the durable subscription with the given name.
626      *
627      * @param subscriptionName - name of the durable subscription
628      * @throws MessagingException
629      */
630     public void unsubscribeDurable(String subscriptionName)
631             throws MessagingException {
632         try {
633             Session session =
634                 connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
635             MessageConsumer consumer = durableSubscriptions.get(subscriptionName);
636             if(consumer != null) {
637                 consumer.close();
638             }
639             session.unsubscribe(subscriptionName);
640         } catch (JMSException jmse) {
641             String errMsg = "Unable to unsubscribe from subscription with name: "
642                           + subscriptionName + " due to exception: "
643                           + jmse.getMessage();
644             logger.debug(errMsg, jmse);
645             throw new MessagingException(errMsg, jmse);
646         }
647     }
