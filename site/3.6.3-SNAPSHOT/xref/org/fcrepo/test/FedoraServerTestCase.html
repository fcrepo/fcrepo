60              throws Exception {
61          InputStream is = client.get(getBaseURL() + location, true, true);
62          Document result = XMLUnit.buildControlDocument(new InputSource(is));
63          is.close();
64          return result;

90      /**
91       * Ingest a specific directory of demo objects.
92       * <p>
93       * Given a path relative to the format-independent demo object hierarchy,
94       * will ingest all files in the hierarchy denoted by the path.
95       * </p>
96       * <h2>example</h2>
97       * <p>
98       * <code>ingestDemoObjects(local-server-demos)</code> will ingest all files
99       * underneath the <code>client/demo/[format]/local-server-demos/</code>
100      * hierarchy
101      * </p>
102      *
103      * @param path format-independent path to a directory within the demo object
104      *             hierarchy.
105      * @throws Exception
106      */
107     public static void ingestDemoObjects(String path) throws Exception {
108         FedoraClient client = FedoraTestCase.getFedoraClient();
109         FedoraAPIAMTOM apia = client.getAPIAMTOM();
110         FedoraAPIMMTOM apim = client.getAPIMMTOM();
111         ingestDemoObjects(path, apia, apim);
112     }
113     public static void ingestDemoObjects(String path, FedoraAPIAMTOM apia, FedoraAPIMMTOM apim) throws Exception {
114 
115         File dir = null;
116 
117         String specificPath = File.separator + path;
118 
119         String ingestFormat;
120         if (testingMETS()) {
121             System.out.println("Ingesting demo objects in METS format from " + specificPath);
122             dir = new File(FEDORA_HOME, "client/demo/mets" + specificPath);
123             ingestFormat = METS_EXT1_1.uri;
124         } else if (testingAtom()) {
125             System.out.println("Ingesting demo objects in Atom format from " + specificPath);
126             dir = new File(FEDORA_HOME, "client/demo/atom" + specificPath);
127             ingestFormat = ATOM1_1.uri;
128         } else if (testingAtomZip()) {
129             System.out.println("Ingesting all demo objects in Atom Zip format from " + specificPath);
130             dir = new File(FEDORA_HOME, "client/demo/atom-zip" + specificPath);
131             ingestFormat = ATOM_ZIP1_1.uri;
132         } else {
133             System.out.println("Ingesting demo objects in FOXML format from " + specificPath);
134             dir = new File(FEDORA_HOME, "client/demo/foxml" + specificPath);
135             ingestFormat = FOXML1_1.uri;
136         }
137 
138         Ingest.multiFromDirectory(dir,
139                                   ingestFormat,
140                                   apia,
141                                   apim,
142                                   null,
143                                   new PrintStream(File.createTempFile("demo",
144                                                                       null)),
145                                   new IngestCounter());
146         // clone some demo objects to managed-content equivalents for reserved datastreams (RELS-*, DC)
147         try {
148             ManagedContentTranslator.createManagedClone(apim, "demo:SmileyPens", "demo:SmileyPens_M");
149             ManagedContentTranslator.createManagedClone(apim, "demo:SmileyBeerGlass", "demo:SmileyBeerGlass_M");
150         } catch (Exception e) { // ignore errors, just log (for cases where ingest repeated before purge done)
151             System.out.println("Could not create managed clone test objects: " + e.getMessage());
152     }
153     }

155     /**
156      * Gets the PIDs of objects in the "demo" pid namespace that are in the
157      * repository
158      *
159      * @return set of PIDs of the specified object type
160      * @throws Exception
161      */
162     public static Set<String> getDemoObjects() throws Exception {
163 
164         FedoraClient client = getFedoraClient();
165         InputStream queryResult;
166         queryResult =
167                 client.get(getBaseURL() + "/search?query=pid~demo:*"
168                            + "&maxResults=1000&pid=true&xml=true", true, true);
169         SearchResultParser parser = new SearchResultParser(queryResult);
170 
171         return parser.getPIDs();
172     }

180     public static void purgeDemoObjects(FedoraAPIMMTOM apim) throws Exception {

182             AutoPurger.purge(apim, pid, null);
