224     /**
225      * Do Access requests require auth based on current test configuration?
226      *
227      * @return true if auth is required for API-requests
228      */
229     private boolean getAuthAccess() {
230         if (authorizeAccess == null) {
231             String property = System.getProperty(authAccessProperty);
232             if (property.equals("true")) {
233                 authorizeAccess = true;
234             } else if (property.equals("false")) {
235                 authorizeAccess = false;
236             } else {
237                 assertTrue("Failed to determine whether to perform authorization on Access requests from: "
238                                    + authAccessProperty,
239                            false);
240                 throw new RuntimeException("Failed to determine whether to perform authorization on Access requests from: "
241                         + authAccessProperty);
242             }
243 
244         }
245         return authorizeAccess;
246 
247     }

770     private String extractPid(String source) {
771         Matcher m = Pattern.compile("^.*/([^/]+$)").matcher(source);
772         String pid = null;
773         if (m.find() && m.groupCount() == 1) {
774             pid = m.group(1);
775         }
776         pid = pid.replaceAll("\n", "").replaceAll("\r", "").replaceAll("%3A", ":");
777         return pid;
778     }

1722     private HttpResponse getOrDelete(String method, boolean authenticate)
1723             throws Exception {
1724         if (url == null || url.length() == 0) {
1725             throw new IllegalArgumentException("url must be a non-empty value");
1726         } else if (!(url.startsWith("http://") || url.startsWith("https://"))) {
1727             url = getBaseURL() + url;
1728         }
1729         HttpMethod httpMethod = null;
1730         try {
1731             if (method.equals("GET")) {
1732                 httpMethod = new GetMethod(url);
1733             } else if (method.equals("DELETE")) {
1734                 httpMethod = new DeleteMethod(url);
1735             } else {
1736                 throw new IllegalArgumentException("method must be one of GET or DELETE.");
1737             }
1738             httpMethod.setDoAuthentication(authenticate);
1739             httpMethod.setFollowRedirects(false);
1740             httpMethod.getParams().setParameter("Connection", "Keep-Alive");
1741             getClient(authenticate).executeMethod(httpMethod);
1742             HttpResponse response = new HttpResponse(httpMethod);
1743 
1744             if (response.getStatusCode() == SC_MOVED_TEMPORARILY) {
1745                 String redir =
1746                         response.getResponseHeader("Location").getValue();
1747                 if (redir != url) {
1748                     url = redir;
1749                     response = getOrDelete(method, authenticate);
1750                 }
1751             }
1752 
1753             return response;
1754         } finally {
1755             if (httpMethod != null) {
1756                 httpMethod.releaseConnection();
1757             }
1758         }
1759     }

1885                         + "xsd");

1888         ArrayList<File> result = new ArrayList<File>();
1889         Pattern p = Pattern.compile("schemaLocation=\"(.+?)\"");
1890         Matcher m = p.matcher(xml);
1891         while (m.find()) {
1892             String[] content = m.group(1).split("\\s+");
1893             for (String frag : content) {
1894                 if (frag.contains(".xsd")) {
