154     /**
155      * Overwrite existing fedora users file with one containing user testuser, role testing
156      * backing up original fedora users file
157      */
158     private void createFedoraUsersTestFile() {
159         System.out.println("Creating test Fedora Users file");
160         // backup existing
161         backupFedoraUsersFile();
162 
163         String sep = System.getProperty("line.seperator");
164         if (sep == null) {
165             sep = "\n";
166         }
167 
168         if (!fedoraUsersBackup.exists()) {
169             throw new RuntimeException("Fedora Users backup file expected, but none present");
170         }
171 
172         // get existing file and add test user and role
173         FileInputStream fis;
174         try {
175             // nb from backup file, which is only ever backed-up once, to avoid re-patching a patched users file
176             fis = new FileInputStream(fedoraUsersBackup);
177             BufferedReader br = new BufferedReader(new InputStreamReader(fis));
178 
179             boolean addedUser = false;
180             String usersTag = "<users>";
181 
182             // read through each line in turn, when we find the <users> tag add the new user, then copy the rest of the file
183             String inp;
184             StringBuilder data = new StringBuilder();
185             while ((inp = br.readLine()) != null) {
186                 if (!addedUser && inp.contains(usersTag)) {
187                     // found start of users section
188                     // add this...
189                     data.append(inp + sep);
190                     // and the test user
191                     data.append("<user name=\"testuser\" password=\"testuser\">" + sep);
192                     data.append("<attribute name=\"fedoraRole\">" + sep);
193                     data.append("<value>testing</value>" + sep);
194                     data.append("</attribute>" + sep);
195                     data.append("</user>" + sep);
196                     addedUser = true;
197 
198                 } else {
199                     data.append(inp + sep);
200                 }
201             }
202 
203             // overwrite existing with new version
204             FileOutputStream fu =
205                 new FileOutputStream(FedoraUsers.fedoraUsersXML);
206             OutputStreamWriter pw = new OutputStreamWriter(fu);
207             pw.write(data.toString());
208             pw.close();
209         } catch (IOException e) {
210             System.out.println("Error generating test fedora-users.xml: " + e.getMessage());
211             throw new RuntimeException(e);
212         }
213 
214     }

217     private void backupFedoraUsersFile() {
218         fedoraUsersBackup =
219                 new File(FedoraUsers.fedoraUsersXML.getAbsolutePath()
220                          + ".backup-fesl");
221         if (!fedoraUsersBackup.exists()) {
222             System.out.println("Backing Up Fedora Users");
223             try {
224                 fedoraUsersBackup.createNewFile();
225                 copyFile(FedoraUsers.fedoraUsersXML, fedoraUsersBackup);
226             } catch (IOException e) {
227                 e.printStackTrace();
228             }
229         }
230     }
231 
232     private void restoreFedoraUsersFile() {
233         if (!fedoraUsersBackup.exists()) {
234             System.out.println("Error - Fedora Users backup file does not exist");
235         } else {
236             System.out.println("Restoring Fedora Users");
237             copyFile(fedoraUsersBackup, FedoraUsers.fedoraUsersXML);
238         }
239     }

241     private boolean copyFile(File src, File dest) {
242         InputStream in;
243         try {
244             in = new FileInputStream(src);
245             OutputStream out = new FileOutputStream(dest);
246             StreamUtility.pipeStream(in, out, 1024);
247             in.close();
248             out.close();
249             return true;
250         } catch (IOException e) {
251             throw new RuntimeException(e);
252         }
