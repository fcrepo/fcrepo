29      private static byte[] getTestObjectBytes(String pid) throws Exception {
30          StringBuffer xml = new StringBuffer();
31          xml.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
32          xml.append("<foxml:digitalObject VERSION=\"1.1\" xmlns:xsi=\""
33                     + XSI.uri + "\"\n");
34  
35          xml.append("           xmlns:foxml=\"" + FOXML.uri + "\"\n");
36          xml.append("           xsi:schemaLocation=\"" + FOXML.uri + " "
37                     + FOXML1_1.xsdLocation + "\"\n");
38          xml.append("\n           PID=\"" + pid + "\">\n");
39          xml.append("  <foxml:objectProperties>\n");
40          xml.append("    <foxml:property NAME=\"" + MODEL.LABEL.uri
41                     + "\" VALUE=\"label\"/>\n");
42          xml.append("  </foxml:objectProperties>\n");
43          xml.append("</foxml:digitalObject>");
44          return xml.toString().getBytes("UTF-8");
45      }
46  
47      @Override
48      public void setUp() throws Exception {
49          System.out.println("Ingesting test object: " + m_pid);
50          m_apim = FedoraTestCase.getFedoraClient().getAPIMMTOM();
51          m_apim.ingest(TypeUtility.convertBytesToDataHandler(getTestObjectBytes(m_pid)), FOXML1_1.uri, "");
52      }
53  
54      @Override
55      public void tearDown() throws Exception {
56          System.out.println("Purging test object: " + m_pid);
57          m_apim.purgeObject(m_pid, "", false);
58      }
