113     // hack to dynamically set protocol based on settings in beSecurity
114     // Settings for fedoraInternalCall-1 should have callSSL=true when server is secure
115     public static String getProtocol() {
116         BufferedReader br = null;
117         try {
118             br =
119                     new BufferedReader(new InputStreamReader(new FileInputStream(BESECURITY)));
120             String line = null;
121             while ((line = br.readLine()) != null) {
122                 if (line.indexOf("role=\"fedoraInternalCall-1\"") > 0
123                         && line.indexOf("callSSL=\"true\"") > 0) {
124                     ssl = "https";
125                     break;
126                 }
127             }
128         } catch (Exception e) {
129             System.out.println("fedora.home: " + FEDORA_HOME);
130             fail("beSecurity file Not found: " + BESECURITY.getAbsolutePath());
131         } finally {
132             try {
133                 if (br != null) {
134                     br.close();
135                     br = null;
136                 }
137             } catch (Exception e) {
138                 System.out.println("Unable to close BufferdReader");
139             }
140         }
141         return ssl;
142     }
