20  public class ExistingTomcat
21          extends Tomcat {
22  
23      private final File installDir;
24      private File commonLib;
25  
26      public ExistingTomcat(Distribution dist, InstallOptions options) {
27          super(dist, options);
28          installDir =
29                  new File(getOptions().getValue(InstallOptions.FEDORA_HOME)
30                          + File.separator + "install" + File.separator);
31      }
32  
33      @Override
34      protected void installTomcat() throws InstallationFailedException {
35          // nothing to do
36      }
37  
38      @Override
39      protected void installServerXML() throws InstallationFailedException {
40          try {
41              File distServerXML = new File(getConf(), "server.xml");
42              TomcatServerXML serverXML =
43                      new TomcatServerXML(distServerXML, getOptions());
44              serverXML.update();
45  
46              File example = new File(installDir, "server.xml");
47              serverXML.write(example.getAbsolutePath());
48              System.out.println("Will not overwrite existing "
49                      + distServerXML.getAbsolutePath() + ".\n"
50                      + "Wrote example server.xml to: \n\t"
51                      + example.getAbsolutePath());
52          } catch (IOException e) {
53              throw new InstallationFailedException(e.getMessage(), e);
54          } catch (DocumentException e) {
55              throw new InstallationFailedException(e.getMessage(), e);
56          }
57      }
58  
59      @Override
60      protected void installIncludedKeystore() throws InstallationFailedException {
61          String keystoreFile =
62                  getOptions().getValue(InstallOptions.KEYSTORE_FILE);
63          if (keystoreFile == null
64                  || !keystoreFile.equals(InstallOptions.INCLUDED)) {
65              // nothing to do
66              return;
67          }
68          try {
69              InputStream is = getDist().get(Distribution.KEYSTORE);
70  
71              File keystore = getIncludedKeystore();
72              if (keystore.exists()) {
73                  System.out
74                          .println("WARNING: A keystore file already exists at: "
75                                  + keystore.getAbsolutePath() + ".");
76                  keystore = new File(installDir, Distribution.KEYSTORE);
77                  System.out
78                          .println("WARNING: The existing keystore will not be overwritten.");
79                  System.out
80                          .println("WARNING: The installer-provided keystore will not be installed, it will be copied to: \n\t"
81                                  + keystore.getAbsolutePath());
82              }
83              if (!FileUtils.copy(is, new FileOutputStream(keystore))) {
84                  throw new InstallationFailedException("Copy to "
85                          + keystore.getAbsolutePath() + " failed.");
86              }
87          } catch (IOException e) {
88              throw new InstallationFailedException(e.getMessage(), e);
89          }
90      }
