5   package org.fcrepo.common.policy;
6   
7   import java.util.Vector;
8   
9   public abstract class XacmlNamespace {
10  
11      public String uri;
12  
13      private final Vector<XacmlNamespace> memberNamespaces =
14              new Vector<XacmlNamespace>();
15  
16      private final Vector<XacmlName> memberNames = new Vector<XacmlName>();
17  
18      protected XacmlNamespace(XacmlNamespace parent, String localName) {
19          uri = (parent == null ? "" : parent.uri + ":") + localName;
20      }
21  
22      protected XacmlNamespace addNamespace(XacmlNamespace namespace) {
23          XacmlNamespace result = null;
24          if (memberNamespaces.add(namespace)) {
25              result = namespace;
26          }
27          return result;
28      }
29  
30      protected XacmlName addName(XacmlName name) {
31          XacmlName result = null;
32          if (memberNames.add(name)) {
33              result = name;
34          }
35          return result;
36      }
37  
38      public void flatRep(Vector<XacmlName> flatRep) {
39          flatRep.addAll(memberNames);
40          for (int i = 0; i < memberNamespaces.size(); i++) {
41              memberNamespaces.get(i).flatRep(flatRep);
42          }
43      }
44  
45  }
