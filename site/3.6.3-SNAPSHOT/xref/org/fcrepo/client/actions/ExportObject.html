5   package org.fcrepo.client.actions;
6   
7   import java.awt.FileDialog;
8   import java.awt.event.ActionEvent;
9   
10  import java.io.File;
11  import java.io.FileOutputStream;
12  
13  import java.util.HashSet;
14  import java.util.Iterator;
15  import java.util.Set;
16  
17  import javax.swing.AbstractAction;
18  import javax.swing.JFileChooser;
19  import javax.swing.JOptionPane;
20  
21  import org.fcrepo.client.Administrator;
22  import org.fcrepo.client.export.ExportOptionsDialog;
23  import org.fcrepo.client.utility.export.AutoExporter;
24  
25  
26  /**
27   * @author Chris Wilper
28   */
29  public class ExportObject
30          extends AbstractAction {
31  
32      private static final long serialVersionUID = 1L;
33  
34      private Set m_pids;
35  
36      private boolean m_prompt;
37  
38      public ExportObject() {
39          super("One Object...");
40          m_prompt = true;
41      }
42  
43      public ExportObject(String pid) {
44          super("Export...");
45          m_pids = new HashSet();
46          m_pids.add(pid);
47      }
48  
49      public ExportObject(Set pids) {
50          super("Export Objects...");
51          m_pids = pids;
52      }
53  
54      public void actionPerformed(ActionEvent ae) {
55          AutoExporter exporter = null;
56          try {
57              exporter = new AutoExporter(Administrator.APIA, Administrator.APIM);
58          } catch (Exception e) {
59              Administrator.showErrorDialog(Administrator.getDesktop(),
60                                            "Export Failure",
61                                            e.getClass().getName() + ": "
62                                                    + e.getMessage(),
63                                            e);
64          }
65          if (exporter != null) {
66              if (m_prompt) {
67                  String pid = JOptionPane.showInputDialog("Enter the PID.");
68                  if (pid == null) {
69                      return;
70                  }
71                  m_pids = new HashSet();
72                  m_pids.add(pid);
73              }
74              Iterator pidIter = m_pids.iterator();
75              if (m_pids.size() == 1) {
76                  // If there's only one pid, get export filename
77                  String pid = (String) pidIter.next();
78                  try {
79                      FileDialog dlg =
80                              new FileDialog(Administrator.INSTANCE,
81                                             "Export object to...",
82                                             FileDialog.SAVE);
83                      if (Administrator.getLastDir() != null) {
84                          dlg.setDirectory(Administrator.getLastDir().getPath());
85                      }
86                      dlg.setVisible(true);
87                      if (dlg.getFile() != null) {
88                          File file =
89                                  new File(new File(dlg.getDirectory()), dlg
90                                          .getFile());
91                          Administrator.setLastDir(file.getParentFile()); // remember the dir for next time
92                          ExportOptionsDialog optsDialog =
93                                  new ExportOptionsDialog("Select Options for Export");
94                          if (optsDialog.getFormatSelection() != null) {
95                              exporter.export(pid,
96                                              optsDialog.getFormatSelection(),
97                                              optsDialog.getContextSelection(),
98                                              new FileOutputStream(file));
99                              JOptionPane.showMessageDialog(Administrator
100                                     .getDesktop(), "Exported " + pid);
101                         }
102                     }
103                 } catch (Exception e) {
104                     Administrator.showErrorDialog(Administrator.getDesktop(),
105                                                   "Export Failure",
106                                                   e.getClass().getName() + ": "
107                                                           + e.getMessage(),
108                                                   e);
109                 }
110             } else {
111                 // If there are multiple pids, select a directory first.
112                 try {
113                     JFileChooser browse;
114                     if (Administrator.getLastDir() == null) {
115                         browse = new JFileChooser();
116                     } else {
117                         browse = new JFileChooser(Administrator.getLastDir());
118                     }
119                     browse.setApproveButtonText("Export");
120                     browse.setApproveButtonMnemonic('E');
121                     browse
122                             .setApproveButtonToolTipText("Exports to the selected directory.");
123                     browse.setDialogTitle("Export to...");
124                     browse.setDialogTitle("Choose export directory...");
125                     browse.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
126                     int returnVal =
127                             browse.showOpenDialog(Administrator.getDesktop());
128                     if (returnVal == JFileChooser.APPROVE_OPTION) {
129                         Administrator.setLastDir(browse.getSelectedFile()); // remember the dir for next time
130                         ExportOptionsDialog optsDialog =
131                                 new ExportOptionsDialog("Select Options for Export");
132                         if (optsDialog.getFormatSelection() != null) {
133                             while (pidIter.hasNext()) {
134                                 String pid = (String) pidIter.next();
135                                 StringBuffer buf = new StringBuffer();
136                                 for (int i = 0; i < pid.length(); i++) {
137                                     char c = pid.charAt(i);
138                                     if (c == ':') {
139                                         buf.append('_');
140                                     } else {
141                                         buf.append(c);
142                                     }
143                                 }
144                                 File outFile =
145                                         new File(browse.getSelectedFile(), buf
146                                                 .toString()
147                                                 + ".xml");
148                                 exporter
149                                         .export(pid,
150                                                 optsDialog.getFormatSelection(),
151                                                 optsDialog
152                                                         .getContextSelection(),
153                                                 new FileOutputStream(outFile));
154                             }
155                             JOptionPane.showMessageDialog(Administrator
156                                     .getDesktop(), "Exported " + m_pids.size()
157                                     + " objects.");
158                         }
159                     }
160                 } catch (Exception e) {
161                     Administrator.showErrorDialog(Administrator.getDesktop(),
162                                                   "Export Failure",
163                                                   e.getClass().getName() + ": "
164                                                           + e.getMessage(),
165                                                   e);
166                 }
167             }
168         }
169     }
