5   package org.fcrepo.client.actions;
6   
7   import org.fcrepo.client.Administrator;
8   import org.fcrepo.client.objecteditor.ObjectEditorFrame;
9   import org.fcrepo.client.utility.AutoPurger;
10  import org.fcrepo.server.utilities.StringUtility;
11  
12  import javax.swing.*;
13  import java.awt.event.ActionEvent;
14  import java.util.HashSet;
15  import java.util.Iterator;
16  import java.util.Set;
17  
18  
19  /**
20   * @author Chris Wilper
21   */
22  public class PurgeObject
23          extends AbstractAction {
24  
25      private static final long serialVersionUID = 1L;
26  
27      private Set m_pids;
28  
29      private boolean m_prompt;
30  
31      private ObjectEditorFrame m_parent;
32  
33      public PurgeObject() {
34          super("Purge Object...");
35          m_prompt = true;
36      }
37  
38      public PurgeObject(ObjectEditorFrame parent, String pid) {
39          super("Purge Object");
40          m_pids = new HashSet();
41          m_pids.add(pid);
42          m_parent = parent;
43      }
44  
45      public PurgeObject(Set pids) {
46          super("Purge Objects");
47          m_pids = pids;
48      }
49  
50      public PurgeObject(String pid) {
51          super("Purge Object");
52          m_pids = new HashSet();
53          m_pids.add(pid);
54      }
55  
56      public void actionPerformed(ActionEvent ae) {
57          boolean failed = false;
58          if (m_prompt) {
59              String pid = JOptionPane.showInputDialog("Enter the PID.");
60              if (pid == null) {
61                  return;
62              }
63              m_pids = new HashSet();
64              m_pids.add(pid);
65          }
66          AutoPurger purger = null;
67          try {
68              purger = new AutoPurger(Administrator.APIM);
69          } catch (Exception e) {
70              Administrator
71                      .showErrorDialog(Administrator.getDesktop(),
72                                       "Purge Failure",
73                                       StringUtility.prettyPrint(e.getClass()
74                                               .getName()
75                                                                 + ": " + e.getMessage(), 70, null),
76                                       e);
77          }
78          if (purger != null) {
79              Iterator pidIter = m_pids.iterator();
80              if (m_pids.size() == 1) {
81                  String pid = (String) pidIter.next();
82                  // just purge one
83                  String reason =
84                          JOptionPane
85                                  .showInputDialog("Why are you permanently removing "
86                                                   + pid + "?");
87                  if (reason != null) {
88                      try {
89                          purger.purge(pid, reason);
90                      } catch (Exception e) {
91                          Administrator.showErrorDialog(Administrator
92                                  .getDesktop(), "Purge Failure", StringUtility
93                                  .prettyPrint(e.getClass().getName() + ": "
94                                               + e.getMessage(), 70, null), e);
95                          failed = true;
96                      }
97                      if (!failed) {
98                          JOptionPane.showMessageDialog(Administrator
99                                  .getDesktop(), "Purge succeeded.");
100                         if (m_parent != null) {
101                             m_parent.dispose();
102                         }
103                     }
104                 }
105             } else {
106                 // purge multiple
107                 String reason =
108                         JOptionPane
109                                 .showInputDialog("Why are you permanently removing these objects?");
110                 if (reason != null) {
111                     while (pidIter.hasNext()) {
112                         try {
113                             String pid = (String) pidIter.next();
114                             purger.purge(pid, reason);
115                         } catch (Exception e) {
116                             Administrator
117                                     .showErrorDialog(Administrator.getDesktop(),
118                                                      "Purge Failure",
119                                                      StringUtility
120                                                              .prettyPrint(e
121                                                              .getClass()
122                                                              .getName()
123                                                                           + ": "
124                                                                           + e
125                                                              .getMessage(),
126                                                                           70,
127                                                                           null),
128                                                      e);
129                             failed = true;
130                         }
131                     }
132                     if (!failed) {
133                         JOptionPane.showMessageDialog(Administrator
134                                 .getDesktop(), "Purge of " + m_pids.size()
135                                                + " objects succeeded.");
136                     }
137                 }
138             }
139         }
140     }
