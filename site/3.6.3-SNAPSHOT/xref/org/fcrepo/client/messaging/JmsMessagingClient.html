69      /**
70       * Creates a messaging client
71       *
72       * <h4>Client ID</h4>
73       * <p>
74       * The clientId provides applications with the opportunity to create
75       * multiple messaging clients and track them based on this identifier.
76       * The clientId is used within the MessagingClient when creating a
77       * connection for durable subscriptions.
78       * </p>
79       *
80       * <h4>Message Listener</h4>
81       * <p>
82       * A listener, the onMessage() method of which will be called when a
83       * message arrives from the messaging provider. See the documentation
84       * for javax.jms.MessageListener for more information.
85       * </p>
86       *
87       * <h4>Connection Properties</h4>
88       *
89       * <p>All of the following properties must be included:</p>
90       * <table border="1">
91       *   <th>Property</th>
92       *   <th>Description</th>
93       *   <th>Example Value</th>
94       *   <tr>
95       *     <td>java.naming.factory.initial (javax.naming.Context.INITIAL_CONTEXT_FACTORY)</td>
96       *     <td>The JNDI initial context factory which will allow lookup of the other attributes</td>
97       *     <td>org.apache.activemq.jndi.ActiveMQInitialContextFactory</td>
98       *   </tr>
99       *   <tr>
100      *     <td>java.naming.provider.url (javax.naming.Context.PROVIDER_URL)</td>
101      *     <td>The JNDI provider URL
102      *     <td>tcp://localhost:61616</td>
103      *   </tr>
104      *   <tr>
105      *     <td>connection.factory.name (org.fcrepo.server.messaging.JMSManager.CONNECTION_FACTORY)</td>
106      *     <td>The JNDI name of the connection factory needed to create a JMS Connection</td>
107      *     <td>ConnectionFactory</td>
108      *   </tr>
109      * </table>
110      * <p>One or more of the following properties must be specified:</p>
111      * <table border="1">
112      *   <th>Property Name</th>
113      *   <th>Description</th>
114      *   <th>Example Value</th>
115      *   <tr>
116      *     <td>topic.X (where X = name of subscription topic, example - topic.fedoraManagement)</td>
117      *     <td>A topic over which notification messages will be provided</td>
118      *     <td>fedora.apim.*</td>
119      *   </tr>
120      *   <tr>
121      *     <td>queue.X (where X = name of subscription queue, example - queue.fedoraManagement)</td>
122      *     <td>A queue through which notification messages will be provided</td>
123      *     <td>fedora.apim.update</td>
124      *   </tr>
125      * </table>
126      *
127      * <h4>Durable</h4>
128      * <p>
129      * Specifies whether the topics included in the connection properties should
130      * have durable consumers. If set to true, all topics listeners will be
131      * constructed as durable subscribers. If set to false, all topic listeners
132      * will be constructed as non-durable subscribers. This does not affect
133      * queue listeners.
134      * </p>
135      * <p>
136      * If there is a need for multiple topics, some of which are durable and some
137      * of which are not, then two MessagingClients should be created.
138      * One client would include topics needing durable subscribers and the other
139      * client would include topics not needing durable subscribers.
140      * A single MessageListener can be registered as the listener for both clients.
141      * </p>
142      *
143      * <h4>Message Selector</h4>
144      * <p>
145      * A JMS message selector allows a client to specify, by header field references
146      * and property references, the messages it is interested in. Only messages
147      * whose header and property values match the selector are delivered.
148      * See the javadoc for javax.jms.Message for more information about message selectors.
149      * </p>
150      *
151      * @param clientId identification value for this messaging client
152      * @param listener the listener which will be called when messages arrive
153      * @param connectionProperties set of properties necessary to connect to JMS provider
154      * @param messageSelector a selection which determines the messages to deliver
155      * @param durable determines if the underlying JMS subscribers are durable
156      * @throws MessagingException if listener is null or required properties are not set
157      */
158     public JmsMessagingClient(String clientId,
159                               MessagingListener listener,
160                               Properties connectionProperties,
161                               String messageSelector,
162                               boolean durable)
163             throws MessagingException {
164 
165         // Check for a null listener
166         if (listener == null) {
167             throw new MessagingException("MessageListener may not be null");
168         }
169 
170         // Check for null properties
171         if (connectionProperties == null) {
172             throw new MessagingException("Connection properties may not be null");
173         }
174 
175         // Check for required property values
176         String initialContextFactory =
177             connectionProperties.getProperty(Context.INITIAL_CONTEXT_FACTORY);
178         String providerUrl =
179             connectionProperties.getProperty(Context.PROVIDER_URL);
180         String connectionFactoryName =
181             connectionProperties.getProperty(JMSManager.CONNECTION_FACTORY_NAME);
182 
183         if (initialContextFactory == null
184             || providerUrl == null
185             || connectionFactoryName == null) {
186             throw new MessagingException("Propery values for "
187                     + "'java.naming.factory.initial', "
188                     + "'java.naming.provider.url', and"
189                     + "'connection.factory.name' must be provided "
190                     + "in order to initialize a messaging client");
191         }
192 
193         // Check for valid client ID if durable subscriptions are required
194         if (durable) {
195             if (clientId == null || clientId.equals("")) {
196                 throw new MessagingException("ClientId must be "
197                         + "specified for durable subscriptions");
198             }
199         }
200 
201         m_clientId = clientId;
202         m_listener = listener;
203         m_connectionProperties = connectionProperties;
204         m_messageSelector = messageSelector;
205         m_durable = durable;
206     }

216     /**
217      * Starts the MessagingClient. This method must be called
218      * in order to receive messages.
219      *
220      * @param wait Set to true to wait until the startup process
221      *             is complete before returning. Set to false to
222      *             allow for asynchronous startup.
223      */
224     public void start(boolean wait) throws MessagingException {
225         Thread connector = new JMSBrokerConnector();
226         connector.start();
227 
228         if(wait) {
229             int maxWait = RETRY_INTERVAL * MAX_RETRIES;
230             int waitTime = 0;
231             while(!isConnected()) {
232                 if(waitTime < maxWait) {
233                     try {
234                         Thread.sleep(100);
235                         waitTime += 100;
236                     } catch (InterruptedException ie) {
237                         Thread.currentThread().interrupt();
238                     }
239                 } else {
240                     throw new MessagingException("Timeout reached waiting " +
241                                            "for messaging client to start.");
242                 }
243             }
244         }
245     }

251     /**
252      * Creates topics and queues and starts listeners
253      */
254     private void createDestinations() throws MessagingException {
255         try {
256             // Create Destinations based on properties
257             Enumeration<?> propertyNames = m_connectionProperties.keys();
258             while (propertyNames.hasMoreElements()) {
259                 String propertyName = (String) propertyNames.nextElement();
260                 if (propertyName.startsWith("topic.")) {
261                     String destinationName =
262                             m_connectionProperties.getProperty(propertyName);
263                     m_jmsManager.createDestination(destinationName,
264                                                    DestinationType.Topic);
265                 } else if (propertyName.startsWith("queue.")) {
266                     String destinationName =
267                             m_connectionProperties.getProperty(propertyName);
268                     m_jmsManager.createDestination(destinationName,
269                                                    DestinationType.Queue);
270                 }
271             }
272 
273             // Get destination list
274             List<Destination> destinations = m_jmsManager.getDestinations();
275 
276             // If there are no Destinations, throw an exception
277             if (destinations.size() == 0) {
278                 throw new MessagingException("No destinations available for "
279                         + "subscription, make sure that there is at least one topic "
280                         + "or queue specified in the connection properties.");
281             }
282 
283             // Subscribe
284             for (Destination destination : destinations) {
285                 if (m_durable && (destination instanceof Topic)) {
286                     m_jmsManager.listenDurable((Topic) destination,
287                                                m_messageSelector,
288                                                this,
289                                                null);
290                 } else {
291                     m_jmsManager.listen(destination, m_messageSelector, this);
292                 }
293             }
294         } catch (MessagingException me) {
295             logger.error("MessagingException encountered attempting to start "
296                     + "Messaging Client: " + m_clientId
297                     + ". Exception message: " + me.getMessage(), me);
298             throw me;
299         }
300     }

302     /**
303      * Stops the MessagingClient, shuts down connections. If the unsubscribe
304      * parameter is set to true, all durable subscriptions will be removed.
305      *
306      * @param unsubscribe
307      */
308     public void stop(boolean unsubscribe) throws MessagingException {
309         try {
310             if (unsubscribe) {
311                 m_jmsManager.unsubscribeAllDurable();
312             }
313             m_jmsManager.close();
314             m_jmsManager = null;
315             m_connected = false;
316         } catch (MessagingException me) {
317             logger.error("Messaging Exception encountered attempting to stop "
318                     + "Messaging Client: " + m_clientId
319                     + ". Exception message: " + me.getMessage(), me);
320             throw me;
321         }
322     }

334     /**
335      * Starts the connection to the JMS Broker. Retries on failure.
336      */
337     private class JMSBrokerConnector extends Thread {
338 
339         @Override
340         public void run() {
341             try {
342                 connect();
343                 createDestinations();
344                 m_connected = true;
345             } catch (MessagingException me) {
346                 throw new RuntimeException(me);
347             }
348         }
349 
350         private void connect() throws MessagingException {
351             int retries = 0;
352 
353             while(m_jmsManager == null && retries < MAX_RETRIES) {
354                 try {
355                     m_jmsManager = new JMSManager(m_connectionProperties, m_clientId);
356                 } catch(MessagingException me) {
357                     Throwable rootCause = me.getCause();
358                     while(rootCause.getCause() != null) {
359                         rootCause = rootCause.getCause();
360                     }
361 
362                     if(rootCause instanceof java.net.ConnectException) {
363                         try {
364                             sleep(RETRY_INTERVAL);
365                         } catch (InterruptedException ie) {
366                             Thread.currentThread().interrupt();
367                         }
368                         retries++;
369                     } else {
370                         throw me;
371                     }
372                 }
373             }
374 
375             if(m_jmsManager == null) {
376                 String errorMessage =
377                     "Unable to start JMS Messaging Client, " + MAX_RETRIES +
378                     " attempts were made, each attempt resulted in a " +
379                     "java.net.ConnectException. The messaging broker at " +
380                     m_connectionProperties.getProperty(Context.PROVIDER_URL) +
381                     " is not available";
382                 throw new RuntimeException(errorMessage);
383             }
384         }
385     }
