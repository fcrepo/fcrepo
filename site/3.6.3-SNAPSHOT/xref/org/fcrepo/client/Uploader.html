20  /**
21   * A client to a Fedora server's upload facility, accessed via a
22   * basic-authenticated multipart POST to the server. See
23   * server.management.UploadServlet for protocol details.

57          fc = new FedoraClient(baseURL, user, pass);

60      /**
61       * Send the data from the stream to the server. This is less efficient than
62       * <i>upload(File)</i>, but if you already have a stream, it's convenient.
63       * This method takes care of temporarily making a File out of the stream,
64       * making the request, and removing the temporary file. Having a File source
65       * for the upload is necessary because the content-length must be sent along
66       * with the request as per the HTTP Multipart POST protocol spec.
67       */
68      public String upload(InputStream in) throws IOException {
69          File tempFile = File.createTempFile("fedora-upload-", null);
70          FileOutputStream out = new FileOutputStream(tempFile);
71          try {
72              StreamUtility.pipeStream(in, out, 8192);
73              return upload(tempFile);
74          } finally {
75              in.close();
76              out.close();
77              if (!tempFile.delete()) {
78                  System.err.println("WARNING: Could not remove temporary file: "
79                          + tempFile.getName());
80                  tempFile.deleteOnExit();
81              }
82          }
83      }

85      /**
86       * Send a file to the server, getting back the identifier.
87       */
88      public String upload(File file) throws IOException {
89          if (Administrator.INSTANCE == null) {
90              return fc.uploadFile(file);
91          } else {
92              // paint initial status to the progress bar
93              String msg =
94                      "Uploading " + file.length() + " bytes to "
95                              + fc.getUploadURL();
96              Dimension d = Administrator.PROGRESS.getSize();
97              Administrator.PROGRESS.setString(msg);
98              Administrator.PROGRESS.setValue(100);
99              Administrator.PROGRESS.paintImmediately(0,
100                                                     0,
101                                                     (int) d.getWidth() - 1,
102                                                     (int) d.getHeight() - 1);
103 
104             // then start the thread, passing parms in
105             HashMap<String, Object> PARMS = new HashMap<String, Object>();
106             PARMS.put("fc", fc);
107             PARMS.put("file", file);
108             SwingWorker worker = new SwingWorker(PARMS) {
109 
110                 @Override
111                 public Object construct() {
112                     try {
113                         FedoraClient fc = (FedoraClient) parms.get("fc");
114                         File file = (File) parms.get("file");
115                         return fc.uploadFile(file);
116                     } catch (IOException e) {
117                         thrownException = e;
118                         return "";
119                     }
120                 }
121             };
122             worker.start();
123 
124             // keep updating status till the worker's finished
125             int ms = 200;
126             while (!worker.done) {
127                 try {
128                     Administrator.PROGRESS.setValue(ms);
129                     Administrator.PROGRESS.paintImmediately(0, 0, (int) d
130                             .getWidth() - 1, (int) d.getHeight() - 1);
131                     Thread.sleep(100);
132                     ms = ms + 100;
133                     if (ms >= 2000) {
134                         ms = 200;
135                     }
136                 } catch (InterruptedException ie) {
137                 }
138             }
139 
140             // reset the status bar to normal
141             Administrator.PROGRESS.setValue(2000);
142             Administrator.PROGRESS.paintImmediately(0,
143                                                     0,
144                                                     (int) d.getWidth() - 1,
145                                                     (int) d.getHeight() - 1);
146             try {
147                 Thread.sleep(100);
148             } catch (InterruptedException ie) {
149             }
150 
151             // report if there was an error; otherwise return the response
152             if (worker.thrownException != null) {
153                 throw (IOException) worker.thrownException;
154             } else {
155                 return (String) worker.getValue();
156             }
157 
158         }
159     }
