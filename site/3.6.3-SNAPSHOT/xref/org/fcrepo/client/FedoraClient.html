121     public FedoraClient(String baseURL, String user, String pass)
122             throws MalformedURLException {
123         m_baseURL = baseURL;
124         m_user = user;
125         m_pass = pass;
126         if (!baseURL.endsWith("/")) {
127             m_baseURL += "/";
128         }
129         URL url = new URL(m_baseURL);
130         m_authScope =
131                 new AuthScope(url.getHost(),
132                               AuthScope.ANY_PORT,
133                               AuthScope.ANY_REALM);
134         m_creds = new UsernamePasswordCredentials(user, pass);
135         m_cManager = new MultiThreadedHttpConnectionManager();
136     }

138     public HttpClient getHttpClient() {
139         m_cManager.getParams()
140                 .setDefaultMaxConnectionsPerHost(MAX_CONNECTIONS_PER_HOST);
141         m_cManager.getParams().setMaxTotalConnections(MAX_TOTAL_CONNECTIONS);
142         m_cManager.getParams().setConnectionTimeout(TIMEOUT_SECONDS * 1000);
143         m_cManager.getParams().setSoTimeout(SOCKET_TIMEOUT_SECONDS * 1000);
144         HttpClient client = new HttpClient(m_cManager);
145         client.getState().setCredentials(m_authScope, m_creds);
146         client.getParams().setAuthenticationPreemptive(true);
147         return client;
148     }

150     /**
151      * Upload the given file to Fedora's upload interface via HTTP POST.
152      *
153      * @return the temporary id which can then be passed to API-M requests as a
154      *         URL. It will look like uploaded://123
155      */
156     public String uploadFile(File file) throws IOException {
157         PostMethod post = null;
158         try {
159             // prepare the post method
160             post = new PostMethod(getUploadURL());
161             post.setDoAuthentication(true);
162             post.getParams().setParameter("Connection", "Keep-Alive");
163 
164             // chunked encoding is not required by the Fedora server,
165             // but makes uploading very large files possible
166             post.setContentChunked(true);
167 
168             // add the file part
169             Part[] parts = {new FilePart("file", file)};
170             post.setRequestEntity(new MultipartRequestEntity(parts, post
171                     .getParams()));
172 
173             // execute and get the response
174             int responseCode = getHttpClient().executeMethod(post);
175             String body = null;
176             try {
177                 body = post.getResponseBodyAsString();
178             } catch (Exception e) {
179                 logger.warn("Error reading response body", e);
180             }
181             if (body == null) {
182                 body = "[empty response body]";
183             }
184             body = body.trim();
185             if (responseCode != HttpStatus.SC_CREATED) {
186                 throw new IOException("Upload failed: "
187                         + HttpStatus.getStatusText(responseCode) + ": "
188                         + replaceNewlines(body, " "));
189             } else {
190                 return replaceNewlines(body, "");
191             }
192         } finally {
193             if (post != null) {
194                 post.releaseConnection();
195             }
196         }
197     }

297     /**
298      * Get an HTTP resource with the response as an InputStream, given a URL.
299      * Note that if the HTTP response has no body, the InputStream will be
300      * empty. The success of a request can be checked with getResponseCode().
301      * Usually you'll want to see a 200. See
302      * http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html for other codes.
303      *
304      * @param url
305      *        A URL that we want to do an HTTP GET upon
306      * @param failIfNotOK
307      *        boolean value indicating if an exception should be thrown if we do
308      *        NOT receive an HTTP 200 response (OK)
309      * @param followRedirects
310      *        boolean value indicating whether HTTP redirects should be handled
311      *        in this method, or be passed along so that they can be handled
312      *        later.
313      * @return HttpInputStream the HTTP response
314      * @throws IOException
315      */
316     public HttpInputStream get(URL url,
317                                boolean failIfNotOK,
318                                boolean followRedirects) throws IOException {
319 
320         String urlString = url.toString();
321         logger.debug("FedoraClient is getting " + urlString);
322         HttpClient client = getHttpClient();
323         GetMethod getMethod = new GetMethod(urlString);
324         getMethod.setDoAuthentication(true);
325         getMethod.setFollowRedirects(followRedirects);
326         HttpInputStream in = new HttpInputStream(client, getMethod, urlString);
327         int status = in.getStatusCode();
328         if (failIfNotOK) {
329             if (status != 200) {
330                 if (followRedirects && 300 <= status && status <= 399) {
331                     // Handle the redirect here !
332                     logger.debug("FedoraClient is handling redirect for HTTP STATUS="
333                             + status);
334                     Header hLoc = in.getResponseHeader("location");
335                     if (hLoc != null) {
336                         logger.debug("FedoraClient is trying redirect location: "
337                                 + hLoc.getValue());
338                         // Try the redirect location, but don't try to handle another level of redirection.
339                         return get(hLoc.getValue(), true, false);
340                     } else {
341                         try {
342                             throw new IOException("Request failed [" + status
343                                     + " " + in.getStatusText() + "]");
344                         } finally {
345                             try {
346                                 in.close();
347                             } catch (Exception e) {
348                                 logger.error("Can't close InputStream: "
349                                         + e.getMessage());
350                             }
351                         }
352                     }
353                 } else {
354                     try {
355                         throw new IOException("Request failed ["
356                                 + in.getStatusCode() + " " + in.getStatusText()
357                                 + "] : " + urlString);
358                     } finally {
359                         try {
360                             in.close();
361                         } catch (Exception e) {
362                             logger.error("Can't close InputStream: "
363                                     + e.getMessage());
364                         }
365                     }
366                 }
367             }
368         }
369         return in;
370     }

372     /**
373      * Get an HTTP resource with the response as a String instead of an
374      * InputStream, given a resource locator that either begins with
375      * 'info:fedora/' , 'http://', or '/'.
376      *
377      * @param locator
378      *        A URL, relative Fedora URL, or Fedora URI that we want to do an
379      *        HTTP GET upon
380      * @param failIfNotOK
381      *        boolean value indicating if an exception should be thrown if we do
382      *        NOT receive an HTTP 200 response (OK)
383      * @param followRedirects
384      *        boolean value indicating whether HTTP redirects should be handled
385      *        in this method, or be passed along so that they can be handled
386      *        later.
387      * @return String the HTTP response as a string
388      * @throws IOException
389      */
390     public String getResponseAsString(String locator,
391                                       boolean failIfNotOK,
392                                       boolean followRedirects)
393             throws IOException {
394 
395         InputStream in = get(locator, failIfNotOK, followRedirects);
396 
397         // Convert the response into a String.
398         try {
399             BufferedReader reader =
400                     new BufferedReader(new InputStreamReader(in));
401             StringBuffer buffer = new StringBuffer();
402             String line = reader.readLine();
403             while (line != null) {
404                 buffer.append(line + "\n");
405                 line = reader.readLine();
406             }
407             return buffer.toString();
408         } finally {
409             try {
410                 in.close();
411             } catch (Exception e) {
412                 logger.error("Can't close InputStream: " + e.getMessage());
413             }
414         }
415     }

417     private String getLocatorAsURL(String locator) throws IOException {
418 
419         String url;
420         if (locator.startsWith(FEDORA_URI_PREFIX)) {
421             url =
422                     m_baseURL + "get/"
423                             + locator.substring(FEDORA_URI_PREFIX.length());
424         } else if (locator.startsWith("http://")
425                 || locator.startsWith("https://")) {
426             url = locator;
427         } else if (locator.startsWith("/")) {
428             // assume it's for something within this Fedora server
429             while (locator.startsWith("/")) {
430                 locator = locator.substring(1);
431             }
432             url = m_baseURL + locator;
433         } else {
434             throw new IOException("Bad locator (must start with '"
435                     + FEDORA_URI_PREFIX + "', 'http[s]://', or '/'");
436         }
437         return url;
438     }

567 
568         ResourceBundle bundle =
569                 ResourceBundle.getBundle("org.fcrepo.client.resources.Client");
570         List<String> list = new ArrayList<String>();
571 
572         String versions = bundle.getString("compatibleServerVersions");
573         if (versions != null && versions.trim().length() > 0) {
574             String[] va = versions.trim().split(" ");
575             for (String element : va) {

581             list.add(getVersion());
582         }
583

587     /**
588      * Get the version reported by the remote Fedora server.
589      */
590     public String getServerVersion() throws IOException {
591         // only do this once -- future invocations will use the known value
592         if (m_serverVersion == null) {
593             // Make the APIA call for describe repository
594             // and make sure that HTTP 302 status is handled.
595             String desc = getResponseAsString("/describe?xml=true", true, true);
596             logger.debug("describeRepository response:\n" + desc);
597             String[] parts = desc.split("<repositoryVersion>");
598             if (parts.length < 2) {
599                 throw new IOException("Could not find repositoryVersion element in content of /describe?xml=true");
600             }
601             int i = parts[1].indexOf("<");
602             if (i == -1) {
603                 throw new IOException("Could not find end of repositoryVersion element in content of /describe?xml=true");
604             }
605             m_serverVersion = parts[1].substring(0, i).trim();
606             logger.debug("Server version is " + m_serverVersion);
607         }
608         return m_serverVersion;
609     }

611     /**
612      * Return the current date as reported by the Fedora server.
613      *
614      * @throws IOException
615      *         if the HTTP Date header is not provided by the server for any
616      *         reason, or it is in the wrong format.
617      */
618     public Date getServerDate() throws IOException {
619         HttpInputStream in = get("/describe", false, false);
620         String dateString = null;
621         try {
622             Header header = in.getResponseHeader("Date");
623             if (header == null) {
624                 throw new IOException("Date was not supplied in HTTP response "
625                         + "header for " + m_baseURL + "describe");
626             }
627             dateString = header.getValue();
628 
629             // This is the date format recommended by RFC2616
630             SimpleDateFormat format =
631                     new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss z",
632                                          Locale.US);
633             format.setTimeZone(TimeZone.getTimeZone("UTC"));
634             return format.parse(dateString);
635 
636         } catch (ParseException e) {
637             throw new IOException("Unparsable date (" + dateString
638                     + ") in HTTP response header for " + m_baseURL + "describe");
639         } finally {
640             in.close();
641         }
642     }

644     public Date getLastModifiedDate(String locator) throws IOException {
645         if (locator.startsWith(FEDORA_URI_PREFIX)) {
646             String query =
647                     "select $date " + "from <#ri> " + "where <" + locator
648                             + "> <" + VIEW.LAST_MODIFIED_DATE.uri + "> $date";
649             Map<String, String> map = new HashMap<String, String>();
650             map.put("lang", "itql");
651             map.put("query", query);
652             TupleIterator tuples = getTuples(map);
653             try {
654                 if (tuples.hasNext()) {
655                     Map<String, Node> row = tuples.next();
656                     Literal dateLiteral = (Literal) row.get("date");
657                     if (dateLiteral == null) {
658                         throw new IOException("A row was returned, but it did not contain a 'date' binding");
659                     }
660                     return DateUtility.parseDateLoose(dateLiteral
661                             .getLexicalForm());
662                 } else {
663                     throw new IOException("No rows were returned");
664                 }
665             } catch (TrippiException e) {
666                 throw new IOException(e.getMessage());
667             } finally {
668                 try {
669                     tuples.close();
670                 } catch (Exception e) {
671                 }
672             }
673         } else {

756     private String encodeParameters(Map<String, String> params) {
757         StringBuffer encoded = new StringBuffer();
758         Iterator<String> iter = params.keySet().iterator();
759         int n = 0;
760         while (iter.hasNext()) {
761             String name = iter.next();
762             if (n > 0) {
763                 encoded.append("&");
764             }
765             n++;
766             encoded.append(name);
767             encoded.append('=');
768             try {
769                 encoded.append(URLEncoder.encode(params.get(name), "UTF-8"));
770             } catch (UnsupportedEncodingException e) { // UTF-8 won't fail
771             }
772         }
773         return encoded.toString();
774     }
