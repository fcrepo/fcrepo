5   package org.fcrepo.client;

7   import java.io.PrintStream;
8   
9   import java.util.Properties;
10  
11  import javax.swing.JTextArea;
12  
13  import org.fcrepo.client.batch.BatchTool;
14  
15  
16  /**
17   * @author Bill Niebel
18   */
19  public class BatchThread
20          extends Thread {
21  
22      private String leadText = "";
23  
24      private Properties properties = null;
25  
26      private final Properties nullProperties = null;
27  
28      private BatchOutput batchOutput = null;
29  
30      private JTextArea jTextArea = null;
31  
32      private PrintStream originalOut = null;
33  
34      private PrintStream originalErr = null;
35  
36      private PrintStream printStream = null;
37  
38      public BatchThread(BatchOutput batchOutput,
39                         JTextArea jTextArea,
40                         String leadText)
41              throws Exception {
42          this.batchOutput = batchOutput;
43          this.jTextArea = jTextArea;
44          this.leadText = leadText;
45          BatchOutputCatcher batchOutputCatcher =
46                  new BatchOutputCatcher(jTextArea);
47          printStream = new PrintStream(batchOutputCatcher, true);
48      }
49  
50      public void setProperties(Properties properties) {
51          this.properties = properties;
52      }
53  
54      @Override
55      public void run() {
56          try {
57              jTextArea.setText(leadText + "\n");
58              originalOut = System.out;
59              originalErr = System.err;
60              System.setOut(printStream);
61              System.setErr(printStream);
62              batchOutput.setVisible(true);
63              BatchTool batchTool =
64                      new BatchTool(properties, nullProperties, nullProperties);
65              batchTool.prep();
66              batchTool.process();
67          } catch (Exception e) {
68          } finally {
69              System.setOut(originalOut);
70              originalOut = null;
71              System.setErr(originalErr);
72              originalErr = null;
73              batchOutput.flush2file(); //2003.12.03 niebel -- duplicate output to file
74          }
75      }
76  }
