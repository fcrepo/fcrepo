20  /**
21   * Turns a datastream input spec into a java object(s).
22   */
23  class DatastreamInputSpecDeserializer
24          extends DefaultHandler {
25  
26      /** The deserialized input spec */
27      private DatastreamInputSpec m_result;
28  
29      // values gathered and built while parsing
30      private String m_label;
31  
32      private List m_bindingRules;
33  
34      private String m_min;
35  
36      private String m_max;
37  
38      private String m_orderMatters;
39  
40      private String m_key;
41  
42      private StringBuffer m_inputLabel;
43  
44      private StringBuffer m_inputInstruction;
45  
46      private StringBuffer m_types;
47  
48      // for character content, which element are we reading?
49      private int m_readingContent;
50  
51      private static int NONE = 0;
52  
53      private static int INPUT_LABEL = 1;
54  
55      private static int TYPES = 2;
56  
57      private static int INPUT_INSTRUCTION = 3;
58  
59      public DatastreamInputSpecDeserializer(InputStream xml)
60              throws IOException {
61          try {
62              SAXParserFactory spf = SAXParserFactory.newInstance();
63              spf.setNamespaceAware(true);
64              SAXParser parser = spf.newSAXParser();
65              parser.parse(xml, this);
66          } catch (Exception e) {
67              if (e.getMessage() != null) {
68                  throw new IOException(e.getMessage());
69              } else {
70                  throw new IOException("Error parsing datastream input spec: "
71                          + e.getClass().getName());
72              }
73          }
74      }
75  
76      public DatastreamInputSpec getResult() {
77          return m_result;
78      }
79  
80      @Override
81      public void startElement(String uri,
82                               String localName,
83                               String qName,
84                               Attributes a) {
85          if (localName.equals("DSInputSpec")) {
86              m_bindingRules = new ArrayList();
87              for (int i = 0; i < a.getLength(); i++) {
88                  String name = a.getLocalName(i);
89                  if (name.equals("label")) {
90                      m_label = a.getValue(i);
91                  }
92              }
93          } else if (localName.equals("DSInput")) {
94              m_inputLabel = new StringBuffer();
95              m_inputInstruction = new StringBuffer();
96              m_types = new StringBuffer();
97              for (int i = 0; i < a.getLength(); i++) {
98                  String name = a.getLocalName(i);
99                  if (name.equals("DSMin")) {
100                     m_min = a.getValue(i);
101                 } else if (name.equals("DSMax")) {
102                     m_max = a.getValue(i);
103                 } else if (name.equals("DSOrdinality")) {
104                     m_orderMatters = a.getValue(i);
105                 } else if (name.equals("wsdlMsgPartName")) {
106                     m_key = a.getValue(i);
107                 }
108             }
109         } else if (localName.equals("DSInputLabel")) {
110             m_readingContent = INPUT_LABEL;
111         } else if (localName.equals("DSMIME")) {
112             m_readingContent = TYPES;
113         } else if (localName.equals("DSInputInstruction")) {
114             m_readingContent = INPUT_INSTRUCTION;
115         }
116     }
117 
118     @Override
119     public void characters(char[] ch, int start, int length) {
120         if (m_readingContent == INPUT_LABEL) {
121             m_inputLabel.append(ch, start, length);
122         } else if (m_readingContent == TYPES) {
123             m_types.append(ch, start, length);
124         } else if (m_readingContent == INPUT_INSTRUCTION) {
125             m_inputInstruction.append(ch, start, length);
126         }
127     }
128 
129     @Override
130     public void endElement(String uri, String localName, String qName)
131             throws SAXException {
132         if (localName.equals("DSInputLabel") || localName.equals("DSMIME")
133                 || localName.equals("DSInputInstruction")) {
134             m_readingContent = NONE;
135             if (localName.equals("DSMIME")) {
136                 // there might be multiple elements, so add a space,
137                 // which can't hurt since later it's trimmed
138                 m_types.append(" ");
139             }
140         } else if (localName.equals("DSInput")) {
141             // first interpret what we read, 
142             // converting to appropriate types and defualt values
143             int min;
144             try {
145                 min = Integer.parseInt(m_min);
146             } catch (Exception e) {
147                 min = 1; // default the same way xsd:minOccurs does
148             }
149             int max;
150             try {
151                 max = Integer.parseInt(m_max);
152             } catch (Exception e) {
153                 if (m_max != null
154                         && (m_max.startsWith("-") || m_max.startsWith("n")
155                                 || m_max.startsWith("N")
156                                 || m_max.startsWith("u")
157                                 || m_max.startsWith("U")
158                                 || m_max.startsWith("i") || m_max
159                                 .startsWith("I"))) {
160                     // covers various spellings of "n/a", "none", "unbounded", 
161                     // "unrestricted", "infinite", and -1
162                     max = -1;
163                 } else {
164                     max = 1; // default the same way xsd:maxOccurs does
165                 }
166             }
167             boolean orderMatters = false;
168             if (m_orderMatters != null
169                     && (m_orderMatters.startsWith("y")
170                             || m_orderMatters.startsWith("Y")
171                             || m_orderMatters.startsWith("t")
172                             || m_orderMatters.startsWith("T") || m_orderMatters
173                             .equals("1"))) {
174                 // covers various spellings of "yes", "true", and 1
175                 orderMatters = true;
176             }
177             String[] types =
178                     m_types.toString().replaceAll(" +", " ").trim().split(" ");
179             // then add it to the list of rules
180             m_bindingRules.add(new DatastreamBindingRule(m_key,
181                                                          m_inputLabel
182                                                                  .toString(),
183                                                          m_inputInstruction
184                                                                  .toString(),
185                                                          min,
186                                                          max,
187                                                          orderMatters,
188                                                          types));
189         } else if (localName.equals("DSInputSpec")) {
190             m_result = new DatastreamInputSpec(m_label, m_bindingRules);
191         }
192     }
