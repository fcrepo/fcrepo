7   import java.util.HashMap;
8   import java.util.Iterator;

10  /**
11   * @author Chris Wilper

13  public class DigitalObject {
14  
15      protected HashMap basisStreams = new HashMap();
16  
17      protected HashMap inlineStreams = new HashMap();
18  
19      private boolean m_dirty = true;
20  
21      private String m_name = "Untitled";
22  
23      public DigitalObject() {
24      }
25  
26      public String getName() {
27          return m_name;
28      }
29  
30      public void setName(String name) {
31          m_dirty = true;
32          m_name = name;
33      }
34  
35      public boolean isDirty() {
36          if (m_dirty) {
37              return true;
38          }
39          Iterator iter;
40          iter = basisStreams.values().iterator();
41          while (iter.hasNext()) {
42              BasisDataStream basis = (BasisDataStream) iter.next();
43              if (basis.isDirty()) {
44                  return true;
45              }
46          }
47          iter = inlineStreams.values().iterator();
48          while (iter.hasNext()) {
49              InlineDataStream inline = (InlineDataStream) iter.next();
50              if (inline.isDirty()) {
51                  return true;
52              }
53          }
54          return false;
55      }
56  
57      public void setClean() {
58          m_dirty = false;
59      }
60  
61      public void setAllClean() {
62          m_dirty = false;
63          Iterator iter;
64          iter = basisStreams.values().iterator();
65          while (iter.hasNext()) {
66              BasisDataStream basis = (BasisDataStream) iter.next();
67              basis.setClean();
68          }
69          iter = inlineStreams.values().iterator();
70          while (iter.hasNext()) {
71              InlineDataStream inline = (InlineDataStream) iter.next();
72              inline.setClean();
73          }
74      }
75  
76  }
