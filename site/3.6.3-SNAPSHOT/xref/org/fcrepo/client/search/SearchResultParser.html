5   package org.fcrepo.client.search;
6   
7   import java.io.IOException;
8   import java.io.InputStream;
9   
10  import java.util.HashSet;
11  import java.util.Set;
12  
13  import javax.xml.parsers.SAXParser;
14  import javax.xml.parsers.SAXParserFactory;
15  
16  import org.xml.sax.Attributes;
17  import org.xml.sax.SAXException;
18  import org.xml.sax.helpers.DefaultHandler;
19  
20  /**
21   * Parses plain old xml results from a Fedora search request.
22   */
23  public class SearchResultParser
24          extends DefaultHandler {
25  
26      private boolean READING_PID;
27  
28      private boolean READING_TOKEN;
29  
30      private StringBuffer m_currentPID;
31  
32      private StringBuffer m_sessionToken;
33  
34      private final Set<String> m_pids = new HashSet<String>();
35  
36      public SearchResultParser(InputStream xml)
37              throws IOException {
38          try {
39              SAXParserFactory spf = SAXParserFactory.newInstance();
40              spf.setNamespaceAware(true);
41              SAXParser parser = spf.newSAXParser();
42              parser.parse(xml, this);
43          } catch (Exception e) {
44              if (e.getMessage() != null) {
45                  throw new IOException(e.getMessage());
46              } else {
47                  throw new IOException("Error parsing search result xml: "
48                          + e.getClass().getName());
49              }
50          }
51      }
52  
53      @Override
54      public void startElement(String uri,
55                               String localName,
56                               String qName,
57                               Attributes a) {
58          if (localName.equals("pid")) {
59              READING_PID = true;
60              m_currentPID = new StringBuffer();
61          } else if (localName.equals("token")) {
62              READING_TOKEN = true;
63              m_sessionToken = new StringBuffer();
64          }
65      }
66  
67      @Override
68      public void characters(char[] ch, int start, int length) {
69          if (READING_PID) {
70              m_currentPID.append(ch, start, length);
71          } else if (READING_TOKEN) {
72              m_sessionToken.append(ch, start, length);
73          }
74      }
75  
76      @Override
77      public void endElement(String uri, String localName, String qName)
78              throws SAXException {
79          if (localName.equals("pid")) {
80              READING_PID = false;
81              m_pids.add(m_currentPID.toString());
82          } else if (localName.equals("token")) {
83              READING_TOKEN = false;
84          }
85      }
86  
87      public Set<String> getPIDs() {
88          return m_pids;
89      }
90  
91      public String getToken() {
92          if (m_sessionToken == null) {
93              return null;
94          }
95          return m_sessionToken.toString();
96      }
97  }
