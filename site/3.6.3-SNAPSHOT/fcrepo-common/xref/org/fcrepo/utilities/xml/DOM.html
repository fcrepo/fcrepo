42      /**
43       * Extracts all textual and CDATA content from the given node and its
44       * children.
45       *
46       * @param node the node to get the content from.
47       * @return the textual content of node.
48       */
49      public static String getElementNodeValue(Node node) {
50          StringWriter sw = new StringWriter(2000);
51          if (node.getNodeType() == Node.ELEMENT_NODE) {
52              NodeList all = node.getChildNodes();
53              for (int i = 0; i < all.getLength(); i++) {
54                  if (all.item(i).getNodeType() == Node.TEXT_NODE ||
55                      all.item(i).getNodeType() == Node.CDATA_SECTION_NODE) {
56                      // TODO: Check if we exceed the limit for getNodeValue
57                      sw.append(all.item(i).getNodeValue());
58                  }
59              }
60          }
61          return sw.toString();
62      }

66      /**
67       * Parses an XML document from a String to a DOM.
68       *
69       * @param xmlString      a String containing an XML document.
70       * @param namespaceAware if {@code true} the parsed DOM will reflect any
71       *                       XML namespaces declared in the document
72       * @return The document in a DOM or {@code null} on errors.
73       */
74      public static Document stringToDOM(String xmlString,
75                                         boolean namespaceAware) {
76          try {
77              InputSource in = new InputSource();
78              in.setCharacterStream(new StringReader(xmlString));
79  
80              DocumentBuilderFactory dbFact = DocumentBuilderFactory.newInstance();
81              dbFact.setNamespaceAware(namespaceAware);
82  
83              return dbFact.newDocumentBuilder().parse(in);
84          } catch (IOException e) {
85              log.warn("I/O error when parsing XML :" + e.getMessage() + "\n"
86                       + xmlString, e);
87          } catch (SAXException e) {
88              log.warn("Parse error when parsing XML :" + e.getMessage() + "\n"
89                       + xmlString, e);
90          } catch (ParserConfigurationException e) {
91              log.warn("Parser configuration error when parsing XML :"
92                       + e.getMessage() + "\n"
93                       + xmlString, e);
94          }
95          return null;
96      }

157     /**
158      * Convert the given DOM to an UTF-8 XML String.
159      *
160      * @param dom                the Document to convert.
161      * @param withXmlDeclaration if trye, an XML-declaration is prepended.
162      * @return the dom as an XML String.
163      * @throws TransformerException if the dom could not be converted.
164      */
165     // TODO: Consider optimizing this with ThreadLocal Transformers
166     public static String domToString(Node dom, boolean withXmlDeclaration)
167             throws TransformerException {
168         Transformer t = TransformerFactory.newInstance().newTransformer();
169         t.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
170         if (withXmlDeclaration) {
171             t.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "no");
172         } else {
173             t.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
174         }
175         t.setOutputProperty(OutputKeys.METHOD, "xml");
176 
177         /* Transformer */
178         StringWriter sw = new StringWriter();
179         t.transform(new DOMSource(dom), new StreamResult(sw));
180 
181         return sw.toString();
182     }
