46      /**
47       * Dates with millisecond precision should come back as given.
48       */
49      @Test
50      public void testMillisecondDatePrecision() throws Exception {
51  
52          String[] DT =
53                  {"1970-01-01T00:00:00.001Z", "1970-01-01T00:00:00.010Z",
54                          "1970-01-01T00:00:00.100Z"};
55          String[] DT_XSD =
56                  {"1970-01-01T00:00:00.001Z", "1970-01-01T00:00:00.01Z",
57                          "1970-01-01T00:00:00.1Z"};
58          Date[] date = {new Date(1L), new Date(10L), new Date(100L)};
59  
60          initRI(1);
61          for (int i = 0; i < DT.length; i++) {
62              testDates(DT[i], date[i], "test:" + i, DT_XSD[i]);
63          }
64      }

66      /**
67       * Test boundary dates.
68       */
69      @Test
70      public void testBoundaryDates() throws Exception {
71          Date EPOCH = new Date(0L);
72          Date ONE_CE = new Date(-62135769600000L);
73          Date ONE_BCE = new Date(-62167392000000L);
74          Date TWO_BCE = new Date(-62198928000000L);
75  
76          String EPOCH_DT = "1970-01-01T00:00:00.000Z";
77          String ONE_CE_DT = "0001-01-01T00:00:00.000Z";
78          String ONE_BCE_DT = "0000-01-01T00:00:00.000Z";
79          String TWO_BCE_DT = "-0001-01-01T00:00:00.000Z";
80  
81          String EPOCH_XSD = "1970-01-01T00:00:00Z";
82          String ONE_CE_XSD = "0001-01-01T00:00:00Z";
83          String ONE_BCE_XSD = "0000-01-01T00:00:00Z";
84          String TWO_BCE_XSD = "-0001-01-01T00:00:00Z";
85  
86          initRI(1);
87          testDates(EPOCH_DT, EPOCH, "test:epoch", EPOCH_XSD);
88          testDates(ONE_CE_DT, ONE_CE, "test:one_ce", ONE_CE_XSD);
89          testDates(ONE_BCE_DT, ONE_BCE, "test:one_bce", ONE_BCE_XSD);
90          testDates(TWO_BCE_DT, TWO_BCE, "test:two_bce", TWO_BCE_XSD);
91      }

93      /**
94       * Test that dateTime, date and xsdDateTime all represent the same date.
95       *
96       * @param dateTime
97       *        dateTime for parsing by DateFormat using the pattern
98       *        "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'".
99       * @param date
100      * @param pid
101      *        PID of the test object
102      * @param xsdDateTime
103      *        canonical lexical representation of the dateTime
104      * @throws Exception
105      */
106     private void testDates(String dateTime,
107                            Date date,
108                            String pid,
109                            String xsdDateTime) throws Exception {
110         Date createDate = _millisFormat.parse(dateTime);
111         assertEquals(date, createDate);
112         DigitalObject obj = getTestObject(pid, pid);
113         obj.setCreateDate(createDate);
114         addObj(obj, true);
115 
116         String query =
117                 String.format("<info:fedora/%s> <%s> *",
118                               pid,
119                               Constants.MODEL.CREATED_DATE.uri);
120         TripleIterator results = spo(query);
121 
122         try {
123             assertTrue(results.hasNext());
124             ObjectNode dateNode = results.next().getObject();
125             assertTrue(dateNode instanceof Literal);
126             Literal dateLiteral = (Literal) dateNode;
127             assertEquals(dateLiteral.getDatatypeURI().toString(),
128                          Constants.RDF_XSD.DATE_TIME.uri);
129             assertEquals(xsdDateTime, dateLiteral.getLexicalForm());
130         } finally {
131             results.close();
132         }
133     }
