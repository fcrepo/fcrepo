162     private void tearDownTriplestore() throws Exception {
163 
164         // delete all triples from the RI
165         File dump = new File(TEST_DIR + "/all-triples.txt");
166         FileOutputStream out = null;
167         try {
168             // write all to temp file
169             TripleIterator triples = _ri.findTriples(null, null, null, -1);
170             out = new FileOutputStream(dump);
171             triples.toStream(out, RDFFormat.TURTLE);
172             try {
173                 out.close();
174             } catch (Exception e) {
175             }
176             out = null;
177 
178             // load all from temp file
179             triples =
180                     _factory.fromStream(new FileInputStream(dump),
181                                               RDFFormat.TURTLE);
182             _ri.delete(triples, true);
183         } finally {
184             if (out != null) {
185                 out.close();
186             }
187             dump.delete();
188         }
189 
190         _ri.close();
191     }

227     // if riLevel is -1, assume original objects have already been added
228     // and we don't need to change the ri level
229     protected void doModifyTest(int riLevel,
230                                 Set<DigitalObject> origObjects,
231                                 DigitalObject modifiedObject) throws Exception {
232 
233         if (riLevel > -1) {
234             initRI(riLevel);
235             addAll(origObjects, true);
236         }
237 
238         DigitalObject origObject = null;
239 
240         // get a set with the modified object in place of its old version
241         Set<DigitalObject> newObjects = new HashSet<DigitalObject>();
242         for (DigitalObject orig : origObjects) {
243             if (orig.getPid().equals(modifiedObject.getPid())) {
244                 origObject = orig;
245             } else {
246                 newObjects.add(orig);
247             }
248         }
249         newObjects.add(modifiedObject);
250 
251         modify(origObject, modifiedObject, true);
252 
253         assertTrue("Did not get expected triples after modify",
254                    sameTriples(getExpectedTriples(riLevel, newObjects),
255                                getActualTriples(),
256                                true));
257     }

312     private void addOrDelAll(Set<DigitalObject> objects,
313                              boolean flush,
314                              boolean add) throws Exception {
315         for (DigitalObject obj : objects) {
316             if (add) {
317                     _ri.addObject(getDOReader(obj));

327     protected Set<Triple> getExpectedTriples(int riLevel,
328                                              Set<DigitalObject> objects)
329             throws Exception {
330 
331         // we can return early in this case
332         if (riLevel == 0) {
333             return new HashSet<Triple>();
334         }
335 
336         // add all to a mock repository reader
337         MockRepositoryReader repo = new MockRepositoryReader();
338         for (DigitalObject obj : objects) {
339             repo.putObject(obj);
340         }
341 
342         // prepare appropriate MethodInfoStore and TripleGenerator
343         TripleGenerator generator = new ModelBasedTripleGenerator();
344 
345         Set<Triple> expected = new HashSet<Triple>();
346 
347         for (DigitalObject obj : objects) {
348             expected.addAll(generator.getTriplesForObject(getDOReader(obj)));
349         }
350 
351         return expected;
352

355     protected boolean sameTriples(Set<Triple> expected,
356                                   Set<Triple> actual,
357                                   boolean logDiffs) {
358         TreeSet<String> eStrings = new TreeSet<String>();
359         for (Triple triple : expected) {
360             eStrings.add(RDFUtil.toString(triple));
361         }
362 
363         TreeSet<String> aStrings = new TreeSet<String>();
364         for (Triple triple : actual) {
365             aStrings.add(RDFUtil.toString(triple));
366         }
367 
368         if (eStrings.equals(aStrings)) {
369             return true;
370         } else {
371             if (logDiffs) {
372                 StringBuffer out = new StringBuffer();
373                 out.append("Triple sets differ.\n");
374                 out.append("Expected set has " + expected.size()
375                         + " triples.\n");
376                 out.append("Actual set has " + actual.size() + " triples.\n\n");
377                 out.append("Expected triples:\n");
378                 for (String t : eStrings) {
379                     out.append("  " + t + "\n");
380                 }
381                 out.append("\nActual triples:\n");
382                 for (String t : aStrings) {
383                     out.append("  " + t + "\n");
384                 }
385                 logger.warn(out.toString());
386             }
387             return false;
388         }
389

396     protected Set<Triple> getActualTriples(SubjectNode subject,
397                                            PredicateNode predicate,
398                                            ObjectNode object) throws Exception {
399         Set<Triple> set = new HashSet<Triple>();
400         TripleIterator iter = _ri.findTriples(subject, predicate, object, -1);
401         while (iter.hasNext()) {
402             set.add(iter.next());
403         }
404         iter.close();
405         return set;

430     /**
431      * Get the METHODMAP xml for a sDef or sDep.
432      */
433     protected static String getMethodMap(Set<ParamDomainMap> methodDefs,
434                                          Map<String, Set<String>> inputKeys,
435                                          boolean forSDef) {
436         StringBuffer xml = new StringBuffer();
437         xml.append("<MethodMap name=\"MethodMap\" xmlns=\"http://fed"
438                 + "ora.comm.nsdlib.org/service/methodmap\">\n");
439         for (ParamDomainMap methodDef : methodDefs) {
440             String method = methodDef.getMethodName();
441             xml.append("  <Method operationName=\"" + method + "\"");
442             if (forSDef) {
443                 xml.append(" wsdlMsgName=\"" + method + "Request\"");
444                 xml.append(" wsdlMsgOutput=\"dissemResponse\"");
445             }
446             xml.append(">\n");
447             for (String paramName : methodDef.keySet()) {
448                 ParamDomain domain = methodDef.get(paramName);
449                 xml.append("    <UserInputParm parmName=\"" + paramName
450                         + "\" passBy=\"VALUE\" defaultValue=\"\" required=\""
451                         + domain.isRequired() + "\">\n");
452                 if (domain.size() > 0) {
453                     xml.append("      <ValidParmValues>\n");
454                     for (String value : domain) {
455                         xml.append("        <ValidParm value=\"" + value
456                                 + "\"/>\n");
457                     }
458                     xml.append("      </ValidParmValues>\n");
459                 }
460                 xml.append("    </UserInputParm>\n");
461             }
462             if (forSDef) {
463                 Set<String> keys = inputKeys.get(method);
464                 if (keys != null) {
465                     for (String key : keys) {
466                         xml.append("    <DatastreamInputParm parmName=\"" + key
467                                 + "\" passBy=\"URL_REF\"/>\n");
468                     }
469                 }
470 
471                 xml.append("    <MethodReturnType wsdlMsgName=\""
472                         + "dissemResponse\" wsdlMsgTOMIME=\""
473                         + "application/octet-stream\"/>\n");
474             }
475             xml.append("  </Method>\n");
476         }
477         xml.append("</MethodMap>");
478         return xml.toString();
479     }

509     /**
510      * Get the WSDL xml for a sDef.
511      */
512     protected static String getWSDL(Set<ParamDomainMap> methodDefs,
513                                     Map<String, Set<String>> inputKeys,
514                                     Map<String, Set<String>> outputTypes) {
515         StringBuffer xml = new StringBuffer();
516 
517         xml.append("<definitions xmlns=\"http://schemas.xmlsoap.org/wsdl/\""
518                 + " xmlns:http=\"http://schemas.xmlsoap.org/wsdl/http/\""
519                 + " xmlns:mime=\"http://schemas.xmlsoap.org/wsdl/mime/\""
520                 + " xmlns:this=\"MyService\""
521                 + " xmlns:xsd=\"http://www.w3.org/2001/XMLSchema\""
522                 + " name=\"Name\" targetNamespace=\"MyService\">\n");
523 
524         //
525         // xsd type definitions
526         //
527 
528         xml.append("  <types>\n");
529         xml.append("    <xsd:schema targetNamespace=\"MyService\">\n");
530         Set<String> addedTypes = new HashSet<String>();
531         for (ParamDomainMap methodDef : methodDefs) {
532             // one type def per distinct user param name
533             for (String name : methodDef.keySet()) {
534                 if (addedTypes.add(name + "Type")) {
535                     addXSDType(name + "Type", xml);
536                 }
537             }
538 
539             // one type def per distinct ds input key
540             for (String key : inputKeys.get(methodDef.getMethodName())) {
541                 if (addedTypes.add(key + "Type")) {
542                     addXSDType(key + "Type", xml);
543                 }
544             }
545         }
546         xml.append("    </xsd:schema>\n");
547         xml.append("  </types>\n");
548 
549         //
550         // message definitions
551         //
552 
553         // one request message per method
554         for (ParamDomainMap methodDef : methodDefs) {
555             String method = methodDef.getMethodName();
556             xml.append("  <message name=\"" + method + "Request\">\n");
557             // one part per user param
558             for (String name : methodDef.keySet()) {
559                 xml.append("    <part name=\"" + name + "\" type=\"this:"
560                         + name + "Type\"/>\n");
561             }
562             // one part per ds input key
563             for (String key : inputKeys.get(method)) {
564                 xml.append("    <part name=\"" + key + "\" type=\"this:" + key
565                         + "Type\"/>\n");
566             }
567             xml.append("  </message>\n");
568         }
569         // one dissemResponse output message
570         xml.append("  <message name=\"dissemResponse\">\n");
571         xml.append("    <part name=\"response\" type=\"xsd:base64Binary\"/>\n");
572         xml.append("  </message>\n");
573 
574         //
575         // port type (per-method input/output messages)
576         //
577         xml.append("  <portType name=\"MyServicePortType\">\n");
578         for (ParamDomainMap methodDef : methodDefs) {
579             String method = methodDef.getMethodName();
580             xml.append("    <operation name=\"" + method + "\">\n");
581             xml.append("      <input message=\"this:" + method
582                     + "Request\"/>\n");
583             xml.append("      <output message=\"this:dissemResponse\"/>\n");
584             xml.append("    </operation>\n");
585         }
586         xml.append("  </portType>\n");
587 
588         //
589         // service location
590         //
591         xml.append("  <service name=\"MyService\">\n");
592         xml.append("    <port binding=\"this:MyService_http\" "
593                 + "name=\"MyService_port\">\n");
594         xml
595                 .append("      <http:address location=\"http://example.org/MyService/\"/>\n");
596         xml.append("    </port>\n");
597         xml.append("  </service>\n");
598 
599         //
600         // operation locations and input/output bindings
601         //
602         xml
603                 .append("  <binding name=\"MyService_http\" type=\"this:MyServicePortType\">\n");
604         xml.append("    <http:binding verb=\"GET\"/>\n");
605         for (ParamDomainMap methodDef : methodDefs) {
606             String method = methodDef.getMethodName();
607             xml.append("    <operation name=\"" + method + "\">\n");
608 
609             // location = ..?userParm1=(userParm1)&key1=KEY1..etc
610             StringBuffer location = new StringBuffer();
611             location.append(method + "?");
612             boolean first = true;
613             for (String name : methodDef.keySet()) {
614                 if (!first) {
615                     location.append("&amp;");
616                 }
617                 location.append(name + "=(" + name + ")");
618                 first = false;
619             }
620             for (String key : inputKeys.get(method)) {
621                 if (!first) {
622                     location.append("&amp;");
623                 }
624                 location.append(key.toLowerCase() + "=(" + key + ")");
625                 first = false;
626             }
627             xml.append("      <http:operation location=\""
628                     + location.toString() + "\"/>\n");
629 
630             // input is always urlReplacement
631             xml.append("      <input><http:urlReplacement/></input>\n");
632 
633             // output lists all possible output mime types
634             xml.append("      <output>\n");
635             for (String mimeType : outputTypes.get(method)) {
636                 xml.append("        <mime:content type=\"" + mimeType
637                         + "\"/>\n");
638             }
639             xml.append("      </output>\n");
640 
641             xml.append("    </operation>\n");
642         }
643         xml.append("  </binding>\n");
644 
645         xml.append("</definitions>\n");
646         return xml.toString();
647     }

1134     /**
1135      * A Thread that continuously flushes the buffer.
1136      */
1137     public class Flusher
1138             extends Thread {
1139 
1140         private final ResourceIndex _ri;
1141 
1142         private final int _sleepMS;
1143 
1144         private boolean _shouldFinish = false;
1145 
1146         private Exception _error;
1147 
1148         /**
1149          * Construct a flusher that sleeps the given number of milliseconds
1150          * between flush attempts.
1151          *
1152          * @param sleepMS
1153          *        milliseconds to sleep. Will simply yield between flush
1154          *        attempts if less than 1.
1155          */
1156         public Flusher(ResourceIndex ri, int sleepMS) {
1157             _ri = ri;
1158             _sleepMS = sleepMS;
1159         }
1160 
1161         /**
1162          * Set signal for flusher to finish and wait for it.
1163          *
1164          * @throws Exception
1165          *         if the flusher encountered an error any time while it was
1166          *         running.
1167          */
1168         public void finish() throws Exception {
1169             _shouldFinish = true;
1170             while (isAlive()) {
1171                 try {
1172                     Thread.sleep(50);
1173                 } catch (InterruptedException e) {
1174                 }
1175             }
1176             if (_error != null) {
1177                 throw _error;
1178             }
1179         }
1180 
1181         /**
1182          * Flush the buffer until the finish signal arrives from another thread.
1183          */
1184         @Override
1185         public void run() {
1186             try {
1187                 while (!_shouldFinish) {
1188                     if (_sleepMS > 0) {
1189                         try {
1190                             Thread.sleep(_sleepMS);
1191                         } catch (InterruptedException e) {
1192                         }
1193                     } else {
1194                         Thread.yield();
1195                     }
1196                     _ri.flushBuffer();
1197                 }
1198             } catch (Exception e) {
1199                 _error = e;
1200             }
1201         }
1202     }
