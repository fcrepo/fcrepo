29      /**
30       * Modify an object's label with the RI at level 0.
31       */
32      @Test
33      public void testModObjOnceLabelLv0() throws Exception {
34          Set<DigitalObject> objects = getTestObjects(1, 0);
35  
36          // add at level 1
37          initRI(1);
38          addAll(objects, true);
39          Set<Triple> origTriples = getExpectedTriples(1, objects);
40  
41          // mod at level 0
42          DigitalObject original = (DigitalObject) objects.toArray()[0];
43          DigitalObject modified = ObjectBuilder.deepCopy(original);
44          modified.setLabel("new label");
45          initRI(0);
46          modify(original, modified, true);
47  
48          assertTrue("Did not get expected orig triples after modify at level 0",
49                     sameTriples(origTriples, getActualTriples(), true));
50      }

66       * Modify an object's label multiple times.

68      @Test
69      public void testModObjMultiLabel() throws Exception {
70  
71          // prep by initting at lv 1 and adding original object
72          initRI(1);
73          DigitalObject previous = getTestObject("test:1", "test1");
74          Set<DigitalObject> origSet = new HashSet<DigitalObject>();
75          origSet.add(previous);
76          addAll(origSet, true);
77  
78          // modify the label multiple times
79          for (int i = 1; i <= 5; i++) {
80              DigitalObject modified = ObjectBuilder.deepCopy(previous);
81              modified.setLabel("new label " + i);
82              doModifyTest(-1, previous, modified);
83              previous = modified;
84          }

88      /**
89       * Modify an object's label multiple times while flushing the buffer many
90       * times from a separate thread.
91       */
92      @Test
93      public void testModObjMultiLabelAsyncFlush() throws Exception {
94  
95          // prep by initting at lv 1 and adding original object
96          initRI(1);
97          DigitalObject original = getTestObject("test:1", "test1");
98          DigitalObject previous = original;
99          Set<DigitalObject> origSet = new HashSet<DigitalObject>();
100         origSet.add(previous);
101         addAll(origSet, true);
102 
103         // hold on to the original triples so we can compare later
104         Set<Triple> origTriples = getExpectedTriples(1, origSet);
105 
106         // modify the label multiple times while flushing
107         startFlushing(0);
108         try {
109             for (int i = 0; i <= 5; i++) {
110                 DigitalObject modified = ObjectBuilder.deepCopy(previous);
111                 modified.setLabel("new label " + i);
112                 modify(previous, modified, false);
113                 previous = modified;
114             }
115             // last change puts obj back into original state
116             modify(previous, original, false);
117         } finally {
118             // this stops async flushing and flushes one last time
119             finishFlushing();
120         }
121 
122         assertTrue("Did not get expected orig triples after multi mod of "
123                            + "one object with async buffer flushing",
124                    sameTriples(origTriples, getActualTriples(), true));
125     }
