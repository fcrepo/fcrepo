74      public void testApplyProperties() throws Exception {
75          ServerConfiguration config =
76                  new ServerConfigurationParser(new FileInputStream(FCFG_BASE))
77                          .parse();
78  
79          String testVal = "9999";
80          String xpath =
81                  "/" + NS_FCFG_PREFIX + ":server/" + NS_FCFG_PREFIX
82                          + ":param[@name='fedoraServerPort'][@value='" + testVal
83                          + "']";
84  
85          // ensure the new property is really new
86          config.serialize(out);
87          assertXpathNotExists(xpath, getDocument(out));
88  
89          // apply the new property and ensure it is present in the serialized
90          // output
91          out.reset();
92          Properties props = new Properties();
93          props.put("server:fedoraServerPort", testVal);
94          config.applyProperties(props);
95          config.serialize(out);
96          assertXpathExists(xpath, getDocument(out));
97      }
98  
99      public void testSerialize() throws Exception {
100         ServerConfiguration config =
101                 new ServerConfigurationParser(new FileInputStream(FCFG_BASE))
102                         .parse();
103         ByteArrayOutputStream out = new ByteArrayOutputStream();
104         config.serialize(out);
105 
106         Document original = builder.parse(FCFG_BASE);
107         Document generated =
108                 builder.parse(new ByteArrayInputStream(out.toByteArray()));
109         XMLUnit.setIgnoreWhitespace(true);
110         Diff diff = new Diff(original, generated);
111         diff.overrideElementQualifier(new ElementNameAndAttributeQualifier());
112         assertXMLEqual(diff, true);
113     }
114 
115     private Document getDocument(ByteArrayOutputStream out) throws Exception {
116         return XMLUnit.buildControlDocument(new String(out.toByteArray(),
117                                                        "UTF-8"));
118     }
