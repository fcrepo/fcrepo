46                  postObjMethod("FedoraDatastreamsTest7");

50                  postDSMethod("FedoraDatastreamsTest7", "ds1",

52          assertEquals(201, getStatus(createDS1Method));
53          final HttpPost createDS2Method =
54                  postDSMethod("FedoraDatastreamsTest7", "ds2",
55                          "marbles for no one");
56          assertEquals(201, getStatus(createDS2Method));
57  
58          final HttpGet getDSesMethod =
59                  new HttpGet(serverAddress + "FedoraDatastreamsTest7");
60          getDSesMethod.addHeader("Accept", "text/n3");
61          final HttpResponse response = client.execute(getDSesMethod);
62          assertEquals(200, response.getStatusLine().getStatusCode());
63          final GraphStore result =
64                  TestHelpers.parseTriples(response.getEntity().getContent());
65          logger.debug("Received triples: \n{}", result.toString());
66          final String subjectURI =
67                  serverAddress + "FedoraDatastreamsTest7";
68  
69          assertTrue("Didn't find the first datastream! ", result.contains(ANY,
70                  createURI(subjectURI), ANY, createURI(subjectURI + "/ds1")));
71          assertTrue("Didn't find the second datastream! ", result.contains(ANY,
72                  createURI(subjectURI), ANY, createURI(subjectURI + "/ds2")));

75      @Test
76      public void testModifyMultipleDatastreams() throws Exception {
77          final HttpPost objMethod = postObjMethod("FedoraDatastreamsTest8");
78  
79          assertEquals(201, getStatus(objMethod));
80  
81          final HttpPost createDSVOIDMethod =
82                  postDSMethod("FedoraDatastreamsTest8", "ds_void",
83                          "marbles for everyone");
84          assertEquals(201, getStatus(createDSVOIDMethod));
85  
86          final HttpPost post =
87                  new HttpPost(serverAddress +
88                          "FedoraDatastreamsTest8/fcr:datastreams?delete=ds_void");
89  
90          final MultipartEntity multiPartEntity = new MultipartEntity();
91          multiPartEntity.addPart("ds1", new StringBody("asdfg"));
92          multiPartEntity.addPart("ds2", new StringBody("qwerty"));
93  
94          post.setEntity(multiPartEntity);
95  
96          final HttpResponse postResponse = client.execute(post);
97  
98          assertEquals(201, postResponse.getStatusLine().getStatusCode());
99  
100         final HttpGet getDSesMethod =
101                 new HttpGet(serverAddress + "FedoraDatastreamsTest8");
102         getDSesMethod.addHeader("Accept", "text/n3");
103         final HttpResponse response = client.execute(getDSesMethod);
104         assertEquals(200, response.getStatusLine().getStatusCode());
105 
106         final String subjectURI =
107                 serverAddress + "FedoraDatastreamsTest8";
108         final GraphStore result =
109                 parseTriples(response.getEntity().getContent());
110         assertTrue("Didn't find the first datastream! ", result.contains(ANY,
111                 createURI(subjectURI), ANY, createURI(subjectURI + "/ds1")));
112         assertTrue("Didn't find the second datastream! ", result.contains(ANY,
113                 createURI(subjectURI), ANY, createURI(subjectURI + "/ds2")));
114         assertFalse("Found the deleted datastream! ", result.contains(Node.ANY,
115                 createURI(subjectURI), ANY, createURI(subjectURI + "/ds_void")));
116 
117     }
