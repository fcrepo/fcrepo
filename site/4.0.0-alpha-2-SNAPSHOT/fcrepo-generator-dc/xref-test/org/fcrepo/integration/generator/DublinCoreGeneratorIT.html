39      @Test
40      public void testJcrPropertiesBasedOaiDc() throws Exception {
41          final int status = getStatus(postObjMethod("DublinCoreTest1"));
42          assertEquals(201, status);
43          final HttpPatch post = new HttpPatch(serverAddress + "DublinCoreTest1");
44          post.setHeader("Content-Type", "application/sparql-update");
45          BasicHttpEntity entity = new BasicHttpEntity();
46          String subjectURI = serverAddress + "DublinCoreTest1";
47          entity.setContent(new ByteArrayInputStream(
48                  ("INSERT { <" + subjectURI + "> <http://purl.org/dc/terms/identifier> \"this is an identifier\" } WHERE {}")
49                          .getBytes()));
50          post.setEntity(entity);
51          assertEquals(204, getStatus(post));
52          final HttpGet getWorstCaseOaiMethod =
53                  new HttpGet(serverOAIAddress + "DublinCoreTest1/oai:dc");
54          getWorstCaseOaiMethod.setHeader("Accept", TEXT_XML);
55          final HttpResponse response = client.execute(getWorstCaseOaiMethod);
56  
57          assertEquals(200, response.getStatusLine().getStatusCode());
58  
59          final String content = EntityUtils.toString(response.getEntity());
60          logger.debug("Got content: {}", content);
61          assertTrue("Didn't find oai_dc!", compile("oai_dc", DOTALL).matcher(
62                  content).find());
63  
64          assertTrue("Didn't find dc:identifier!", compile("dc:identifier",
65                  DOTALL).matcher(content).find());
66      }

68      @Test
69      public void testWellKnownPathOaiDc() throws Exception {
70          HttpResponse response =
71                  client.execute(postObjMethod("DublinCoreTest2"));
72          assertEquals(201, response.getStatusLine().getStatusCode());
73          response =
74                  client.execute(postDSMethod("DublinCoreTest2", "DC",
75                          "marbles for everyone"));
76          int status = response.getStatusLine().getStatusCode();
77          if (status != 201) {
78              System.err.println(EntityUtils.toString(response.getEntity()));
79          }
80          assertEquals(201, status);
81  
82          final HttpGet getWorstCaseOaiMethod =
83                  new HttpGet(serverOAIAddress + "DublinCoreTest2/oai:dc");
84          getWorstCaseOaiMethod.setHeader("Accept", TEXT_XML);
85          response = client.execute(getWorstCaseOaiMethod);
86          assertEquals(200, response.getStatusLine().getStatusCode());
87          final String content = EntityUtils.toString(response.getEntity());
88          assertTrue("Didn't find our datastream!", compile(
89                  "marbles for everyone", DOTALL).matcher(content).find());
90      }
