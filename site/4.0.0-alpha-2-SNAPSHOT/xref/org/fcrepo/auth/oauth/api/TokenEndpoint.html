74  @Component
75  @Path("/token")
76  public class TokenEndpoint extends AbstractResource {
77  
78      public static final String INVALID_CLIENT_DESCRIPTION =
79          "Client authentication failed (e.g., unknown client, " +
80          "no client authentication included, " +
81          "or unsupported authentication method).";
82  
83      private static final Logger LOGGER = getLogger(TokenEndpoint.class);
84  
85      @Autowired(required = false)
86      private TokenRequestValidations requestValidator;
87  
88      @PostConstruct
89      void defaultValidations() {
90          if (requestValidator == null) {
91              requestValidator = new DefaultTokenRequestValidations(sessions);
92          }
93      }
94  
95      /**
96       * @param request An HTTP request
97       * @return A token-bearing HTTP response
98       * @throws OAuthSystemException
99       * @throws RepositoryException
100      */
101     @POST
102     @Consumes(APPLICATION_FORM_URLENCODED)
103     @Produces(APPLICATION_JSON)
104     public Response getToken(@Context final HttpServletRequest request) throws
105         OAuthSystemException, RepositoryException {
106         LOGGER.debug("Received request for token carried on request: {}",
107                 request);
108         OAuthTokenRequest oauthRequest = null;
109 
110         final OAuthIssuer oauthIssuerImpl =
111                 new OAuthIssuerImpl(new MD5Generator());
112 
113         try {
114             oauthRequest = new OAuthTokenRequest(request);
115 
116             if (!requestValidator.isValidClient(oauthRequest)) {
117                 final OAuthResponse response =
118                         OAuthASResponse.errorResponse(SC_BAD_REQUEST).setError(
119                                 INVALID_CLIENT).setErrorDescription(
120                                 INVALID_CLIENT_DESCRIPTION).buildJSONMessage();
121                 return status(response.getResponseStatus()).entity(
122                         response.getBody()).build();
123             }
124 
125             if (!requestValidator.isValidSecret(oauthRequest)) {
126                 final OAuthResponse response =
127                         OAuthASResponse
128                                 .errorResponse(SC_UNAUTHORIZED)
129                                 .setError(UNAUTHORIZED_CLIENT)
130                                 .setErrorDescription(INVALID_CLIENT_DESCRIPTION)
131                                 .buildJSONMessage();
132                 return status(response.getResponseStatus()).entity(
133                         response.getBody()).build();
134             }
135 
136             if (oauthRequest.getParam(OAUTH_GRANT_TYPE).equals(
137                     AUTHORIZATION_CODE.toString())) {
138                 if (!requestValidator.isValidAuthCode(oauthRequest)) {
139                     final OAuthResponse response =
140                             errorResponse(SC_BAD_REQUEST).setError(
141                                     INVALID_GRANT).setErrorDescription(
142                                     "invalid authorization code")
143                                     .buildJSONMessage();
144                     return status(response.getResponseStatus()).entity(
145                             response.getBody()).build();
146                 }
147             } else if (oauthRequest.getParam(OAUTH_GRANT_TYPE).equals(
148                     GrantType.PASSWORD.toString())) {
149                 if (!requestValidator.isValidCredentials(oauthRequest)) {
150                     final OAuthResponse response =
151                             errorResponse(SC_BAD_REQUEST).setError(
152                                     INVALID_GRANT).setErrorDescription(
153                                     "invalid username or password")
154                                     .buildJSONMessage();
155                     return status(response.getResponseStatus()).entity(
156                             response.getBody()).build();
157                 }
158             } else if (oauthRequest.getParam(OAuth.OAUTH_GRANT_TYPE).equals(
159                     GrantType.REFRESH_TOKEN.toString())) {
160                 // refresh token is not supported in this implementation
161                 final OAuthResponse response =
162                         errorResponse(SC_BAD_REQUEST).setError(INVALID_GRANT)
163                                 .setErrorDescription(
164                                         "invalid username or password")
165                                 .buildJSONMessage();
166                 return status(response.getResponseStatus()).entity(
167                         response.getBody()).build();
168             }
169 
170             final String token = oauthIssuerImpl.accessToken();
171             LOGGER.debug("Created token: {}", token);
172             saveToken(token, oauthRequest.getClientId(), oauthRequest
173                     .getUsername());
174             final OAuthResponse response =
175                     tokenResponse(SC_OK).setAccessToken(token).setExpiresIn(
176                             "3600").buildJSONMessage();
177             return status(response.getResponseStatus()).entity(
178                     response.getBody()).build();
179 
180         } catch (final OAuthProblemException e) {
181             final OAuthResponse res =
182                     errorResponse(SC_BAD_REQUEST).error(e).buildJSONMessage();
183             return status(res.getResponseStatus()).entity(res.getBody())
184                     .build();
185         }
186     }

188     /**
189      * Stores a token for later use by the configured {@link OAuthRSProvider}
190      * 
191      * @param token
192      * @param client
193      * @param username
194      * @throws RepositoryException
195      */
196     private void saveToken(final String token, final String client,
197             final String username) throws RepositoryException {
198         final Session session = sessions.getInternalSession(OAUTH_WORKSPACE);
199         try {
200             final Node tokenNode =
201                     jcrTools.findOrCreateNode(session, "/tokens/" + token);
202             tokenNode.setProperty(CLIENT_PROPERTY, client);
203             tokenNode.setProperty(PRINCIPAL_PROPERTY, username);
204             session.save();
205         } finally {
206             session.logout();
207         }
208 
209     }
