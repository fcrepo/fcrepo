107     public LegacyMethod(final Event jcrEvent, final Node resource)
108         throws RepositoryException {
109         this(EntryFactory.newEntry());
110 
111         final boolean isDatastreamNode =
112                 FedoraTypesUtils.isFedoraDatastream.apply(resource);
113         final boolean isObjectNode =
114                 FedoraTypesUtils.isFedoraObject.apply(resource) &&
115                         !isDatastreamNode;
116 
117         if (isDatastreamNode || isObjectNode) {
118             setMethodName(mapMethodName(jcrEvent.getType(), isObjectNode));
119             final String returnValue = getReturnValue(jcrEvent, resource);
120             setContent(getEntryContent(getMethodName(), returnValue));
121             if (isDatastreamNode) {
122                 setPid(resource.getParent().getName());
123                 setDsId(resource.getName());
124             } else {
125                 setPid(resource.getName());
126             }
127         } else {
128             setMethodName(null);
129         }
130         final String userID =
131                 jcrEvent.getUserID() == null ? "unknown" : jcrEvent.getUserID();
132         setUserId(userID);
133         setModified(new Date(jcrEvent.getDate()));
134     }

230     private void setLabelledCategory(String label, String val) {
231         final List<Category> vals = delegate.getCategories(FEDORA_ID_SCHEME);
232         Category found = null;
233         if (vals != null && !vals.isEmpty()) {
234             for (Category c : vals) {
235                 if (label.equals(c.getLabel())) {
236                     found = c.setTerm(val);
237                 }
238             }
239         }
240         if (found == null) {
241             delegate.addCategory(FEDORA_ID_SCHEME, val, label);
242         }
243     }

245     private String getLabelledCategory(String label) {
246         final List<Category> categories =
247                 delegate.getCategories(FEDORA_ID_SCHEME);
248         for (final Category c : categories) {
249             if (label.equals(c.getLabel())) {
250                 return c.getTerm();
251             }
252         }
253         return null;
254     }

303     private static String getEntryContent(final String methodName,
304             final String returnVal) {
305         final String datatype =
306                 (String) FEDORA_TYPES.get(methodName + ".datatype");
307         return objectToString(returnVal, datatype);
308     }
309 
310     protected static String objectToString(final String obj,
311             final String xsdType) {
312         if (obj == null) {
313             return "null";
314         }
315         String term;
316         // TODO Most of these types are not yet relevant to FCR4, but we can
317         // borrow their serializations as necessary
318         // several circumstances yield null canonical names
319         switch (xsdType) {
320             case "fedora-types:ArrayOfString":
321                 term = "[UNSUPPORTED" + xsdType + "]";
322                 break;
323             case "xsd:boolean":
324                 term = obj;
325                 break;
326             case "xsd:nonNegativeInteger":
327                 term = obj;
328                 break;
329             case "fedora-types:RelationshipTuple":
330                 term = "[UNSUPPORTED" + xsdType + "]";
331                 break;
332             default:
333                 term = obj;
334                 term = term.replaceAll("\"", "'");
335                 break;
336         }
337 
338         return term;
339     }

341     protected static String getReturnValue(final Event jcrEvent,
342             final Node jcrNode) throws RepositoryException {
343         switch (jcrEvent.getType()) {
344             case NODE_ADDED:
345                 return jcrNode.getName();
346             case NODE_REMOVED:
347             case PROPERTY_ADDED:
348             case PROPERTY_CHANGED:
349             case PROPERTY_REMOVED:
350                 return convertDateToXSDString(jcrEvent.getDate());
351             default:
352                 return null;
353         }
354     }
355 
356     protected static String mapMethodName(final int eventType,
357             final boolean isObjectNode) {
358         switch (eventType) {
359             case NODE_ADDED:
360                 return isObjectNode ? INGEST_METHOD : ADD_DS_METHOD;
361             case NODE_REMOVED:
362                 return isObjectNode ? PURGE_OBJ_METHOD : PURGE_DS_METHOD;
363             case PROPERTY_ADDED:
364                 return isObjectNode ? MODIFY_OBJ_METHOD : MODIFY_DS_METHOD;
365             case PROPERTY_CHANGED:
366                 return isObjectNode ? MODIFY_OBJ_METHOD : MODIFY_DS_METHOD;
367             case PROPERTY_REMOVED:
368                 return isObjectNode ? MODIFY_OBJ_METHOD : MODIFY_DS_METHOD;
369         }
370         return null;
371     }

379     public static boolean canParse(final Message jmsMessage) {
380         try {
381             return EntryFactory.FORMAT.equals(jmsMessage.getJMSType()) &&
382                     METHOD_NAMES.contains(jmsMessage
383                             .getStringProperty("methodName"));
384         } catch (final JMSException e) {
385             LOGGER.info("Could not parse message: {}", jmsMessage);
386             throw propagate(e);
387         }
