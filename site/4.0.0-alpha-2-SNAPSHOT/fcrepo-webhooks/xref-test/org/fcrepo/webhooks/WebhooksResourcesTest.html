60          mockSubjects = new DefaultGraphSubjects(mock(Session.class));

62  
63      @Test
64      public void shouldDecorateModeRootNodesWithRepositoryWideLinks()
65          throws RepositoryException {
66  
67          final NodeType mockNodeType = mock(NodeType.class);
68          when(mockNodeType.isNodeType(FedoraJcrTypes.ROOT)).thenReturn(true);
69          when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
70          when(mockNode.getPath()).thenReturn("/");
71  
72          Resource graphSubject = mockSubjects.getGraphSubject(mockNode);
73  
74          final Model model =
75                  testObj.createModelForResource(mockResource, uriInfo,
76                          mockSubjects);
77  
78          assertTrue(model.contains(graphSubject, HAS_SUBSCRIPTION_SERVICE));
79      }
