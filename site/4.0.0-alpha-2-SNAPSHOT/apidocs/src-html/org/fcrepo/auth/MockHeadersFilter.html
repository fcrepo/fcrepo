017package org.fcrepo.auth;
018
019import java.io.IOException;
020import java.security.Principal;
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.Enumeration;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027import java.util.Vector;
028
029import javax.servlet.Filter;
030import javax.servlet.FilterChain;
031import javax.servlet.FilterConfig;
032import javax.servlet.ServletException;
033import javax.servlet.ServletRequest;
034import javax.servlet.ServletResponse;
035import javax.servlet.http.HttpServletRequest;
036import javax.servlet.http.HttpServletRequestWrapper;
037
038import org.apache.http.auth.BasicUserPrincipal;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041
042/**
043 * This is a servlet filter that makes requests appear to be authenticated and
044 * adds headers for tests.
045 * 
046 * @author Gregory Jansen
047 */
048public class MockHeadersFilter implements Filter {
049
050    private static Logger logger = LoggerFactory.getLogger(MockHeadersFilter.class);
051
052    private Map<String, List<String>> mockHeaders =
053            new HashMap<String, List<String>>();
054
055    /*
056     * (non-Javadoc)
057     * @see javax.servlet.Filter#destroy()
058     */
059    @Override
060    public void destroy() {
061    }
062
063    /*
064     * (non-Javadoc)
065     * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest,
066     * javax.servlet.ServletResponse, javax.servlet.FilterChain)
067     */
068    @Override
069    public void doFilter(final ServletRequest req,
070        final ServletResponse res, final FilterChain chain)
071        throws IOException, ServletException {
072        logger.debug("in filter");
073        if (req instanceof HttpServletRequest) {
074            final HttpServletRequest hreq = (HttpServletRequest) req;
075            final MockedHTTPServletRequest wreq =
076                    new MockedHTTPServletRequest(hreq);
077            chain.doFilter(wreq, res);
078        } else {
079            chain.doFilter(req, res);
080        }
081    }
082
083    /*
084     * (non-Javadoc)
085     * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
086     */
087    @Override
088    public void init(final FilterConfig config) throws ServletException {
089        final Enumeration<String> params = config.getInitParameterNames();
090        while (params.hasMoreElements()) {
091            final String param = params.nextElement();
092            final String val = config.getInitParameter(param);
093            final List<String> headers = new ArrayList<String>();
094            Collections.addAll(headers, val.split("\\|"));
095            this.mockHeaders.put(param, headers);
096        }
097        logger.debug("config loaded with " + mockHeaders.size() +
098                " headers mocked.");
099    }
100
101    private class MockedHTTPServletRequest extends
102            HttpServletRequestWrapper {
103
104        /**
105         * @param request
106         */
107        public MockedHTTPServletRequest(final HttpServletRequest request) {
108            super(request);
109        }
110
111        private Principal userPrincipal = null;
112
113        /*
114         * (non-Javadoc)
115         * @see
116         * javax.servlet.http.HttpServletRequestWrapper
117         * #getAuthType ()
118         */
119        @Override
120        public String getAuthType() {
121            return BASIC_AUTH;
122        }
123
124        /*
125         * (non-Javadoc)
126         * @see
127         * javax.servlet.http.HttpServletRequestWrapper#
128         * getRemoteUser()
129         */
130        @Override
131        public String getRemoteUser() {
132            return mockHeaders.get("REMOTE_USER").get(0);
133        }
134
135        /*
136         * (non-Javadoc)
137         * @see
138         * javax.servlet.http.HttpServletRequestWrapper#
139         * getUserPrincipal()
140         */
141        @Override
142        public Principal getUserPrincipal() {
143            if (userPrincipal == null) {
144                userPrincipal =
145                        new BasicUserPrincipal(mockHeaders.get(
146                                "REMOTE_USER").get(0));
147            }
148            return userPrincipal;
149        }
150
151        /*
152         * (non-Javadoc)
153         * @see
154         * javax.servlet.http.HttpServletRequestWrapper
155         * #getHeader (java.lang.String)
156         */
157        @Override
158        public String getHeader(final String name) {
159            if (mockHeaders.containsKey(name)) {
160                return mockHeaders.get(name).get(0);
161            }
162            return super.getHeader(name);
163        }
164
165        /*
166         * (non-Javadoc)
167         * @see
168         * javax.servlet.http.HttpServletRequestWrapper#isUserInRole(java.lang
169         * .String)
170         */
171        @Override
172        public boolean isUserInRole(final String role) {
173            return ServletContainerAuthenticationProvider.FEDORA_USER
174                    .equals(role);
175        }
176
177        /*
178         * (non-Javadoc)
179         * @see
180         * javax.servlet.http.HttpServletRequestWrapper#
181         * getHeaderNames()
182         */
183        @Override
184        public Enumeration<String> getHeaderNames() {
185            final List<String> result = new ArrayList<String>();
186            final Enumeration<String> orig = super.getHeaderNames();
187            while (orig.hasMoreElements()) {
188                result.add(orig.nextElement());
189            }
190            result.addAll(mockHeaders.keySet());
191            final Vector<String> v = new Vector<String>(result);
192            return v.elements();
193        }
194
195        /*
196         * (non-Javadoc)
197         * @see
198         * javax.servlet.http.HttpServletRequestWrapper
199         * #getHeaders (java.lang.String)
200         */
201        @Override
202        public Enumeration<String> getHeaders(final String name) {
203            if (mockHeaders.containsKey(name)) {
204                return new Vector<String>(mockHeaders.get(name))
205                        .elements();
206            }
207            return super.getHeaders(name);
208        }
209    }
