053/**
054 * Common behaviors across FedoraObject and Datastream types; also used
055 * when the exact type of an object is irrelevant
056 */
057public class FedoraResource extends JcrTools implements FedoraJcrTypes {
058
059    private static final Logger LOGGER = getLogger(FedoraResource.class);
060
061    private static final GraphProperties DEFAULT_PROPERTY_FACTORY =
062            new JcrGraphProperties();
063
064    protected Node node;
065
066    private GraphProperties properties;
067
068    /**
069     * Construct a FedoraObject without a backing JCR Node
070     */
071    public FedoraResource() {
072        node = null;
073        this.properties = DEFAULT_PROPERTY_FACTORY;
074    }
075
076    /**
077     * Construct a FedoraObject from an existing JCR Node
078     * @param node an existing JCR node to treat as an fcrepo object
079     */
080    public FedoraResource(final Node node) {
081        super(false);
082        this.node = node;
083        this.properties = DEFAULT_PROPERTY_FACTORY;
084    }
085
086    /**
087     * Create or find a FedoraObject at the given path
088     * @param session the JCR session to use to retrieve the object
089     * @param path the absolute path to the object
090     * @throws RepositoryException
091     */
092    public FedoraResource(final Session session, final String path,
093                          final String nodeType)
094        throws RepositoryException {
095        super(false);
096        this.node = findOrCreateNode(
097                session, path, JcrConstants.NT_FOLDER, nodeType);
098
099        if (!hasMixin(node)) {
100            node.addMixin(FEDORA_RESOURCE);
101        }
102
103        if (node.isNew()) {
104            node.setProperty(JCR_LASTMODIFIED, Calendar.getInstance());
105        }
106        this.properties = DEFAULT_PROPERTY_FACTORY;
107    }
108
109    /**
110     * Is the given node a Fedora resource
111     * (because it has a fedora:resource mixin)?
112     *
113     * @param node
114     * @return
115     * @throws RepositoryException
116     */
117    public static boolean hasMixin(final Node node) throws RepositoryException {
118        return isFedoraResource.apply(node);
119    }
120
121    /**
122     * Does the resource have a jcr:content child node?
123     * @return
124     * @throws RepositoryException
125     */
126    public boolean hasContent() throws RepositoryException {
127        return node.hasNode(JcrConstants.JCR_CONTENT);
128    }
129
130    /**
131     * @return The JCR node that backs this object.
132     */
133    public Node getNode() {
134        return node;
135    }
136
137    /**
138     * Get the path to the JCR node
139     * @return
140     * @throws RepositoryException
141     */
142    public String getPath() throws RepositoryException {
143        return node.getPath();
144    }
145
146    /**
147     * Get the date this datastream was created
148     * @return
149     * @throws RepositoryException
150     */
151    public Date getCreatedDate() throws RepositoryException {
152        if (node.hasProperty(JCR_CREATED)) {
153            return new Date(node.getProperty(JCR_CREATED).getDate()
154                            .getTimeInMillis());
155        } else {
156            LOGGER.info("Node {} does not have a createdDate", node);
157            return null;
158        }
159    }
160
161    /**
162     * Get the date this datastream was last modified
163     * @return
164     * @throws RepositoryException
165     */
166    public Date getLastModifiedDate() throws RepositoryException {
167        if (node.hasProperty(JCR_LASTMODIFIED)) {
168            return new Date(node.getProperty(JCR_LASTMODIFIED).getDate()
169                            .getTimeInMillis());
170        } else {
171            LOGGER.info(
172                        "Could not get last modified date property for node {}",
173                        node);
174        }
175
176        final Date createdDate = getCreatedDate();
177
178        if (createdDate != null) {
179            LOGGER.info(
180                        "Using created date for last modified date for node {}",
181                        node);
182            return createdDate;
183        }
184
185        return null;
186    }
187
188    /**
189     * Get the total size of this object and its datastreams
190     * @return size in bytes
191     * @throws RepositoryException
192     */
193    public long getSize() throws RepositoryException {
194        return getObjectSize(node);
195    }
196
197    /**
198     * Get the mixins this object uses
199     * @return a collection of mixin names
200     * @throws javax.jcr.RepositoryException
201     */
202    public Collection<String> getModels() throws RepositoryException {
203        return map(node.getMixinNodeTypes(), nodetype2name);
204    }
205
206    /**
207     * Update the properties Dataset with a SPARQL Update query. The updated
208     * properties may be serialized to the JCR store.
209     *
210     * After applying the statement, clients SHOULD check the result
211     * of #getDatasetProblems, which may include problems when attempting to
212     * serialize the data to JCR.
213     *
214     * @param subjects
215     * @param sparqlUpdateStatement
216     * @throws RepositoryException
217     */
218    public Dataset updatePropertiesDataset(final GraphSubjects subjects,
219                                        final String sparqlUpdateStatement)
220        throws RepositoryException {
221        final Dataset dataset = getPropertiesDataset(subjects, 0, 0);
222        final UpdateRequest request = UpdateFactory.create(sparqlUpdateStatement, dataset.getContext().getAsString(GraphProperties.URI_SYMBOL));
223        dataset.getDefaultModel().setNsPrefixes(request.getPrefixMapping());
224        UpdateAction.execute(request, dataset);
225        return dataset;
226    }
227
228    /**
229     * Serialize the JCR properties as an RDF Dataset
230     *
231     *
232     * @param subjects
233     * @param offset
234     * @param limit
235     * @return
236     * @throws RepositoryException
237     */
238    public Dataset getPropertiesDataset(final GraphSubjects subjects,
239            final long offset, final int limit)
240        throws RepositoryException {
241
242        if (this.properties != null) {
243            return this.properties.getProperties(
244                    node, subjects, offset, limit);
245        } else {
246            return null;
247        }
248    }
249
250    /**
251     * Serialize the JCR properties of this object as an RDF Dataset
252     * @return
253     * @throws RepositoryException
254     */
255    public Dataset getPropertiesDataset(final GraphSubjects subjects)
256        throws RepositoryException {
257        return getPropertiesDataset(subjects, 0, -1);
258    }
259
260    /**
261     * Serialize the JCR versions information as an RDF dataset
262     * @param subjects
263     * @return
264     * @throws RepositoryException
265     */
266    public Dataset getVersionDataset(final GraphSubjects subjects)
267        throws RepositoryException {
268        final Model model = JcrRdfTools.withContext(subjects, node.getSession()).getJcrPropertiesModel(FedoraTypesUtils.getVersionHistory(node), subjects.getGraphSubject(node));
269
270        final Dataset dataset = DatasetFactory.create(model);
271
272        String uri = subjects.getGraphSubject(node).getURI();
273        com.hp.hpl.jena.sparql.util.Context context = dataset.getContext();
274        context.set(GraphProperties.URI_SYMBOL,uri);
275
276        return dataset;
277    }
278
279    /**
280     * Tag the current version of the Node with a version label that
281     * can be retrieved by name later.
282     *
283     * @param label
284     * @throws RepositoryException
285     */
286    public void addVersionLabel(final String label) throws RepositoryException {
287        final VersionHistory versionHistory = getVersionHistory(node);
288        versionHistory.addVersionLabel(getBaseVersion(node).getName(), label,
289                                       true);
290    }
291
292    /**
293     * Check if a resource was created in this session
294     * @return
295     */
296    public boolean isNew() {
297        return node.isNew();
298    }
299
300    /**
301     * Replace the properties of this object with the properties from the given
302     * model
303     *
304     * @param subjects
305     * @param inputModel
306     * @return
307     * @throws RepositoryException
308     */
309    public Dataset replacePropertiesDataset(final GraphSubjects subjects, final Model inputModel) throws RepositoryException {
310        final Dataset propertiesDataset = getPropertiesDataset(subjects, 0, -2);
311        final Model model = propertiesDataset.getDefaultModel();
312
313        final Model removed = model.difference(inputModel);
314        model.remove(removed.listStatements());
315
316        final Model created = inputModel.difference(model);
317        model.add(created.listStatements());
318
319        return propertiesDataset;
320    }
321
322    /**
323     * Construct an ETag value from the last modified date and path.
324     *
325     * JCR has a mix:etag type, but it only takes into account binary
326     * properties. We actually want whole-object etag data.
327     *
328     * TODO : construct and store an ETag value on object modify
329     *
330     * @return
331     * @throws RepositoryException
332     */
333    public String getEtagValue() throws RepositoryException {
334        final Date lastModifiedDate = getLastModifiedDate();
335
336        if (lastModifiedDate != null) {
337            return DigestUtils.shaHex(node.getPath() + lastModifiedDate.toString());
338        } else {
339            return "";
340        }
341    }
