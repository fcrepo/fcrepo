075    /**
076     * Predicate for determining whether this {@link Node} is a Fedora object.
077     */
078    public static Predicate<Node> isFedoraResource = new Predicate<Node>() {
079
080        @Override
081        public boolean apply(final Node node) {
082            checkArgument(node != null, "null cannot be a Fedora object!");
083            try {
084                return map(node.getMixinNodeTypes(), nodetype2name).contains(
085                        FEDORA_RESOURCE);
086            } catch (final RepositoryException e) {
087                throw propagate(e);
088            }
089        }
090    };

176    /**
177     * Check if a node is "internal" and should not be exposed e.g. via the REST
178     * API
179     */
180    public static Predicate<Node> isInternalNode = new Predicate<Node>() {
181
182        @Override
183        public boolean apply(final Node n) {
184            checkArgument(n != null,
185                    "null is neither multiple or not multiple!");
186            try {
187                final NodeType primaryNodeType = n.getPrimaryNodeType();
188                return primaryNodeType != null &&
189                        primaryNodeType.isNodeType("mode:system");
190            } catch (final RepositoryException e) {
191                throw propagate(e);
192            }
193        }

214    /**
215     * Map a JCR property to an RDF property with the right namespace URI and
216     * local name
217     */
218    public static Function<Property, com.hp.hpl.jena.rdf.model.Property> getPredicateForProperty =
219            new Function<Property, com.hp.hpl.jena.rdf.model.Property>() {
220
221                @Override
222                public com.hp.hpl.jena.rdf.model.Property apply(
223                        final Property property) {
224                    LOGGER.trace("Creating predicate for property: {}",
225                            property);
226                    try {
227                        if (property instanceof Namespaced) {
228                            final Namespaced nsProperty = (Namespaced) property;
229                            final String uri = nsProperty.getNamespaceURI();
230                            return createProperty(
231                                    getRDFNamespaceForJcrNamespace(uri),
232                                    nsProperty.getLocalName());
233                        } else {
234                            return createProperty(property.getName());
235                        }
236                    } catch (final RepositoryException e) {
237                        throw propagate(e);
238                    }
239
240                }
241            };

249    /**
250     * Creates a JCR {@link Binary}
251     *
252     * @param n a {@link Node}
253     * @param i an {@link InputStream}
254     * @return a JCR {@link Binary}
255     */

257        try {
258            checkArgument(n != null,
259                    "null cannot have a Binary created for it!");
260            checkArgument(i != null,
261                    "null cannot have a Binary created from it!");

263        } catch (final RepositoryException e) {
264            throw propagate(e);
265        }

302    /**
303     * Get the property definition information (containing type and multi-value
304     * information)
305     *
306     * @param node the node to use for inferring the property definition
307     * @param propertyName the property name to retrieve a definition for
308     * @return a JCR PropertyDefinition, if available, or null
309     * @throws javax.jcr.RepositoryException
310     */
311    public static PropertyDefinition getDefinitionForPropertyName(
312            final Node node, final String propertyName)
313        throws RepositoryException {
314        final PropertyDefinition[] propertyDefinitions =
315                getNodeTypeManager(node).getNodeType(FEDORA_RESOURCE)
316                        .getPropertyDefinitions();
317
318        for (final PropertyDefinition p : propertyDefinitions) {
319            if (p.getName().equals(propertyName)) {
320                return p;
321            }
322        }
323        return null;
324    }

326    /**
327     * Convenience method for transforming arrays into {@link Collection}s
328     * through a mapping {@link Function}.
329     *
330     * @param input A Collection<F>.
331     * @param f A Function<F,T>.
332     * @return An ImmutableSet copy of input after transformation by f
333     */
334    public static <F, T> Collection<T> map(final F[] input,
335            final Function<F, T> f) {
336        return transform(copyOf(input), f);
337    }

388    /**
389     * @return a double of the size of the fedora:datastream binary content
390     * @throws RepositoryException
391     */
392    public static long getRepositoryCount(final Repository repository)
393        throws RepositoryException {
394        final Session session = repository.login();
395        try {
396            final QueryManager queryManager =
397                    session.getWorkspace().getQueryManager();
398
399            final String querystring =
400                    "SELECT [" + JcrConstants.JCR_PATH + "] FROM [" +
401                            FedoraJcrTypes.FEDORA_OBJECT + "]";
402
403            final QueryResult queryResults =
404                    queryManager.createQuery(querystring, JCR_SQL2).execute();
405
406            return queryResults.getRows().getSize();
407        } finally {
408
409            session.logout();
410        }
411    }

413    /**
414     * @return a double of the size of the fedora:datastream binary content
415     * @throws RepositoryException
416     */
417    public static long getRepositorySize(final Repository repository)
418        throws RepositoryException {
419        final Session session = repository.login();
420        long sum = 0;
421        final QueryManager queryManager =
422                session.getWorkspace().getQueryManager();
423
424        final String querystring =
425                "SELECT [" + FedoraJcrTypes.CONTENT_SIZE + "] FROM [" +
426                        FedoraJcrTypes.FEDORA_BINARY + "]";
427
428        final QueryResult queryResults =
429                queryManager.createQuery(querystring, JCR_SQL2).execute();
430
431        for (final RowIterator rows = queryResults.getRows(); rows.hasNext();) {
432            final Value value =
433                    rows.nextRow().getValue(FedoraJcrTypes.CONTENT_SIZE);
434            sum += value.getLong();
435        }
436
437        session.logout();
438
439        return sum;
440    }
