017package org.fcrepo.http.api;
018
019import static org.fcrepo.http.commons.domain.RDFMediaType.POSSIBLE_RDF_VARIANTS;
020import static org.fcrepo.http.commons.test.util.PathSegmentImpl.createPathList;
021import static org.fcrepo.http.commons.test.util.TestHelpers.getUriInfoImpl;
022import static org.fcrepo.http.commons.test.util.TestHelpers.mockSession;
023import static org.fcrepo.http.commons.test.util.TestHelpers.setField;
024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertNotNull;
026import static org.mockito.Matchers.any;
027import static org.mockito.Matchers.anyInt;
028import static org.mockito.Matchers.anyLong;
029import static org.mockito.Matchers.anyString;
030import static org.mockito.Mockito.verify;
031import static org.mockito.Mockito.when;
032import static org.mockito.MockitoAnnotations.initMocks;
033
034import java.io.IOException;
035
036import javax.jcr.RepositoryException;
037import javax.jcr.Session;
038import javax.ws.rs.core.MediaType;
039import javax.ws.rs.core.Request;
040import javax.ws.rs.core.Response;
041import javax.ws.rs.core.Variant;
042
043import org.fcrepo.http.api.FedoraVersions;
044import org.fcrepo.kernel.FedoraResource;
045import org.fcrepo.http.commons.api.rdf.HttpGraphSubjects;
046import org.fcrepo.kernel.services.NodeService;
047import org.fcrepo.http.commons.test.util.TestHelpers;
048import org.junit.Before;
049import org.junit.Test;
050import org.mockito.Mock;
051
052import com.hp.hpl.jena.query.Dataset;
053
054public class FedoraVersionsTest {
055
056    private FedoraVersions testObj;
057
058    @Mock
059    private NodeService mockNodes;
060
061    private Session mockSession;
062
063    @Mock
064    private FedoraResource mockResource;
065
066    @Mock
067    private Request mockRequest;
068
069    @Mock
070    private Variant mockVariant;
071
072    @Mock
073    private Dataset mockDataset;
074
075    @Before
076    public void setUp() throws Exception {
077        initMocks(this);
078        testObj = new FedoraVersions();
079        mockSession = mockSession(testObj);
080        setField(testObj, "nodeService", mockNodes);
081        setField(testObj, "uriInfo", getUriInfoImpl());
082        setField(testObj, "session", mockSession);
083    }
084
085    @Test
086    public void testGetVersionList() throws RepositoryException {
087        final String pid = "FedoraVersioningTest";
088        when(mockRequest.selectVariant(POSSIBLE_RDF_VARIANTS)).thenReturn(
089                mockVariant);
090        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
091                mockResource);
092        when(mockResource.getVersionDataset(any(HttpGraphSubjects.class)))
093                .thenReturn(mockDataset);
094        when(mockVariant.getMediaType()).thenReturn(
095                new MediaType("text", "turtle"));
096
097        final Response response =
098                testObj.getVersionList(createPathList(pid), mockRequest,
099                        getUriInfoImpl());
100        assertNotNull(response);
101        assertEquals(200, response.getStatus());
102    }
103
104    @Test
105    public void testAddVersionLabel() throws RepositoryException {
106        final String pid = "FedoraVersioningTest";
107        final String versionLabel = "FedoraVersioningTest1/fcr:versions/v0.0.1";
108        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
109                mockResource);
110
111        final Response response =
112                testObj.addVersionLabel(createPathList(pid), versionLabel);
113        verify(mockResource).addVersionLabel(anyString());
114        assertNotNull(response);
115    }
116
117    @Test
118    public void testGetVersion() throws RepositoryException, IOException {
119        final String pid = "FedoraVersioningTest";
120        final String versionLabel = "v0.0.1";
121        when(
122                mockNodes.getObject(any(Session.class), any(String.class),
123                        any(String.class))).thenReturn(mockResource);
124        testObj.getVersion(createPathList(pid), versionLabel, TestHelpers
125                .getUriInfoImpl());
126        verify(mockResource).getPropertiesDataset(any(HttpGraphSubjects.class),
127                anyLong(), anyInt());
128    }
129
130}
