710    private void validateHTML(final String path) throws Exception {
711        final HttpGet getMethod = new HttpGet(serverAddress + path);
712        getMethod.addHeader("Accept", "text/html");
713        final HttpResponse response = client.execute(getMethod);
714        assertEquals(OK.getStatusCode(), response.getStatusLine()
715                .getStatusCode());
716        final String content = EntityUtils.toString(response.getEntity());
717        logger.debug("Retrieved HTML view:\n" + content);
718
719        final HtmlParser htmlParser = new HtmlParser(ALLOW);
720        htmlParser.setDoctypeExpectation(NO_DOCTYPE_ERRORS);
721        htmlParser.setErrorHandler(new HTMLErrorHandler());
722        htmlParser.setContentHandler(new TreeBuilder());
723        try (final InputStream htmlStream =
724                new ByteArrayInputStream(content.getBytes())) {
725            htmlParser.parse(new InputSource(htmlStream));
726        }
727        logger.info("HTML found to be valid.");
728    }
