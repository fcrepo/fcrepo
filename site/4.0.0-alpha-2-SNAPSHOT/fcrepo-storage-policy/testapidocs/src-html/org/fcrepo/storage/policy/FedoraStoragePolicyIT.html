017package org.fcrepo.storage.policy;
018
019import static org.fcrepo.storage.policy.FedoraStoragePolicy.POLICY_RESOURCE;
020import static org.junit.Assert.assertEquals;
021
022import org.apache.http.Header;
023import org.apache.http.HttpResponse;
024import org.apache.http.client.methods.HttpDelete;
025import org.apache.http.client.methods.HttpGet;
026import org.apache.http.client.methods.HttpPost;
027import org.apache.http.entity.StringEntity;
028import org.apache.tika.io.IOUtils;
029import org.fcrepo.kernel.services.policy.StoragePolicy;
030import org.junit.After;
031import org.junit.Before;
032import org.junit.Test;
033
034import java.io.IOException;
035import java.util.HashSet;
036import java.util.Set;
037
038import static javax.ws.rs.core.MediaType.APPLICATION_FORM_URLENCODED;
039import static org.junit.Assert.assertNotNull;
040import static org.junit.Assert.fail;
041
042// Runs integration test for restful interface for storage policies
043public class FedoraStoragePolicyIT extends AbstractResourceIT {
044
045    private static final String MIME_KEY = "mix:mimeType";
046    private static final String MIME = "image/tiff";
047    private static final String STORE = "cloud-store";
048
049    private Set<String> policyKeys;
050
051    @Before
052    public void setUp() {
053        policyKeys = new HashSet<>();
054    }
055
056    @After
057    public void tearDown() {
058        for (String policyKey : policyKeys) {
059            deletePolicy(policyKey);
060        }
061    }
062   
063    @Test
064    public void testPolicyCreateByPost() throws Exception {
065        final HttpPost objMethod = HttpPostObjMethod(POLICY_RESOURCE);
066        StringEntity input = new StringEntity(MIME_KEY + " " + MIME + " " + STORE,
067                                              "UTF-8");
068        input.setContentType(APPLICATION_FORM_URLENCODED);
069        objMethod.setEntity(input);
070        final HttpResponse response = client.execute(objMethod);
071
072        String body = IOUtils.toString(response.getEntity().getContent());
073        assertEquals(body, 201, response.getStatusLine().getStatusCode());
074
075        policyKeys.add(MIME_KEY);
076
077        Header[] headers = response.getHeaders("Location");
078        assertNotNull(headers);
079        assertEquals(1, headers.length);
080        assertEquals(objMethod.getURI()
081                             .toString()
082                             .replace(POLICY_RESOURCE, MIME_KEY),
083                     headers[0].getValue());
084    }
085
086    @Test
087    public void testGetStoragePolicy() throws Exception {
088        // Test no policy
089        final HttpGet getMethod0 = HttpGetObjMethod(MIME_KEY);
090        final HttpResponse response0 = client.execute(getMethod0);
091        assertNotNull(response0);
092        assertEquals(IOUtils.toString(response0.getEntity().getContent()),
093                     404,
094                     response0.getStatusLine().getStatusCode());
095
096        // Add a policy
097        testPolicyCreateByPost();
098
099        // Test Get Storage Policy
100        final HttpGet getMethod1 = HttpGetObjMethod(MIME_KEY);
101        final HttpResponse response1 = client.execute(getMethod1);
102        assertNotNull(response1);
103
104        String body = IOUtils.toString(response1.getEntity().getContent());
105        assertEquals(body, 200, response1.getStatusLine().getStatusCode());
106        assertEquals(MIME + ":" + STORE, body);
107    }
108
109    @Test
110    public void testGetStoragePolicies() throws Exception {
111        // Add a policy
112        testPolicyCreateByPost();
113
114        // Test Get Storage Policy
115        final HttpGet getMethod1 = HttpGetObjMethod(POLICY_RESOURCE);
116        final HttpResponse response1 = client.execute(getMethod1);
117        assertNotNull(response1);
118
119        String body = IOUtils.toString(response1.getEntity().getContent());
120        assertEquals(body, 200, response1.getStatusLine().getStatusCode());
121
122        StoragePolicy policy = new MimeTypeStoragePolicy(MIME, STORE);
123        assertEquals("policies=[" + policy.toString() + "]", body);
124    }
125
126    @Test
127    public void testInvalidPoliciesCreateByPost() throws Exception {
128        final HttpPost objMethod = HttpPostObjMethod(POLICY_RESOURCE);
129        StringEntity input = new StringEntity("mix:newType " + MIME + " " + STORE,
130                                              "UTF-8");
131        input.setContentType(APPLICATION_FORM_URLENCODED);
132        objMethod.setEntity(input);
133        final HttpResponse response = client.execute(objMethod);
134        assertEquals(response.getStatusLine().getStatusCode(), 500);     
135    }
136    
137    @Test
138    public void testPolicyDelete() throws Exception {
139        testPolicyCreateByPost();
140
141        final HttpResponse response = deletePolicy(MIME_KEY);
142        assertEquals(204, response.getStatusLine().getStatusCode());
143
144        final HttpGet objGetMethod = HttpGetObjMethod(POLICY_RESOURCE);
145        final HttpResponse getResponse = client.execute(objGetMethod);
146        assertEquals("No Policies Found",
147                     IOUtils.toString(getResponse.getEntity().getContent())
148        );
149    }
150
151    private HttpResponse deletePolicy(String policyKey) {
152        final HttpDelete objMethod = HttpDeleteObjMethod(policyKey);
153        try {
154            return client.execute(objMethod);
155
156        } catch (IOException e) {
157            fail(e.getMessage());
158            return null; // never
159        }
160    }
161   
162}
