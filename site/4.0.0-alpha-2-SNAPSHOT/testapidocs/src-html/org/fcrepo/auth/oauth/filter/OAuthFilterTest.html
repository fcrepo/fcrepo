016package org.fcrepo.auth.oauth.filter;
017
018import static org.apache.oltu.oauth2.common.OAuth.OAUTH_CLIENT_ID;
019import static org.mockito.Matchers.any;
020import static org.mockito.Matchers.eq;
021import static org.mockito.Mockito.times;
022import static org.mockito.Mockito.verify;
023import static org.mockito.Mockito.when;
024import static org.mockito.MockitoAnnotations.initMocks;
025
026import java.io.IOException;
027import java.util.HashSet;
028
029import javax.servlet.FilterChain;
030import javax.servlet.ServletException;
031import javax.servlet.http.HttpServletRequest;
032import javax.servlet.http.HttpServletRequestWrapper;
033import javax.servlet.http.HttpServletResponse;
034
035import org.apache.oltu.oauth2.common.OAuth;
036import org.apache.oltu.oauth2.common.exception.OAuthProblemException;
037import org.apache.oltu.oauth2.common.message.types.ParameterStyle;
038import org.apache.oltu.oauth2.rsfilter.OAuthClient;
039import org.apache.oltu.oauth2.rsfilter.OAuthDecision;
040import org.apache.oltu.oauth2.rsfilter.OAuthRSProvider;
041import org.junit.Before;
042import org.junit.Test;
043import org.mockito.Mock;
044
045
046public class OAuthFilterTest {
047
048    private static final String DUMMY_REALM = "DUMMY-REALM";
049    private static final String DUMMY_TOKEN = "DUMMY-TOKEN";
050
051    @Mock
052    private HttpServletRequest mockRequest;
053    
054    @Mock
055    private HttpServletResponse mockResponse;
056    
057    @Mock
058    private FilterChain mockFilterChain;
059    
060    @Mock
061    private OAuthRSProvider mockProvider;
062    
063    @Mock
064    private OAuthDecision mockDecision;
065    
066    @Mock
067    private OAuthClient mockClient;
068    
069    private OAuthFilter testObj;
070    
071    @Before
072    public void setUp(){
073        initMocks(this);
074        testObj = new OAuthFilter();
075        testObj.setProvider(mockProvider);
076        testObj.setRealm(DUMMY_REALM);
077    }
078    
079    @Test
080    public void testValidRequest()
081        throws IOException, ServletException, OAuthProblemException {
082        when(mockRequest.getRequestURI()).thenReturn("foo://bar");
083        /**
084         *  the default validation method is via a header
085         *  of the format "Authorization: Bearer $TOKEN"
086         */
087        when(mockRequest.getHeader(OAuth.HeaderType.AUTHORIZATION))
088            .thenReturn("Bearer " + DUMMY_TOKEN);
089        when(mockProvider.validateRequest(DUMMY_REALM, DUMMY_TOKEN, mockRequest))
090            .thenReturn(mockDecision);
091        when(mockDecision.getOAuthClient()).thenReturn(mockClient);
092        when(mockProvider.validateRequest(
093            eq(DUMMY_REALM), any(String.class), eq(mockRequest)))
094            .thenReturn(mockDecision);
095        testObj.doFilter(mockRequest, mockResponse, mockFilterChain);
096        // a successful request should set the client id attribute
097        verify(mockRequest).setAttribute(eq(OAUTH_CLIENT_ID), any(String.class));
098        // a successful request should not send an error back to client
099        verify(mockResponse, times(0)).sendError(any(Integer.class));
100        // a successful request should call the rest of the filter chain
101        verify(mockFilterChain)
102            .doFilter(any(HttpServletRequestWrapper.class), eq(mockResponse));
103    }
104    
105
106    /**
107     * The token-bearing header is expected to be of the format
108     * Authorization : Bearer $TOKEN
109     * Our OAuth library sends no elaborating error type back
110     * in this circumstance, resulting in a generic 401 (Unauthorized)
111     * @throws IOException
112     * @throws ServletException
113     */
114    @Test
115    public void testBadAuthorizationHeader() throws IOException, ServletException {
116        when(mockRequest.getRequestURI()).thenReturn("foo://bar");
117        when(mockRequest.getHeader(OAuth.HeaderType.AUTHORIZATION))
118        .thenReturn("Broken " + DUMMY_TOKEN);
119        testObj.doFilter(mockRequest, mockResponse, mockFilterChain);
120        // a failed request should not set the client id attribute
121        verify(mockRequest, times(0)).setAttribute(eq(OAUTH_CLIENT_ID), any(String.class));
122        // a failed request should send an error response
123        verify(mockResponse).sendError(401);
124        // a failed request should not call the rest of the filter chain
125        verify(mockFilterChain, times(0))
126            .doFilter(any(HttpServletRequestWrapper.class), eq(mockResponse));
127    }
128    
129    /**
130     * The token-bearing header is must be present for a default
131     * configuration. Our OAuth library sends no elaborating error type back
132     * when it is missing, resulting in a generic 401 (Unauthorized)
133     * @throws IOException
134     * @throws ServletException
135     */
136    @Test
137    public void testMissingAuthorizationHeader()
138        throws IOException, ServletException {
139        when(mockRequest.getRequestURI()).thenReturn("foo://bar");
140        testObj.doFilter(mockRequest, mockResponse, mockFilterChain);
141        // a failed request should not set the client id attribute
142        verify(mockRequest, times(0)).setAttribute(eq(OAUTH_CLIENT_ID), any(String.class));
143        // a failed request should send an error back to client
144        verify(mockResponse).sendError(401);
145        // a failed request should not call the rest of the filter chain
146        verify(mockFilterChain, times(0))
147            .doFilter(any(HttpServletRequestWrapper.class), eq(mockResponse));
148    }
149    
150    /**
151     * A misconfigured filter that attempts to parse tokens in multiple
152     * ways is treated as an invalid request by our OAuth library, and
153     * should result in a 400 (Bad Request)
154     * @throws IOException
155     * @throws ServletException
156     * @throws OAuthProblemException 
157     */
158    @Test
159    public void testMultipleStyles()
160        throws IOException, ServletException, OAuthProblemException {
161        when(mockRequest.getRequestURI()).thenReturn("foo://bar");
162        when(mockRequest.getHeader(OAuth.HeaderType.AUTHORIZATION))
163        .thenReturn("Bearer " + DUMMY_TOKEN);
164        when(mockRequest.getParameter(OAuth.OAUTH_BEARER_TOKEN))
165        .thenReturn(DUMMY_TOKEN);
166        when(mockRequest.getQueryString())
167        .thenReturn("access_token=" + DUMMY_TOKEN);
168        when(mockProvider.validateRequest(DUMMY_REALM, DUMMY_TOKEN, mockRequest))
169        .thenReturn(mockDecision);
170        when(mockDecision.getOAuthClient()).thenReturn(mockClient);
171        HashSet<ParameterStyle> styles = new HashSet<ParameterStyle>(2);
172        styles.add(ParameterStyle.HEADER);
173        styles.add(ParameterStyle.QUERY);
174        testObj.setParameterStyles(styles);
175        testObj.doFilter(mockRequest, mockResponse, mockFilterChain);
176        // a failed request should not set the client id attribute
177        verify(mockRequest, times(0)).setAttribute(eq(OAUTH_CLIENT_ID), any(String.class));
178        // a failed request should send an error back to client
179        verify(mockResponse).sendError(400);
180        // a failed request should not call the rest of the filter chain
181        verify(mockFilterChain, times(0))
182            .doFilter(any(HttpServletRequestWrapper.class), eq(mockResponse));
183    }
184
185    /**
186     * A misconfigured filter that has no token parsing strategies
187     * is treated as an invalid request by our OAuth library, and
188     * should result in a 400 (Bad Request)
189     * @throws IOException
190     * @throws ServletException
191     */
192    @Test
193    public void testNoStyles()
194        throws IOException, ServletException {
195        when(mockRequest.getRequestURI()).thenReturn("foo://bar");
196        testObj.setParameterStyles(new HashSet<ParameterStyle>());
197        testObj.doFilter(mockRequest, mockResponse, mockFilterChain);
198        // a failed request should not set the client id attribute
199        verify(mockRequest, times(0)).setAttribute(eq(OAUTH_CLIENT_ID), any(String.class));
200        // a failed request should send an error back to client
201        verify(mockResponse).sendError(400);
202        // a failed request should not call the rest of the filter chain
203        verify(mockFilterChain, times(0))
204            .doFilter(any(HttpServletRequestWrapper.class), eq(mockResponse));
205    }
