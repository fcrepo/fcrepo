6   package org.fcrepo.kernel.api.services;
7   
8   import java.time.Instant;
9   
10  import org.fcrepo.kernel.api.RdfStream;
11  import org.fcrepo.kernel.api.Transaction;
12  import org.fcrepo.kernel.api.identifiers.FedoraId;
13  
14  /**
15   * Service used to manage membership properties of resources
16   *
17   * @author bbpennel
18   */
19  public interface MembershipService {
20  
21      /**
22       * Return an RdfStream of membership relations of which the provided resource is the subject.
23       *
24       * @param transaction transaction
25       * @param fedoraId the resource to get membership relations for.
26       * @return RdfStream of membership relations.
27       */
28      RdfStream getMembership(final Transaction transaction, final FedoraId fedoraId);
29  
30      /**
31       * Update membership properties based on the creation of the specified resource
32       *
33       * @param transaction transaction
34       * @param fedoraId ID of the object created
35       */
36      void resourceCreated(final Transaction transaction, final FedoraId fedoraId);
37  
38      /**
39       * Update membership properties based on the modification of the specified resource
40       *
41       * @param transaction transaction
42       * @param fedoraId ID of the object modified
43       */
44      void resourceModified(final Transaction transaction, final FedoraId fedoraId);
45  
46      /**
47       * Update membership properties based on the deletion of the specified resource
48       *
49       * @param transaction transaction
50       * @param fedoraId ID of the object deleted
51       */
52      void resourceDeleted(final Transaction transaction, final FedoraId fedoraId);
53  
54      /**
55       * Regenerate the membership history for specified Direct or Indirect container.
56       *
57       * @param transaction transaction
58       * @param containerId ID of the container
59       */
60      void populateMembershipHistory(final Transaction transaction, final FedoraId containerId);
61  
62      /**
63       * Get the timestamp of the most recent member added or removed, or null if none.
64       * @param transaction transaction or null if none
65       * @param fedoraId the resource id
66       * @return the timestamp or null
67       */
68      Instant getLastUpdatedTimestamp(final Transaction transaction, final FedoraId fedoraId);
69  
70      /**
71       * Commit any pending membership changes.
72       * @param transaction the transaction
73       */
74      void commitTransaction(final Transaction transaction);
75  
76      /**
77       * Rollback any pending membership changes.
78       * @param transaction the transaction
79       */
80      void rollbackTransaction(final Transaction transaction);
81  
82      /**
83       * Truncates the membership index. This should only be called when rebuilding the index.
84       */
85      void reset();
86  }
