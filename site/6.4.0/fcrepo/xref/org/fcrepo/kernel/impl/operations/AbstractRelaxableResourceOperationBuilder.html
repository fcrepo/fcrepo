6   package org.fcrepo.kernel.impl.operations;
7   
8   import org.apache.jena.rdf.model.Model;
9   import org.fcrepo.config.ServerManagedPropsMode;
10  import org.fcrepo.kernel.api.Transaction;
11  import org.fcrepo.kernel.api.identifiers.FedoraId;
12  import org.fcrepo.kernel.api.operations.RelaxableResourceOperationBuilder;
13  
14  import java.time.Instant;
15  
16  import static org.fcrepo.kernel.api.utils.RelaxedPropertiesHelper.getCreatedBy;
17  import static org.fcrepo.kernel.api.utils.RelaxedPropertiesHelper.getCreatedDate;
18  import static org.fcrepo.kernel.api.utils.RelaxedPropertiesHelper.getModifiedBy;
19  import static org.fcrepo.kernel.api.utils.RelaxedPropertiesHelper.getModifiedDate;
20  
21  /**
22   * Abstract builder for constructing relaxable resource operations
23   * @author bbpennel
24   */
25  public abstract class AbstractRelaxableResourceOperationBuilder extends AbstractResourceOperationBuilder
26                                                                  implements RelaxableResourceOperationBuilder {
27      protected String lastModifiedBy;
28  
29      protected String createdBy;
30  
31      protected Instant lastModifiedDate;
32  
33      protected Instant createdDate;
34  
35      protected ServerManagedPropsMode serverManagedPropsMode;
36  
37      protected AbstractRelaxableResourceOperationBuilder(final Transaction transaction, final FedoraId rescId,
38                                                          final ServerManagedPropsMode serverManagedPropsMode) {
39          super(transaction, rescId);
40          this.serverManagedPropsMode = serverManagedPropsMode;
41      }
42  
43      @Override
44      public RelaxableResourceOperationBuilder relaxedProperties(final Model model) {
45          // Has no affect if the server is not in relaxed mode
46          if (model != null && serverManagedPropsMode == ServerManagedPropsMode.RELAXED) {
47              final var resc = model.getResource(rescId.asResourceId().getFullDescribedId());
48  
49              final var createdDateVal = getCreatedDate(resc);
50              if (createdDateVal != null) {
51                  this.createdDate = createdDateVal.toInstant();
52              }
53              final var createdByVal = getCreatedBy(resc);
54              if (createdByVal != null) {
55                  this.createdBy = createdByVal;
56              }
57              final var modifiedDate = getModifiedDate(resc);
58              if (modifiedDate != null) {
59                  this.lastModifiedDate = modifiedDate.toInstant();
60              }
61              final var modifiedBy = getModifiedBy(resc);
62              if (modifiedBy != null) {
63                  this.lastModifiedBy = modifiedBy;
64              }
65          }
66  
67          return this;
68      }
69  }
