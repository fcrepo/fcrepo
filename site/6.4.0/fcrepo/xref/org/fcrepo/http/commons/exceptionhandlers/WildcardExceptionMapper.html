6   package org.fcrepo.http.commons.exceptionhandlers;
7   
8   import org.fcrepo.kernel.api.exception.SessionMissingException;
9   import org.slf4j.Logger;
10  
11  import javax.ws.rs.core.Response;
12  import javax.ws.rs.ext.ExceptionMapper;
13  import javax.ws.rs.ext.Provider;
14  
15  import static com.google.common.base.Throwables.getStackTraceAsString;
16  import static javax.ws.rs.core.Response.serverError;
17  import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
18  import static org.slf4j.LoggerFactory.getLogger;
19  
20  /**
21   * Catch all the exceptions!
22   *
23   * @author lsitu
24   * @author awoods
25   * @author cbeer
26   * @author fasseg
27   */
28  @Provider
29  public class WildcardExceptionMapper implements
30          ExceptionMapper<Exception>, ExceptionDebugLogging {
31  
32      Boolean showStackTrace = false;
33  
34      private static final Logger LOGGER =
35          getLogger(WildcardExceptionMapper.class);
36  
37      @Override
38      public Response toResponse(final Exception e) {
39          if (e.getCause() instanceof SessionMissingException) {
40              return new SessionMissingExceptionMapper()
41                      .toResponse((SessionMissingException) e.getCause());
42          }
43  
44          LOGGER.warn("Unmapped exception", e);
45          return serverError().entity(
46                  showStackTrace ? getStackTraceAsString(e) : null).type(TEXT_PLAIN_WITH_CHARSET).build();
47      }
48  
49      /**
50       * Set whether the full stack trace should be returned as part of the
51       * error response. This may be a bad idea if the stack trace is exposed
52       * to the public.
53       * @param showStackTrace the boolean value of showing stack trace
54       */
55      public void setShowStackTrace(final Boolean showStackTrace) {
56          this.showStackTrace = showStackTrace;
57      }
58  }
