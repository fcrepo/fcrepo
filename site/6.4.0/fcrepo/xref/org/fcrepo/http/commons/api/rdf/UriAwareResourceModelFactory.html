6   package org.fcrepo.http.commons.api.rdf;
7   
8   import javax.ws.rs.core.UriInfo;
9   
10  import org.fcrepo.kernel.api.models.FedoraResource;
11  
12  import org.apache.jena.rdf.model.Model;
13  
14  /**
15   * Helper to generate an RDF model for a FedoraResourceImpl that (likely) creates
16   * relations from our resource to other HTTP components
17   *
18   * @author awoods
19   */
20  public interface UriAwareResourceModelFactory {
21  
22      /**
23       * Given a resource, the UriInfo and a way to generate graph subjects,
24       * create a model with triples to inject into an RDF response for the
25       * resource (e.g. to add HATEOAS links)
26       *
27       * @param resource the resource
28       * @param uriInfo the uri info
29       * @return model containing triples for the given resource
30       */
31      Model createModelForResource(final FedoraResource resource, final UriInfo uriInfo);
32  }
