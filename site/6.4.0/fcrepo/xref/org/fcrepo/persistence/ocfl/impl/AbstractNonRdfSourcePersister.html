25  import static org.fcrepo.kernel.api.operations.ResourceOperationType.CREATE;

27  /**
28   * This class implements the persistence of a NonRDFSource
29   *
30   * @author whikloj
31   * @since 6.0.0
32   */
33  abstract class AbstractNonRdfSourcePersister extends AbstractPersister {
34  
35      private static final Logger log = LoggerFactory.getLogger(AbstractNonRdfSourcePersister.class);
36  
37      /**
38       * Constructor
39       */
40      protected AbstractNonRdfSourcePersister(final Class<? extends ResourceOperation> resourceOperation,
41                                    final ResourceOperationType resourceOperationType,
42                                    final FedoraToOcflObjectIndex index) {
43          super(resourceOperation, resourceOperationType, index);
44      }
45  
46      /**
47       * This method handles the shared logic for writing the resource specified in the operation parameter to
48       * the OCFL Object Session.
49       *
50       * @param operation The operation to perform the persistence routine on
51       * @param objectSession The ocfl object session
52       * @param rootIdentifier The fedora object root identifier associated with the resource to be persisted.
53       * @param isArchivalPart indicates if the resource is an AG part resource, ignored on update
54       * @throws PersistentStorageException thrown if writing fails
55       */
56      protected void persistNonRDFSource(final ResourceOperation operation,
57                                         final OcflObjectSession objectSession,
58                                         final FedoraId rootIdentifier,
59                                         final boolean isArchivalPart)
60              throws PersistentStorageException {
61          final var resourceId = operation.getResourceId();
62  
63          log.debug("persisting NonRDFSource {} to OCFL", resourceId);
64  
65          final var nonRdfSourceOperation = (NonRdfSourceOperation) operation;
66  
67          final var headers = new ResourceHeadersAdapter(
68                  createHeaders(objectSession,
69                          nonRdfSourceOperation,
70                          resourceId.equals(rootIdentifier),
71                          isArchivalPart ? rootIdentifier : null));
72  
73          // This covers the case where a resource may have been temporarily deleted if it was changed from an internal
74          // non-RDF resource to an external non-RDF resource
75          headers.setDeleted(false);
76  
77          if (forExternalBinary(nonRdfSourceOperation)) {
78              objectSession.writeResource(headers.asStorageHeaders(), null);
79          } else {
80              final var providedDigests = nonRdfSourceOperation.getContentDigests();
81  
82              // Wrap binary stream in digest computing wrapper, requesting
83              final var multiDigestWrapper = new MultiDigestInputStreamWrapper(
84                      nonRdfSourceOperation.getContentStream(),
85                      providedDigests,
86                      Collections.emptyList());
87              final var contentStream = multiDigestWrapper.getInputStream();
88  
89              objectSession.writeResource(headers.asStorageHeaders(), new BufferedInputStream(contentStream));
90  
91              // Verify that the content matches the provided digests
92              if (!CollectionUtils.isEmpty(providedDigests)) {
93                  multiDigestWrapper.checkFixity();
94              }
95          }
96      }
97  
98      /**
99       * Constructs a ResourceHeaders object populated with the properties provided by the
100      * operation, and merged with existing properties if appropriate.
101      *
102      * @param objSession the object session
103      * @param op the operation being persisted
104      * @param objectRoot flag indicating whether or not headerPath represents the object root resource
105      * @param archivalGroupId for AG parts, the id of the containg AG, otherwise null
106      * @return populated resource headers
107      */
108     private ResourceHeadersImpl createHeaders(final OcflObjectSession objSession,
109                                               final NonRdfSourceOperation op,
110                                               final boolean objectRoot,
111                                               final FedoraId archivalGroupId) throws PersistentStorageException {
112 
113         final var headers = createCommonHeaders(objSession, op, objectRoot, archivalGroupId);
114 
115         ResourceHeaderUtils.populateBinaryHeaders(headers,
116                 op.getMimeType(),
117                 op.getFilename(),
118                 op.getContentSize(),
119                 op.getContentDigests());
120 
121         if (forExternalBinary(op)) {
122             // Clear any content path for externals
123             headers.setContentPath(null);
124             ResourceHeaderUtils.populateExternalBinaryHeaders(headers,
125                     op.getContentUri().toString(),
126                     op.getExternalHandling());
127         } else {
128             if (!CREATE.equals(op.getType())) {
129                 // If performing an update of an internal binary, ensure that no external properties are retained
130                 headers.setExternalHandling(null);
131                 headers.setExternalUrl(null);
132             }
133         }
134 
135         return headers;
136     }
137 
138     /**
139      * @param op the operation
140      * @return Returns true if the operation involved persisting an external binary
141      */
142     protected boolean forExternalBinary(final NonRdfSourceOperation op) {
143         return op.getContentUri() != null && op.getExternalHandling() != null;
144     }
145 }
