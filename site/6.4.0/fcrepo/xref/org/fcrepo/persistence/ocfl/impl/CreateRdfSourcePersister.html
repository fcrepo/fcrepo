10  import org.fcrepo.kernel.api.operations.RdfSourceOperation;
11  import org.fcrepo.kernel.api.operations.ResourceOperation;
12  import org.fcrepo.persistence.api.exceptions.PersistentItemConflictException;
13  import org.fcrepo.persistence.api.exceptions.PersistentStorageException;
14  import org.fcrepo.persistence.ocfl.api.FedoraToOcflObjectIndex;
15  import org.fcrepo.storage.ocfl.OcflObjectSession;
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  
19  import static org.fcrepo.kernel.api.operations.ResourceOperationType.CREATE;
20  
21  /**
22   * This class implements the persistence of a new RDFSource
23   *
24   * @author dbernstein
25   * @since 6.0.0
26   */
27  class CreateRdfSourcePersister extends AbstractRdfSourcePersister {
28  
29      private static final Logger log = LoggerFactory.getLogger(CreateRdfSourcePersister.class);
30  
31      /**
32       * Constructor
33       * @param index The FedoraToOCFLObjectIndex
34       */
35      protected CreateRdfSourcePersister(final FedoraToOcflObjectIndex index) {
36          super(RdfSourceOperation.class, CREATE, index);
37      }
38  
39      @Override
40      public void persist(final OcflPersistentStorageSession session, final ResourceOperation operation)
41              throws PersistentStorageException {
42  
43          final var resourceId = operation.getResourceId();
44          log.debug("persisting {} to {}", resourceId, session);
45  
46          final CreateResourceOperation createResourceOp = ((CreateResourceOperation)operation);
47          final boolean archivalGroup = createResourceOp.isArchivalGroup();
48  
49          final var archivalGroupId = findArchivalGroupInAncestry(resourceId, session);
50          final FedoraId rootObjectId;
51  
52          if (archivalGroup) {
53              //if archival group, ensure that there are no archival group ancestors
54              if (archivalGroupId.isPresent()) {
55                  throw new PersistentItemConflictException("Nesting an ArchivalGroup within an ArchivalGroup is not " +
56                          "permitted");
57              }
58              rootObjectId = resourceId;
59          } else {
60              rootObjectId = archivalGroupId.orElseGet(resourceId::asBaseId);
61          }
62  
63          final String ocflObjectId = mapToOcflId(operation.getTransaction(), rootObjectId);
64          final OcflObjectSession ocflObjectSession = session.findOrCreateSession(ocflObjectId);
65          persistRDF(ocflObjectSession, operation, rootObjectId.asBaseId(), archivalGroupId.isPresent());
66          ocflIndex.addMapping(operation.getTransaction(), resourceId.asResourceId(), rootObjectId.asBaseId(),
67                  ocflObjectId);
68      }
69  }
