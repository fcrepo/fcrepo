6   package org.fcrepo.persistence.ocfl.api;
7   
8   import org.fcrepo.kernel.api.operations.ResourceOperation;
9   import org.fcrepo.persistence.api.exceptions.PersistentStorageException;
10  import org.fcrepo.persistence.ocfl.impl.OcflPersistentStorageSession;
11  
12  /**
13   * @author dbernstein
14   * @since 6.0.0
15   */
16  public interface Persister {
17  
18      /**
19       * The method returns true if the operation can be persisted by this persister.
20       * @param operation the operation to persist
21       * @return true or false
22       */
23      boolean handle(ResourceOperation operation);
24  
25      /**
26       * The persistence handling for the given operation.
27       *
28       * @param session The persistent storage session
29       * @param operation The operation and associated data need to perform the operation.
30       * @throws PersistentStorageException on failure
31       */
32      void persist(final OcflPersistentStorageSession session,
33              final ResourceOperation operation)
34              throws PersistentStorageException;
35  }
