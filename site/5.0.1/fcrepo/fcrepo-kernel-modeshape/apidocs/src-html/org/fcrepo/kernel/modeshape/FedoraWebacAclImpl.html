018package org.fcrepo.kernel.modeshape;
019
020import javax.jcr.Node;
021import javax.jcr.RepositoryException;
022
023import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
024import org.fcrepo.kernel.api.models.FedoraWebacAcl;
025
026/**
027 * @author lsitu
028 * @since Apr. 19, 2018
029 */
030public class FedoraWebacAclImpl extends ContainerImpl implements FedoraWebacAcl {
031
032    /**
033     * Construct a {@link org.fcrepo.kernel.api.models.FedoraWebacAcl} from an existing JCR Node
034     * @param node an existing JCR node to be treated as a fcrepo object
035     */
036    public FedoraWebacAclImpl(final Node node) {
037        super(node);
038    }
039
040    @Override
041    public boolean isOriginalResource() {
042        return false;
043    }
044
045    /**
046     * Check if the JCR node has a webac:Acl mixin
047     * @param node the JCR node
048     * @return true if the JCR node has the webac:Acl mixin
049     */
050    public static boolean hasMixin(final Node node) {
051        try {
052            return node.isNodeType(FEDORA_WEBAC_ACL);
053        } catch (final RepositoryException e) {
054            throw new RepositoryRuntimeException(e);
055        }
056    }
057}
