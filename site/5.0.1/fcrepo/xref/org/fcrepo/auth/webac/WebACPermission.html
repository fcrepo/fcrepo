18  package org.fcrepo.auth.webac;
19  
20  import java.net.URI;
21  
22  import org.apache.shiro.authz.Permission;
23  
24  /**
25   * A WebAC permission represents a particular mode of access (e.g., acl:read) to a particular resource. Both the mode
26   * and resource are URIs. One WebAC permission implies another if and only if their mode and resource URIs are both
27   * equal to the other's.
28   *
29   * @author peichman
30   */
31  public class WebACPermission implements Permission {
32  
33      private final URI resource;
34  
35      private final URI mode;
36  
37      /**
38       * @param mode ACL access mode
39       * @param resource resource to be accessed
40       */
41      public WebACPermission(final URI mode, final URI resource) {
42          this.mode = mode;
43          this.resource = resource;
44      }
45  
46      /**
47       * One WebACPermission implies another if they are equal (i.e., have the same mode and resource URIs).
48       *
49       * @param p permission to compare to
50       */
51      @Override
52      public boolean implies(final Permission p) {
53          return equals(p);
54      }
55  
56      /**
57       * One WebACPermission equals another if they have the same mode and resource URIs.
58       *
59       * @param o object to compare to
60       */
61      @Override
62      public boolean equals(final Object o) {
63          if (o instanceof WebACPermission) {
64              final WebACPermission./../org/fcrepo/auth/webac/WebACPermission.html#WebACPermission">WebACPermission perm = (WebACPermission) o;
65              return perm.getResource().equals(resource) && perm.getMode().equals(mode);
66          } else {
67              return false;
68          }
69      }
70  
71      @Override
72      public int hashCode() {
73          final int prime = 31;
74          int result = 1;
75          result = prime * result + ((mode == null) ? 0 : mode.hashCode());
76          result = prime * result + ((resource == null) ? 0 : resource.hashCode());
77          return result;
78      }
79  
80      /**
81       * @return the mode
82       */
83      private URI getMode() {
84          return mode;
85      }
86  
87      /**
88       * @return the resource
89       */
90      private URI getResource() {
91          return resource;
92      }
93  
94      @Override
95      public String toString() {
96          return "[" + mode.toString() + " " + resource.toString() + "]";
97      }
98  
99  }
