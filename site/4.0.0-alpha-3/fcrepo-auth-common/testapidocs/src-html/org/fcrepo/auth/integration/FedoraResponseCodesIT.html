017package org.fcrepo.auth.integration;

019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertTrue;
021
022import java.io.IOException;
023
024import org.apache.http.HttpResponse;
025import org.apache.http.client.methods.HttpGet;
026import org.apache.http.client.methods.HttpPost;
027import org.apache.http.client.methods.HttpPut;
028import org.apache.http.util.EntityUtils;
029import org.junit.Before;

031
032public class FedoraResponseCodesIT extends AbstractResourceIT {
033
034    @Before
035    public void setup() throws IOException {
036        final HttpPost objMethod = postObjMethod("Permit");
037        assertEquals(201, getStatus(objMethod));
038    }
039
040    @Test
041    public void testAllowedAddDatastream() throws Exception {
042        final HttpPost objMethod =
043                postObjMethod("FedoraDatastreamsTest2Permit");
044
045        assertEquals(201, getStatus(objMethod));
046        final HttpPost method =
047                postDSMethod("FedoraDatastreamsTest2Permit", "zxcpermit",
048                        "foo");
049        final HttpResponse response = client.execute(method);
050        final String location =
051                response.getFirstHeader("Location").getValue();
052        assertEquals(201, response.getStatusLine().getStatusCode());
053        assertEquals(
054                "Got wrong URI in Location header for datastream creation!",
055                serverAddress +
056                        "FedoraDatastreamsTest2Permit/zxcpermit/fcr:content",
057                location);
058    }
059
060    @Test
061    public void testDeniedAddDatastream() throws Exception {
062        final HttpPost objMethod =
063                postObjMethod("FedoraDatastreamsTest2Permit");
064
065        assertEquals(201, getStatus(objMethod));
066        final HttpPost obj2Method = postObjMethod(
067                "FedoraDatastreamsTest2Permit/FedoraDatastreamsTest2Deny");
068        assertEquals(201, getStatus(obj2Method));
069        final HttpPost method = postDSMethod(
070                "FedoraDatastreamsTest2Permit/FedoraDatastreamsTest2Deny",
071                "zxc", "foo");
072        final HttpResponse response = client.execute(method);
073        assertEquals(403, response.getStatusLine().getStatusCode());
074    }
075
076    @Test
077    public void testAllowedAddDeepDatastream() throws Exception {
078        final HttpPost method =
079                postDSMethod(
080                        "FedoraDatastreamsTest2_permit/does_permit/not_permit/exist_permit/yet_permit",
081                        "zxc_permit", "foo");
082        final HttpResponse response = client.execute(method);
083        final String location =
084                response.getFirstHeader("Location").getValue();
085        assertEquals(201, response.getStatusLine().getStatusCode());
086        assertEquals(
087                "Got wrong URI in Location header for datastream creation!",
088                serverAddress +
089                        "FedoraDatastreamsTest2_permit/does_permit/not_permit/exist_permit/yet_permit/zxc_permit/fcr:content",
090                location);
091    }
092
093    @Test
094    public void testDeniedAddDeepDatastream() throws Exception {
095        final HttpPost method =
096                postDSMethod(
097                        "FedoraDatastreamsTest2_permit/does_permit/not_permit/exist_permit/yet_permit/allowed_child",
098                        "zxc", "foo");

101    }

104    public void testAllowedPutDatastream() throws Exception {
105        final HttpPost objMethod =
106                postObjMethod("FedoraDatastreamsTestPut_permit");

108        final HttpPut method =
109                putDSMethod("FedoraDatastreamsTestPut_permit",
