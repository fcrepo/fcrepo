018import java.util.ArrayList;
019import java.util.Collections;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023
024/**
025 * @author Scott Prater
026 */
027public abstract class AbstractCommonRolesIT extends AbstractRolesIT {
028
029    protected final static List<RolesPepTestObjectBean> test_objs =
030            defineTestObjects();
031
032    protected static Map<String, List<String>> t_roles =
033            new HashMap<String, List<String>>();
034
035    protected static Map<String, List<String>> admin_role =
036            new HashMap<String, List<String>>();
037
038    static {
039        t_roles.put("exampleadmin", Collections.singletonList("admin"));
040        t_roles.put("examplereader", Collections.singletonList("reader"));
041        t_roles.put("examplewriter", Collections.singletonList("writer"));
042        admin_role.put("exampleadmin", Collections.singletonList("admin"));
043    }
044
045    protected String test_json_roles = makeJson(t_roles);
046
047    protected String admin_json_role = makeJson(admin_role);
048
049    private static List<RolesPepTestObjectBean> defineTestObjects() {
050        final List<RolesPepTestObjectBean> test_objs =
051                new ArrayList<RolesPepTestObjectBean>();
052        final RolesPepTestObjectBean objA = new RolesPepTestObjectBean();
053        /* parent object */
054        objA.setPath("testcommonobj1");
055        test_objs.add(objA);
056
057        final RolesPepTestObjectBean objB = new RolesPepTestObjectBean();
058        /* child object */
059        objA.setPath("testcommonobj1/testchildobj1");
060        test_objs.add(objB);
061
062        return test_objs;
063
064    }
065}
