041/**
042 * @author Mike Durbin

044public class VersionServiceTest {
045
046
047    private VersionService testObj;
048
049    private TransactionService txService;
050
051    @Mock
052    private Session s;
053
054    @Mock
055    private Workspace mockWorkspace;
056
057    @Mock
058    private VersionManager mockVM;
059
060    @Before
061    public void setup() throws Exception {
062        txService = new TransactionService();
063        initMocks(this);
064        testObj = new VersionService();
065        testObj.txService = txService;
066        txService.versionService = testObj;
067
068        s = mock(Session.class);
069        mockWorkspace = mock(Workspace.class);
070        when(mockWorkspace.getName()).thenReturn("default");
071        when(s.getWorkspace()).thenReturn(mockWorkspace);
072        mockVM = mock(VersionManager.class);
073        when(mockWorkspace.getVersionManager()).thenReturn(mockVM);
074
075        // add a node that's versioned (but not auto-versioned)
076        Node versionedNode = mock(Node.class);
077        when(versionedNode.getPath()).thenReturn("/example-versioned");
078        when(versionedNode.getSession()).thenReturn(s);
079        when(versionedNode.isNodeType(VersionService.VERSIONABLE)).thenReturn(true);
080        when(s.getNode("/example-versioned")).thenReturn(versionedNode);
081
082        // add a node that's autoversioned
083        Node autoversionedNode = mock(Node.class);
084        when(autoversionedNode.getPath()).thenReturn("/example-auto-versioned");
085        when(autoversionedNode.getSession()).thenReturn(s);
086        when(autoversionedNode.isNodeType(VersionService.VERSIONABLE)).thenReturn(true);
087        when(s.getNode("/example-auto-versioned")).thenReturn(autoversionedNode);
088        Property autoVersionProperty = mock(Property.class);
089        Value autoVersionValue = mock(Value.class);
090        when(autoVersionValue.getString()).thenReturn(VersionService.AUTO_VERSION);
091        when(autoVersionProperty.isMultiple()).thenReturn(true);
092        when(autoVersionProperty.getValues()).thenReturn(new Value[] { autoVersionValue });
093        when(autoversionedNode.hasProperty(VersionService.VERSION_POLICY)).thenReturn(true);
094        when(autoversionedNode.getProperty(VersionService.VERSION_POLICY)).thenReturn(autoVersionProperty);
095
096
097        // add a node that's unversioned
098        Node unversionedNode = mock(Node.class);
099        when(unversionedNode.getPath()).thenReturn("/example-unversioned");
100        when(unversionedNode.getSession()).thenReturn(s);
101        when(unversionedNode.isNodeType(VersionService.VERSIONABLE)).thenReturn(false);
102        when(s.getNode("/example-unversioned")).thenReturn(unversionedNode);
103    }
104
105    private void markAsAutoVersioned(Node versionedNode) throws RepositoryException {
106        Property autoVersionProperty = mock(Property.class);
107        Value autoVersionValue = mock(Value.class);
108        when(autoVersionValue.getString()).thenReturn(VersionService.AUTO_VERSION);
109        when(autoVersionProperty.getValue()).thenReturn(autoVersionValue);
110        when(versionedNode.getProperty(VersionService.VERSION_POLICY)).thenReturn(autoVersionProperty);
111    }
112
113    @Test
114    public void testCheckpoint() throws Exception {
115        // request a version be created
116        testObj.nodeUpdated(s, "/example-versioned");
117
118        // ensure that it was
119        verify(mockVM, never()).checkpoint("/example-versioned");
120    }
121
122    @Test

124        // request a version be created
125        testObj.nodeUpdated(s, "/example-unversioned");
126
127        // ensure that it was
128        verify(mockVM, never()).checkpoint("/example-unversioned");
129    }
130
131    @Test
132    public void testCheckpointAutoVersioned() throws Exception {
133        // request a version be created
134        testObj.nodeUpdated(s, "/example-auto-versioned");
135
136        // ensure that it was
137        verify(mockVM, only()).checkpoint("/example-auto-versioned");
138    }
139
140    @Test
141    public void testDeferredCheckpointVersioned() throws Exception {
142        // start a transaction
143        Transaction t = txService.beginTransaction(s);
144        s = t.getSession();
145        when(s.getNamespaceURI(TransactionService.FCREPO4_TX_ID))
146                .thenReturn(t.getId());
147
148        assertNotNull("Transaction must have started!",
149                txService.getTransaction(
150                        s.getNode("/example-auto-versioned").getSession()));
151
152        // request a version be created
153        testObj.nodeUpdated(s, "/example-versioned");
154
155        // ensure that no version was created (because the transaction is still open)
156        verify(mockVM, never()).checkpoint("/example-versioned");
157
158        // close the transaction
159        txService.commit(t.getId());
160
161        // ensure that no version was made because none was explicitly requested
162        verify(mockVM, never()).checkpoint("/example-versioned");
163    }
