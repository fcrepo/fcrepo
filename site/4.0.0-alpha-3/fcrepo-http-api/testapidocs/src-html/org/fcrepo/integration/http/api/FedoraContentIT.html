091    @Test
092    public void testPutDatastreamWithContentDisposition() throws Exception {
093
094        final String pid = randomUUID().toString();
095
096        final HttpPost objMethod = postObjMethod(pid);
097        assertEquals(201, getStatus(objMethod));
098        final HttpPut method =
099            putDSMethod(pid, "zxc", "foo");
100        method.addHeader("Content-Disposition", "inline; filename=\"some-name\"");
101        final HttpResponse response = client.execute(method);
102        final String location = response.getFirstHeader("Location").getValue();
103        assertEquals(201, response.getStatusLine().getStatusCode());
104
105        final HttpGet getObjMethod = new HttpGet(serverAddress + pid + "/zxc");
106        final GraphStore results = getGraphStore(getObjMethod);
107        assertTrue("Didn't find original name!",
108                      results.contains(Node.ANY, NodeFactory.createURI(location), NodeFactory.createURI("http://www.loc.gov/premis/rdf/v1#hasOriginalName"), NodeFactory.createLiteral("some-name")));
109
110        final HttpGet getContentMethod = new HttpGet(location);
111        final HttpResponse contentResponse = client.execute(getContentMethod);
112
113        final ContentDisposition contentDisposition = new ContentDisposition(contentResponse.getFirstHeader("Content-Disposition").getValue());
114
115        assertEquals("some-name", contentDisposition.getFileName());
116    }

118    @Test
119    public void testMutateDatastream() throws Exception {
120        final HttpPost createObjectMethod =
121                postObjMethod("FedoraDatastreamsTest3");
122        assertEquals("Couldn't create an object!", 201,
123                getStatus(createObjectMethod));
124
125        final HttpPost createDataStreamMethod =
126                postDSMethod("FedoraDatastreamsTest3", "ds1", "foo");
127        assertEquals("Couldn't create a datastream!", 201,
128                getStatus(createDataStreamMethod));
129
130        final HttpPut mutateDataStreamMethod =
131                putDSMethod("FedoraDatastreamsTest3", "ds1", "bar");
132        mutateDataStreamMethod.setEntity(new StringEntity(faulkner1, "UTF-8"));
133        final HttpResponse response = client.execute(mutateDataStreamMethod);
134        final int status = response.getStatusLine().getStatusCode();
135        if (status != 204) {
136            logger.error(EntityUtils.toString(response.getEntity()));
137        }
138        assertEquals("Couldn't mutate a datastream!", 204, status);
139
140        final HttpGet retrieveMutatedDataStreamMethod =
141                new HttpGet(serverAddress +
142                        "FedoraDatastreamsTest3/ds1/fcr:content");
143        assertTrue("Datastream didn't accept mutation!", faulkner1
144                .equals(EntityUtils.toString(client.execute(
145                        retrieveMutatedDataStreamMethod).getEntity())));
146    }

148    @Test
149    public void testGetDatastreamContent() throws Exception {
150        final HttpPost createObjMethod =
151                postObjMethod("FedoraDatastreamsTest6");
152        assertEquals(201, getStatus(createObjMethod));
153
154        final HttpPost createDSMethod =
155                postDSMethod("FedoraDatastreamsTest6", "ds1",
156                        "marbles for everyone");
157        assertEquals(201, getStatus(createDSMethod));
158        final HttpGet method_test_get =
159                new HttpGet(serverAddress +
160                        "FedoraDatastreamsTest6/ds1/fcr:content");
161        assertEquals(200, getStatus(method_test_get));
162        final HttpResponse response = client.execute(method_test_get);
163        logger.debug("Returned from HTTP GET, now checking content...");
164        assertTrue("Got the wrong content back!", "marbles for everyone"
165                .equals(EntityUtils.toString(response.getEntity())));
166
167        assertEquals("urn:sha1:ba6cb22191300aebcfcfb83de9635d6b224677df",
168                response.getFirstHeader("ETag").getValue().replace("\"", ""));
169
170        final ContentDisposition contentDisposition = new ContentDisposition(response.getFirstHeader("Content-Disposition").getValue());
171
172        assertEquals("attachment", contentDisposition.getType());
173        assertEquals("ds1", contentDisposition.getFileName());
174
175        logger.debug("Content was correct.");
176    }

178    @Test
179    public void testRefetchingDatastreamContent() throws Exception {
180
181        final HttpPost createObjMethod =
182                postObjMethod("FedoraDatastreamsTest61");
183        assertEquals(201, getStatus(createObjMethod));
184
185        final HttpPost createDSMethod =
186                postDSMethod("FedoraDatastreamsTest61", "ds1",
187                        "marbles for everyone");
188        assertEquals(201, getStatus(createDSMethod));
189
190        final SimpleDateFormat format =
191                new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss zzz", Locale.US);
192        format.setTimeZone(getTimeZone("GMT"));
193
194        final HttpGet method_test_get =
195                new HttpGet(serverAddress +
196                        "FedoraDatastreamsTest61/ds1/fcr:content");
197        method_test_get.setHeader("If-None-Match",
198                "\"urn:sha1:ba6cb22191300aebcfcfb83de9635d6b224677df\"");
199        method_test_get.setHeader("If-Modified-Since", format
200                .format(new Date()));
201
202        assertEquals(304, getStatus(method_test_get));
203
204    }
