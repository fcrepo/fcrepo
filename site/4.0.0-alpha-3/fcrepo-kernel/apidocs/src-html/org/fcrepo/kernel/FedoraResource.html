093    /**
094     * Create or find a FedoraObject at the given path
095     * @param session the JCR session to use to retrieve the object
096     * @param path the absolute path to the object
097     * @throws RepositoryException
098     */
099    public FedoraResource(final Session session, final String path,
100        final String nodeType) throws RepositoryException {
101        this();
102        initializeNewResourceProperties(session, path, nodeType);
103    }
104
105    private void initializeNewResourceProperties(final Session session,
106                                                 final String path,
107                                                 final String nodeType) throws RepositoryException {
108        this.node = findOrCreateNode(
109                session, path, NT_FOLDER, nodeType);
110
111        if (node.isNew()) {
112
113            if (!hasMixin(node) && !isFrozen(node)) {
114                node.addMixin(FEDORA_RESOURCE);
115            }
116
117            node.setProperty(JCR_LASTMODIFIED, Calendar.getInstance());
118        }
119    }

169    /**
170     * Get the date this datastream was created
171     * @return
172     * @throws RepositoryException
173     */
174    public Date getCreatedDate() throws RepositoryException {
175        if (node.hasProperty(JCR_CREATED)) {
176            return new Date(node.getProperty(JCR_CREATED).getDate()
177                            .getTimeInMillis());
178        } else {
179            LOGGER.debug("Node {} does not have a createdDate", node);
180            return null;
181        }
182    }
183
184    /**
185     * Get the date this datastream was last modified
186     * @return
187     * @throws RepositoryException
188     */
189    public Date getLastModifiedDate() throws RepositoryException {
190        if (node.hasProperty(JCR_LASTMODIFIED)) {
191            return new Date(node.getProperty(JCR_LASTMODIFIED).getDate()
192                            .getTimeInMillis());
193        } else {
194            LOGGER.debug(
195                        "Could not get last modified date property for node {}",
196                        node);
197        }
198
199        final Date createdDate = getCreatedDate();
200
201        if (createdDate != null) {
202            LOGGER.trace(
203                        "Using created date for last modified date for node {}",
204                        node);
205            return createdDate;
206        }
207
208        return null;
209    }

220    /**
221     * Get the mixins this object uses
222     * @return a collection of mixin names
223     * @throws javax.jcr.RepositoryException
224     */
225    public Collection<String> getModels() throws RepositoryException {
226        if (isFrozen.apply(node)) {
227            return Lists.newArrayList(
228                Iterators.transform(
229                    property2values.apply(node.getProperty(FROZEN_MIXIN_TYPES)), value2string));
230        } else {
231            return map(node.getMixinNodeTypes(), nodetype2name);
232        }
233    }

235    /**
236     * Update the properties Dataset with a SPARQL Update query. The updated
237     * properties may be serialized to the JCR store.
238     *
239     * After applying the statement, clients SHOULD check the result
240     * of #getDatasetProblems, which may include problems when attempting to
241     * serialize the data to JCR.
242     *
243     * @param subjects
244     * @param sparqlUpdateStatement
245     * @throws RepositoryException
246     */
247    public Dataset updatePropertiesDataset(final GraphSubjects subjects,
248            final String sparqlUpdateStatement) throws RepositoryException {
249        final Dataset dataset = getPropertiesDataset(subjects);
250        final UpdateRequest request =
251            create(sparqlUpdateStatement, dataset.getContext().getAsString(
252                    URI_SYMBOL));
253        dataset.getDefaultModel().setNsPrefixes(request.getPrefixMapping());
254        execute(request, dataset);
255        return dataset;
256    }

258    /**
259     * Return the JCR properties of this object as a Jena {@link Dataset}
260     *
261     * @param graphSubjects
262     * @param offset
263     * @param limit
264     * @return
265     * @throws RepositoryException
266     */
267    public Dataset getPropertiesDataset(final GraphSubjects graphSubjects,
268        final int offset, final int limit)
269        throws RepositoryException {
270
271        final JcrRdfTools jcrRdfTools =
272            JcrRdfTools.withContext(graphSubjects, getNode().getSession());
273
274        final RdfStream propertiesStream =
275            jcrRdfTools.getJcrTriples(getNode());
276
277        propertiesStream.concat(jcrRdfTools.getTreeTriples(getNode()));
278
279        final Dataset dataset = DatasetFactory.create(propertiesStream.limit(limit).skip(offset).asModel());
280
281        final Model problemsModel = createDefaultModel();
282
283        final JcrPropertyStatementListener listener =
284            JcrPropertyStatementListener.getListener(graphSubjects, node
285                    .getSession(), problemsModel);
286
287        dataset.getDefaultModel().register(listener);
288
289        dataset.addNamedModel(PROBLEMS_MODEL_NAME, problemsModel);
290
291        dataset.getContext().set(URI_SYMBOL,
292                graphSubjects.getGraphSubject(getNode()));
293
294
295
296        return dataset;
297    }

373    /**
374     * Replace the properties of this object with the properties from the given
375     * model
376     *
377     * @param graphSubjects
378     * @param inputModel
379     * @return
380     * @throws RepositoryException
381     */
382    public RdfStream replaceProperties(final GraphSubjects graphSubjects,
383        final Model inputModel) throws RepositoryException {
384        final RdfStream originalTriples = getTriples(graphSubjects);
385
386        final RdfStream replacementStream = RdfStream.fromModel(inputModel);
387
388        final Set<Triple> replacementTriples =
389            copyOf(replacementStream.iterator());
390
391        final DifferencingIterator<Triple> differencer =
392            new DifferencingIterator<>(replacementTriples, originalTriples);
393
394        new RdfRemover(graphSubjects, getNode().getSession(), replacementStream
395                .withThisContext(differencer)).consume();
396
397        new RdfAdder(graphSubjects, getNode().getSession(), replacementStream
398                .withThisContext(differencer.notCommon())).consume();
399
400        return replacementStream.withThisContext(Iterables.concat(differencer
401                .common(), differencer.notCommon()));
402    }

404    /**
405     * Construct an ETag value from the last modified date and path. JCR has a
406     * mix:etag type, but it only takes into account binary properties. We
407     * actually want whole-object etag data. TODO : construct and store an ETag
408     * value on object modify
409     *
410     * @return
411     * @throws RepositoryException
412     */
413    public String getEtagValue() throws RepositoryException {
414        final Date lastModifiedDate = getLastModifiedDate();
415
416        if (lastModifiedDate != null) {
417            return shaHex(node.getPath() + lastModifiedDate);
418        } else {
419            return "";
420        }
421    }
