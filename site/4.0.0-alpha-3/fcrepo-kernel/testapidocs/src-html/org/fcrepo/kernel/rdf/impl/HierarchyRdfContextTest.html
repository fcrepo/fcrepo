075    @Test
076    public void testParentTriples() throws RepositoryException, IOException {
077        nodeIsContainer();
078        when(mockNode.hasNodes()).thenReturn(false);
079        // not really a child of the root node, but this is just for test
080        when(mockNode.getDepth()).thenReturn(1);
081        final Model results = getResults();
082        logRdf("Retrieved RDF for testParentTriples(): ", results);
083        assertTrue("Node doesn't have a parent!", results.contains(testSubject,
084                HAS_PARENT, testParentSubject));
085        assertTrue("Node doesn't have a parent!", results.contains(
086                testParentSubject, HAS_CHILD, testSubject));
087    }

089    @Test
090    public void shouldIncludeChildNodeInformation() throws RepositoryException,
091                                                   IOException {
092        nodeIsContainer();
093        when(mockNode.hasNodes()).thenReturn(true);
094        when(mockNode.getDepth()).thenReturn(0);
095        buildChildNodes();
096        when(mockNode.getNodes()).thenReturn(
097                nodeIterator(mockChildNode, mockChildNode2, mockChildNode3,
098                        mockChildNode4, mockChildNode5));
099        // we exhaust the original mock from setUp() so that the preceding
100        // iterator is returned instead
101        mockNode.getNodes();
102
103        final Model actual = getResults();
104        logRdf("Retrieved RDF for shouldIncludeChildNodeInformation() as follows: ",
105                actual);
106        assertEquals("Didn't find enough children!", 5, Iterators.size(actual
107                .listObjectsOfProperty(HAS_CHILD)));
108        assertEquals("Found too many parents!", 1, Iterators.size(actual
109                .listObjectsOfProperty(HAS_PARENT)));
110
111    }

124    @Test
125    public void testNotContainer() throws RepositoryException, IOException {
126        nodeIsNotContainer();
127        when(mockNode.hasNodes()).thenReturn(false);
128        final Model actual = getResults();
129        logRdf("Created RDF for testNotContainer()", actual);
130        assertTrue(actual.contains(testPage, type, PAGE));
131        assertFalse(actual.contains(testPage, MEMBERS_INLINED, actual
132                .createTypedLiteral(true)));
133        assertFalse(actual.contains(testSubject, type, CONTAINER));
134
135        assertFalse(actual.contains(testSubject, MEMBERSHIP_SUBJECT,
136                testSubject));
137        assertFalse(actual.contains(testSubject, MEMBERSHIP_PREDICATE,
138                HAS_CHILD));
139        assertFalse(actual.contains(testSubject, MEMBERSHIP_OBJECT,
140                MEMBER_SUBJECT));
141
142    }

144    @Test
145    public void testForLDPTriples() throws RepositoryException, IOException {
146        nodeIsContainer();
147        when(mockNode.hasNodes()).thenReturn(false);
148        final Model results = getResults();
149
150        logRdf("Created RDF for testForLDPTriples()", results);
151
152        // check for LDP-specified Page information
153        assertTrue("Didn't find page described as LDP Page!", results.contains(
154                testPage, type, PAGE));
155        assertTrue("Didn't find page described as LDP Page of correct node!",
156                results.contains(testPage, PAGE_OF, testSubject));
157        assertTrue("Didn't find page described as having inlined members!",
158                results.contains(testPage, MEMBERS_INLINED,
159                        createPlainLiteral("true")));
160
161        // check for LDP-specified node information
162        assertTrue(
163                "Didn't find node described as being the subject of membership!",
164                results.contains(testSubject, MEMBERSHIP_SUBJECT, testSubject));
165        assertTrue("Didn't find node described as being an LDP Container!",
166                results.contains(testSubject, type, CONTAINER));
167        assertTrue(
168                "Didn't find node described as have an LDP membership object!",
169                results.contains(testSubject, MEMBERSHIP_OBJECT, MEMBER_SUBJECT));
170        assertTrue(
171                "Didn't find node described as using the correct LDP membership predicate!",
172                results.contains(testSubject, MEMBERSHIP_PREDICATE, HAS_CHILD));
173    }

261        when(mockContentNode.getSession()).thenReturn(mockSession);
262        when(mockContentNode.getPath())
263                .thenReturn(MOCK_NODE_PATH + JCR_CONTENT);
264        when(mockContentNode.getPrimaryNodeType()).thenReturn(mockNodeType);
265        when(mockContentNode.getMixinNodeTypes()).thenReturn(new NodeType[] {});
266        when(mockBinary.getKey()).thenReturn(new BinaryKey(testBinaryKey));
267        when(mockBinaryProperty.getBinary()).thenReturn(mockBinary);
268        when(mockContentNode.getProperty(JCR_DATA)).thenReturn(
269                mockBinaryProperty);
270        when(mockBinaryProperty.getName()).thenReturn(JCR_DATA);
271        when(mockBinaryProperty.getParent()).thenReturn(mockContentNode);
272        when(mockCacheEntry.getExternalIdentifier()).thenReturn(
273                testExternalIdentifier);
274        when(
275                mockLowLevelStorageService
276                        .getLowLevelCacheEntries(mockContentNode)).thenReturn(
277                of(mockCacheEntry));
278
279        when(mockContentNode.getProperties()).thenReturn(
280                new TestPropertyIterator(mockBinaryProperty));
281    }
