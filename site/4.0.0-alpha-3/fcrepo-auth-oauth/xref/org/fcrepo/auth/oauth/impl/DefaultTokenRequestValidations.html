41   * Default audits and validations for an OAuth token request
42   */
43  public class DefaultTokenRequestValidations implements TokenRequestValidations {
44  
45      private static final Logger LOGGER =
46          getLogger(DefaultTokenRequestValidations.class);
47  
48      private SessionFactory sessions;
49  
50      /**
51       * Instantiate the default audits and validations for an OAuth
52       * token request, using information stored in a jcr workspace
53       * accessible via the given session factory.
54       * @param sessions
55       */
56      public DefaultTokenRequestValidations(SessionFactory sessions) {
57          this.sessions = sessions;
58      }
59  
60      /**
61       * @param oauthRequest
62       * @return An answer to whether or not this request matches up with an
63       *         authorization code issued at the {@link AuthzEndpoint}
64       * @throws RepositoryException
65       */
66      @Override
67      public boolean isValidAuthCode(OAuthTokenRequest oauthRequest)
68          throws RepositoryException {
69          final String client = oauthRequest.getClientId();
70          LOGGER.debug("Request has authorization client: {}", client);
71          final String code = oauthRequest.getCode();
72          if (code == null) {
73              return false;
74          }
75          final Set<String> scopes = oauthRequest.getScopes();
76          final Session session = sessions.getInternalSession(OAUTH_WORKSPACE);
77          try {
78              final Node authCodeNode =
79                      session.getNode("/authorization-codes/" + code);
80              LOGGER.debug("Found authorization code node stored: {}",
81                      authCodeNode.getPath());
82              // if the client is right
83              if (authCodeNode.getProperty(CLIENT_PROPERTY).getString().equals(
84                      client)) {
85                  if (authCodeNode.getProperty(SCOPES_PROPERTY) != null) {
86  
87                      final Set<String> storedScopes =
88                          newHashSet(
89                              map(authCodeNode.getProperty(SCOPES_PROPERTY)
90                                  .getValues(),
91                                  value2string));
92                      // and if there is at least one scope in common
93                      return (storedScopes.isEmpty() || !intersection(storedScopes, scopes).isEmpty());
94                  } else {
95                      return true;
96                  }
97              }
98          } catch (final PathNotFoundException e) {
99              // this wasn't a code we stored
100             LOGGER.trace("Missing authorization token for code {}", code, e);
101             return false;
102         } finally {
103             session.logout();
104         }
105         throw new AccessControlException(
106                 "Could not establish validity or invalidity of" +
107                  " authorization code! Code:" + code);
108     }
