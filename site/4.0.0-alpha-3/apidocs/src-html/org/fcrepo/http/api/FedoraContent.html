070/**
071 * Content controller for adding, reading, and manipulating
072 * binary streams
073 */
074@Component
075@Scope("prototype")
076@Path("/{path: .*}/fcr:content")
077public class FedoraContent extends AbstractResource {
078
079    public static final int REQUESTED_RANGE_NOT_SATISFIABLE = 416;
080
081    public static final int PARTIAL_CONTENT = 206;
082
083    @InjectedSession
084    protected Session session;
085
086    private static final Logger LOGGER = getLogger(FedoraContent.class);
087
088    /**
089     * Create an anonymous DS with a newly minted name and content from request
090     * body
091     *
092     * @param pathList
093     * @throws RepositoryException
094     */
095    @POST
096    @Timed
097    public Response create(@PathParam("path")
098            final List<PathSegment> pathList,
099            @HeaderParam("Slug") final String slug,
100            @HeaderParam("Content-Disposition") final String contentDisposition,
101            @QueryParam("checksum") final String checksum,
102            @HeaderParam("Content-Type") final MediaType requestContentType,
103                    final InputStream requestBodyStream)
104        throws IOException, InvalidChecksumException,
105                   RepositoryException, URISyntaxException, ParseException {
106        final MediaType contentType =
107                requestContentType != null ? requestContentType
108                        : APPLICATION_OCTET_STREAM_TYPE;
109
110
111        final String newDatastreamPath;
112        final String path = toPath(pathList);
113
114        if (nodeService.exists(session, path)) {
115            final String pid;
116
117            if (slug != null) {
118                pid = slug;
119            }  else {
120                pid = pidMinter.mintPid();
121            }
122
123            newDatastreamPath = path + "/" + pid;
124        } else {
125            newDatastreamPath = path;
126        }
127
128
129        LOGGER.debug("Attempting to ingest fcr:content with path: {}", newDatastreamPath);
130
131        try {
132
133            if (nodeService.exists(session, newDatastreamPath)) {
134                return status(SC_CONFLICT)
135                           .entity(path + " is an existing resource!").build();
136            }
137
138            final URI checksumURI;
139
140            if (checksum != null && !checksum.equals("")) {
141                checksumURI = new URI(checksum);
142            } else {
143                checksumURI = null;
144            }
145
146            final String originalFileName;
147
148            if (contentDisposition != null) {
149                final ContentDisposition disposition = new ContentDisposition(contentDisposition);
150                originalFileName = disposition.getFileName();
151            } else {
152                originalFileName = null;
153            }
154
155
156            final Node datastreamNode =
157                    datastreamService.createDatastreamNode(session, newDatastreamPath,
158                            contentType.toString(), originalFileName, requestBodyStream,
159                            checksumURI);
160
161            final HttpGraphSubjects subjects =
162                    new HttpGraphSubjects(session, FedoraNodes.class,
163                            uriInfo);
164
165            session.save();
166            versionService.nodeUpdated(datastreamNode);
167            return created(
168                    new URI(subjects.getGraphSubject(
169                            datastreamNode.getNode(JCR_CONTENT)).getURI()))
170                    .build();
171
172        } finally {
173            session.logout();
174        }
175    }
176
177    /**
178     * Modify an existing datastream's content
179     *
180     * @param pathList
181     * @param requestContentType Content-Type header
182     * @param requestBodyStream Binary blob
183     * @return 201 Created
184     * @throws RepositoryException
185     * @throws IOException
186     * @throws InvalidChecksumException
187     */
188    @PUT
189    @Timed
190    public Response modifyContent(@PathParam("path") final List<PathSegment> pathList,
191                                  @QueryParam("checksum") final String checksum,
192                                  @HeaderParam("Content-Disposition") final String contentDisposition,
193                                  @HeaderParam("Content-Type") final MediaType requestContentType,
194                                  final InputStream requestBodyStream,
195                                  @Context final Request request)
196        throws RepositoryException, IOException, InvalidChecksumException, URISyntaxException, ParseException {
197
198        try {
199            final String path = toPath(pathList);
200            final MediaType contentType =
201                    requestContentType != null ? requestContentType
202                            : APPLICATION_OCTET_STREAM_TYPE;
203
204            if (nodeService.exists(session, path)) {
205
206                final Datastream ds =
207                        datastreamService.getDatastream(session, path);
208
209                final EntityTag etag =
210                        new EntityTag(ds.getContentDigest().toString());
211                final Date date = ds.getLastModifiedDate();
212                final Date roundedDate = new Date();
213                roundedDate
214                        .setTime(date.getTime() - date.getTime() % 1000);
215                final ResponseBuilder builder =
216                        request.evaluatePreconditions(roundedDate, etag);
217
218                if (builder != null) {
219                    throw new WebApplicationException(builder.build());
220                }
221            }
222
223            LOGGER.debug("create Datastream {}", path);
224
225            final URI checksumURI;
226
227            if (checksum != null && !checksum.equals("")) {
228                checksumURI = new URI(checksum);
229            } else {
230                checksumURI = null;
231            }
232
233            final String originalFileName;
234
235            if (contentDisposition != null) {
236                final ContentDisposition disposition = new ContentDisposition(contentDisposition);
237                originalFileName = disposition.getFileName();
238            } else {
239                originalFileName = null;
240            }
241
242            final Node datastreamNode =
243                datastreamService.createDatastreamNode(session, path,
244                    contentType.toString(), originalFileName, requestBodyStream, checksumURI);
245
246            final boolean isNew = datastreamNode.isNew();
247            session.save();
248            versionService.nodeUpdated(datastreamNode);
249
250            if (isNew) {
251                final HttpGraphSubjects subjects =
252                        new HttpGraphSubjects(session, FedoraNodes.class,
253                                uriInfo);
254
255                return created(
256                        new URI(
257                                subjects.getGraphSubject(
258                                        datastreamNode
259                                                .getNode(JCR_CONTENT))
260                                        .getURI())).build();
261            } else {
262                return noContent().build();
263            }
264        } finally {
265            session.logout();
266        }
267
268    }

270    /**
271     * Get the binary content of a datastream
272     *
273     * @param pathList
274     * @return Binary blob
275     * @throws RepositoryException
276     */
277    @GET
278    @Timed
279    public Response getContent(@PathParam("path")
280        final List<PathSegment> pathList, @HeaderParam("Range")
281        final String rangeValue, @Context
282        final Request request) throws RepositoryException, IOException {
283
284        try {
285            final String path = toPath(pathList);
286            final Datastream ds =
287                    datastreamService.getDatastream(session, path);
288
289            final EntityTag etag =
290                    new EntityTag(ds.getContentDigest().toString());
291            final Date date = ds.getLastModifiedDate();
292            final Date roundedDate = new Date();
293            roundedDate.setTime(date.getTime() - date.getTime() % 1000);
294            ResponseBuilder builder =
295                    request.evaluatePreconditions(roundedDate, etag);
296
297            final CacheControl cc = new CacheControl();
298            cc.setMaxAge(0);
299            cc.setMustRevalidate(true);
300
301            if (builder == null) {
302
303                final InputStream content = ds.getContent();
304
305                if (rangeValue != null && rangeValue.startsWith("bytes")) {
306
307                    final Range range = Range.convert(rangeValue);
308
309                    final long contentSize = ds.getContentSize();
310
311                    final String endAsString;
312
313                    if (range.end() == -1) {
314                        endAsString = Long.toString(contentSize - 1);
315                    } else {
316                        endAsString = Long.toString(range.end());
317                    }
318
319                    final String contentRangeValue =
320                            String.format("bytes %s-%s/%s", range.start(),
321                                    endAsString, contentSize);
322
323                    if (range.end() > contentSize ||
324                            (range.end() == -1 && range.start() > contentSize)) {
325                        builder =
326                                status(
327                                        REQUESTED_RANGE_NOT_SATISFIABLE)
328                                        .header("Content-Range",
329                                                contentRangeValue);
330                    } else {
331                        final RangeRequestInputStream rangeInputStream =
332                                new RangeRequestInputStream(content, range
333                                        .start(), range.size());
334
335                        builder =
336                                status(PARTIAL_CONTENT).entity(
337                                        rangeInputStream)
338                                        .header("Content-Range",
339                                                contentRangeValue);
340                    }
341
342                } else {
343                    builder = ok(content);
344                }
345            }
346
347            final HttpGraphSubjects subjects =
348                    new HttpGraphSubjects(session, FedoraNodes.class,
349                            uriInfo);
350
351            final ContentDisposition contentDisposition = ContentDisposition.type("attachment")
352                                                              .fileName(ds.getFilename())
353                                                              .creationDate(ds.getCreatedDate())
354                                                              .modificationDate(ds.getLastModifiedDate())
355                                                              .size(ds.getContentSize())
356                                                              .build();
357
358            return builder.type(ds.getMimeType()).header(
359                    "Link",
360                    subjects.getGraphSubject(ds.getNode()) +
361                            ";rel=\"meta\"").header("Accept-Ranges",
362                    "bytes").cacheControl(cc).lastModified(date).tag(etag)
363                    .header("Content-Disposition", contentDisposition)
364                    .build();
365        } finally {
366            session.logout();
367        }
368    }
