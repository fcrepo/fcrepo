017package org.fcrepo.http.commons.responses;
018
019import static java.util.Collections.singletonList;
020import static org.apache.jena.riot.WebContent.contentTypeToLang;
021import static org.fcrepo.http.commons.domain.RDFMediaType.N3;
022import static org.fcrepo.http.commons.domain.RDFMediaType.N3_ALT1;
023import static org.fcrepo.http.commons.domain.RDFMediaType.N3_ALT2;
024import static org.fcrepo.http.commons.domain.RDFMediaType.NTRIPLES;
025import static org.fcrepo.http.commons.domain.RDFMediaType.RDF_JSON;
026import static org.fcrepo.http.commons.domain.RDFMediaType.RDF_XML;
027import static org.fcrepo.http.commons.domain.RDFMediaType.TURTLE;
028import static org.fcrepo.http.commons.responses.RdfSerializationUtils.setCachingHeaders;
029import static org.slf4j.LoggerFactory.getLogger;
030
031import java.io.IOException;
032import java.io.OutputStream;
033import java.lang.annotation.Annotation;
034import java.lang.reflect.Type;
035
036import javax.ws.rs.Produces;
037import javax.ws.rs.core.MediaType;
038import javax.ws.rs.core.MultivaluedMap;
039import javax.ws.rs.ext.MessageBodyWriter;
040import javax.ws.rs.ext.Provider;
041
042import org.slf4j.Logger;
043import org.springframework.stereotype.Component;
044
045import com.hp.hpl.jena.query.Dataset;
046
047/**
048 * Helper for writing Datasets out in a variety of serialization formats
049 * with cache headers.
050 */
051@Provider
052@Component
053@Produces({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES})
054public class RdfProvider implements MessageBodyWriter<Dataset> {
055
056    private static final Logger LOGGER = getLogger(RdfProvider.class);
057
058    @Override
059    public void writeTo(final Dataset rdf, final Class<?> type,
060            final Type genericType, final Annotation[] annotations,
061            final MediaType mediaType,
062            final MultivaluedMap<String, Object> httpHeaders,
063            final OutputStream entityStream) throws IOException {
064
065        LOGGER.debug("Writing a response for: {} with MIMEtype: {}", rdf,
066                        mediaType);
067
068        // add standard headers
069        httpHeaders.put("Content-type", singletonList((Object) mediaType.toString()));
070
071        setCachingHeaders(httpHeaders, rdf);
072
073        new GraphStoreStreamingOutput(rdf, mediaType).write(entityStream);
074    }
075
076    @Override
077    public boolean isWriteable(final Class<?> type, final Type genericType,
078            final Annotation[] annotations, final MediaType mediaType) {
079
080        // we can return a result for any MIME type that Jena can serialize
081        final Boolean appropriateMimeType =
082                contentTypeToLang(mediaType.toString()) != null;
083        return appropriateMimeType && Dataset.class.isAssignableFrom(type);
084    }
085
086    @Override
087    public long getSize(final Dataset rdf, final Class<?> type,
088            final Type genericType, final Annotation[] annotations,
089            final MediaType mediaType) {
090        // we don't know in advance how large the result might be
091        return -1;
092    }
093
094}
