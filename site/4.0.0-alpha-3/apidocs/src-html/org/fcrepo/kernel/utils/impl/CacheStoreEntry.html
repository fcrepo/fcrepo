016package org.fcrepo.kernel.utils.impl;
017
018import static java.util.Objects.hash;
019import static org.slf4j.LoggerFactory.getLogger;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.OutputStream;
024import java.util.Properties;
025
026import org.apache.poi.util.IOUtils;
027import org.fcrepo.kernel.utils.LowLevelCacheEntry;
028import org.fcrepo.kernel.utils.infinispan.StoreChunkInputStream;
029import org.fcrepo.kernel.utils.infinispan.StoreChunkOutputStream;
030import org.infinispan.loaders.AbstractCacheStoreConfig;
031import org.infinispan.loaders.CacheStore;
032import org.infinispan.loaders.CacheStoreConfig;
033import org.infinispan.loaders.file.FileCacheStoreConfig;
034import org.modeshape.jcr.value.BinaryKey;
035import org.modeshape.jcr.value.binary.BinaryStoreException;
036import org.slf4j.Logger;
037
038/**
039 * An entry in a specific BinaryStore
040 */
041public class CacheStoreEntry extends LowLevelCacheEntry {
042
043    private static final Logger LOGGER = getLogger(CacheStoreEntry.class);
044
045    private final CacheStore store;
046    private final String cacheName;
047
048    /**
049     * An entry in an infinispan CacheStore
050     */
051    public CacheStoreEntry(final CacheStore store,
052            final String cacheName,
053            final BinaryKey key) {
054        super(key);
055        this.store = store;
056        this.cacheName = cacheName;
057    }
058
059    /**
060     * Get the low-level cache store this entry is contained in
061     * @return
062     */
063    public CacheStore getLowLevelStore() {
064        return store;
065    }
066
067    /**
068     * Get a raw input stream from the underlying store
069     * @return the content for this entry
070     * @throws BinaryStoreException
071     */
072    @Override
073    public InputStream getInputStream() throws BinaryStoreException {
074        return new StoreChunkInputStream(store, key +
075                                             DATA_SUFFIX);
076    }
077
078    /**
079     * Send a raw input stream to the underlying store for this entry; used for
080     * fixing e.g. fixity failures.
081     *
082     * @param stream binary content to REPLACE the content in the store
083     * @throws BinaryStoreException
084     * @throws IOException
085     */
086    @Override
087    public void storeValue(final InputStream stream)
088        throws BinaryStoreException, IOException {
089        // TODO this is probably an auditable action.
090        LOGGER.info("Doing a low-level write to store {} for key {}",
091                    getExternalIdentifier(), key);
092
093        final OutputStream outputStream =
094                new StoreChunkOutputStream(store, key +
095                                           DATA_SUFFIX);
096        IOUtils.copy(stream, outputStream);
097        outputStream.close();
098    }
099
100    /**
101     * Generate a human-readable identifier for the location of this entry
102     *
103     * @return
104     */
105    @Override
106    public String getExternalIdentifier() {
107
108        final CacheStoreConfig config = store.getCacheStoreConfig();
109
110        String ispnExternalId = null;
111
112        if (config instanceof AbstractCacheStoreConfig) {
113            ispnExternalId = getStoreIdentifier((AbstractCacheStoreConfig)config);
114            if (ispnExternalId != null) {
115                return ispnExternalId;
116            }
117        }
118
119        if (config instanceof FileCacheStoreConfig) {
120            ispnExternalId = getStoreIdentifier((FileCacheStoreConfig) config);
121        }
122
123        // ChainingCacheStore's have a null config
124        if (config == null) {
125            return getExternalId() + "/" + store.getClass().getName() + ":" +
126                    cacheName + ":" +
127                    AbstractCacheStoreConfig.class.getPackage() +
128                    "ChainingCacheStoreConfig";
129        }
130
131        if (ispnExternalId == null) {
132            ispnExternalId = config.toString();
133        }
134
135        return getExternalId() + "/" + store.getClass().getName() + ":" +
136            cacheName + ":" +
137            config.getCacheLoaderClassName() +
138            ":" + ispnExternalId;
139    }
140
141    private static String getStoreIdentifier(final AbstractCacheStoreConfig config) {
142        final Properties properties =
143                config.getProperties();
144        if (properties.containsKey("id")) {
145            return properties.getProperty("id");
146        }
147        return null;
148    }
149
150    private static String getStoreIdentifier(final FileCacheStoreConfig config) {
151        return config.getLocation();
152    }
153
154
155    /**
156     * Two LowLevelCacheEntries are the same if they have the same key,
157     * come from the same CacheStore, with the same cache name,
158     * and have the same underlying store configuration
159     * @param other
160     * @return
161     */
162    @Override
163    public boolean equals(final Object other) {
164        if (other instanceof CacheStoreEntry) {
165            final CacheStoreEntry that = (CacheStoreEntry) other;
166
167            return key.equals(that.getKey()) &&
168                           cacheName.equals(that.cacheName) &&
169                           ((store == null && that.store == null) ||
170                                    (store != null && store.equals(that.store)));
171        } else {
172            return false;
173        }
174    }
175
176    @Override
177    public int hashCode() {
178        return hash(store, key);
179    }
180
181}
