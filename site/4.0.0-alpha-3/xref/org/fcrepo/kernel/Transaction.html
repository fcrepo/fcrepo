17  package org.fcrepo.kernel;
18  
19  import org.fcrepo.kernel.services.VersionService;
20  
21  import static java.lang.System.currentTimeMillis;
22  import static java.util.UUID.randomUUID;
23  import static org.fcrepo.kernel.Transaction.State.COMMITED;
24  import static org.fcrepo.kernel.Transaction.State.DIRTY;
25  import static org.fcrepo.kernel.Transaction.State.ROLLED_BACK;
26  
27  import java.util.Calendar;
28  import java.util.Collections;
29  import java.util.Date;
30  import java.util.Set;
31  import java.util.concurrent.ConcurrentHashMap;
32  import javax.jcr.RepositoryException;
33  import javax.jcr.Session;
34  
35  /**
36   * A Fedora Transaction wraps a JCR session with some expiration logic.
37   * Whenever the transaction's session is requested, the expiration is extended
38   */
39  public class Transaction {
40  
41      // the default timeout is 3 minutes
42      public static final long DEFAULT_TIMEOUT = 3L * 60L * 1000L;
43  
44      public static final String TIMEOUT_SYSTEM_PROPERTY = "fcrepo4.tx.timeout";
45  
46      /**
47       * Information about the state of the transaction
48       */
49      public static enum State {
50          DIRTY, NEW, COMMITED, ROLLED_BACK
51      }
52  
53      private final Session session;
54  
55      private final String id;
56  
57      private final Date created;
58  
59      private Calendar expires;
60  
61      private State state = State.NEW;
62  
63      private Set<String> versionedPaths;
64  
65      /**
66       * Create a transaction for the given Session
67       * @param session
68       */
69      public Transaction(final Session session) {
70          super();
71          this.session = session;
72          this.created = new Date();
73          this.id = randomUUID().toString();
74          this.expires = Calendar.getInstance();
75          this.updateExpiryDate();
76      }
77  
78      /**
79       * Get the transaction-aware session
80       * @return
81       */
82      public Session getSession() {
83          updateExpiryDate();
84          return TxAwareSession.newInstance(session, id);
85      }
86  
87      /**
88       * Get the date this transaction was created
89       * @return
90       */
91      public Date getCreated() {
92          return new Date(created.getTime());
93      }
94  
95      /**
96       * Get the transaction identifier
97       * @return
98       */
99      public String getId() {
100         return id;
101     }
102 
103     /**
104      * Get the state of this transaction
105      * @return
106      * @throws RepositoryException
107      */
108     public State getState() throws RepositoryException {
109         if (this.session != null && this.session.hasPendingChanges()) {
110             return DIRTY;
111         }
112         return state;
113     }
114 
115     /**
116      * Get the Date when this transaction is expired and can be
117      * garbage-collected
118      * @return
119      */
120     public Date getExpires() {
121         return expires.getTime();
122     }
123 
124     /**
125      * Adds a path at which a new version should be made upon successful
126      * completion of this transaction.  Subsequent calls with the same path
127      * have no effect, as the entire transaction is meant to be atomic and only
128      * one new version can result from it.
129      * @param absPath the object path to the resource to have a version
130      *                checkpoint made
131      */
132     public void addPathToVersion(String absPath) {
133         if (versionedPaths == null) {
134             versionedPaths = Collections.newSetFromMap(
135                     new ConcurrentHashMap<String, Boolean>());
136         }
137         versionedPaths.add(absPath);
138     }
139 
140     /**
141      * "Commit" the transaction by saving the backing-session
142      * @param vService a versionService
143      * @throws RepositoryException
144      */
145     public void commit(VersionService vService) throws RepositoryException {
146 
147         this.session.save();
148         if (this.versionedPaths != null) {
149             if (vService == null) {
150                 throw new IllegalStateException("Versioned Paths were added," +
151                         " but no VersionService was provided!");
152             }
153             vService.createVersion(session.getWorkspace(), versionedPaths);
154         }
155         this.state = COMMITED;
156         this.expire();
157     }
158 
159     /**
160      * End the session, and mark for reaping
161      * @throws RepositoryException
162      */
163     public void expire() throws RepositoryException {
164         this.session.logout();
165         this.expires.setTimeInMillis(currentTimeMillis());
166     }
167 
168     /**
169      * Discard all unpersisted changes and expire
170      * @throws RepositoryException
171      */
172     public void rollback() throws RepositoryException {
173         this.state = ROLLED_BACK;
174         this.session.refresh(false);
175         this.expire();
176     }
177 
178     /**
179      * Roll forward the expiration date for recent activity
180      */
181     public void updateExpiryDate() {
182         long duration;
183         if (System.getProperty(TIMEOUT_SYSTEM_PROPERTY) != null) {
184             duration =
185                     Long.parseLong(System.getProperty(TIMEOUT_SYSTEM_PROPERTY));
186         } else {
187             duration = DEFAULT_TIMEOUT;
188         }
189         this.expires.setTimeInMillis(currentTimeMillis() + duration);
190     }
191 }
