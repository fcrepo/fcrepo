48  /**
49   * Assemble {@link com.hp.hpl.jena.graph.Triple}s derived from the {@link NodeType}s in a repository.
50   *
51   * @author cbeer
52   */
53  public class NodeTypeRdfContext extends RdfStream {
54  
55      private static final Logger LOGGER = getLogger(NodeTypeRdfContext.class);
56  
57      private static final Predicate<ItemDefinition> isWildcardResidualDefinition = x -> x.getName().equals("*");
58  
59      /**
60       * Convert the NodeTypeManager to an RDF stream, including both primary and
61       * mixin node types.
62       *
63       * @param nodeTypeManager the node type manager
64       * @throws RepositoryException if repository exception occurred
65       */
66      @SuppressWarnings("unchecked")
67      public NodeTypeRdfContext(final NodeTypeManager nodeTypeManager) throws RepositoryException {
68          super();
69  
70          concat(new NodeTypeRdfContext(nodeTypeManager.getPrimaryNodeTypes()));
71          concat(new NodeTypeRdfContext(nodeTypeManager.getMixinNodeTypes()));
72      }
73  
74      /**
75       * Convert a NodeType iterator into an RDF stream
76       *
77       * @param nodeTypeIterator the node type iterator
78       * @throws RepositoryException if repository exception occurred
79       */
80      public NodeTypeRdfContext(final Iterator<NodeType> nodeTypeIterator)
81          throws RepositoryException {
82          super();
83  
84          while (nodeTypeIterator.hasNext()) {
85              concat(new NodeTypeRdfContext(nodeTypeIterator.next()));
86          }
87      }
88  
89      /**
90       * Convert a NodeType into an RDF stream by capturing the supertypes, node
91       * definitions, and property definitions of the type as RDFS triples.
92       *
93       * @param nodeType the node type
94       * @throws RepositoryException if repository exception occurred
95       */
96      public NodeTypeRdfContext(final NodeType nodeType)
97          throws RepositoryException {
98          super();
99  
100         final Node nodeTypeResource = getResource(nodeType).asNode();
101         final String nodeTypeName = nodeType.getName();
102 
103         LOGGER.trace("Adding triples for nodeType: {} with URI: {}",
104                 nodeTypeName, nodeTypeResource.getURI());
105 
106         concat(Collections2.transform(copyOf(nodeType.getDeclaredSupertypes()),
107                     uncheck((final NodeType x) -> {
108                         final Node supertypeNode = getResource(x).asNode();
109                         LOGGER.trace(
110                                 "Adding triple for nodeType: {} with subclass: {}",
111                                 nodeTypeName, supertypeNode.getURI());
112                         return create(nodeTypeResource,
113                                 subClassOf.asNode(), supertypeNode);
114                     })::apply));
115 
116         concat(Iterators.concat(
117             Iterators.transform(Iterators.filter(
118                 forArray(nodeType.getDeclaredChildNodeDefinitions()),
119                 isWildcardResidualDefinition.negate()::test),
120                 (new NodeDefinitionToTriples(nodeTypeResource))::apply)));
121 
122         concat(Iterators.concat(
123             Iterators.transform(Iterators.filter(
124                 forArray(nodeType.getDeclaredPropertyDefinitions()),
125                 isWildcardResidualDefinition.negate()::test),
126                 (new PropertyDefinitionToTriples(nodeTypeResource))::apply)));
127 
128         concat(create(nodeTypeResource, type.asNode(), Class.asNode()), create(
129                 nodeTypeResource, label.asNode(), createLiteral(nodeTypeName)));
130     }
131

133 }
