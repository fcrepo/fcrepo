16  package org.fcrepo.kernel.api.exception;
17  
18  import org.fcrepo.kernel.api.models.FedoraResource;
19  
20  /**
21   * Exception when a Tombstone {@link org.fcrepo.kernel.api.models.FedoraResource}
22   * is used where a real object is expected
23   *
24   * @author cabeer
25   * @since 10/16/14
26   */
27  public class TombstoneException extends RepositoryRuntimeException {
28  
29      private static final long serialVersionUID = 1L;
30  
31      private final FedoraResource fedoraResource;
32      private final String uri;
33  
34      /**
35       * Construct a new tombstone exception for a resource
36       * @param fedoraResource the fedora resource
37       */
38      public TombstoneException(final FedoraResource fedoraResource) {
39          this(fedoraResource, null);
40      }
41  
42      /**
43       * Create a new tombstone exception with a URI to the tombstone resource
44       * @param fedoraResource the fedora resource
45       * @param uri the uri to the tombstone resource
46       */
47      public TombstoneException(final FedoraResource fedoraResource, final String uri) {
48          super("Discovered tombstone resource at " + fedoraResource);
49          this.fedoraResource = fedoraResource;
50          this.uri = uri;
51      }
52  
53      /**
54       * Get the tombstone resource
55       * @return the tombstone resource
56       */
57      public FedoraResource getResource() {
58          return fedoraResource;
59      }
60  
61      /**
62       * Get a URI to the tombstone resource
63       * @return the URI to the tombstone resource
64       */
65      public String getURI() {
66          return uri;
67      }
68  }
