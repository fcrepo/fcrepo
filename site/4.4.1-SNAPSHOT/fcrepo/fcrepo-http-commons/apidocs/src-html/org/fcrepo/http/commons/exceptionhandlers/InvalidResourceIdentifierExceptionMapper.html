016package org.fcrepo.http.commons.exceptionhandlers;
017
018import org.fcrepo.kernel.api.exception.InvalidResourceIdentifierException;
019import org.slf4j.Logger;
020
021import javax.ws.rs.core.Response;
022import javax.ws.rs.ext.ExceptionMapper;
023import javax.ws.rs.ext.Provider;
024
025import static javax.ws.rs.core.Response.Status.BAD_REQUEST;
026import static javax.ws.rs.core.Response.status;
027import static org.slf4j.LoggerFactory.getLogger;
028
029/**
030 * The class translates {@link org.fcrepo.kernel.api.exception.InvalidResourceIdentifierException}s to its proper
031 * response code.
032 *
033 * @author awoods
034 * @since July 14, 2015.
035 */
036@Provider
037public class InvalidResourceIdentifierExceptionMapper implements ExceptionMapper<InvalidResourceIdentifierException> {
038
039    private static final Logger LOGGER = getLogger(InvalidResourceIdentifierExceptionMapper.class);
040
041    @Override
042    public Response toResponse(final InvalidResourceIdentifierException e) {
043        LOGGER.debug("InvalidResourceIdentifierExceptionMapper caught exception: {}", e.getMessage());
044        return status(BAD_REQUEST).entity(e.getMessage()).build();
045    }
046}
