017package org.fcrepo.kernel.services;
018
019import static org.junit.Assert.assertEquals;
020import static org.mockito.Mockito.mock;
021import static org.mockito.Mockito.verify;
022import static org.mockito.Mockito.when;
023import static org.mockito.MockitoAnnotations.initMocks;
024import static org.powermock.api.mockito.PowerMockito.mockStatic;
025
026import java.io.InputStream;
027import java.util.Set;
028
029import javax.jcr.Node;
030import javax.jcr.NodeIterator;
031import javax.jcr.RepositoryException;
032import javax.jcr.Session;
033import javax.jcr.Workspace;
034import javax.jcr.nodetype.NodeTypeIterator;
035
036import org.junit.Before;
037import org.junit.Test;
038import org.junit.runner.RunWith;
039import org.mockito.Mock;
040import org.modeshape.jcr.api.nodetype.NodeTypeManager;
041import org.powermock.core.classloader.annotations.PowerMockIgnore;
042import org.powermock.core.classloader.annotations.PrepareForTest;
043import org.powermock.modules.junit4.PowerMockRunner;
044
045@RunWith(PowerMockRunner.class)
046@PowerMockIgnore({"org.slf4j.*", "javax.xml.parsers.*", "org.apache.xerces.*"})
047@PrepareForTest({ServiceHelpers.class})
048public class NodeServiceImplTest {
049
050    @Mock
051    private NodeTypeIterator mockNTI;
052
053    @Mock
054    private NodeTypeManager mockNodeTypeManager;
055
056    @Mock
057    private Session mockSession;
058
059    @Mock
060    private Node mockRoot;
061
062    @Mock
063    private Node mockObjNode;
064
065    @Mock
066    private Workspace mockWorkspace;
067
068    private NodeService testObj;
069
070    @Before
071    public void setUp() throws RepositoryException {
072        initMocks(this);
073        testObj = new NodeServiceImpl();
074        when(mockSession.getRootNode()).thenReturn(mockRoot);
075
076        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
077        when(mockWorkspace.getNodeTypeManager()).thenReturn(mockNodeTypeManager);
078        when(mockNodeTypeManager.getAllNodeTypes()).thenReturn(mockNTI);
079    }
080
081    @SuppressWarnings("unchecked")
082    @Test
083    public void testGetObjectNames() throws RepositoryException {
084        final String objPath = "";
085        when(mockObjNode.getName()).thenReturn("foo");
086        when(mockObjNode.isNodeType("nt:folder")).thenReturn(true);
087        final NodeIterator mockIter = mock(NodeIterator.class);
088        when(mockIter.hasNext()).thenReturn(true, false);
089        when(mockIter.nextNode()).thenReturn(mockObjNode).thenThrow(
090                IndexOutOfBoundsException.class);
091        when(mockRoot.getNodes()).thenReturn(mockIter);
092        when(mockSession.getNode(objPath)).thenReturn(mockRoot);
093        final Set<String> actual = testObj.getObjectNames(mockSession, "");
094        verify(mockSession).getNode(objPath);
095        assertEquals(1, actual.size());
096        assertEquals("foo", actual.iterator().next());
097    }
098
099    @Test
100    public void testDeleteOBject() throws RepositoryException {
101        final String objPath = "foo";
102        final Node mockObjectsNode = mock(Node.class);
103        when(mockSession.getRootNode()).thenReturn(mockRoot);
104        when(mockRoot.getNode("objects")).thenReturn(mockObjectsNode);
105        when(mockSession.getNode(objPath)).thenReturn(mockObjNode);
106        mockStatic(ServiceHelpers.class);
107        testObj.deleteObject(mockSession, "foo");
108        verify(mockSession).getNode(objPath);
109        verify(mockObjNode).remove();
110    }
111
112    @Test
113    public void testCopyObject() throws RepositoryException {
114        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
115        testObj.copyObject(mockSession, "foo", "bar");
116        verify(mockWorkspace).copy("foo", "bar");
117    }
118
119    @Test
120    public void testMoveObject() throws RepositoryException {
121        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
122        testObj.moveObject(mockSession, "foo", "bar");
123        verify(mockWorkspace).move("foo", "bar");
124    }
125
126    @Test
127    public void testExists() throws RepositoryException {
128        final String existsPath = "/foo/bar/exists";
129        when(mockSession.nodeExists(existsPath)).thenReturn(true);
130        assertEquals(true, testObj.exists(mockSession, existsPath));
131        assertEquals(false, testObj.exists(mockSession, "/foo/bar"));
132    }
133
134    @Test
135    public void testGetAllNodeTypes() throws RepositoryException {
136        final NodeTypeIterator actual = testObj.getAllNodeTypes(mockSession);
137        assertEquals(mockNTI, actual);
138    }
139
140    @Test
141    public void testGetNodeTypes() throws Exception {
142        when(mockNodeTypeManager.getPrimaryNodeTypes()).thenReturn(mock(NodeTypeIterator.class));
143        when(mockNodeTypeManager.getMixinNodeTypes()).thenReturn(mock(NodeTypeIterator.class));
144        testObj.getNodeTypes(mockSession);
145
146        verify(mockNodeTypeManager).getPrimaryNodeTypes();
147        verify(mockNodeTypeManager).getMixinNodeTypes();
148    }
149
150    @Test
151    public void testRegisterNodeTypes() throws Exception {
152        final InputStream mockInputStream = mock(InputStream.class);
153        testObj.registerNodeTypes(mockSession, mockInputStream);
154
155        verify(mockNodeTypeManager).registerNodeTypes(mockInputStream, true);
156    }
157}
