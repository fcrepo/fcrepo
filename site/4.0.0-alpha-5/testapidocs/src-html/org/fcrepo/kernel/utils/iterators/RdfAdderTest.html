061
062public class RdfAdderTest {
063
064    private RdfAdder testAdder;
065
066    private static final Model m = createDefaultModel();
067
068    private static final String propertyNamespacePrefix = "ex";
069
070    private static final String propertyNamespaceUri =
071        "http://www.example.com#";
072
073    private static final String propertyBaseName = "example-property";
074
075    private static final String propertyLongName = propertyNamespaceUri
076            + propertyBaseName;
077
078    private static final String propertyShortName = propertyNamespacePrefix
079            + ":" + propertyBaseName;
080
081    private static final Map<String, String> mockNamespaceMap = ImmutableMap
082            .of(propertyNamespacePrefix, propertyNamespaceUri, "rdf", type.getNameSpace());
083
084    private static final String description = "Description.";
085
086    private static final Triple descriptiveTriple = create(createAnon(),
087            createURI(propertyLongName), createLiteral(description));
088
089    private static final Statement descriptiveStmnt = m
090            .asStatement(descriptiveTriple);
091
092    private static final Resource mockNodeSubject = descriptiveStmnt
093            .getSubject();
094
095    private static final String mixinLongName = type.getNameSpace() + "someType";
096
097    private static final String mixinShortName = "rdf" + ":" + "someType";
098
099    private static final Resource mixinObject = createResource(mixinLongName);
100
101    private static final Triple mixinTriple = create(mockNodeSubject.asNode(),
102            type.asNode(), mixinObject.asNode());
103
104
105    private static final Statement mixinStmnt = m.asStatement(mixinTriple);
106
107
108    @Test
109    public void testAddingProperty() throws Exception {
110        testAdder = new RdfAdder(mockGraphSubjects, mockSession, testStream);
111        when(mockNode.setProperty(propertyShortName, mockValue, UNDEFINED)).thenReturn(mockProperty);
112        testAdder.operateOnProperty(descriptiveStmnt, mockNode);
113        verify(mockNode).setProperty(propertyShortName, mockValue, UNDEFINED);
114
115    }
116
117    @Test
118    public void testAddingModelWithStreamNamespace() throws Exception {
119        testAdder = new RdfAdder(mockGraphSubjects, mockSession, testStream);
120        testAdder.operateOnMixin(mixinStmnt.getObject().asResource(), mockNode);
121        verify(mockNode).addMixin(anyString());
122    }
123
124    @Test(expected = MalformedRdfException.class)
125    public void testAddingWithBadNamespace() throws Exception {
126        // we drop our stream namespace map
127        testStream = new RdfStream(mockTriples);
128        when(
129                mockSession
130                        .getNamespacePrefix(getJcrNamespaceForRDFNamespace(type
131                                .getNameSpace()))).thenThrow(new NamespaceException("Expected."));
132        testAdder = new RdfAdder(mockGraphSubjects, mockSession, testStream);
133        testAdder.operateOnMixin(mixinStmnt.getObject().asResource(), mockNode);
134    }
135
136    @Test
137    public void testAddingWithRepoNamespace() throws Exception {
138        // we drop our stream namespace map
139        testStream = new RdfStream(mockTriples);
140        when(
141                mockSession
142                        .getNamespacePrefix(getJcrNamespaceForRDFNamespace(type
143                                .getNameSpace()))).thenReturn("rdf");
144        testAdder = new RdfAdder(mockGraphSubjects, mockSession, testStream);
145        testAdder.operateOnMixin(mixinStmnt.getObject().asResource(), mockNode);
146    }
147
148    @Test(expected = MalformedRdfException.class)
149    public void testAddingWithBadMixinOnNode() throws Exception {
150        when(mockNode.canAddMixin(mixinShortName)).thenReturn(false);
151        testAdder = new RdfAdder(mockGraphSubjects, mockSession, testStream);
152        testAdder.operateOnMixin(mixinStmnt.getObject().asResource(), mockNode);
153    }
154
155    @Test(expected = MalformedRdfException.class)
156    public void testAddingWithBadMixinForRepo() throws Exception {
157        when(mockNodeTypeManager.hasNodeType(mixinShortName)).thenReturn(false);
158        testAdder = new RdfAdder(mockGraphSubjects, mockSession, testStream);
159        testAdder.operateOnMixin(mixinStmnt.getObject().asResource(), mockNode);
160    }
161
162    @Before
163    public void setUp() throws RepositoryException {
164        initMocks(this);
165        when(mockNode.getSession()).thenReturn(mockSession);
166        when(mockNode.getName()).thenReturn("mockNode");
167        when(mockNode.getPath()).thenReturn("/mockNode");
168        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
169        when(mockSession.getValueFactory()).thenReturn(mockValueFactory);
170        when(
171                mockSession
172                        .getNamespacePrefix(getJcrNamespaceForRDFNamespace(type
173                                .getNameSpace()))).thenReturn("rdf");
174        when(mockValueFactory.createValue(description, STRING)).thenReturn(mockValue);
175        when(mockWorkspace.getNamespaceRegistry()).thenReturn(
176                mockNamespaceRegistry);
177        when(mockNamespaceRegistry.getURI(propertyNamespacePrefix)).thenReturn(
178                propertyNamespaceUri);
179        when(mockNamespaceRegistry.getURI("rdf")).thenReturn(
180                type.getNameSpace());
181        when(mockNamespaceRegistry.isRegisteredUri(propertyNamespaceUri))
182                .thenReturn(true);
183        when(mockNamespaceRegistry.isRegisteredUri(type.getNameSpace()))
184        .thenReturn(true);
185        when(mockNamespaceRegistry.getPrefix(propertyNamespaceUri)).thenReturn(
186                propertyNamespacePrefix);
187        when(mockNamespaceRegistry.getPrefix(type.getNameSpace())).thenReturn(
188                "rdf");
189        when(mockWorkspace.getNodeTypeManager())
190                .thenReturn(mockNodeTypeManager);
191        when(mockNodeTypeManager.getNodeType(FEDORA_RESOURCE)).thenReturn(
192                mockNodeType);
193        when(mockNodeTypeManager.hasNodeType(mixinShortName)).thenReturn(true);
194        when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
195        when(mockNode.getMixinNodeTypes()).thenReturn(new NodeType[] {});
196        when(mockNode.canAddMixin(mixinShortName)).thenReturn(true);
197        when(mockNodeType.getPropertyDefinitions()).thenReturn(
198                new PropertyDefinition[] {mockPropertyDefinition});
199        when(mockPropertyDefinition.isMultiple()).thenReturn(false);
200        when(mockPropertyDefinition.getName()).thenReturn(propertyShortName);
201        when(mockPropertyDefinition.getRequiredType()).thenReturn(STRING);
202        when(mockGraphSubjects.getSubject(mockNode.getPath())).thenReturn(mockNodeSubject);
203        when(mockTriples.hasNext()).thenReturn(true, true, false);
204        when(mockTriples.next()).thenReturn(descriptiveTriple, mixinTriple);
205        testStream = new RdfStream(mockTriples);
206        testStream.namespaces(mockNamespaceMap);
207    }
