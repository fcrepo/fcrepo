061public class NodeTypeRdfContextTest {

063
064    @Mock
065    private NamespacedNodeType mockNodeType, mockNodeTypeA, mockNodeTypeB;
066
067    private static final String mockNodeTypePrefix = "jcr";
068
069    private static final String mockNodeTypeName = "someType";
070
071    private static final String jcrNamespace = "http://www.jcp.org/jcr/1.0";
072
073    @Mock
074    private NamespacedPropertyDefinition mockProperty;
075
076    @Mock
077    private NamespacedNodeDefinition mockNodeDefinitionA, mockNodeDefinitionB;
078
079    @Before
080    public void setUp() throws RepositoryException {
081        initMocks(this);
082
083        initNodeTypeMocks(mockNodeType, jcrNamespace, mockNodeTypeName);
084        when(mockNodeType.getName()).thenReturn(mockNodeTypePrefix + ":" + mockNodeTypeName);
085        when(mockProperty.getName()).thenReturn("a:a");
086        when(((Namespaced) mockProperty).getNamespaceURI()).thenReturn("a#");
087        when(((Namespaced) mockProperty).getLocalName()).thenReturn("a");
088
089        when(mockNodeTypeB.getName()).thenReturn("b:b");
090        initNodeTypeMocks(mockNodeTypeB, "b#", "b");
091
092        when(mockNodeTypeA.getName()).thenReturn("a:a");
093        initNodeTypeMocks(mockNodeTypeA, "a#", "a");
094        initNamespacedMocks(mockNodeDefinitionA, "a#", "a");
095        when(mockNodeDefinitionA.getName()).thenReturn("a:a");
096
097        initNamespacedMocks(mockNodeDefinitionB, "b#", "b");
098        when(mockNodeDefinitionB.getName()).thenReturn("b:b");
099
100        when(mockNodeDefinitionA.getRequiredPrimaryTypes()).thenReturn(new NodeType[]{});
101        when(mockNodeDefinitionB.getRequiredPrimaryTypes()).thenReturn(new NodeType[]{});
102
103    }
104
105    @Test
106    public void testShouldMapASimpleNodeTypeToRdf() throws RepositoryException {
107        final Model actual = new NodeTypeRdfContext(mockNodeType).asModel();
108        assertTrue(actual.contains(createResource(REPOSITORY_NAMESPACE
109                + mockNodeTypeName), type, Class));
110        assertTrue(actual.contains(createResource(REPOSITORY_NAMESPACE
111                + mockNodeTypeName), label, mockNodeTypePrefix + ":"
112                + mockNodeTypeName));
113    }
114
115    @Test
116    public void testShouldMapNodeTypeIteratorToRdf() throws RepositoryException {
117        @SuppressWarnings("rawtypes")
118        final List mockNodeTypeList = singletonList(mockNodeType);
119        @SuppressWarnings("unchecked")
120        final Model actual = new NodeTypeRdfContext(mockNodeTypeList).asModel();
121        assertTrue(actual.contains(createResource(REPOSITORY_NAMESPACE + mockNodeTypeName),
122                type, Class));
123    }
124
125    @Test
126    public void testShouldMapNodeTypeManagerToRdf() throws RepositoryException {
127        final NodeTypeManager mockNodeTypeManager = mock(NodeTypeManager.class);
128        final NodeTypeIterator mockNodeTypeIterator = mock(NodeTypeIterator.class);
129        when(mockNodeTypeIterator.next()).thenReturn(mockNodeType);
130        when(mockNodeTypeIterator.hasNext()).thenReturn(true, false);
131        when(mockNodeTypeManager.getPrimaryNodeTypes()).thenReturn(mockNodeTypeIterator);
132
133        final NodeTypeIterator mockMixinTypeIterator = mock(NodeTypeIterator.class);
134
135        when(mockMixinTypeIterator.next()).thenReturn(mockNodeTypeB);
136        when(mockMixinTypeIterator.hasNext()).thenReturn(true, false);
137        when(mockNodeTypeManager.getMixinNodeTypes()).thenReturn(mockMixinTypeIterator);
138
139        final Model actual = new NodeTypeRdfContext(mockNodeTypeManager).asModel();
140        assertTrue(actual.contains(ResourceFactory.createResource(REPOSITORY_NAMESPACE + mockNodeTypeName), type, Class));
141        assertTrue(actual.contains(ResourceFactory.createResource("b#b"), type, Class));
142    }
143
144    @Test
145    public void testShouldIncludeSupertypeInformation() throws RepositoryException {
146
147        when(mockNodeType.getDeclaredSupertypes()).thenReturn(new NodeType[] { mockNodeTypeA, mockNodeTypeB });
148        final Model actual = new NodeTypeRdfContext(mockNodeType).asModel();
149        assertTrue(actual.contains(getResource((NodeType)mockNodeType), subClassOf, getResource((NodeType)mockNodeTypeA)));
150        assertTrue(actual.contains(getResource((NodeType)mockNodeType), subClassOf, getResource((NodeType)mockNodeTypeB)));
151    }
152
153    @Test
154    public void testShouldIncludeChildNodeDefinitions() throws RepositoryException {
155
156        when(mockNodeType.getDeclaredChildNodeDefinitions()).thenReturn(new NodeDefinition[] {mockNodeDefinitionA, mockNodeDefinitionB});
157
158        final Model actual = new NodeTypeRdfContext(mockNodeType).asModel();
159        assertTrue(actual.contains(
160                getResource((NodeDefinition) mockNodeDefinitionA), domain,
161                getResource((NodeType) mockNodeType)));
162        assertTrue(actual.contains(
163                getResource((NodeDefinition) mockNodeDefinitionB), domain,
164                getResource((NodeType) mockNodeType)));
165
166    }
167
168    @Test
169    public void testShouldIncludeChildNodeRangeWhenTheChildNodeDeclaresRequiredType() throws RepositoryException {
170
171        when(mockNodeDefinitionA.getRequiredPrimaryTypes()).thenReturn(new NodeType[] { mockNodeTypeB });
172        when(mockNodeType.getDeclaredChildNodeDefinitions()).thenReturn(new NodeDefinition[] { mockNodeDefinitionA });
173
174        final Model actual  = new NodeTypeRdfContext(mockNodeType).asModel();
175
176        assertTrue(actual.contains(
177                getResource((NodeDefinition) mockNodeDefinitionA), domain,
178                getResource((NodeType) mockNodeType)));
179        assertTrue(actual.contains(
180                getResource((NodeDefinition) mockNodeDefinitionA), range,
181                getResource((NodeType) mockNodeTypeB)));
182
183    }
184
185    @Test
186    public void testShouldSkipChildNodesAsResidualSet() throws RepositoryException, IOException {
187
188        when(mockNodeDefinitionA.getName()).thenReturn("*");
189        when(mockNodeType.getDeclaredChildNodeDefinitions()).thenReturn(
190                new NodeDefinition[] {mockNodeDefinitionA});
191
192        final Model actual = new NodeTypeRdfContext(mockNodeType).asModel();
193        logRdf("Retrieved RDF for testShouldSkipChildNodesAsResidualSet():",
194                actual);
195        assertFalse(actual.listResourcesWithProperty(domain,
196                getResource((NodeType)mockNodeType)).hasNext());
197    }

199    @Test
200    public void testShouldIncludePropertyDefinitions() throws RepositoryException {
201        when(mockNodeType.getDeclaredPropertyDefinitions()).thenReturn(new PropertyDefinition[] { mockProperty });
202
203        final Model actual = new NodeTypeRdfContext(mockNodeType).asModel();
204
205        assertTrue(actual.contains(
206                getResource((PropertyDefinition) mockProperty), domain,
207                getResource((NodeType) mockNodeType)));
208    }

211    @Test
212    public void testShouldIncludePropertyDefinitionsRequiredTypeAsRange() throws RepositoryException {
213        when(mockProperty.getRequiredType()).thenReturn(REFERENCE);
214        when(mockNodeType.getDeclaredPropertyDefinitions()).thenReturn(
215                new PropertyDefinition[] {mockProperty});
216
217        final Model actual = new NodeTypeRdfContext(mockNodeType).asModel();
218
219        assertTrue(actual.contains(getResource((PropertyDefinition)mockProperty), domain,
220                getResource((NodeType) mockNodeType)));
221        assertTrue(actual.contains(getResource((PropertyDefinition)mockProperty), range,
222                createResource(XSDanyURI.getURI())));
223    }

225    @Test
226    public void testShouldExcludePropertyDefinitionsForResidualSets() throws RepositoryException {
227        when(mockProperty.getName()).thenReturn("*");
228        when(mockNodeType.getDeclaredPropertyDefinitions()).thenReturn(new PropertyDefinition[] { mockProperty });
229
230        final Model actual = new NodeTypeRdfContext(mockNodeType).asModel();
231
232        assertFalse(actual.listResourcesWithProperty(domain, getResource((NodeType)mockNodeType)).hasNext());
233    }

235    private void initNodeTypeMocks(final NodeType mockNodeType, final String mockNamespaceUri, final String mockNodeTypeName) throws RepositoryException {
236
237        initNamespacedMocks((Namespaced) mockNodeType, mockNamespaceUri, mockNodeTypeName);
238        when(mockNodeType.getDeclaredSupertypes()).thenReturn(new NodeType[] {});
239        when(mockNodeType.getDeclaredChildNodeDefinitions()).thenReturn(new NodeDefinition[] {});
240        when(mockNodeType.getDeclaredPropertyDefinitions()).thenReturn(new PropertyDefinition[] {});
241
242    }
243
244    private static void initNamespacedMocks(final Namespaced namedspacedObject, final String mockNamespaceUri, final String mockNodeTypeName) throws RepositoryException {
245        when(namedspacedObject.getNamespaceURI()).thenReturn(mockNamespaceUri);
246        when(namedspacedObject.getLocalName()).thenReturn(mockNodeTypeName);
247    }

256    }

258    private static void logRdf(final String message, final Model model) throws IOException {
259        LOGGER.debug(message);
260        try (Writer w = new StringWriter()) {
261            model.write(w);
262            LOGGER.debug("\n" + w.toString());
263        }
264    }

266    private static final Logger LOGGER = getLogger(NodeRdfContextTest.class);
