017package org.fcrepo.auth.roles.basic;
018
019import static java.util.Arrays.asList;
020import static java.util.Collections.singleton;
021import static java.util.Collections.singletonMap;
022import static org.fcrepo.auth.common.FedoraAuthorizationDelegate.FEDORA_ALL_PRINCIPALS;
023import static org.fcrepo.auth.common.FedoraAuthorizationDelegate.FEDORA_USER_PRINCIPAL;
024import static org.fcrepo.http.commons.test.util.TestHelpers.setField;
025import static org.junit.Assert.assertFalse;
026import static org.junit.Assert.assertTrue;
027import static org.mockito.Mockito.when;
028import static org.mockito.MockitoAnnotations.initMocks;
029
030import org.fcrepo.auth.roles.common.AccessRolesProvider;
031import org.fcrepo.auth.roles.common.Constants.JcrName;
032import org.fcrepo.http.commons.session.SessionFactory;
033import org.junit.Before;
034import org.junit.Test;
035import org.mockito.Mock;
036import org.modeshape.jcr.value.Path;
037
038import javax.jcr.RepositoryException;
039import javax.jcr.Session;
040
041import java.security.Principal;
042import java.util.Collections;
043import java.util.List;
044import java.util.Map;
045import java.util.Set;
046
047/**
048 * @author Mike Daines
049 */
050public class BasicRolesAuthorizationDelegateTest {
051
052    private static final String[] READ_ACTION = {"read"};
053
054    private static final String[] WRITE_ACTION = {"write"};
055
056    private BasicRolesAuthorizationDelegate authorizationDelegate;
057
058    @Mock
059    private AccessRolesProvider accessRolesProvider;
060
061    @Mock
062    private SessionFactory sessionFactory;
063
064    @Mock
065    private Session mockSession;
066
067    @Mock
068    private Principal principal;
069
070    private Set<Principal> allPrincipals;
071
072    @Mock
073    private Path adminablePath;
074
075    @Mock
076    private Path writablePath;
077
078    @Mock
079    private Path readablePath;
080
081    @Mock
082    private Path unreadablePath;
083
084    @Mock
085    private Path unrecognizablePath;
086
087    @Mock
088    private Path authzPath;
089
090    @Before
091    public void setUp() throws RepositoryException {
092        initMocks(this);
093
094        authorizationDelegate = new BasicRolesAuthorizationDelegate();
095        setField(authorizationDelegate, "accessRolesProvider",
096                accessRolesProvider);
097        setField(authorizationDelegate, "sessionFactory", sessionFactory);
098
099        when(sessionFactory.getInternalSession()).thenReturn(mockSession);
100
101        when(principal.getName()).thenReturn("user");
102        allPrincipals = singleton(principal);
103
104        when(mockSession.getAttribute(FEDORA_USER_PRINCIPAL)).thenReturn(
105                principal);
106        when(mockSession.getAttribute(FEDORA_ALL_PRINCIPALS)).thenReturn(
107                allPrincipals);
108
109        // ACLs for paths
110
111        final Map<String, List<String>> adminAcl =
112                singletonMap("user", asList("admin"));
113        final Map<String, List<String>> writerAcl =
114                singletonMap("user", asList("writer"));
115        final Map<String, List<String>> readerAcl =
116                singletonMap("user", asList("reader"));
117        final Map<String, List<String>> emptyAcl =
118                singletonMap("user", Collections.<String> emptyList());
119        final Map<String, List<String>> unrecognizableAcl =
120                singletonMap("user", asList("something_else"));
121
122        when(accessRolesProvider.findRolesForPath(adminablePath, mockSession))
123                .thenReturn(adminAcl);
124        when(accessRolesProvider.findRolesForPath(writablePath, mockSession))
125                .thenReturn(writerAcl);
126        when(accessRolesProvider.findRolesForPath(readablePath, mockSession))
127                .thenReturn(readerAcl);
128        when(accessRolesProvider.findRolesForPath(unreadablePath, mockSession))
129                .thenReturn(emptyAcl);
130        when(
131                accessRolesProvider.findRolesForPath(unrecognizablePath,
132                        mockSession)).thenReturn(unrecognizableAcl);
133        when(accessRolesProvider.findRolesForPath(authzPath, mockSession))
134                .thenReturn(writerAcl);
135
136        // Identify authzPath as an ACL node
137
138        final String authzDetection = "/{" + JcrName.NS_URI + "}";
139        when(authzPath.toString()).thenReturn("/blah" + authzDetection);
140    }
141
142    @Test
143    public void testPermitRemoveChildNodesForRemoveChildNodesAction() {
144        assertTrue(authorizationDelegate.hasPermission(mockSession,
145                unreadablePath, new String[] {"remove_child_nodes"}));
146    }
147
148    @Test
149    public void testPermitAnythingForAdminablePath() {
150        assertTrue("Should permit write action for path with admin role",
151                authorizationDelegate.hasPermission(mockSession, adminablePath,
152                        WRITE_ACTION));
153        assertTrue("Should permit read action for path with admin role",
154                authorizationDelegate.hasPermission(mockSession, adminablePath,
155                        READ_ACTION));
156        assertTrue(
157                "Should permit another arbitrary action for path with admin role",
158                authorizationDelegate.hasPermission(mockSession, adminablePath,
159                        new String[] {"whatever"}));
160    }
161
162    @Test
163    public void testPermitReadAndWriteForWritablePath() {
164        assertTrue("Should permit write for path with writer role",
165                authorizationDelegate.hasPermission(mockSession, writablePath,
166                        WRITE_ACTION));
167        assertTrue("Should permit read for path with writer role",
168                authorizationDelegate.hasPermission(mockSession, writablePath,
169                        READ_ACTION));
170    }
171
172    @Test
173    public void testDenyWriteForReadablePath() {
174        assertFalse("Should deny write for path with reader role",
175                authorizationDelegate.hasPermission(mockSession, readablePath,
176                        WRITE_ACTION));
177        assertTrue("Should permit read for path with reader role",
178                authorizationDelegate.hasPermission(mockSession, readablePath,
179                        READ_ACTION));
180    }
181
182    @Test
183    public void testDenyReadAndWriteForUnreadablePath() {
184        assertFalse("Should deny write for path with no roles",
185                authorizationDelegate.hasPermission(mockSession,
186                        unreadablePath, WRITE_ACTION));
187        assertFalse("Should deny read for path with no roles",
188                authorizationDelegate.hasPermission(mockSession,
189                        unreadablePath, READ_ACTION));
190    }
191
192    @Test
193    public void testDenyAllForUnrecognizableRole() {
194        assertFalse("Should deny write for path with unrecognizable role",
195                authorizationDelegate.hasPermission(mockSession,
196                        unrecognizablePath, WRITE_ACTION));
197        assertFalse("Should deny read for path with unrecognizable role",
198                authorizationDelegate.hasPermission(mockSession,
199                        unrecognizablePath, READ_ACTION));
200    }
201
202    @Test
203    public void testDenyWriteToWriterForAuthzPath() {
204        assertFalse("Should deny write for ACL path", authorizationDelegate
205                .hasPermission(mockSession, authzPath, WRITE_ACTION));
206    }
207
208}
