336          *

339          * line is simply to test that writes can be completed after a

425     /**
426      * Verifies that one version exists with each supplied value.  This method
427      * makes assertions that each of the provided values is the content of a
428      * version node and nothing else.  Order isn't important, and no assumption
429      * is made about whether extra versions exist.
430      */
431     private void verifyVersions(final GraphStore graph, final Node subject, final String ... values) throws IOException {
432         final ArrayList<String> remainingValues = newArrayList(values);
433         final Iterator<Quad> versionIt = graph.find(ANY, subject, HAS_VERSION.asNode(), ANY);
434         while (versionIt.hasNext() && !remainingValues.isEmpty()) {
435             final String value =
436                 EntityUtils.toString(client
437                         .execute(new HttpGet(versionIt.next().getObject().getURI() + "/fcr:content")).getEntity());
438             remainingValues.remove(value);
439         }
440         if (!remainingValues.isEmpty()) {
441             fail(remainingValues.get(0) + " was not preserved in the version history!");
442         }
443     }

461     private static void patchLiteralProperty(final String url, final String predicate, final String literal) throws IOException {
462         final HttpPatch updateObjectGraphMethod =
463                 new HttpPatch(url);
464         updateObjectGraphMethod.addHeader("Content-Type",
465                 "application/sparql-update");
466         final BasicHttpEntity e = new BasicHttpEntity();
467         e.setContent(new ByteArrayInputStream(
468                 ("INSERT DATA { <> <" + predicate + "> \"" + literal + "\" } ")
469                         .getBytes()));
470         updateObjectGraphMethod.setEntity(e);

472         assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine()
