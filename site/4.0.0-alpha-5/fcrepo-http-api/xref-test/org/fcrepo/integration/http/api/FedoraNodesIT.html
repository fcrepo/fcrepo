1117     private static List<String> headerValues( HttpResponse response,
1118             String headerName ) {
1119         final List<String> values = new ArrayList<String>();
1120         for ( Header header : response.getHeaders(headerName) ) {
1121             for ( String elem : header.getValue().split(",") ) {
1122                 values.add( elem.trim() );
1123             }
1124         }
1125         return values;
