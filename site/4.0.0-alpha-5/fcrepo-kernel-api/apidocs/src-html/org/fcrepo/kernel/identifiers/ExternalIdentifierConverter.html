017package org.fcrepo.kernel.identifiers;
018
019import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
020
021import java.util.List;
022
023import javax.annotation.PostConstruct;
024import javax.inject.Inject;
025
026import com.google.common.base.Converter;
027import com.google.common.collect.Lists;
028import com.hp.hpl.jena.rdf.model.Resource;
029
030/**
031 * Translates internal {@link String} identifiers to external {@link Resource}
032 * identifiers, by passing the input through the translation chain in one
033 * direction or the other, then converting the results into a {@link Resource}.
034 * Subclasses should override doRdfForward() and doRdfReverse().
035 *
036 * @author ajs6f
037 * @date Apr 1, 2014
038 */
039public class ExternalIdentifierConverter extends IdentifierConverter<Resource> {
040
041    @Inject
042    protected List<InternalIdentifierConverter> translationChain;
043
044    protected Converter<String, String> forward = identity();
045
046    protected  Converter<String, String> reverse = identity();
047
048    @Override
049    protected Resource doForward(final String a) {
050        return doRdfForward(forward.convert(a));
051    }
052
053    @Override
054    protected String doBackward(final Resource a) {
055        return reverse.convert(doRdfBackward(a));
056    }
057
058    protected Resource doRdfForward(final String a) {
059        return createResource(a);
060    }
061
062    protected String doRdfBackward(final Resource a) {
063        return a.toString();
064    }
065
066    /**
067     * We fold the list of translators once in each direction and store the
068     * resulting calculation.
069     */
070    @PostConstruct
071    public void accumulateTranslations() {
072        for (final InternalIdentifierConverter t : translationChain) {
073            forward = forward.andThen(t);
074        }
075        for (final InternalIdentifierConverter t : Lists.reverse(translationChain)) {
076            reverse = reverse.andThen(t.reverse());
077        }
078    }
079
080    /**
081     * @param chain the translation chain to use
082     */
083    public void setTranslationChain(final List<InternalIdentifierConverter> chain) {
084        this.translationChain = chain;
085        accumulateTranslations();
086    }
087
088}
