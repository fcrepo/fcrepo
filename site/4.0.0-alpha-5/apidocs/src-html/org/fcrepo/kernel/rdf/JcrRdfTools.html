228    /**
229     * Get an {@link RdfStream} for the given JCR NodeIterator
230     *
231     * @param nodeIterator
232     * @param iteratorSubject
233     * @return
234     * @throws RepositoryException
235     */
236    public RdfStream getJcrPropertiesModel(final Iterator<Node> nodeIterator,
237            final Resource iteratorSubject) throws RepositoryException {
238
239        final RdfStream results = new RdfStream();
240        while (nodeIterator.hasNext()) {
241            final Node node = nodeIterator.next();
242            results.concat(new PropertiesRdfContext(node, graphSubjects,
243                    llstore));
244            if (iteratorSubject != null) {
245                results.concat(singleton(create(iteratorSubject.asNode(), HAS_MEMBER_OF_RESULT.asNode(), graphSubjects
246                        .getSubject(node.getPath()).asNode())));
247            }
248        }
249        return results;
250    }

333    /**
334     * Decides whether the RDF representation of this {@link Node} will receive LDP Container status.
335     *
336     * @param node
337     * @return
338     * @throws RepositoryException
339     */
340    public static boolean isContainer(final Node node) throws RepositoryException {
341        return HAS_CHILD_NODE_DEFINITIONS.apply(node.getPrimaryNodeType())
342                || any(ImmutableList.copyOf(node.getMixinNodeTypes()),
343                        HAS_CHILD_NODE_DEFINITIONS);
344    }
345
346    static Predicate<NodeType> HAS_CHILD_NODE_DEFINITIONS =
347        new Predicate<NodeType>() {
348
349            @Override
350            public boolean apply(final NodeType input) {
351                return input.getChildNodeDefinitions().length > 0;
352            }
353        };

412    /**
413     * Create a JCR value from an RDF node with the given JCR type
414     * @param valueFactory
415     * @param data
416     * @param type
417     * @return
418     * @throws RepositoryException
419     */
420    public Value createValue(final ValueFactory valueFactory, final RDFNode data, final int type)
421        throws RepositoryException {
422        assert (valueFactory != null);
423
424        if (data.isURIResource()
425                && (type == REFERENCE || type == WEAKREFERENCE)) {
426            // reference to another node (by path)
427            final Node nodeFromGraphSubject = session.getNode(graphSubjects.getPathFromSubject(data.asResource()));
428            return valueFactory.createValue(nodeFromGraphSubject,
429                    type == WEAKREFERENCE);
430        } else if (data.isURIResource() || type == URI) {
431            // some random opaque URI
432            return valueFactory.createValue(data.toString(), PropertyType.URI);
433        } else if (data.isResource()) {
434            // a non-URI resource (e.g. a blank node)
435            return valueFactory.createValue(data.toString(), UNDEFINED);
436        } else if (data.isLiteral() && type == UNDEFINED) {
437            // the JCR schema doesn't know what this should be; so introspect
438            // the RDF and try to figure it out
439            final Literal literal = data.asLiteral();
440            final RDFDatatype dataType = literal.getDatatype();
441            final Object rdfValue = literal.getValue();
442
443            if (rdfValue instanceof Boolean) {
444                return valueFactory.createValue((Boolean) rdfValue);
445            } else if (rdfValue instanceof Byte
446                    || (dataType != null && dataType.getJavaClass() == Byte.class)) {
447                return valueFactory.createValue(literal.getByte());
448            } else if (rdfValue instanceof Double) {
449                return valueFactory.createValue((Double) rdfValue);
450            } else if (rdfValue instanceof Float) {
451                return valueFactory.createValue((Float) rdfValue);
452            } else if (rdfValue instanceof Long
453                    || (dataType != null && dataType.getJavaClass() == Long.class)) {
454                return valueFactory.createValue(literal.getLong());
455            } else if (rdfValue instanceof Short
456                    || (dataType != null && dataType.getJavaClass() == Short.class)) {
457                return valueFactory.createValue(literal.getShort());
458            } else if (rdfValue instanceof Integer) {
459                return valueFactory.createValue((Integer) rdfValue);
460            } else if (rdfValue instanceof XSDDateTime) {
461                return valueFactory.createValue(((XSDDateTime) rdfValue)
462                        .asCalendar());
463            } else {
464                return valueFactory.createValue(literal.getString(), STRING);
465            }
466
467        } else {
468            LOGGER.debug("Using default JCR value creation for RDF literal: {}",
469                    data);
470            return valueFactory.createValue(data.asLiteral().getString(), type);
471        }
472    }

542    /**
543     * Get the JCR property name for an RDF predicate
544     * @param namespaceRegistry
545     * @param predicate
546     * @param namespaceMapping
547     * @return
548     * @throws RepositoryException
549     */
550    public String getPropertyNameFromPredicate(final NamespaceRegistry namespaceRegistry,
551                                               final com.hp.hpl.jena.rdf.model.Property predicate,
552                                               final Map<String, String> namespaceMapping) throws RepositoryException {
553
554        final String prefix;
555
556        final String namespace =
557            getJcrNamespaceForRDFNamespace(predicate.getNameSpace());
558
559        assert (namespaceRegistry != null);
560
561        if (namespaceRegistry.isRegisteredUri(namespace)) {
562            LOGGER.debug("Discovered namespace: {} in namespace registry.",namespace);
563            prefix = namespaceRegistry.getPrefix(namespace);
564        } else {
565            LOGGER.debug("Didn't discover namespace: {} in namespace registry.",namespace);
566            final ImmutableBiMap<String, String> nsMap =
567                ImmutableBiMap.copyOf(namespaceMapping);
568            if (nsMap.containsValue(namespace)) {
569                LOGGER.debug("Discovered namespace: {} in namespace map: {}.", namespace,
570                        nsMap);
571                prefix = nsMap.inverse().get(namespace);
572                namespaceRegistry.registerNamespace(prefix, namespace);
573            } else {
574                prefix = namespaceRegistry.registerNamespace(namespace);
575            }
576        }
577
578        final String localName = predicate.getLocalName();
579
580        final String propertyName = prefix + ":" + localName;
581
582        LOGGER.debug("Took RDF predicate {} and translated it to JCR property {}", predicate, propertyName);
583
584        return propertyName;
585
586    }

607    /**
608     * Given a node type and a property name, figure out an appropraite jcr value type
609     * @param nodeType
610     * @param propertyName
611     * @return
612     */
613    public int getPropertyType(final NodeType nodeType, final String propertyName) {
614        final PropertyDefinition[] propertyDefinitions = nodeType.getPropertyDefinitions();
615        int type = UNDEFINED;
616        for (final PropertyDefinition propertyDefinition : propertyDefinitions) {
617            if (propertyDefinition.getName().equals(propertyName)) {
618                if (type != UNDEFINED) {
619                    return UNDEFINED;
620                }
621
622                type = propertyDefinition.getRequiredType();
623            }
624        }
625
626        return type;
627    }

629    /**
630     * Map a JCR property to an RDF property with the right namespace URI and
631     * local name
632     */
633    public static Function<Property, com.hp.hpl.jena.rdf.model.Property> getPredicateForProperty =
634            new Function<Property, com.hp.hpl.jena.rdf.model.Property>() {
635
636                @Override
637                public com.hp.hpl.jena.rdf.model.Property apply(
638                        final Property property) {
639                    LOGGER.trace("Creating predicate for property: {}",
640                            property);
641                    try {
642                        if (property instanceof Namespaced) {
643                            final Namespaced nsProperty = (Namespaced) property;
644                            final String uri = nsProperty.getNamespaceURI();
645                            return createProperty(
646                                    getRDFNamespaceForJcrNamespace(uri),
647                                    nsProperty.getLocalName());
648                        }
649                        return createProperty(property.getName());
650                    } catch (final RepositoryException e) {
651                        throw propagate(e);
652                    }
653
654                }
655            };
656}
