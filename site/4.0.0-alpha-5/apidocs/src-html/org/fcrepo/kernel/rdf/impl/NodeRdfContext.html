044/**
045 * {@link RdfStream} that holds contexts related to a specific {@link Node}.
046 *
047 * @author ajs6f
048 * @date Oct 10, 2013
049 */
050public class NodeRdfContext extends RdfStream {
051
052    private final Node node;
053
054    private final IdentifierTranslator graphSubjects;
055
056    private final com.hp.hpl.jena.graph.Node subject;
057
058    private final LowLevelStorageService lowLevelStorageService;
059
060    private static final Logger LOGGER = getLogger(NodeRdfContext.class);
061
062    /**
063     * Default constructor.
064     *
065     * @param node
066     * @param graphSubjects
067     * @throws RepositoryException
068     */
069    public NodeRdfContext(final Node node, final IdentifierTranslator graphSubjects,
070        final LowLevelStorageService lowLevelStorageService)
071        throws RepositoryException {
072        super();
073        this.node = node;
074        this.graphSubjects = graphSubjects;
075        this.subject = graphSubjects.getSubject(node.getPath()).asNode();
076
077        // TODO fix GraphProperties to allow for LowLevelStorageServices to pass
078        // through it
079        // this is horribly ugly. LowLevelStorageServices are supposed to be
080        // managed beans.
081        // but the contract of GraphProperties isn't wide enough to pass one in,
082        // so rather than
083        // alter GraphProperties right now, I'm just spinning one on the fly.
084        if (lowLevelStorageService == null) {
085            this.lowLevelStorageService = new LowLevelStorageService();
086            this.lowLevelStorageService.setRepository(node.getSession()
087                    .getRepository());
088        } else {
089            this.lowLevelStorageService = lowLevelStorageService;
090        }
091
092        //include rdf:type for primaryType, mixins, and their supertypes
093        concatRdfTypes();
094    }
095
096    /**
097     * @return The {@link Node} in question
098     */
099    public Node node() {
100        return node;
101    }
102
103    /**
104     * @return local {@link IdentifierTranslator}
105     */
106    public IdentifierTranslator graphSubjects() {
107        return graphSubjects;
108    }
109
110    /**
111     * @return the RDF subject at the center of this context
112     */
113    public com.hp.hpl.jena.graph.Node subject() {
114        return subject;
115    }
116
117    /**
118     * @return the {@link LowLevelStorageService} in scope
119     */
120    public LowLevelStorageService lowLevelStorageService() {
121        return lowLevelStorageService;
122    }
123
124    private Function<NodeType, Triple> nodetype2triple() {
125        return new Function<NodeType, Triple>() {
126
127            @Override
128            public Triple apply(final NodeType nodeType) {
129                try {
130                    final String fullTypeName = nodeType.getName();
131                    LOGGER.trace("Translating JCR mixin name: {}", fullTypeName);
132                    final String prefix = fullTypeName.split(":")[0];
133                    final String typeName = fullTypeName.split(":")[1];
134                    final String namespace = getJcrUri(prefix);
135                    LOGGER.trace("with JCR namespace: {}", namespace);
136                    final com.hp.hpl.jena.graph.Node rdfType =
137                        createURI(getRDFNamespaceForJcrNamespace(namespace)
138                                + typeName);
139                    LOGGER.trace("into RDF resource: {}", rdfType);
140                    return create(subject(), type.asNode(), rdfType);
141                } catch (final RepositoryException e) {
142                    throw propagate(e);
143                }
144            }
145
146        };
147    }
148
149    private String getJcrUri(final String prefix) throws RepositoryException {
150        return node().getSession().getWorkspace().getNamespaceRegistry()
151                .getURI(prefix);
152    }
153
154    private void concatRdfTypes() throws RepositoryException {
155        final NodeType primaryNodeType = node.getPrimaryNodeType();
156        final NodeType[] mixinNodeTypesArr = node.getMixinNodeTypes();
157        final Set<NodeType> primarySupertypes = ImmutableSet.<NodeType>builder()
158                .add(primaryNodeType.getSupertypes()).build();
159        final Set<NodeType> mixinNodeTypes = ImmutableSet.<NodeType>builder().add(mixinNodeTypesArr).build();
160        final ImmutableList.Builder<NodeType> nodeTypesB = ImmutableList.<NodeType>builder()
161                .add(primaryNodeType)
162                .addAll(primarySupertypes)
163                .addAll(mixinNodeTypes);
164        final ImmutableSet.Builder<NodeType> mixinSupertypes = ImmutableSet.<NodeType>builder();
165        for (final NodeType mixinNodeType : mixinNodeTypes) {
166            mixinSupertypes.addAll(ImmutableSet.<NodeType>builder().add(mixinNodeType.getSupertypes()).build());
167        }
168        nodeTypesB.addAll(mixinSupertypes.build());
169        final ImmutableList<NodeType> nodeTypes = nodeTypesB.build();
170        final Iterator<NodeType> nodeTypesIt = nodeTypes.iterator();
171        concat(Iterators.transform(nodeTypesIt,nodetype2triple()));
172    }
