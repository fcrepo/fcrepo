63      @Test
64      public void testPolicyCreateByPost() throws Exception {
65          final HttpPost objMethod = HttpPostObjMethod(POLICY_RESOURCE);
66          StringEntity input = new StringEntity(MIME_KEY + " " + MIME + " " + STORE,
67                                                "UTF-8");
68          input.setContentType(APPLICATION_FORM_URLENCODED);
69          objMethod.setEntity(input);
70          final HttpResponse response = client.execute(objMethod);
71  
72          String body = IOUtils.toString(response.getEntity().getContent());
73          assertEquals(body, 201, response.getStatusLine().getStatusCode());
74  
75          policyKeys.add(MIME_KEY);
76  
77          Header[] headers = response.getHeaders("Location");
78          assertNotNull(headers);
79          assertEquals(1, headers.length);
80          assertEquals(objMethod.getURI()
81                               .toString()
82                               .replace(POLICY_RESOURCE, MIME_KEY),
83                       headers[0].getValue());
84      }
