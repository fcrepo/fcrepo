017package org.fcrepo.kernel;
018
019import static org.fcrepo.kernel.services.ServiceHelpers.getObjectSize;
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertNotNull;
022import static org.junit.Assert.fail;
023import static org.mockito.Mockito.mock;
024import static org.mockito.Mockito.verify;
025import static org.mockito.Mockito.when;
026import static org.mockito.MockitoAnnotations.initMocks;
027import static org.powermock.api.mockito.PowerMockito.mockStatic;
028
029import java.util.Calendar;
030import java.util.Collection;
031
032import javax.jcr.Node;
033import javax.jcr.Property;
034import javax.jcr.RepositoryException;
035import javax.jcr.Session;
036import javax.jcr.nodetype.NodeType;
037
038import org.fcrepo.jcr.FedoraJcrTypes;
039import org.fcrepo.kernel.services.ServiceHelpers;
040import org.junit.After;
041import org.junit.Before;
042import org.junit.Test;
043import org.junit.runner.RunWith;
044import org.mockito.Mock;
045import org.powermock.core.classloader.annotations.PowerMockIgnore;
046import org.powermock.core.classloader.annotations.PrepareForTest;
047import org.powermock.modules.junit4.PowerMockRunner;
048
049import com.google.common.base.Predicate;
050
051@RunWith(PowerMockRunner.class)
052@PowerMockIgnore({"org.slf4j.*", "javax.xml.parsers.*", "org.apache.xerces.*"})
053@PrepareForTest({ServiceHelpers.class})
054public class FedoraObjectImplTest implements FedoraJcrTypes {
055
056    private static final String testPid = "testObj";
057
058    private static final String mockUser = "mockUser";
059
060    @Mock
061    private Session mockSession;
062
063    @Mock
064    private Node mockRootNode;
065
066    @Mock
067    private Node mockObjNode;
068
069    @Mock
070    private Property mockProp;
071
072    @Mock
073    private Predicate<Node> mockPredicate;
074
075    private FedoraObject testFedoraObject;
076
077    private NodeType[] mockNodetypes;
078
079    @Before
080    public void setUp() {
081        initMocks(this);
082        final String relPath = "/" + testPid;
083        final NodeType[] types = new NodeType[0];
084        try {
085            when(mockObjNode.getName()).thenReturn(testPid);
086            when(mockObjNode.getSession()).thenReturn(mockSession);
087            when(mockObjNode.getMixinNodeTypes()).thenReturn(types);
088            final NodeType mockNodeType = mock(NodeType.class);
089            when(mockNodeType.getName()).thenReturn("nt:folder");
090            when(mockObjNode.getPrimaryNodeType()).thenReturn(mockNodeType);
091            when(mockSession.getRootNode()).thenReturn(mockRootNode);
092            when(mockRootNode.getNode(relPath)).thenReturn(mockObjNode);
093            when(mockSession.getUserID()).thenReturn(mockUser);
094            testFedoraObject = new FedoraObjectImpl(mockObjNode);
095
096            mockNodetypes = new NodeType[2];
097            mockNodetypes[0] = mock(NodeType.class);
098            mockNodetypes[1] = mock(NodeType.class);
099
100            when(mockObjNode.getMixinNodeTypes()).thenReturn(mockNodetypes);
101
102            when(mockPredicate.apply(mockObjNode)).thenReturn(true);
103
104        } catch (final RepositoryException e) {
105            e.printStackTrace();
106            fail(e.getMessage());
107        }
108
109    }
110
111    @After
112    public void tearDown() {
113        mockSession = null;
114        mockRootNode = null;
115        mockObjNode = null;
116    }
117
118    @Test
119    public void testGetName() throws RepositoryException {
120        assertEquals(testFedoraObject.getName(), testPid);
121    }
122
123    @Test
124    public void testGetNode() {
125        assertEquals(testFedoraObject.getNode(), mockObjNode);
126    }
127
128    @Test
129    public void testGetCreated() throws RepositoryException {
130        when(mockProp.getDate()).thenReturn(Calendar.getInstance());
131        when(mockObjNode.hasProperty(JCR_CREATED)).thenReturn(true);
132        when(mockObjNode.getProperty(JCR_CREATED)).thenReturn(mockProp);
133        testFedoraObject.getCreatedDate();
134        verify(mockObjNode).getProperty(JCR_CREATED);
135    }
136
137    @Test
138    public void testGetLastModified() throws RepositoryException {
139        when(mockObjNode.hasProperty(JCR_LASTMODIFIED)).thenReturn(true);
140        when(mockObjNode.getProperty(JCR_LASTMODIFIED)).thenReturn(mockProp);
141        when(mockProp.getDate()).thenReturn(Calendar.getInstance());
142        testFedoraObject.getLastModifiedDate();
143        verify(mockObjNode).getProperty(JCR_LASTMODIFIED);
144    }
145
146    @Test
147    public void testGetSize() throws RepositoryException {
148        mockStatic(ServiceHelpers.class);
149        // obviously not a real value
150        when(getObjectSize(mockObjNode)).thenReturn(-8L);
151        final long actual = testFedoraObject.getSize();
152        assertEquals(-8, actual);
153    }
154
155    @Test
156    public void testGetModels() throws RepositoryException {
157        final Collection<String> actual = testFedoraObject.getModels();
158        assertNotNull(actual);
159    }

161}
