360    @Test
361    public void testReplacePropertiesDataset() throws Exception {
362
363        mockStatic(JcrRdfTools.class);
364        final DefaultIdentifierTranslator defaultGraphSubjects = new DefaultIdentifierTranslator();
365        when(JcrRdfTools.withContext(defaultGraphSubjects, mockSession)).thenReturn(mockJcrRdfTools);
366
367        when(mockNode.getPath()).thenReturn("/xyz");
368
369        final Model propertiesModel = createDefaultModel();
370        propertiesModel.add(propertiesModel.createResource("a"),
371                               propertiesModel.createProperty("b"),
372                               "c");
373
374
375        propertiesModel.add(propertiesModel.createResource("i"),
376                               propertiesModel.createProperty("j"),
377                               "k");
378
379        propertiesModel.add(propertiesModel.createResource("x"),
380                               propertiesModel.createProperty("y"),
381                               "z");
382        final RdfStream propertiesStream = RdfStream.fromModel(propertiesModel);
383        when(mockJcrRdfTools.getJcrTriples(mockNode)).thenReturn(propertiesStream);
384
385        final RdfStream treeStream = new RdfStream();
386        when(mockJcrRdfTools.getTreeTriples(mockNode)).thenReturn(treeStream);
387        final Model problemsModel = createDefaultModel();
388        when(getProblemsModel()).thenReturn(problemsModel);
389
390        final Model replacementModel = createDefaultModel();
391
392        replacementModel.add(replacementModel.createResource("a"),
393                                replacementModel.createProperty("b"),
394                               "n");
395
396
397        replacementModel.add(replacementModel.createResource("i"),
398                                replacementModel.createProperty("j"),
399                               "k");
400
401        final Model replacements = testObj.replaceProperties(defaultGraphSubjects, replacementModel).asModel();
402
403        assertTrue(replacements.containsAll(replacementModel));
404
405        assertFalse(problemsModel.contains(propertiesModel.createResource("x"),
406                                              propertiesModel.createProperty("y"),
407                                              "z"));
408    }
409    @Test
410    public void shouldGetEtagForAnObject() throws RepositoryException {
411        final Property mockMod = mock(Property.class);
412        final Calendar modDate = Calendar.getInstance();
413        modDate.set(2013, JULY, 30, 0, 0, 0);
414        when(mockNode.getPath()).thenReturn("some-path");
415        when(mockNode.hasProperty(JCR_LASTMODIFIED)).thenReturn(true);
416        when(mockNode.getProperty(JCR_LASTMODIFIED)).thenReturn(mockMod);
417        when(mockMod.getDate()).thenReturn(modDate);
418
419        assertEquals(shaHex("some-path"
420                + testObj.getLastModifiedDate().toString()), testObj
421                .getEtagValue());
