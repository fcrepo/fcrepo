25  import javax.jcr.Repository;

30  import static org.junit.Assert.assertTrue;
31  import static org.mockito.Mockito.when;
32  import static org.mockito.MockitoAnnotations.initMocks;
33  
34  public class WorkspaceRdfContextTest {
35  
36  
37      @Mock
38      private IdentifierTranslator subjects;
39  
40      @Mock
41      private Session session;
42  
43      @Mock
44      private Repository repository;
45  
46      @Mock
47      private Workspace mockWorkspace;
48  
49  
50      @Before
51      public void setUp() {
52          initMocks(this);
53          subjects = new DefaultIdentifierTranslator();
54          when(session.getRepository()).thenReturn(repository);
55          when(session.getWorkspace()).thenReturn(mockWorkspace);
56      }
57  
58      @Test
59      public void testWorkspaceTriples() throws RepositoryException {
60          when(mockWorkspace.getAccessibleWorkspaceNames()).thenReturn(new String[] { "default", "a", "b" } );
61          final WorkspaceRdfContext testObj = new WorkspaceRdfContext(session, subjects);
62  
63          final Model model = testObj.asModel();
64  
65          assertTrue(model.contains(subjects.getSubject("/"),
66                                        RdfLexicon.HAS_DEFAULT_WORKSPACE,
67                                       subjects.getSubject("/workspace:default" )));
68  
69          assertTrue(model.contains(subjects.getSubject("/workspace:default" ),
70                                       RdfLexicon.DC_TITLE,
71                                       "default"));
72  
73          assertTrue(model.contains(subjects.getSubject("/"),
74                                       RdfLexicon.HAS_WORKSPACE,
75                                       subjects.getSubject("/workspace:a")));
76  
77          assertTrue(model.contains(subjects.getSubject("/"),
78                                       RdfLexicon.HAS_WORKSPACE,
79                                       subjects.getSubject("/workspace:b" )));
80  
81      }
