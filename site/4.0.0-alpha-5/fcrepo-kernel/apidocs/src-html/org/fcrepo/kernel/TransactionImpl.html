017package org.fcrepo.kernel;
018
019import static java.lang.System.currentTimeMillis;
020import static java.util.UUID.randomUUID;
021import static org.fcrepo.kernel.Transaction.State.NEW;
022import static org.fcrepo.kernel.Transaction.State.COMMITED;
023import static org.fcrepo.kernel.Transaction.State.DIRTY;
024import static org.fcrepo.kernel.Transaction.State.ROLLED_BACK;
025
026import java.util.Calendar;
027import java.util.Collections;
028import java.util.Date;
029import java.util.Set;
030import java.util.concurrent.ConcurrentHashMap;
031
032import javax.jcr.RepositoryException;
033import javax.jcr.Session;
034
035import org.fcrepo.kernel.services.VersionService;
036
037/**
038 * A Fedora Transaction wraps a JCR session with some expiration logic.
039 * Whenever the transaction's session is requested, the expiration is extended
040 */
041public class TransactionImpl implements Transaction {
042
043    // the default timeout is 3 minutes
044    public static final long DEFAULT_TIMEOUT = 3L * 60L * 1000L;
045
046    public static final String TIMEOUT_SYSTEM_PROPERTY = "fcrepo4.tx.timeout";
047
048    private final Session session;
049
050    private final String id;
051
052    private final String userName;
053
054    private final Date created;
055
056    private final Calendar expires;
057
058    private State state = NEW;
059
060    private Set<String> versionedPaths;
061
062    /**
063     * Create a transaction for the given Session
064     * @param session
065     */
066
067    public TransactionImpl(final Session session, final String userName) {
068        super();
069        this.session = session;
070        this.created = new Date();
071        this.id = randomUUID().toString();
072        this.expires = Calendar.getInstance();
073        this.updateExpiryDate();
074        this.userName = userName;
075    }
076
077    /* (non-Javadoc)
078     * @see org.fcrepo.kernel.Transaction#getSession()
079     */
080    @Override
081    public Session getSession() {
082        updateExpiryDate();
083        return TxAwareSession.newInstance(session, id);
084    }
085
086    /* (non-Javadoc)
087     * @see org.fcrepo.kernel.Transaction#getCreated()
088     */
089    @Override
090    public Date getCreated() {
091        return new Date(created.getTime());
092    }
093
094    /* (non-Javadoc)
095     * @see org.fcrepo.kernel.Transaction#getId()
096     */
097    @Override
098    public String getId() {
099        return id;
100    }
101
102    /* (non-Javadoc)
103     * @see org.fcrepo.kernel.Transaction#getState()
104     */
105    @Override
106    public State getState() throws RepositoryException {
107        if (this.session != null && this.session.hasPendingChanges()) {
108            return DIRTY;
109        }
110        return state;
111    }
112
113    /* (non-Javadoc)
114     * @see org.fcrepo.kernel.Transaction#getExpires()
115     */
116    @Override
117    public Date getExpires() {
118        return expires.getTime();
119    }
120
121    /* (non-Javadoc)
122     * @see org.fcrepo.kernel.Transaction#addPathToVersion(java.lang.String)
123     */
124    @Override
125    public void addPathToVersion(final String absPath) {
126        if (versionedPaths == null) {
127            versionedPaths = Collections.newSetFromMap(
128                    new ConcurrentHashMap<String, Boolean>());
129        }
130        versionedPaths.add(absPath);
131    }
132
133    /* (non-Javadoc)
134     * @see org.fcrepo.kernel.Transaction#commit(org.fcrepo.kernel.services.VersionService)
135     */
136    @Override
137    public void commit(final VersionService vService) throws RepositoryException {
138
139        this.session.save();
140        if (this.versionedPaths != null) {
141            if (vService == null) {
142                throw new IllegalStateException("Versioned Paths were added," +
143                        " but no VersionService was provided!");
144            }
145            vService.createVersion(session.getWorkspace(), versionedPaths);
146        }
147        this.state = COMMITED;
148        this.expire();
149    }
150
151    /* (non-Javadoc)
152     * @see org.fcrepo.kernel.Transaction#expire()
153     */
154    @Override
155    public void expire() {
156        this.session.logout();
157        this.expires.setTimeInMillis(currentTimeMillis());
158    }
159
160    /**
161     * Checks if this transaction is associated with a specific user.
162     * If both user names are null, it is assumed that the transaction is
163     * anonymous and thus not bound to any user.
164     * @param userName the user
165     * @return true if the userName is associated with this transaction
166     */
167    @Override
168    public boolean isAssociatedWithUser(final String userName) {
169        boolean associatedWith = false;
170        if (this.userName == null) {
171            if (userName == null) {
172                associatedWith = true;
173            }
174        } else {
175            associatedWith = this.userName.equals(userName);
176        }
177        return associatedWith;
178    }
179
180    /* (non-Javadoc)
181     * @see org.fcrepo.kernel.Transaction#rollback()
182     */
183    @Override
184    public void rollback() throws RepositoryException {
185        this.state = ROLLED_BACK;
186        this.session.refresh(false);
187        this.expire();
188    }
189
190    /* (non-Javadoc)
191     * @see org.fcrepo.kernel.Transaction#updateExpiryDate()
192     */
193    @Override
194    public void updateExpiryDate() {
195        long duration;
196        if (System.getProperty(TIMEOUT_SYSTEM_PROPERTY) != null) {
197            duration =
198                    Long.parseLong(System.getProperty(TIMEOUT_SYSTEM_PROPERTY));
199        } else {
200            duration = DEFAULT_TIMEOUT;
201        }
202        this.expires.setTimeInMillis(currentTimeMillis() + duration);
203    }
204}
