056public class BaseHtmlProviderTest {
057
058    final BaseHtmlProvider baseHtmlProvider = new BaseHtmlProvider();
059
060    Dataset testData = new DatasetImpl(createDefaultModel());
061
062    {
063        testData.asDatasetGraph().getDefaultGraph().add(
064                new Triple(createURI("test:subject"),
065                        createURI("test:predicate"),
066                        createLiteral("test:object")));
067        testData.asDatasetGraph().getDefaultGraph().add(
068                new Triple(createURI("test:subject"), primaryTypePredicate,
069                        createLiteral("nt:file")));
070
071    }
072
073    @Test
074    public void testIsWriteable() {
075        assertTrue(
076                "Gave false response to HtmlProvider.isWriteable() that contained a legitimate combination of parameters!",
077                baseHtmlProvider.isWriteable(Dataset.class, Dataset.class,
078                        null, TEXT_HTML_TYPE));
079        assertFalse(
080                "HtmlProvider.isWriteable() should return false if asked to serialize anything other than Dataset!",
081                baseHtmlProvider.isWriteable(BaseHtmlProvider.class,
082                        BaseHtmlProvider.class, null, TEXT_HTML_TYPE));
083        assertFalse(
084                "HtmlProvider.isWriteable() should return false to text/plain!",
085                baseHtmlProvider.isWriteable(Dataset.class, Dataset.class,
086                        null, TEXT_PLAIN_TYPE));
087    }
088
089    @Test
090    public void testGetSize() {
091        assertEquals("Returned wrong size from HtmlProvider!", baseHtmlProvider
092                .getSize(null, null, null, null, null), -1);
093
094    }
095
096    @SuppressWarnings({"unchecked", "rawtypes"})
097    @Test
098    public void testWriteTo() throws WebApplicationException,
099            IllegalArgumentException, IOException {
100        final Template mockTemplate = mock(Template.class);
101        final ByteArrayOutputStream outStream = new ByteArrayOutputStream();
102
103        doAnswer(new Answer<Object>() {
104
105            @Override
106            public Object answer(final InvocationOnMock invocation) {
107                outStream.write("abcdefighijk".getBytes(), 0, 10);
108                return "I am pretending to merge a template for you.";
109            }
110        }).when(mockTemplate).merge(isA(Context.class), isA(Writer.class));
111        setField(baseHtmlProvider, "templatesMap", singletonMap("nt:file",
112                mockTemplate));
113        baseHtmlProvider.writeTo(testData, Dataset.class, mock(Type.class),
114                new Annotation[] {}, MediaType.valueOf("text/html"),
115                (MultivaluedMap) new MultivaluedMapImpl(), outStream);
116        final byte[] results = outStream.toByteArray();
117        assertTrue("Got no output from serialization!", results.length > 0);
118
119    }
120
121    @SuppressWarnings({"unchecked", "rawtypes"})
122    @Test
123    public void testWriteToWithAnnotation() throws WebApplicationException,
124            IllegalArgumentException, IOException {
125        final Template mockTemplate = mock(Template.class);
126        final ByteArrayOutputStream outStream = new ByteArrayOutputStream();
127
128        doAnswer(new Answer<Object>() {
129
130            @Override
131            public Object answer(final InvocationOnMock invocation) {
132                outStream.write("abcdefighijk".getBytes(), 0, 10);
133                return "I am pretending to merge a template for you.";
134            }
135        }).when(mockTemplate).merge(isA(Context.class), isA(Writer.class));
136
137        setField(baseHtmlProvider, "templatesMap",
138                of("some:file", mockTemplate));
139        final HtmlTemplate mockAnnotation = mock(HtmlTemplate.class);
140        when(mockAnnotation.value()).thenReturn("some:file");
141        baseHtmlProvider.writeTo(testData, Dataset.class, mock(Type.class),
142                new Annotation[] {mockAnnotation}, MediaType
143                        .valueOf("text/html"),
144                (MultivaluedMap) new MultivaluedMapImpl(), outStream);
145        final byte[] results = outStream.toByteArray();
146        assertTrue("Got no output from serialization!", results.length > 0);
147
148    }
