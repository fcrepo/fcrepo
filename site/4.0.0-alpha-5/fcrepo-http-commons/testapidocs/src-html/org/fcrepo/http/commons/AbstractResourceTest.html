017package org.fcrepo.http.commons;
018
019import static org.fcrepo.http.commons.AbstractResource.toPath;
020import static org.fcrepo.http.commons.test.util.PathSegmentImpl.createPathList;
021import static org.fcrepo.http.commons.test.util.TestHelpers.setField;
022import static org.junit.Assert.assertEquals;
023import static org.mockito.MockitoAnnotations.initMocks;
024import java.util.List;
025
026import javax.ws.rs.core.PathSegment;
027import javax.ws.rs.core.UriInfo;
028
029import org.fcrepo.kernel.identifiers.PidMinter;
030import org.fcrepo.kernel.services.NodeService;
031import org.fcrepo.kernel.utils.NamespaceTools;
032import org.junit.Before;
033import org.junit.Test;
034import org.junit.runner.RunWith;
035import org.mockito.Mock;
036import org.modeshape.jcr.api.NamespaceRegistry;
037import org.powermock.core.classloader.annotations.PowerMockIgnore;
038import org.powermock.core.classloader.annotations.PrepareForTest;
039import org.powermock.modules.junit4.PowerMockRunner;
040
041@RunWith(PowerMockRunner.class)
042@PowerMockIgnore({"org.slf4j.*", "javax.xml.parsers.*", "org.apache.xerces.*"})
043@PrepareForTest({NamespaceTools.class})
044public class AbstractResourceTest {
045
046    private AbstractResource testObj;
047
048    @Mock
049    private NodeService mockNodes;
050
051    @Mock
052    private PidMinter mockPids;
053
054    @Mock
055    private UriInfo mockUris;
056
057    @Mock
058    private NamespaceRegistry mockNames;
059
060    @Before
061    public void setUp() {
062        initMocks(this);
063        testObj = new AbstractResource() {};
064    }
065
066    @Test
067    public void testSetPidMinter() throws Exception {
068        setField(testObj, "pidMinter", mockPids);
069        assertEquals(mockPids, testObj.pidMinter);
070    }
071
072    @Test
073    public void testSetNodeService() throws Exception {
074        setField(testObj, "nodeService", mockNodes);
075        assertEquals(mockNodes, testObj.nodeService);
076    }
077
078    @Test
079    public void testSetUriInfo() throws Exception {
080        setField(testObj, "uriInfo", mockUris);
081        assertEquals(mockUris, testObj.uriInfo);
082    }
083
084    @Test
085    public void testToPath() {
086        final List<PathSegment> pathList =
087                createPathList("foo", "", "bar", "baz");
088        // empty path segments ('//') should be suppressed
089        final String expected = "/foo/bar/baz";
090        final String actual = toPath(pathList);
091        assertEquals(expected, actual);
092    }
093
094    @Test
095    public void testToPathWorkspace() {
096        final List<PathSegment> pathList =
097                createPathList("workspace:abc", "bar", "baz");
098        // workspaces should be ignored
099        final String expected = "/bar/baz";
100        final String actual = toPath(pathList);
101        assertEquals(expected, actual);
102    }
103
104    @Test
105    public void testToPathWorkspaceInSomeOtherSegment() {
106        final List<PathSegment> pathList =
107                createPathList("asdf", "workspace:abc", "bar", "baz");
108        // workspaces should be ignored
109        final String expected = "/asdf/workspace:abc/bar/baz";
110        final String actual = toPath(pathList);
111        assertEquals(expected, actual);
112    }
113
114    @Test
115    public void testToPathWorkspaceWithEmptyPrefix() {
116        final List<PathSegment> pathList =
117                createPathList("", "workspace:abc", "bar", "baz");
118        // workspaces should be ignored
119        final String expected = "/bar/baz";
120        final String actual = toPath(pathList);
121        assertEquals(expected, actual);
122    }
123
124    @Test
125    public void testToPathTransaction() {
126        final List<PathSegment> pathList =
127                createPathList("tx:abc", "bar", "baz");
128        // workspaces should be ignored
129        final String expected = "/bar/baz";
130        final String actual = toPath(pathList);
131        assertEquals(expected, actual);
132    }
133
134    @Test
135    public void testToPathTxInSomeOtherSegment() {
136        final List<PathSegment> pathList =
137                createPathList("asdf", "tx:abc", "bar", "baz");
138        // workspaces should be ignored
139        final String expected = "/asdf/tx:abc/bar/baz";
140        final String actual = toPath(pathList);
141        assertEquals(expected, actual);
142    }
143
144    @Test
145    public void testToPathTxWithEmptyPrefix() {
146        final List<PathSegment> pathList =
147                createPathList("", "tx:abc", "bar", "baz");
148        // workspaces should be ignored
149        final String expected = "/bar/baz";
150        final String actual = toPath(pathList);
151        assertEquals(expected, actual);
152    }
153
154    @Test
155    public void testToPathUuid() {
156        final List<PathSegment> pathList = createPathList("[foo]");
157        final String expected = "[foo]";
158        final String actual = toPath(pathList);
159        assertEquals(expected, actual);
160    }
161
162    @Test
163    public void testToPathEmpty() {
164        final List<PathSegment> pathList = createPathList();
165        // empty path segments ('//') should be suppressed
166        final String expected = "/";
167        final String actual = toPath(pathList);
168        assertEquals(expected, actual);
169    }
170}
