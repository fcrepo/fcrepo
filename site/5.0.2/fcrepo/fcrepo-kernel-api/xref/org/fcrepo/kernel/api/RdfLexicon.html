199     // Properties which are managed by the server but are not from managed namespaces
200     private static final Set<Property> serverManagedProperties;
201     static {
202         final ImmutableSet.Builder<Property> b = ImmutableSet.builder();
203         b.addAll(fixityProperties).addAll(ldpManagedProperties);
204         serverManagedProperties = b.build();
205     }
206 
207     private static final Predicate<Property> hasFedoraNamespace =
208         p -> !p.isAnon() && p.getNameSpace().startsWith(REPOSITORY_NAMESPACE);
209 
210     private static Predicate<Property> hasMementoNamespace =
211         p -> !p.isAnon() && p.getNameSpace().startsWith(MEMENTO_NAMESPACE);
212 
213     // Server managed properties which may be overridden by clients when the server is in "relaxed" mode
214     private static final Set<Property> relaxableProperties = of(LAST_MODIFIED_BY, LAST_MODIFIED_DATE, CREATED_BY,
215             CREATED_DATE);
216 
217     // Detects if a server managed property is allowed to be updated in "relaxed" mode
218     public static final Predicate<Property> isRelaxed =
219             p -> relaxableProperties.contains(p)
220                     && ("relaxed".equals(System.getProperty(SERVER_MANAGED_PROPERTIES_MODE)));
221 
222     /**
223      * Detects whether an RDF property is managed by the repository.
224      */
225     public static final Predicate<Property> isManagedPredicate =
226             hasFedoraNamespace.or(hasMementoNamespace).or(p -> serverManagedProperties.contains(p));
