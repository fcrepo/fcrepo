20  import static org.fcrepo.kernel.api.FedoraTypes.FEDORA_RESOURCE;
21  import static org.fcrepo.kernel.api.FedoraTypes.FEDORA_TIME_MAP;
22  import static org.fcrepo.kernel.api.FedoraTypes.MEMENTO_ORIGINAL;
23  import static org.fcrepo.kernel.api.RdfLexicon.LDPCV_TIME_MAP;
24  import static org.modeshape.jcr.api.JcrConstants.NT_FOLDER;
25  import static org.slf4j.LoggerFactory.getLogger;

27  import javax.jcr.Node;
28  import javax.jcr.RepositoryException;
29  import org.fcrepo.kernel.api.FedoraSession;
30  import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
31  import org.fcrepo.kernel.api.exception.ResourceTypeException;
32  import org.fcrepo.kernel.api.models.FedoraTimeMap;
33  import org.fcrepo.kernel.api.services.TimeMapService;
34  import org.fcrepo.kernel.modeshape.FedoraTimeMapImpl;
35  import org.slf4j.Logger;
36  import org.springframework.stereotype.Component;

38  
39  /**
40   * Service for creating and retrieving {@link org.fcrepo.kernel.api.models.FedoraTimeMap} without using the JCR API.
41   *
42   * @author bbpennel
43   */
44  @Component
45  public class TimeMapServiceImpl extends AbstractService implements TimeMapService {
46  
47      private static final Logger LOGGER = getLogger(TimeMapServiceImpl.class);
48  
49      @Override
50      public FedoraTimeMap find(final FedoraSession session, final String path) {
51          final String ldpcvPath = getLdpcvPath(path);
52  
53          return cast(findNode(session, ldpcvPath));
54      }
55  
56      @Override
57      public FedoraTimeMap findOrCreate(final FedoraSession session, final String path) {
58          try {
59              // Add fedora:timemap to path if not present
60              final String ldpcvPath = getLdpcvPath(path);
61  
62              final Node node = findOrCreateNode(session, ldpcvPath, NT_FOLDER);
63  
64              if (node.isNew()) {
65                  LOGGER.debug("Created TimeMap LDPCv {}", node.getPath());
66  
67                  // add mixin type fedora:Resource
68                  if (node.canAddMixin(FEDORA_RESOURCE)) {
69                      node.addMixin(FEDORA_RESOURCE);
70                  }
71  
72                  // add mixin type fedora:TimeMap
73                  if (node.canAddMixin(FEDORA_TIME_MAP)) {
74                      node.addMixin(FEDORA_TIME_MAP);
75                  }
76  
77                  // Set reference from timegate/map to original resource
78                  node.setProperty(MEMENTO_ORIGINAL, node.getParent());
79              }
80  
81              return new FedoraTimeMapImpl(node);
82          } catch (final RepositoryException e) {
83              throw new RepositoryRuntimeException(e);
84          }
85      }
86  
87      @Override
88      public boolean exists(final FedoraSession session, final String path) {
89          final String ldpcvPath = getLdpcvPath(path);
90          return super.exists(session, ldpcvPath);
91      }
92  
93      private static final String getLdpcvPath(final String path) {
94          if (path.endsWith("/" + LDPCV_TIME_MAP)) {
95              return path;
96          } else {
97              return path.replaceFirst("/*$", "") + "/" + LDPCV_TIME_MAP;
98          }
99      }
100 
101     private FedoraTimeMap cast(final Node node) {
102         assertIsType(node);
103         return new FedoraTimeMapImpl(node);
104     }
105 
106     private static void assertIsType(final Node node) {
107         if (!FedoraTimeMapImpl.hasMixin(node)) {
108             throw new ResourceTypeException(node + " can not be used as a timemap");
109         }
110     }
111 }
