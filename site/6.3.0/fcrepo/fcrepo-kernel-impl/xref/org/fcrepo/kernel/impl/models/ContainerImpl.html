6   package org.fcrepo.kernel.impl.models;
7   
8   import org.fcrepo.kernel.api.Transaction;
9   import org.fcrepo.kernel.api.cache.UserTypesCache;
10  import org.fcrepo.kernel.api.identifiers.FedoraId;
11  import org.fcrepo.kernel.api.models.Container;
12  import org.fcrepo.kernel.api.models.FedoraResource;
13  import org.fcrepo.kernel.api.models.ResourceFactory;
14  import org.fcrepo.persistence.api.PersistentStorageSessionManager;
15  
16  import java.net.URI;
17  import java.util.List;
18  import java.util.stream.Stream;
19  
20  import static org.fcrepo.kernel.api.RdfLexicon.CONTAINER;
21  import static org.fcrepo.kernel.api.RdfLexicon.FEDORA_CONTAINER;
22  import static org.fcrepo.kernel.api.RdfLexicon.RDF_SOURCE;
23  
24  
25  /**
26   * Implementation of an LDP Container resource
27   *
28   * @author bbpennel
29   */
30  public class ContainerImpl extends FedoraResourceImpl implements Container {
31  
32      private static final URI RDF_SOURCE_URI = URI.create(RDF_SOURCE.toString());
33      private static final URI CONTAINER_URI = URI.create(CONTAINER.toString());
34      private static final URI FEDORA_CONTAINER_URI = URI.create(FEDORA_CONTAINER.toString());
35  
36      /**
37       * Construct the container
38       *
39       * @param fedoraID internal identifier
40       * @param transaction transaction
41       * @param pSessionManager session manager
42       * @param resourceFactory resource factory
43       * @param userTypesCache the user types cache
44       */
45      public ContainerImpl(final FedoraId fedoraID,
46                           final Transaction transaction,
47                           final PersistentStorageSessionManager pSessionManager,
48                           final ResourceFactory resourceFactory,
49                           final UserTypesCache userTypesCache) {
50          super(fedoraID, transaction, pSessionManager, resourceFactory, userTypesCache);
51      }
52  
53      @Override
54      public FedoraResource getDescribedResource() {
55          return this;
56      }
57  
58      @Override
59      public List<URI> getSystemTypes(final boolean forRdf) {
60          var types = resolveSystemTypes(forRdf);
61  
62          if (types == null) {
63              types = super.getSystemTypes(forRdf);
64              types.add(RDF_SOURCE_URI);
65              types.add(CONTAINER_URI);
66              types.add(FEDORA_CONTAINER_URI);
67          }
68  
69          return types;
70      }
71  
72      @Override
73      public Stream<FedoraResource> getChildren(final Boolean recursive) {
74          return resourceFactory.getChildren(transaction, fedoraId);
75      }
76  }
