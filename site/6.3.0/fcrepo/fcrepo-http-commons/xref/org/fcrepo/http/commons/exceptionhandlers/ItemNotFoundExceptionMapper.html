6   package org.fcrepo.http.commons.exceptionhandlers;
7   
8   import org.fcrepo.kernel.api.exception.ItemNotFoundException;
9   import org.slf4j.Logger;
10  
11  import javax.ws.rs.core.Response;
12  import javax.ws.rs.ext.ExceptionMapper;
13  import javax.ws.rs.ext.Provider;
14  
15  import static org.slf4j.LoggerFactory.getLogger;
16  
17  
18  /**
19   * Catch ItemNotFoundException
20   *
21   * @author pwinckles
22   */
23  @Provider
24  public class ItemNotFoundExceptionMapper implements
25          ExceptionMapper<ItemNotFoundException>, ExceptionDebugLogging {
26  
27      private static final Logger LOGGER =
28          getLogger(ItemNotFoundExceptionMapper.class);
29  
30      @Override
31      public Response toResponse(final ItemNotFoundException e) {
32          debugException(this, e, LOGGER);
33          return Response.status(Response.Status.NOT_FOUND).
34                  entity("Error: " + e.getMessage()).build();
35      }
36  }
