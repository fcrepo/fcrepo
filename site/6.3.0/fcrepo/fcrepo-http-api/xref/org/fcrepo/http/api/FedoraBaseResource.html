6   package org.fcrepo.http.api;
7   
8   import org.fcrepo.common.db.DbTransactionExecutor;
9   import org.fcrepo.http.commons.AbstractResource;
10  import org.fcrepo.http.commons.api.rdf.HttpIdentifierConverter;
11  import org.fcrepo.http.commons.session.TransactionProvider;
12  import org.fcrepo.kernel.api.Transaction;
13  import org.fcrepo.kernel.api.TransactionManager;
14  import org.fcrepo.kernel.api.exception.PathNotFoundException;
15  import org.fcrepo.kernel.api.identifiers.FedoraId;
16  import org.fcrepo.kernel.api.models.FedoraResource;
17  import org.fcrepo.kernel.api.models.ResourceFactory;
18  import org.fcrepo.kernel.api.models.ResourceHelper;
19  import org.slf4j.Logger;
20  
21  import javax.inject.Inject;
22  import javax.servlet.http.HttpServletRequest;
23  import javax.ws.rs.core.Context;
24  import javax.ws.rs.core.SecurityContext;
25  import java.security.Principal;
26  
27  import static org.slf4j.LoggerFactory.getLogger;
28  
29  /**
30   * @author cabeer
31   * @since 10/5/14
32   */
33  abstract public class FedoraBaseResource extends AbstractResource {
34  
35      private static final Logger LOGGER = getLogger(FedoraBaseResource.class);
36  
37      @Context
38      protected SecurityContext securityContext;
39  
40      @Inject
41      protected ResourceFactory resourceFactory;
42  
43      @Inject
44      protected ResourceHelper resourceHelper;
45  
46      @Context protected HttpServletRequest servletRequest;
47  
48      @Inject
49      protected TransactionManager txManager;
50  
51      @Inject
52      protected DbTransactionExecutor dbTransactionExecutor;
53  
54      private TransactionProvider txProvider;
55  
56      private Transaction transaction;
57  
58      protected HttpIdentifierConverter identifierConverter;
59  
60      protected HttpIdentifierConverter identifierConverter() {
61          if (identifierConverter == null) {
62              identifierConverter = new HttpIdentifierConverter(
63                      uriInfo.getBaseUriBuilder().clone().path(FedoraLdp.class));
64          }
65  
66          return identifierConverter;
67      }
68  
69      /**
70       * Gets a fedora resource by id. Uses the provided transaction if it is uncommitted,
71       * or uses a new transaction.
72       *
73       * @param transaction the fedora transaction
74       * @param fedoraId    identifier of the resource
75       * @return the requested FedoraResource
76       */
77      protected FedoraResource getFedoraResource(final Transaction transaction, final FedoraId fedoraId)
78              throws PathNotFoundException {
79          return resourceFactory.getResource(transaction, fedoraId);
80      }
81  
82      /**
83       * @param transaction the transaction in which to check
84       * @param fedoraId identifier of the object to check
85       * @param includeDeleted Whether to check for deleted resources too.
86       * @return Returns true if an object with the provided id exists
87       */
88      protected boolean doesResourceExist(final Transaction transaction, final FedoraId fedoraId,
89                                          final boolean includeDeleted) {
90          return resourceHelper.doesResourceExist(transaction, fedoraId, includeDeleted);
91      }
92  
93      /**
94       *
95       * @param transaction the transaction in which to check
96       * @param fedoraId identifier of the object to check
97       * @return Returns true if object does not exist but whose ID starts other resources that do exist.
98       */
99      protected boolean isGhostNode(final Transaction transaction, final FedoraId fedoraId) {
100         return resourceHelper.isGhostNode(transaction, fedoraId);
101     }
102 
103     protected String getUserPrincipal() {
104         final Principal p = securityContext.getUserPrincipal();
105         return p == null ? null : p.getName();
106     }
107 
108     protected Transaction transaction() {
109         if (transaction == null) {
110             txProvider = new TransactionProvider(txManager, servletRequest,
111                     uriInfo.getBaseUri(), fedoraPropsConfig.getJmsBaseUrl());
112             transaction = txProvider.provide();
113         }
114         return transaction;
115     }
116 
117     /**
118      * Executes the runnable within a DB transaction that will retry entire block on MySQL deadlock exceptions.
119      * If the runnable throws an exception, after completing any retires, then the Fedora transaction is marked
120      * as failed.
121      *
122      * @param action the code to execute
123      */
124     protected void doInDbTxWithRetry(final Runnable action) {
125         try {
126             dbTransactionExecutor.doInTxWithRetry(action);
127         } catch (RuntimeException e) {
128             transaction().fail();
129             throw e;
130         }
131     }
132 
133     /**
134      * Executes the runnable within a DB transaction. The block is NOT retried on MySQL deadlock exceptions.
135      * If the runnable throws an exception, then the Fedora transaction is marked as failed.
136      *
137      * @param action the code to execute
138      */
139     protected void doInDbTx(final Runnable action) {
140         try {
141             dbTransactionExecutor.doInTx(action);
142         } catch (RuntimeException e) {
143             transaction().fail();
144             throw e;
145         }
146     }
147 
148 }
