006package org.fcrepo.http.api;
007
008import static org.fcrepo.kernel.api.RdfLexicon.EXTERNAL_CONTENT;
009import static org.slf4j.LoggerFactory.getLogger;
010
011import java.util.List;
012import java.util.stream.Collectors;
013
014import org.fcrepo.kernel.api.exception.ExternalMessageBodyException;
015import org.slf4j.Logger;
016
017/**
018 * Constructs ExternalContentHandler objects from link headers
019 *
020 * @author bbpennel
021 */
022public class ExternalContentHandlerFactory {
023
024    private static final Logger LOGGER = getLogger(ExternalContentHandlerFactory.class);
025
026    private ExternalContentPathValidator validator;
027
028    /**
029     * Looks for ExternalContent link header and if it finds one it will return a new ExternalContentHandler object
030     * based on the found Link header. If multiple external content headers were found or the URI provided in the
031     * header is not a valid external content path, then an ExternalMessageBodyException will be thrown.
032     *
033     * @param links links from the request header
034     * @return External Content Handler Object if Link header found, else null
035     * @throws ExternalMessageBodyException thrown if more than one external content link was provided, or if the URL
036     *         of the header was not a valid external content path.
037     */
038    public ExternalContentHandler createFromLinks(final List<String> links) throws ExternalMessageBodyException {
039        if (links == null) {
040            return null;
041        }
042
043        final List<String> externalContentLinks = links.stream()
044                .filter(x -> x.contains(EXTERNAL_CONTENT.toString()))
045                .collect(Collectors.toList());
046
047        if (externalContentLinks.size() > 1) {
048            // got a problem, you can only have one ExternalContent links
049            throw new ExternalMessageBodyException("More then one External Content Link header in request");
050        } else if (externalContentLinks.size() == 1) {
051            final String link = externalContentLinks.get(0);
052            final String uri = getUriString(link);
053            // Validate that the URI is valid according to allowed set of external uris
054            try {
055                validator.validate(uri);
056            } catch (final ExternalMessageBodyException e) {
057                LOGGER.warn("Rejected invalid external path {}", uri);
058                throw e;
059            }
060
061            return new ExternalContentHandler(link);
062        }
063
064        return null;
065    }
066
067    private static String getUriString(final String link) {
068        final String value = link.trim();
069        if (value.startsWith("<")) {
070            final int gtIndex = value.indexOf('>');
071            if (gtIndex != -1) {
072                return value.substring(1, gtIndex).trim();
073            } else {
074                throw new IllegalArgumentException("Missing token > in " + value);
075            }
076        } else {
077            throw new IllegalArgumentException("Missing starting token < in " + value);
078        }
079    }
080
081    /**
082     * Set the external content path validator
083     *
084     * @param validator validator
085     */
086    public void setValidator(final ExternalContentPathValidator validator) {
087        this.validator = validator;
088    }
089}
