006package org.fcrepo.http.api;
007
008import static javax.ws.rs.core.MediaType.APPLICATION_OCTET_STREAM_TYPE;
009import static org.apache.http.HttpHeaders.CONTENT_LENGTH;
010import static org.apache.http.HttpHeaders.CONTENT_TYPE;
011import static org.apache.http.HttpStatus.SC_OK;
012import static org.slf4j.LoggerFactory.getLogger;
013
014import org.apache.http.Header;
015import org.apache.http.client.methods.HttpHead;
016import org.apache.http.client.methods.CloseableHttpResponse;
017import org.apache.http.impl.client.CloseableHttpClient;
018import org.apache.http.impl.client.HttpClients;
019import org.fcrepo.kernel.api.exception.ExternalContentAccessException;
020import org.fcrepo.kernel.api.exception.ExternalMessageBodyException;
021import javax.ws.rs.core.Link;
022
023import java.io.BufferedInputStream;
024import java.io.FileInputStream;
025import java.io.IOException;
026import java.io.InputStream;
027import java.net.URI;
028import java.nio.file.Files;
029import java.nio.file.Path;
030import java.nio.file.Paths;
031import java.util.Map;
032
033import org.fcrepo.kernel.api.models.ExternalContent;
034import org.slf4j.Logger;
035
036/**
037 * This class is a helper for dealing with the External Content Link header and External Content itself, in the case
038 * of handling="copy". This class will verify that an External Content Link header is formatted correctly and
039 * help parse it, delivering parts of it when asked.
040 *
041 * @author bseeger
042 * @since 5/7/2018
043 */
044public class ExternalContentHandler implements ExternalContent {
045
046    private static final Logger LOGGER = getLogger(FedoraLdp.class);
047
048    private final static String HANDLING = "handling";
049    private final static String EXT_CONTENT_TYPE = "type";
050
051    private final Link link;
052    private final String handling;
053    private String contentType;
054    private Long contentSize;
055
056    /* link header for external content should look like this:
057          Link: <http://example.org/some/content>;
058          rel="http://fedora.info/definitions/fcrepo#ExternalContent";
059          handling="proxy";
060          type="image/tiff"
061    */
062
063    /**
064     *  Construct an ExternalContentHandler (helper)
065     *
066     *  @param linkHeader actual link header from request
067     */
068    protected ExternalContentHandler(final String linkHeader) {
069        // if it parses, then we're mostly good to go.
070        link = parseLinkHeader(linkHeader);
071
072        final Map<String, String> map = link.getParams();
073        // handling will be in the map, where as content type may not be
074        handling = map.get(HANDLING).toLowerCase();
075        // Retrieve details directly from the content
076        retrieveContentDetails();
077        final var type = map.get(EXT_CONTENT_TYPE) != null ? map.get(EXT_CONTENT_TYPE).toLowerCase() : null;
078        if (type != null) {
079            contentType = type;
080        } else if (contentType == null) {
081            LOGGER.debug("Defaulting to octet stream for media type");
082            contentType = APPLICATION_OCTET_STREAM_TYPE.toString();
083        }
084
085        if (contentSize == null) {
086            contentSize = -1L;
087        }
088    }
089
090    @Override
091    public String getContentType() {
092        return contentType;
093    }
094
095    @Override
096    public long getContentSize() {
097        return contentSize;
098    }
099
100    @Override
101    public String getHandling() {
102        return handling;
103    }
104
105    @Override
106    public String getURL() {
107        return link.getUri().toString();
108    }
109
110    @Override
111    public URI getURI() {
112        return link.getUri();
113    }
114
115    @Override
116    public boolean isCopy() {
117        return COPY.equals(handling);
118    }
119
120    @Override
121    public boolean isRedirect() {
122        return REDIRECT.equals(handling);
123    }
124
125    @Override
126    public boolean isProxy() {
127        return PROXY.equals(handling);
128    }
129
130    @Override
131    public InputStream fetchExternalContent() {
132
133        final URI uri = link.getUri();
134        final String scheme = uri.getScheme();
135        LOGGER.debug("scheme is {}", scheme);
136        if (scheme != null) {
137            try {
138                if (scheme.equals("file")) {
139                    return new BufferedInputStream(new FileInputStream(uri.getPath()));
140                } else if (scheme.equals("http") || scheme.equals("https")) {
141                    return new BufferedInputStream(uri.toURL().openStream());
142                }
143            } catch (final IOException e) {
144                throw new ExternalContentAccessException("Failed to read external content from " + uri, e);
145            }
146        }
147        return null;
148    }
149
150    /**
151     * Validate that an external content link header is appropriately formatted
152     * @param link to be validated
153     * @return Link object if the header is formatted correctly, else null
154     * @throws ExternalMessageBodyException on error
155     */
156    private Link parseLinkHeader(final String link) throws ExternalMessageBodyException {
157        final Link realLink;
158
159        try {
160            realLink = Link.valueOf(link);
161            final String handling = realLink.getParams().get(HANDLING);
162            if (handling == null || !handling.matches("(?i)" + PROXY + "|" + COPY + "|" + REDIRECT)) {
163                // error
164                throw new ExternalMessageBodyException(
165                        "Link header formatted incorrectly: 'handling' parameter incorrect or missing");
166            }
167        } catch (final Exception e) {
168            throw new ExternalMessageBodyException("External content link header url is malformed");
169        }
170        return realLink;
171    }
172
173    private void retrieveContentDetails() {
174        final URI uri = getURI();
175        final String scheme = uri.getScheme().toLowerCase();
176
177        if ("file".equals(scheme)) {
178            final Path path = Paths.get(uri);
179            try {
180                contentSize = Files.size(path);
181            } catch (final IOException e) {
182                throw new ExternalMessageBodyException("Unable to access external binary at URI " + uri, e);
183            }
184        } else if ("http".equals(scheme) || "https".equals(scheme)) {
185            try (final CloseableHttpClient httpClient = HttpClients.createDefault()) {
186                final HttpHead httpHead = new HttpHead(uri);
187                httpHead.setHeader("Accept-Encoding", "identity");
188                try (final CloseableHttpResponse response = httpClient.execute(httpHead)) {
189                    if (response.getStatusLine().getStatusCode() != SC_OK) {
190                        throw new ExternalMessageBodyException("Unable to access external binary at URI " + uri
191                                + " received response " + response.getStatusLine().getStatusCode());
192                    }
193
194                    final Header typeHeader = response.getFirstHeader(CONTENT_TYPE);
195                    if (typeHeader != null) {
196                        contentType = typeHeader.getValue();
197                    }
198                    final Header sizeHeader = response.getFirstHeader(CONTENT_LENGTH);
199                    if (sizeHeader != null) {
200                        contentSize = Long.parseLong(sizeHeader.getValue());
201                    }
202                }
203            } catch (final IOException e) {
204                throw new ExternalMessageBodyException("Unable to access external binary at URI " + uri, e);
205            }
206        }
207    }
208}
