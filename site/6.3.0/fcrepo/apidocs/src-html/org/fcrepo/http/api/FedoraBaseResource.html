006package org.fcrepo.http.api;
007
008import org.fcrepo.common.db.DbTransactionExecutor;
009import org.fcrepo.http.commons.AbstractResource;
010import org.fcrepo.http.commons.api.rdf.HttpIdentifierConverter;
011import org.fcrepo.http.commons.session.TransactionProvider;
012import org.fcrepo.kernel.api.Transaction;
013import org.fcrepo.kernel.api.TransactionManager;
014import org.fcrepo.kernel.api.exception.PathNotFoundException;
015import org.fcrepo.kernel.api.identifiers.FedoraId;
016import org.fcrepo.kernel.api.models.FedoraResource;
017import org.fcrepo.kernel.api.models.ResourceFactory;
018import org.fcrepo.kernel.api.models.ResourceHelper;
019import org.slf4j.Logger;
020
021import javax.inject.Inject;
022import javax.servlet.http.HttpServletRequest;
023import javax.ws.rs.core.Context;
024import javax.ws.rs.core.SecurityContext;
025import java.security.Principal;
026
027import static org.slf4j.LoggerFactory.getLogger;
028
029/**
030 * @author cabeer
031 * @since 10/5/14
032 */
033abstract public class FedoraBaseResource extends AbstractResource {
034
035    private static final Logger LOGGER = getLogger(FedoraBaseResource.class);
036
037    @Context
038    protected SecurityContext securityContext;
039
040    @Inject
041    protected ResourceFactory resourceFactory;
042
043    @Inject
044    protected ResourceHelper resourceHelper;
045
046    @Context protected HttpServletRequest servletRequest;
047
048    @Inject
049    protected TransactionManager txManager;
050
051    @Inject
052    protected DbTransactionExecutor dbTransactionExecutor;
053
054    private TransactionProvider txProvider;
055
056    private Transaction transaction;
057
058    protected HttpIdentifierConverter identifierConverter;
059
060    protected HttpIdentifierConverter identifierConverter() {
061        if (identifierConverter == null) {
062            identifierConverter = new HttpIdentifierConverter(
063                    uriInfo.getBaseUriBuilder().clone().path(FedoraLdp.class));
064        }
065
066        return identifierConverter;
067    }
068
069    /**
070     * Gets a fedora resource by id. Uses the provided transaction if it is uncommitted,
071     * or uses a new transaction.
072     *
073     * @param transaction the fedora transaction
074     * @param fedoraId    identifier of the resource
075     * @return the requested FedoraResource
076     */
077    protected FedoraResource getFedoraResource(final Transaction transaction, final FedoraId fedoraId)
078            throws PathNotFoundException {
079        return resourceFactory.getResource(transaction, fedoraId);
080    }
081
082    /**
083     * @param transaction the transaction in which to check
084     * @param fedoraId identifier of the object to check
085     * @param includeDeleted Whether to check for deleted resources too.
086     * @return Returns true if an object with the provided id exists
087     */
088    protected boolean doesResourceExist(final Transaction transaction, final FedoraId fedoraId,
089                                        final boolean includeDeleted) {
090        return resourceHelper.doesResourceExist(transaction, fedoraId, includeDeleted);
091    }
092
093    /**
094     *
095     * @param transaction the transaction in which to check
096     * @param fedoraId identifier of the object to check
097     * @return Returns true if object does not exist but whose ID starts other resources that do exist.
098     */
099    protected boolean isGhostNode(final Transaction transaction, final FedoraId fedoraId) {
100        return resourceHelper.isGhostNode(transaction, fedoraId);
101    }
102
103    protected String getUserPrincipal() {
104        final Principal p = securityContext.getUserPrincipal();
105        return p == null ? null : p.getName();
106    }
107
108    protected Transaction transaction() {
109        if (transaction == null) {
110            txProvider = new TransactionProvider(txManager, servletRequest,
111                    uriInfo.getBaseUri(), fedoraPropsConfig.getJmsBaseUrl());
112            transaction = txProvider.provide();
113        }
114        return transaction;
115    }
116
117    /**
118     * Executes the runnable within a DB transaction that will retry entire block on MySQL deadlock exceptions.
119     * If the runnable throws an exception, after completing any retires, then the Fedora transaction is marked
120     * as failed.
121     *
122     * @param action the code to execute
123     */
124    protected void doInDbTxWithRetry(final Runnable action) {
125        try {
126            dbTransactionExecutor.doInTxWithRetry(action);
127        } catch (RuntimeException e) {
128            transaction().fail();
129            throw e;
130        }
131    }
132
133    /**
134     * Executes the runnable within a DB transaction. The block is NOT retried on MySQL deadlock exceptions.
135     * If the runnable throws an exception, then the Fedora transaction is marked as failed.
136     *
137     * @param action the code to execute
138     */
139    protected void doInDbTx(final Runnable action) {
140        try {
141            dbTransactionExecutor.doInTx(action);
142        } catch (RuntimeException e) {
143            transaction().fail();
144            throw e;
145        }
146    }
147
148}
