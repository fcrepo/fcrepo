006package org.fcrepo.http.api;
007
008import static javax.ws.rs.core.MediaType.APPLICATION_JSON;
009import static javax.ws.rs.core.Response.Status.BAD_REQUEST;
010import static javax.ws.rs.core.Response.ok;
011import static javax.ws.rs.core.Response.status;
012import static org.apache.commons.lang3.ObjectUtils.isEmpty;
013import static org.slf4j.LoggerFactory.getLogger;
014
015import java.util.List;
016
017import javax.ws.rs.GET;
018import javax.ws.rs.Path;
019import javax.ws.rs.Produces;
020import javax.ws.rs.QueryParam;
021import javax.ws.rs.core.Response;
022
023import io.micrometer.core.annotation.Timed;
024import org.fcrepo.stats.api.AggregatedRepositoryStatsResults;
025import org.fcrepo.stats.api.RepositoryStats;
026import org.fcrepo.stats.api.RepositoryStatsParameters;
027import org.slf4j.Logger;
028import org.springframework.beans.factory.annotation.Autowired;
029import org.springframework.beans.factory.annotation.Qualifier;
030import org.springframework.context.annotation.Scope;
031
032/**
033 * An HTTP endpoint for retrieving statistics related to the repository.
034 *
035 * @author dbernstein
036 * @since 2022-01-27
037 */
038@Timed
039@Scope("request")
040@Path("/fcr:stats")
041public class FedoraRepositoryStats extends FedoraBaseResource {
042
043    private static final Logger LOGGER = getLogger(FedoraRepositoryStats.class);
044
045    @Autowired
046    @Qualifier("stats")
047    private RepositoryStats repositoryStats;
048
049    /**
050     * Default JAX-RS entry point
051     */
052    public FedoraRepositoryStats() {
053        super();
054    }
055
056
057    /**
058     * Query summary info
059     *
060     * @return
061     */
062    @GET
063    @Produces({APPLICATION_JSON + ";qs=1.0",
064            APPLICATION_JSON})
065    public Response getStats() {
066        final var builder = ok();
067        final var statsParams = new RepositoryStatsParameters();
068        final var totalResourceCount = repositoryStats.getResourceCount(statsParams);
069        final var binaryResources = repositoryStats.getByMimeTypes(statsParams);
070        final var rdfTypes = repositoryStats.getByRdfType(statsParams);
071        final var aggregatedStats = new AggregatedRepositoryStatsResults();
072        aggregatedStats.setResourceCount(totalResourceCount.getResourceCount());
073        aggregatedStats.setBinaries(binaryResources);
074        aggregatedStats.setAllResources(rdfTypes);
075        return builder.entity(aggregatedStats).build();
076
077    }
078
079    /**
080     * Method for querying rdf type stats
081     *
082     * @param rdfTypes A list of rdf types by which to limit the results. By default show all rdf types.
083     * @return
084     */
085    @GET
086    @Path("/rdf-types")
087    @Produces({APPLICATION_JSON + ";qs=1.0",
088            APPLICATION_JSON})
089    public Response getStatsByRdfType(@QueryParam(value = "rdf_type") final List<String> rdfTypes) {
090        if (listHasSingleBlankEntry(rdfTypes)) {
091            return status(BAD_REQUEST).build();
092        }
093        final var builder = ok();
094        final var statsParams = new RepositoryStatsParameters();
095        statsParams.setRdfTypes(rdfTypes);
096        final var results = repositoryStats.getByRdfType(statsParams);
097        return builder.entity(results).build();
098    }
099
100    private boolean listHasSingleBlankEntry(final List<String> list) {
101        return !isEmpty(list) && list.size() == 1 && list.get(0).trim().equals("");
102    }
103
104    /**
105     * Method for querying binary stats
106     *
107     * @param mimeTypes A list of mime types by which to limit the results. By default show all binary mime types.
108     * @return
109     */
110    @GET
111    @Path("/binaries")
112    @Produces({APPLICATION_JSON + ";qs=1.0",
113            APPLICATION_JSON})
114    public Response getBinaryStats(@QueryParam(value = "mime_type") final List<String> mimeTypes ) {
115        if (listHasSingleBlankEntry(mimeTypes)) {
116            return status(BAD_REQUEST).build();
117        }
118
119        final var builder = ok();
120        final var statsParams = new RepositoryStatsParameters();
121        statsParams.setMimeTypes(mimeTypes);
122        final var results = repositoryStats.getByMimeTypes(statsParams);
123        return builder.entity(results).build();
124    }
125}
