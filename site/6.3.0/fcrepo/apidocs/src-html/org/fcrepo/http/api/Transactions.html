006package org.fcrepo.http.api;
007
008import static java.util.Date.from;
009import static javax.ws.rs.core.Response.created;
010import static javax.ws.rs.core.Response.noContent;
011import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
012import static org.fcrepo.http.commons.session.TransactionConstants.ATOMIC_EXPIRES_HEADER;
013import static org.fcrepo.http.commons.session.TransactionConstants.EXPIRES_RFC_1123_FORMATTER;
014import static org.fcrepo.http.commons.session.TransactionConstants.TX_COMMIT_REL;
015import static org.fcrepo.http.commons.session.TransactionConstants.TX_PREFIX;
016import static org.fcrepo.kernel.api.FedoraTypes.FEDORA_ID_PREFIX;
017import static org.slf4j.LoggerFactory.getLogger;
018
019import java.net.URI;
020import java.net.URISyntaxException;
021
022import javax.ws.rs.DELETE;
023import javax.ws.rs.GET;
024import javax.ws.rs.POST;
025import javax.ws.rs.PUT;
026import javax.ws.rs.Path;
027import javax.ws.rs.PathParam;
028import javax.ws.rs.core.Link;
029import javax.ws.rs.core.Response;
030import javax.ws.rs.core.Response.Status;
031
032import io.micrometer.core.annotation.Timed;
033import org.fcrepo.kernel.api.Transaction;
034import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
035import org.fcrepo.kernel.api.exception.TransactionClosedException;
036import org.fcrepo.kernel.api.exception.TransactionNotFoundException;
037import org.slf4j.Logger;
038import org.springframework.context.annotation.Scope;
039
040/**
041 * The rest interface for transaction management. The interfaces
042 * allows for creation, commit and rollback of transactions.
043 *
044 * @author awoods
045 * @author gregjan
046 * @author mohideen
047 */
048@Timed
049@Scope("prototype")
050@Path("/fcr:tx")
051public class Transactions extends FedoraBaseResource {
052
053    private static final Logger LOGGER = getLogger(Transactions.class);
054
055    /**
056     * Get the status of an existing transaction
057     *
058     * @param txId id of the transaction
059     * @return 204 no content if status retrieved, 410 gone if transaction doesn't exist.
060     */
061    @GET
062    @Path("{transactionId}")
063    public Response getTransactionStatus(@PathParam("transactionId") final String txId) {
064        // Retrieve the tx provided via the path
065        final Transaction tx;
066        try {
067            tx = txManager.get(txId);
068        } catch (final TransactionNotFoundException e) {
069            return Response.status(Status.NOT_FOUND).build();
070        } catch (final TransactionClosedException e) {
071            return Response.status(Status.GONE)
072                    .entity(e.getMessage())
073                    .type(TEXT_PLAIN_WITH_CHARSET)
074                    .build();
075        }
076
077        LOGGER.info("Checking transaction status'{}'", tx.getId());
078
079        return Response.status(Status.NO_CONTENT)
080                .header(ATOMIC_EXPIRES_HEADER, EXPIRES_RFC_1123_FORMATTER.format(tx.getExpires()))
081                .build();
082    }
083
084    /**
085     * Refresh an existing transaction
086     *
087     * @param txId id of the transaction
088     * @return 204 no content if successfully refreshed, 410 gone if transaction doesn't exist.
089     */
090    @POST
091    @Path("{transactionId}")
092    public Response refreshTransaction(@PathParam("transactionId") final String txId) {
093        // Retrieve the tx provided via the path
094        final Transaction tx;
095        try {
096            tx = txManager.get(txId);
097        } catch (final TransactionNotFoundException e) {
098            return Response.status(Status.NOT_FOUND).build();
099        } catch (final TransactionClosedException e) {
100            return Response.status(Status.GONE)
101                    .entity(e.getMessage()).type(TEXT_PLAIN_WITH_CHARSET)
102                    .build();
103        }
104
105        tx.refresh();
106        LOGGER.info("Refreshed transaction '{}'", tx.getId());
107
108        return Response.status(Status.NO_CONTENT)
109                .header(ATOMIC_EXPIRES_HEADER, EXPIRES_RFC_1123_FORMATTER.format(tx.getExpires()))
110                .build();
111    }
112
113    /**
114     * Create a new transaction resource and add it to the registry
115     *
116     * @return 201 with the transaction id and expiration date
117     * @throws URISyntaxException if URI syntax exception occurred
118     */
119    @POST
120    public Response createTransaction() throws URISyntaxException {
121        final Transaction tx = transaction();
122        tx.setShortLived(false);
123
124        LOGGER.info("Created transaction '{}'", tx.getId());
125        final var externalId = identifierConverter()
126                .toExternalId(FEDORA_ID_PREFIX + "/" + TX_PREFIX + tx.getId());
127        final var res = created(new URI(externalId));
128        res.expires(from(tx.getExpires()));
129
130        final var commitUri = URI.create(externalId);
131        final var commitLink = Link.fromUri(commitUri).rel(TX_COMMIT_REL).build();
132        res.links(commitLink);
133
134        return res.build();
135    }
136
137    /**
138     * Commit a transaction resource
139     *
140     * @param txId the transaction id
141     * @return 204
142     */
143    @PUT
144    @Path("{transactionId}")
145    public Response commit(@PathParam("transactionId") final String txId) {
146        try {
147            final Transaction transaction = txManager.get(txId);
148            LOGGER.info("Committing transaction '{}'", transaction.getId());
149            transaction.commit();
150            return noContent().build();
151        } catch (final TransactionNotFoundException e) {
152            return Response.status(Status.NOT_FOUND).build();
153        } catch (final TransactionClosedException e) {
154            return Response.status(Status.GONE)
155                    .entity(e.getMessage())
156                    .type(TEXT_PLAIN_WITH_CHARSET)
157                    .build();
158        } catch (final RepositoryRuntimeException e) {
159            return Response.status(Status.CONFLICT)
160                    .entity(e.getMessage())
161                    .type(TEXT_PLAIN_WITH_CHARSET)
162                    .build();
163        }
164    }
165
166    /**
167     * Rollback a transaction
168     *
169     * @param txId the transaction id
170     * @return 204
171     */
172    @DELETE
173    @Path("{transactionId}")
174    public Response rollback(@PathParam("transactionId") final String txId) {
175        try {
176            final Transaction transaction = txManager.get(txId);
177            LOGGER.info("Rollback transaction '{}'", transaction.getId());
178            transaction.rollback();
179            return noContent().build();
180        } catch (final TransactionNotFoundException e) {
181            return Response.status(Status.NOT_FOUND).build();
182        } catch (final TransactionClosedException e) {
183            return Response.status(Status.GONE)
184                    .entity(e.getMessage())
185                    .type(TEXT_PLAIN_WITH_CHARSET)
186                    .build();
187        } catch (final RepositoryRuntimeException e) {
188            return Response.status(Status.CONFLICT)
189                    .entity(e.getMessage())
190                    .type(TEXT_PLAIN_WITH_CHARSET)
191                    .build();
192        }
193    }
194
195}
