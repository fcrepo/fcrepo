7   package org.fcrepo.config;
8   
9   /**
10   * @author pwinckles
11   */
12  public enum ServerManagedPropsMode {
13  
14      STRICT("strict"),
15      RELAXED("relaxed");
16  
17      private final String value;
18  
19      ServerManagedPropsMode(final String value) {
20          this.value = value;
21      }
22  
23      public String getValue() {
24          return value;
25      }
26  
27      public static ServerManagedPropsMode fromString(final String value) {
28          for (final var mode : values()) {
29              if (mode.value.equalsIgnoreCase(value)) {
30                  return mode;
31              }
32          }
33          throw new IllegalArgumentException("Unknown server managed properties mode: " + value);
34      }
35  
36      @Override
37      public String toString() {
38          return value;
39      }
40  
41  }
