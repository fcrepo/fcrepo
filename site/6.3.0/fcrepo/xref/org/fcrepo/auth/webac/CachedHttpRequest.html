6   package org.fcrepo.auth.webac;

8   import java.io.BufferedReader;
9   import java.io.ByteArrayInputStream;
10  import java.io.IOException;
11  import java.io.InputStreamReader;
12  
13  import javax.servlet.ReadListener;
14  import javax.servlet.ServletInputStream;
15  import javax.servlet.ServletRequest;
16  import javax.servlet.http.HttpServletRequest;
17  import javax.servlet.http.HttpServletRequestWrapper;
18  
19  import org.apache.commons.io.IOUtils;
20  
21  /**
22   * An extension of HttpServletRequestWrapper that caches the InputStream as
23   * byte array and overrides the getInputStream to return a new InputStream
24   * object each time based on the cached byte array.
25   * 
26   * @author mohideen
27   */
28  public class CachedHttpRequest extends HttpServletRequestWrapper {
29  
30      private byte[] cachedContent;
31  
32      private BufferedReader reader;
33  
34      /**
35       * Create a new CachedSparqlRequest for the given servlet request.
36       * @param request the original servlet request
37       */
38      public CachedHttpRequest(final ServletRequest request) {
39          super((HttpServletRequest) request);
40      }
41  
42      @Override
43      public ServletInputStream getInputStream() throws IOException {
44          if (getRequest().getInputStream() != null) {
45              if (this.cachedContent == null) {
46                  this.cachedContent = IOUtils.toByteArray(getRequest().getInputStream());
47              }
48              return new CustomServletInputStream(cachedContent);
49          }
50          return null;
51      }
52  
53      @Override
54      public BufferedReader getReader() throws IOException {
55          if (this.reader == null) {
56              this.reader = new BufferedReader(new InputStreamReader(getInputStream(),
57                      getCharacterEncoding()));
58          }
59          return this.reader;
60      }
61  
62      private static class CustomServletInputStream extends ServletInputStream {
63  
64          private final ByteArrayInputStream buffer;
65  
66          public CustomServletInputStream(final byte[] contents) {
67              this.buffer = new ByteArrayInputStream(contents);
68          }
69  
70          @Override
71          public int read() {
72              return buffer.read();
73          }
74  
75          @Override
76          public boolean isFinished() {
77              return buffer.available() == 0;
78          }
79  
80          @Override
81          public boolean isReady() {
82              return true;
83          }
84  
85          @Override
86          public void setReadListener(final ReadListener listener) {
87              throw new RuntimeException("Not implemented");
88          }
89      }
90  }
