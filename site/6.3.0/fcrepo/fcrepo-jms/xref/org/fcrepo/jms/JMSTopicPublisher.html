6   package org.fcrepo.jms;
7   
8   import javax.jms.Destination;
9   import javax.jms.JMSException;
10  
11  /**
12   * Machinery to publish JMS messages when an EventBus
13   * message is received.
14   *
15   * @author barmintor
16   * @author awoods
17   */
18  public class JMSTopicPublisher extends AbstractJMSPublisher {
19  
20      private final String topicName;
21  
22      /**
23       * Create a JMS Topic with the default name of "fedora"
24       */
25      public JMSTopicPublisher() {
26          this("fedora");
27      }
28  
29      /**
30       * Create a JMS Topic with a configurable name
31       *
32       * @param topicName the name of the topic
33       */
34      public JMSTopicPublisher(final String topicName) {
35          this.topicName = topicName;
36      }
37  
38      protected Destination createDestination() throws JMSException {
39          return jmsSession.createTopic(topicName);
40      }
41  }
