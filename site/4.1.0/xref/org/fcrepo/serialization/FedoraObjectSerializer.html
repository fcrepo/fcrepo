16  package org.fcrepo.serialization;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.io.OutputStream;
21  
22  import javax.jcr.RepositoryException;
23  import javax.jcr.Session;
24  
25  import org.fcrepo.kernel.models.FedoraResource;
26  import org.fcrepo.kernel.exception.InvalidChecksumException;
27  import org.springframework.stereotype.Component;
28  
29  /**
30   * Interface for serializing/deserializing a FedoraObject
31   * to some transportable format
32   *
33   * @author cbeer
34   */
35  @Component
36  public interface FedoraObjectSerializer {
37      // Key for jcr/xml serialization
38      final String JCR_XML = "jcr/xml";
39      /**
40       * Get the key for the serializer (that will be
41       * used at the REST API to identify the format)
42       * @return serializer key
43       */
44      String getKey();
45  
46      /**
47       * Get the media type for the serialized output
48       * (so we can send the right mime type as appropriate)
49       * @return serializer output media type
50       */
51      String getMediaType();
52  
53      /**
54       * Determines whether this FedoraObjectSerializer can be used to
55       * serialize the given resource.  A FedoraObjectSerializer
56       * implementation may use any arbitrary criteria to determine
57       * which resources it can serialize, so this method should be
58       * invoked to avoid an InvalidSerializationFormatException from
59       * {@link #serialize}.
60       * @param resource
61       * @return
62       */
63      boolean canSerialize(final FedoraResource resource);
64  
65      /**
66       * Serialize a FedoraObject into some format with options for recurse
67       * and skipBinary, and write it to the given OutputStream
68       *
69       * @param obj
70       * @param out
71       * @param skipBinary
72       * @param recurse
73       * @throws RepositoryException
74       * @throws IOException
75       * @throws org.fcrepo.serialization.InvalidSerializationFormatException
76       */
77      void serialize(final FedoraResource obj, final OutputStream out, final boolean skipBinary, final boolean recurse)
78                      throws RepositoryException, IOException, InvalidSerializationFormatException;
79  
80      /**
81       * Read the given InputStream and de-serialize the content
82       * into new nodes in the given session using the given path
83       * as the parent node
84       *
85       * @param session
86       * @param path
87       * @param stream
88       * @throws IOException
89       * @throws RepositoryException
90       * @throws InvalidChecksumException
91       * @throws org.fcrepo.serialization.InvalidSerializationFormatException
92       */
93      void deserialize(final Session session, final String path,
94              final InputStream stream) throws IOException, RepositoryException,
95              InvalidChecksumException, InvalidSerializationFormatException;
96  
97  }
