16  package org.fcrepo.http.commons.domain;
17  
18  import org.glassfish.jersey.message.internal.HttpHeaderReader;
19  
20  import javax.servlet.http.HttpServletResponse;
21  import java.text.ParseException;
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  /**
26   * Parse a single prefer tag, value and any optional parameters
27   *
28   * @author cabeer
29   */
30  public class PreferTag implements Comparable<PreferTag> {
31      private final String tag;
32      private String value = "";
33      private Map<String, String> params = new HashMap<>();
34  
35      /**
36       * Create an empty PreferTag
37       * @return
38       */
39      public static PreferTag emptyTag() {
40          try {
41              return new PreferTag((String)null);
42          } catch (final ParseException e) {
43              throw new AssertionError(e);
44          }
45      }
46  
47      /**
48       * Create a new PreferTag from an existing tag
49       * @param preferTag
50       */
51      public PreferTag(final PreferTag preferTag) {
52          tag = preferTag.getTag();
53          value = preferTag.getValue();
54          params = preferTag.getParams();
55      }
56  
57      /**
58       * Parse the prefer tag and parameters out of the header
59       * @param reader
60       * @throws ParseException
61       */
62      public PreferTag(final HttpHeaderReader reader) throws ParseException {
63  
64          // Skip any white space
65          reader.hasNext();
66  
67          if (reader.hasNext()) {
68              tag = reader.nextToken();
69  
70              if (reader.hasNextSeparator('=', true)) {
71                  reader.next();
72  
73                  value = reader.nextTokenOrQuotedString();
74              }
75  
76              if (reader.hasNext()) {
77                  params = HttpHeaderReader.readParameters(reader);
78              }
79          } else {
80              tag = "";
81          }
82      }
83  
84      /**
85       * Create a blank prefer tag
86       * @param inputTag
87       */
88      public PreferTag(final String inputTag) throws ParseException {
89          this(HttpHeaderReader.newInstance(inputTag));
90      }
91  
92      /**
93       * Get the tag name
94       * @return tag name
95       */
96      public String getTag() {
97          return tag;
98      }
99  
100     /**
101      * Get the default value for the tag
102      * @return default value for the tag
103      */
104     public String getValue() {
105         return value;
106     }
107 
108     /**
109      * Get any additional parameters for the prefer tag
110      * @return additional parameters for the prefer tag
111      */
112     public Map<String,String> getParams() {
113         return params;
114     }
115 
116     /**
117      * Add appropriate response headers to indicate that the incoming preferences were acknowledged
118      * @param servletResponse
119      */
120     public void addResponseHeaders(final HttpServletResponse servletResponse) {
121         if (!value.equals("minimal")) {
122             servletResponse.addHeader("Preference-Applied", "return=representation");
123         } else {
124             servletResponse.addHeader("Preference-Applied", "return=minimal");
125         }
126         servletResponse.addHeader("Vary", "Prefer");
127     }
128 
129     /**
130      * We consider tags with the same name to be equal, because <a
131      * href="http://tools.ietf.org/html/rfc7240#page-4">the definition of Prefer headers</a> does not permit that tags
132      * with the same name be consumed except by selecting for the first appearing tag.
133      *
134      * @see java.lang.Comparable#compareTo(java.lang.Object)
135      */
136     @Override
137     public int compareTo(final PreferTag otherTag) {
138         return getTag().compareTo(otherTag.getTag());
139     }
140 
141     @Override
142     public boolean equals(final Object obj) {
143         if ((obj != null) && (obj instanceof PreferTag)) {
144             return getTag().equals(((PreferTag) obj).getTag());
145         } else {
146             return false;
147         }
148     }
149 }
