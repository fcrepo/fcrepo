39  /**
40   * A very simple {@link IdentifierConverter} which translates JCR paths into
41   * un-dereference-able Fedora subjects (by replacing JCR-specific names with
42   * Fedora names). Should not be used except in "embedded" deployments in which
43   * no publication of translated identifiers is expected!
44   *
45   * @author barmintor
46   * @author ajs6f
47   * @since May 15, 2013
48   */
49  public class DefaultIdentifierTranslator extends IdentifierConverter<Resource, FedoraResource> {
50  
51  
52      private static final NodeResourceConverter nodeResourceConverter = new NodeResourceConverter();
53  
54      /**
55       * Default namespace to use for node URIs
56       */
57      public static final String RESOURCE_NAMESPACE = "info:fedora/";
58      private final Session session;
59  
60      /**
61       * Construct the graph with a placeholder context resource
62       */
63      public DefaultIdentifierTranslator(final Session session) {
64          this.session = session;
65          setTranslationChain();
66      }
67  
68  
69      protected Converter<String, String> forward = identity();
70      protected Converter<String, String> reverse = identity();
71  
72      private void setTranslationChain() {
73  
74          for (final Converter<String, String> t : minimalTranslationChain) {
75              forward = forward.andThen(t);
76          }
77          for (final Converter<String, String> t : Lists.reverse(minimalTranslationChain)) {
78              reverse = reverse.andThen(t.reverse());
79          }
80      }
81  
82  
83      @SuppressWarnings("unchecked")
84      private static final List<Converter<String, String>> minimalTranslationChain =
85              newArrayList((Converter<String, String>) new NamespaceConverter(),
86                      (Converter<String, String>) new HashConverter()
87              );
88  
89      @Override
90      protected FedoraResource doForward(final Resource subject) {
91          try {
92              if (!inDomain(subject)) {
93                  throw new RepositoryRuntimeException("Subject " + subject + " is not in this repository");
94              }
95  
96              return nodeResourceConverter.convert(session.getNode(asString(subject)));
97          } catch (final RepositoryException e) {
98              throw new RepositoryRuntimeException(e);
99          }
100     }
101 
102     @Override
103     protected Resource doBackward(final FedoraResource resource) {
104         final String absPath = resource.getPath();
105 
106         return toDomain(absPath);
107     }
108 
109     @Override
110     public boolean inDomain(final Resource subject) {
111         return subject.isURIResource() && subject.getURI().startsWith(RESOURCE_NAMESPACE);
112     }
113 
114     @Override
115     public Resource toDomain(final String absPath) {
116         final String relativePath;
117 
118         if (absPath.startsWith("/")) {
119             relativePath = absPath.substring(1);
120         } else {
121             relativePath = absPath;
122         }
123         return createResource(RESOURCE_NAMESPACE + reverse.convert(relativePath));
124     }
125 
126     @Override
127     public String asString(final Resource subject) {
128         if (!inDomain(subject)) {
129             return null;
130         }
131 
132         final String path = subject.getURI().substring(RESOURCE_NAMESPACE.length() - 1);
133 
134         final String absPath = forward.convert(path);
135 
136         if (absPath.isEmpty()) {
137             return "/";
138         }
139         return absPath;
140     }
