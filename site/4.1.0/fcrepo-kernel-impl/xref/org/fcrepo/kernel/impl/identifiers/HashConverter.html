16  package org.fcrepo.kernel.impl.identifiers;
17  
18  import org.fcrepo.kernel.identifiers.InternalIdentifierConverter;
19  
20  /**
21   * @author cabeer
22   * @since 10/9/14
23   */
24  public class HashConverter extends InternalIdentifierConverter {
25  
26      @Override
27      protected String doForward(final String externalId) {
28          final int i = externalId.indexOf("#");
29          if (i >= 0) {
30              return externalId.substring(0, i) + "/#/" + externalId.substring(i + 1).replace("/", "%2F");
31          }
32          return externalId;
33      }
34  
35      @Override
36      protected String doBackward(final String internalId) {
37  
38          final int i = internalId.indexOf("/#/");
39  
40          if (i >= 0) {
41              return internalId.substring(0, i) + "#" + internalId.substring(i + 3).replace("%2F", "/");
42          }
43          return internalId;
44      }
45  }
