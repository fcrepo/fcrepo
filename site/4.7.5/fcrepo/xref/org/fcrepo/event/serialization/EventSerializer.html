18  package org.fcrepo.event.serialization;
19  
20  import static org.apache.jena.datatypes.xsd.XSDDatatype.XSDdateTime;
21  import static org.apache.jena.rdf.model.ModelFactory.createDefaultModel;
22  import static org.apache.jena.rdf.model.ResourceFactory.createProperty;
23  import static org.apache.jena.rdf.model.ResourceFactory.createResource;
24  import static org.apache.jena.rdf.model.ResourceFactory.createTypedLiteral;
25  import static org.apache.jena.vocabulary.RDF.type;
26  import static org.apache.jena.vocabulary.DCTerms.identifier;
27  import static org.apache.jena.vocabulary.DCTerms.isPartOf;
28  import static org.fcrepo.kernel.api.RdfLexicon.PROV_NAMESPACE;
29  import static org.fcrepo.kernel.api.observer.OptionalValues.BASE_URL;
30  import static org.fcrepo.kernel.api.observer.OptionalValues.USER_AGENT;
31  
32  import org.apache.jena.rdf.model.Model;
33  import org.apache.jena.rdf.model.Resource;
34  import org.fcrepo.kernel.api.observer.FedoraEvent;
35  
36  /**
37   * A basic serialization API for Fedora events
38   * @author acoburn
39   */
40  public interface EventSerializer {
41  
42      /**
43       * Convert an event to an Rdf Model
44       * @param evt the Fedora event
45       * @return an RDF model representing the event
46       */
47      public static Model toModel(final FedoraEvent evt) {
48          final String FOAF = "http://xmlns.com/foaf/0.1/";
49  
50          final Model model = createDefaultModel();
51          final String baseUrl = evt.getInfo().get(BASE_URL);
52          final String userAgent = evt.getInfo().get(USER_AGENT);
53          final Resource root = model.createResource((baseUrl != null ? baseUrl : "info:fedora" ) + evt.getPath());
54  
55          evt.getResourceTypes().forEach(rdfType -> {
56              root.addProperty(type, createResource(rdfType));
57          });
58  
59          if (baseUrl != null) {
60              root.addProperty(isPartOf, createResource(baseUrl));
61          }
62  
63          final Resource activity = model.createResource()
64                  .addProperty(type, createResource(PROV_NAMESPACE + "Activity"))
65                  .addProperty(identifier, createResource(evt.getEventID()))
66                  .addLiteral(createProperty(PROV_NAMESPACE, "atTime"),
67                          createTypedLiteral(evt.getDate().toString(), XSDdateTime));
68  
69          evt.getTypes().stream().map(rdfType -> rdfType.getType()).forEach(rdfType -> {
70              activity.addProperty(type, createResource(rdfType));
71          });
72  
73          root.addProperty(createProperty(PROV_NAMESPACE, "wasGeneratedBy"), activity);
74  
75          root.addProperty(createProperty(PROV_NAMESPACE, "wasAttributedTo"),
76                  model.createResource()
77                      .addProperty(type, createResource(PROV_NAMESPACE + "Person"))
78                      .addLiteral(createProperty(FOAF, "name"), evt.getUserID()));
79  
80          if (userAgent != null) {
81              root.addProperty(createProperty(PROV_NAMESPACE, "wasAttributedTo"),
82                      model.createResource()
83                          .addProperty(type, createResource(PROV_NAMESPACE + "SoftwareAgent"))
84                          .addLiteral(createProperty(FOAF, "name"), userAgent));
85          }
86  
87          return model;
88      }
89  
90      /**
91       * Serialize a FedoraEvent into a JSON String
92       * @param evt the Fedora event
93       * @return a JSON string
94       */
95      String serialize(final FedoraEvent evt);
96  }
