016package org.fcrepo.auth.roles.common;
017
018import static javax.ws.rs.core.MediaType.APPLICATION_JSON;
019
020import java.util.List;
021import java.util.Map;
022import java.util.Set;
023
024import javax.jcr.Node;
025import javax.jcr.RepositoryException;
026import javax.jcr.Session;
027import javax.servlet.http.HttpServletRequest;
028import javax.ws.rs.Consumes;
029import javax.ws.rs.DELETE;
030import javax.ws.rs.GET;
031import javax.ws.rs.POST;
032import javax.ws.rs.Path;
033import javax.ws.rs.PathParam;
034import javax.ws.rs.Produces;
035import javax.ws.rs.QueryParam;
036import javax.ws.rs.WebApplicationException;
037import javax.ws.rs.core.Context;
038import javax.ws.rs.core.PathSegment;
039import javax.ws.rs.core.Response;
040import javax.ws.rs.core.Response.Status;
041import javax.ws.rs.core.UriInfo;
042
043import org.fcrepo.http.commons.AbstractResource;
044import org.fcrepo.http.commons.session.InjectedSession;
045import org.fcrepo.kernel.FedoraResource;
046import org.slf4j.Logger;
047import org.slf4j.LoggerFactory;
048import org.springframework.beans.factory.annotation.Autowired;
049import org.springframework.context.annotation.Scope;
050import org.springframework.stereotype.Component;
051
052import com.codahale.metrics.annotation.Timed;
053
054/**
055 * RESTful interface to create and manage access roles
056 *
057 * @author Gregory Jansen
058 * @since Sep 5, 2013
059 */
060@Component
061@Scope("prototype")
062@Path("/{path: .*}/fcr:accessroles")
063public class AccessRoles extends AbstractResource {
064
065    private static final Logger LOGGER = LoggerFactory
066            .getLogger(AccessRoles.class);
067
068    @InjectedSession
069    protected Session session;
070
071    @Autowired
072    private AccessRolesProvider accessRolesProvider = null;
073
074    @Context
075    protected HttpServletRequest request;
076
077
078    /**
079     * @return the accessRolesProvider
080     */
081    private AccessRolesProvider getAccessRolesProvider() {
082        return accessRolesProvider;
083    }
084
085    /**
086     * Retrieve the roles assigned to each principal on this specific path.
087     *
088     * @param pathList
089     * @return JSON representation of assignment map
090     * @throws RepositoryException
091     */
092    @GET
093    @Produces(APPLICATION_JSON)
094    @Timed
095    public Response get(@PathParam("path")
096        final List<PathSegment> pathList, @QueryParam("effective")
097        final String effective) throws RepositoryException {
098        final String path = toPath(pathList);
099        LOGGER.debug("Get access roles for: {}", path);
100        LOGGER.debug("effective: {}", effective);
101        Response.ResponseBuilder response;
102        try {
103            final Node node = nodeService.getObject(session, path).getNode();
104            final Map<String, List<String>> data =
105                    this.getAccessRolesProvider().getRoles(node,
106                            (effective != null));
107            if (data == null) {
108                LOGGER.debug("no content response");
109                response = Response.noContent();
110            } else {
111                response = Response.ok(data);
112            }
113        } finally {
114            session.logout();
115        }
116        return response.build();
117    }
118
119    /**
120     * Apply new role assignments at the specified node.
121     *
122     * @param pathList
123     * @param data
124     * @return response
125     * @throws RepositoryException
126     */
127    @POST
128    @Consumes(APPLICATION_JSON)
129    @Timed
130    public Response post(@PathParam("path")
131        final List<PathSegment> pathList, final Map<String, Set<String>> data)
132        throws RepositoryException {
133        final String path = toPath(pathList);
134        LOGGER.debug("POST Received request param: {}", request);
135        Response.ResponseBuilder response;
136
137        try {
138            validatePOST(data);
139
140            final FedoraResource resource =
141                    nodeService.getObject(session, path);
142            this.getAccessRolesProvider().postRoles(resource.getNode(), data);
143            session.save();
144            LOGGER.debug("Saved access roles {}", data);
145            response =
146                    Response.created(getUriInfo().getBaseUriBuilder()
147                            .path(path).path("fcr:accessroles").build());
148
149        } catch (final IllegalArgumentException e) {
150            throw new WebApplicationException(e, Response.status(Status.BAD_REQUEST).build());
151        } finally {
152            session.logout();
153        }
154
155        return response.build();
156    }
157
158    /**
159     * @param data
160     */
161    private void validatePOST(final Map<String, Set<String>> data) {
162        if (data.isEmpty()) {
163            throw new IllegalArgumentException(
164                    "Posted access roles must include role assignments");
165        }
166        for (final Map.Entry<String, Set<String>> entry : data.entrySet()) {
167            if (entry.getKey() == null || entry.getValue() == null || entry.getValue().isEmpty()) {
168                throw new IllegalArgumentException(
169                        "Assignments must include principal name and one or more roles");
170            }
171            if (entry.getKey().trim().length() == 0) {
172                throw new IllegalArgumentException(
173                        "Principal names cannot be an empty strings or whitespace.");
174            }
175            for (final String r : entry.getValue()) {
176                if (r.trim().length() == 0) {
177                    throw new IllegalArgumentException(
178                            "Role names cannot be an empty strings or whitespace.");
179                }
180            }
181        }
182    }
183
184    /**
185     * Delete the access roles and node type.
186     */
187    @DELETE
188    @Timed
189    public Response deleteNodeType(@PathParam("path")
190        final List<PathSegment> pathList) throws RepositoryException {
191        final String path = toPath(pathList);
192        try {
193            final Node node = nodeService.getObject(session, path).getNode();
194            this.getAccessRolesProvider().deleteRoles(node);
195            session.save();
196            return Response.noContent().build();
197        } finally {
198            session.logout();
199        }
200    }
201
202    private UriInfo getUriInfo() {
203        return this.uriInfo;
204    }
205
206}
