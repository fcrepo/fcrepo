345     private static Predicate<Property> hasJcrNamespace =
346         new Predicate<Property>() {
347 
348             @Override
349             public boolean apply(final Property p) {
350                 return !p.isAnon() && p.getNameSpace().equals(JCR_NAMESPACE);
351 
352             }
353         };
354 
355     private static Predicate<Property> hasFedoraNamespace =
356         new Predicate<Property>() {
357 
358             @Override
359             public boolean apply(final Property p) {
360                 return !p.isAnon()
361                         && p.getNameSpace().startsWith(REPOSITORY_NAMESPACE);
362 
363             }
364         };

366     /**
367      * Detects whether an RDF property is managed by the repository.
368      */
369     @SuppressWarnings("unchecked")
370     public static final Predicate<Property> isManagedPredicate = or(
371             in(managedProperties), hasJcrNamespace, hasFedoraNamespace);

373     /**

375     **/
376     public static Predicate<String> isManagedPredicateURI =
377         new Predicate<String>() {
378             @Override
379             public boolean apply(final String uri) {
