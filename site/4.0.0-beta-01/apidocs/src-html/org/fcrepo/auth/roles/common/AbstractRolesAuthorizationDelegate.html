016package org.fcrepo.auth.roles.common;
017
018import java.security.Principal;
019import java.util.Arrays;
020import java.util.HashSet;
021import java.util.List;
022import java.util.Map;
023import java.util.Set;
024
025import javax.jcr.Node;
026import javax.jcr.NodeIterator;
027import javax.jcr.PathNotFoundException;
028import javax.jcr.RepositoryException;
029import javax.jcr.Session;
030
031import org.fcrepo.auth.common.FedoraAuthorizationDelegate;
032import org.fcrepo.http.commons.session.SessionFactory;
033import org.fcrepo.kernel.exception.RepositoryRuntimeException;
034import org.modeshape.jcr.value.Path;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037import org.springframework.beans.factory.annotation.Autowired;
038
039/**
040 * Policy enforcement point for roles-based authentication
041 * @author Gregory Jansen
042 */
043public abstract class AbstractRolesAuthorizationDelegate implements FedoraAuthorizationDelegate {
044
045    private static final Logger LOGGER = LoggerFactory
046            .getLogger(AbstractRolesAuthorizationDelegate.class);
047
048    protected static final String AUTHZ_DETECTION = "/{" +
049            Constants.JcrName.NS_URI + "}";
050
051    private static final String[] REMOVE_ACTIONS = {"remove"};
052
053    @Autowired
054    private AccessRolesProvider accessRolesProvider = null;
055
056    @Autowired
057    private SessionFactory sessionFactory = null;
058
059    /**
060     * Gather effectives roles
061     *
062     * @param principals effective principals
063     * @return set of effective content roles
064     */
065    public static Set<String>
066    resolveUserRoles(final Map<String, List<String>> acl,
067                    final Set<Principal> principals) {
068        final Set<String> roles = new HashSet<>();
069        for (final Principal p : principals) {
070            final List<String> matchedRoles = acl.get(p.getName());
071            if (matchedRoles != null) {
072                LOGGER.debug("request principal matched role assignment: {}", p.getName());
073                roles.addAll(matchedRoles);
074            }
075        }
076        return roles;
077    }
078
079    @Override
080    public boolean hasPermission(final Session session, final Path absPath,
081            final String[] actions) {
082        final Set<String> roles;
083
084        final Principal userPrincipal = getUserPrincipal(session);
085        if (userPrincipal == null) {
086            return false;
087        }
088
089        final Set<Principal> allPrincipals = getPrincipals(session);
090        if (allPrincipals == null) {
091            return false;
092        }
093
094        try {
095            final Session internalSession = sessionFactory.getInternalSession();
096            final Map<String, List<String>> acl =
097                    accessRolesProvider.findRolesForPath(absPath,
098                            internalSession);
099            roles = resolveUserRoles(acl, allPrincipals);
100            LOGGER.debug("roles for this request: {}", roles);
101        } catch (final RepositoryException e) {
102            throw new RepositoryRuntimeException("Cannot look up node information on " + absPath +
103                    " for permissions check.", e);
104        }
105
106        if (LOGGER.isDebugEnabled()) {
107            LOGGER.debug("{}\t{}\t{}", roles, actions, absPath);
108            if (actions.length > 1) { // have yet to see more than one
109                LOGGER.debug("FOUND MULTIPLE ACTIONS: {}", Arrays
110                        .toString(actions));
111            }
112        }
113
114        if (actions.length == 1 && "remove_child_nodes".equals(actions[0])) {
115            // in roles-based ACLs, the permission to remove children is
116            // conferred by earlier check for "remove_node" on the child node
117            // itself.
118            return true;
119        }
120
121        if (!rolesHavePermission(session, absPath.toString(), actions, roles)) {
122            return false;
123        }
124
125        if (actions.length == 1 && "remove".equals(actions[0])) {
126            // you must be able to delete all the children
127            // TODO make recursive/ACL-query-based check configurable
128            return canRemoveChildrenRecursive(session, absPath.toString(),
129                    allPrincipals, roles);
130        }
131        return true;
132    }
133
134    private static Principal getUserPrincipal(final Session session) {
135        final Object value = session.getAttribute(FEDORA_USER_PRINCIPAL);
136        if (value instanceof Principal) {
137            return (Principal) value;
138        }
139        return null;
140    }
141
142    @SuppressWarnings("unchecked")
143    private static Set<Principal> getPrincipals(final Session session) {
144        final Object value = session.getAttribute(FEDORA_ALL_PRINCIPALS);
145        if (value instanceof Set<?>) {
146            return (Set<Principal>) value;
147        }
148        return null;
149    }
150
151    /**
152     * @param userSession the user session
153     * @param parentPath the parent path
154     * @param allPrincipals all principals
155     * @param parentRoles the roles on the parent
156     * @return true if permitted
157     */
158    private boolean canRemoveChildrenRecursive(final Session userSession,
159                                               final String parentPath,
160                                               final Set<Principal> allPrincipals,
161                                               final Set<String> parentRoles) {
162        try {
163            final Session internalSession = sessionFactory.getInternalSession();
164            LOGGER.debug("Recursive child remove permission checks for: {}",
165                         parentPath);
166            final Node parent = internalSession.getNode(parentPath);
167            if (!parent.hasNodes()) {
168                return true;
169            }
170            final NodeIterator ni = parent.getNodes();
171            while (ni.hasNext()) {
172                final Node n = ni.nextNode();
173                // are there unique roles?
174                final Set<String> roles;
175                Map<String, List<String>> acl = null;
176                try {
177                    acl = accessRolesProvider.getRoles(n, false);
178                } catch (final PathNotFoundException ignored) {
179                    LOGGER.trace("Path not found when removing roles", ignored);
180                }
181                if (acl != null) {
182                    roles = resolveUserRoles(acl, allPrincipals);
183                } else {
184                    roles = parentRoles;
185                }
186                if (rolesHavePermission(userSession, n.getPath(),
187                        REMOVE_ACTIONS,
188                        roles)) {
189
190                    if (!canRemoveChildrenRecursive(userSession, n.getPath(),
191                            allPrincipals, roles)) {
192                        return false;
193                    }
194                } else {
195                    LOGGER.info("Remove permission denied at {} with roles {}", n.getPath(), roles);
196                    return false;
197                }
198            }
199            return true;
200        } catch (final RepositoryException e) {
201            throw new RepositoryRuntimeException(
202                    "Cannot lookup child permission check information for " +
203                            parentPath, e);
204        }
205    }
206
207    /**
208     * Subclasses must override this method to determine permissions based on
209     * supplied roles.
210     * 
211     * @param userSession the user session
212     * @param absPath path to the object
213     * @param actions requested action
214     * @param roles effective roles for this request and content
215     * @return true if role has permission
216     */
217    public abstract boolean rolesHavePermission(final Session userSession, final String absPath,
218            final String[] actions, final Set<String> roles);
219
220}
