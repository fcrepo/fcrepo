016package org.fcrepo.kernel.impl.utils.impl;
017
018import static com.google.common.base.Throwables.propagate;
019import static java.util.concurrent.TimeUnit.MILLISECONDS;
020import static org.fcrepo.kernel.utils.FixityResult.FixityState.BAD_CHECKSUM;
021import static org.fcrepo.kernel.utils.FixityResult.FixityState.BAD_SIZE;
022import static org.fcrepo.kernel.utils.FixityResult.FixityState.MISSING_STORED_FIXITY;
023import static org.fcrepo.kernel.utils.FixityResult.FixityState.SUCCESS;
024
025import com.google.common.collect.ImmutableSet;
026import org.fcrepo.kernel.impl.services.ServiceHelpers;
027import org.fcrepo.kernel.utils.ContentDigest;
028import org.fcrepo.kernel.utils.FixityResult;
029import org.infinispan.distexec.DistributedExecutorService;
030import org.modeshape.jcr.value.BinaryKey;
031import org.modeshape.jcr.value.binary.infinispan.InfinispanBinaryStore;
032import org.slf4j.Logger;
033
034import javax.jcr.Property;
035import javax.jcr.RepositoryException;
036import java.net.URI;
037import java.util.Collection;
038import java.util.Iterator;
039import java.util.List;
040import java.util.concurrent.ExecutionException;
041import java.util.concurrent.Future;
042import java.util.concurrent.TimeoutException;
043
044import static org.slf4j.LoggerFactory.getLogger;
045
046/**
047 * @author cabeer
048 */
049public class InfinispanCacheStoreEntry extends LocalBinaryStoreEntry {
050    private static final Logger LOGGER = getLogger(InfinispanCacheStoreEntry.class);
051    private static final String DATA_SUFFIX = "-data";
052
053    /**
054     *
055     * @param store
056     * @param property
057     */
058    public InfinispanCacheStoreEntry(final InfinispanBinaryStore store, final Property property) {
059        super(store, property);
060    }
061
062    @Override
063    public Collection<FixityResult> checkFixity(final URI checksum, final long size) throws RepositoryException {
064        final BinaryKey key = binaryKey();
065        final ImmutableSet.Builder<FixityResult> fixityResults = new ImmutableSet.Builder<>();
066
067        if (store().hasBinary(key)) {
068            final String dataKey = dataKeyFor(key);
069
070            final DistributedFixityCheck task = new DistributedFixityCheck(dataKey);
071            final List<Future<Collection<FixityResult>>> futures
072                = clusterExecutor().submitEverywhere(task, dataKey + "-0");
073
074            while (!futures.isEmpty()) {
075                final Iterator<Future<Collection<FixityResult>>> iterator = futures.iterator();
076                while (iterator.hasNext()) {
077                    final Future<Collection<FixityResult>> future = iterator.next();
078                    try {
079                        final Collection<FixityResult> result = future.get(100, MILLISECONDS);
080                        iterator.remove();
081                        for (final FixityResult fixityResult : result) {
082                            setFixityStatus(fixityResult, size, checksum);
083                        }
084
085                        fixityResults.addAll(result);
086                    } catch (final TimeoutException e) {
087                        LOGGER.trace("Going to retry cluster transform after timeout", e);
088                    } catch (InterruptedException | ExecutionException e) {
089                        throw propagate(e);
090                    }
091                }
092            }
093        }
094        return fixityResults.build();
095    }
096
097    private String dataKeyFor(final BinaryKey key) {
098        return key + DATA_SUFFIX;
099    }
100
101    private DistributedExecutorService clusterExecutor() {
102        return ServiceHelpers.getClusterExecutor((InfinispanBinaryStore)store());
103    }
104    private void setFixityStatus(final FixityResult result, final long dsSize, final URI dsChecksum) {
105        if (dsChecksum.equals(ContentDigest.missingChecksum()) || dsSize == -1L) {
106            result.getStatus().add(MISSING_STORED_FIXITY);
107        }
108
109        if (!result.matches(dsChecksum)) {
110            result.getStatus().add(BAD_CHECKSUM);
111        }
112
113        if (!result.matches(dsSize)) {
114            result.getStatus().add(BAD_SIZE);
115        }
116
117        if (result.matches(dsSize, dsChecksum)) {
118            result.getStatus().add(SUCCESS);
119        }
120    }
121}
