016package org.fcrepo.integration;
017
018import static org.fcrepo.kernel.RdfLexicon.REPOSITORY_NAMESPACE;
019
020import com.hp.hpl.jena.query.QueryExecution;
021import com.hp.hpl.jena.query.ResultSet;
022import org.fcrepo.kernel.FedoraObject;
023import org.fcrepo.kernel.impl.rdf.impl.DefaultIdentifierTranslator;
024import org.fcrepo.kernel.services.ObjectService;
025import org.fcrepo.transform.transformations.SparqlQueryTransform;
026import org.junit.Before;
027import org.junit.Test;
028import org.junit.runner.RunWith;
029import org.springframework.test.context.ContextConfiguration;
030import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
031
032import javax.inject.Inject;
033import javax.jcr.Repository;
034import javax.jcr.RepositoryException;
035import javax.jcr.Session;
036import java.io.ByteArrayInputStream;
037import java.io.InputStream;
038
039import static org.junit.Assert.assertTrue;
040
041
042/**
043 * <p>SparqlQueryTransformIT class.</p>
044 *
045 * @author cbeer
046 */
047@RunWith(SpringJUnit4ClassRunner.class)
048@ContextConfiguration({"/spring-test/master.xml"})
049public class SparqlQueryTransformIT {
050
051    @Inject
052    Repository repo;
053
054    @Inject
055    ObjectService objectService;
056    private SparqlQueryTransform testObj;
057
058    @Before
059    public void setUp() {
060
061    }
062
063    @Test
064    public void shouldDoStuff() throws RepositoryException {
065        final Session session = repo.login();
066
067        final FedoraObject object = objectService.createObject(session, "/testObject");
068
069        final String s = "SELECT ?x ?uuid\n" +
070                "WHERE { ?x  <" + REPOSITORY_NAMESPACE + "uuid> ?uuid }";
071        final InputStream stringReader = new ByteArrayInputStream(s.getBytes());
072
073        testObj = new SparqlQueryTransform(stringReader);
074
075        final QueryExecution qexec = testObj.apply(object.getPropertiesDataset(new DefaultIdentifierTranslator()));
076
077        assert(qexec != null);
078
079        try {
080            final ResultSet results = qexec.execSelect();
081
082            assert(results != null);
083            assertTrue(results.hasNext());
084        } finally {
085            qexec.close();
086        }
087    }
088}
