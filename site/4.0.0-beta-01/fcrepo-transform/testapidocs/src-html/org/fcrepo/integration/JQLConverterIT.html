258    @Test
259    public void testComplexQuery() throws RepositoryException {
260
261        final String sparql = "PREFIX  ns:  <http://libraries.ucsd.edu/ark:/20775/>"
262                + " SELECT DISTINCT ?subject ?object WHERE  {" +
263                "?subject ns:bb2765355h 'bf2765355h' . ?subject ns:bb3652744n ?object . FILTER regex(" +
264                "?object, \"r\", \"i\") .FILTER (?object >= 'abc' && ?object < 'efg' || !(?object = 'efg')) } " +
265                " ORDER BY DESC(?subject) ?object LIMIT 10 OFFSET 20";
266
267        final JQLConverter testObj = new JQLConverter(session, subjects, sparql);
268
269        final String statement = testObj.getStatement();
270
271        final String namespacePrefix = session.getNamespacePrefix("http://libraries.ucsd.edu/ark:/20775/");
272
273        final String expectedQuery =
274                "SELECT DISTINCT [fedoraResource_subject].[jcr:path] AS subject, " +
275                        "[fedoraResource_subject].[ns001:bb3652744n] AS object FROM [fedora:resource] AS " +
276                        "[fedoraResource_subject] WHERE ((([fedoraResource_subject].[ns001:bb2765355h] = " +
277                        "'bf2765355h' AND [fedoraResource_subject].[ns001:bb3652744n] IS NOT NULL) AND " +
278                        "[fedoraResource_subject].[ns001:bb3652744n] LIKE 'r') AND " +
279                        "(([fedoraResource_subject].[ns001:bb3652744n] >= 'abc' AND " +
280                        "[fedoraResource_subject].[ns001:bb3652744n] < 'efg') OR NOT " +
281                        "([fedoraResource_subject].[ns001:bb3652744n] = 'efg'))) ORDER BY " +
282                        "[fedoraResource_subject].[jcr:path] DESC, " +
283                        "[fedoraResource_subject].[ns001:bb3652744n] ASC LIMIT 10 OFFSET 20";
284
285        assertEquals(expectedQuery.replaceAll("ns001", namespacePrefix), statement);
286    }

288    @Test
289    public void testConstantSubjectQuery() throws RepositoryException {
290        final String path = "/foo";
291        final String selector = "fedoraResource_" + path.replace("/", "_");
292        final String baseUri = subjects.getBaseUri();
293        final String subjectUri = (baseUri.endsWith("/") ? baseUri.substring(0, baseUri.length() - 1) : baseUri) + path;
294        final String sparql = "PREFIX fcrepo: <http://fedora.info/definitions/v4/repository#> "
295                + "select ?date where { <" + subjectUri + "> fcrepo:created ?date }";
296        final String expectedQuery =
297                "SELECT [" + selector + "].[jcr:created] AS date " +
298                        "FROM [fedora:resource] AS [" + selector + "] " +
299                        "WHERE ([" + selector + "].[jcr:path] = '" + path + "' AND " +
300                        "[" + selector + "].[jcr:created] IS NOT NULL)";
301        final JQLConverter testObj  = new JQLConverter(session, subjects, sparql);
302
303        assertEquals(expectedQuery, testObj.getStatement());
304    }
