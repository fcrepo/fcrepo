016package org.fcrepo.kernel.services;
017
018import org.fcrepo.kernel.Lock;
019
020import javax.jcr.RepositoryException;
021import javax.jcr.Session;
022
023/**
024 * @author Mike Durbin
025 */
026public interface LockService extends Service {
027
028    /**
029     * Acquires a lock on the node at the given path.
030     * @param session the session wishing to lock the node/subgraph
031     * @param path the path to be affected
032     * @param deep if true, indicates that the entire subgraph starting
033     *             at the given path is to be locked
034     * @return a Lock object that contains a lock token that will allow
035     *         other sessions to access resources locked by this call.
036     */
037    Lock acquireLock(Session session, String path, boolean deep) throws RepositoryException;
038
039    /**
040     * Gets the lock at the given path.  The returned lock will only
041     * contain identifying information for sessions holding the lock.
042     */
043    Lock getLock(Session session, String path) throws RepositoryException;
044
045    /**
046     * Releases the lock on the node at the given path.  This may only be
047     * successful if the session holding the lock.
048     */
049    void releaseLock(Session session, String path) throws RepositoryException;
050}
