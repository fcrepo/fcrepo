030import com.google.common.base.Function;

040/**
041 * A stream of RDF triples along with some useful context.
042 *
043 * @author ajs6f
044 * @since Oct 9, 2013
045 */
046public class RdfStream extends ForwardingIterator<Triple> implements
047        Iterable<Triple> {
048
049    private Map<String, String> namespaces = new HashMap<>();
050
051    protected Iterator<Triple> triples;
052
053    protected Session context;
054
055    protected Node topic;
056
057    private static final Triple[] NONE = new Triple[] {};
058
059    /**
060     * Constructor that begins the stream with proffered triples.
061     *
062     * @param triples
063     */
064    public <Tr extends Triple, T extends Iterator<Tr>> RdfStream(final T triples) {
065        super();
066        this.triples = Iterators.transform(triples, cast());
067    }
068
069    /**
070     * Constructor that begins the stream with proffered triples.
071     *
072     * @param triples
073     */
074    public <Tr extends Triple, T extends Iterable<Tr>> RdfStream(final T triples) {
075        this(triples.iterator());
076    }
077
078    /**
079     * Constructor that begins the stream with proffered triples.
080     *
081     * @param triples
082     */
083    public <Tr extends Triple, T extends Collection<Tr>> RdfStream(
084            final T triples) {
085        this(triples.iterator());
086    }
087
088    /**
089     * Constructor that begins the stream with proffered triples.
090     *
091     * @param triples
092     */
093    @SafeVarargs
094    public <T extends Triple> RdfStream(final T... triples) {
095        this(Iterators.forArray(triples));
096    }
097
098    /**
099     * Constructor that begins the stream with proffered statements.
100     *
101     * @param statements
102     */
103    @SafeVarargs
104    public <T extends Statement> RdfStream(final T... statements) {
105        this(Iterators.transform(Iterators.forArray(statements),
106                statement2triple));
107    }
108
109    /**
110     * Constructor that begins the stream with proffered triple.
111     *
112     * @param triple
113     */
114    public <T extends Triple> RdfStream(final T triple) {
115        this(Iterators.forArray(new Triple[] {triple}));
116    }
117
118    /**
119     * Constructor that begins the stream without any triples.
120     */
121    public RdfStream() {
122        this(NONE);
123    }
124
125    /**
126     * Returns the proffered {@link Triple}s with the context of this RdfStream.
127     *
128     * @param stream
129     * @return proffered Triples with the context of this RDFStream
130     */
131    public <Tr extends Triple, T extends Iterator<Tr>> RdfStream withThisContext(final T stream) {
132        return new RdfStream(stream).namespaces(namespaces()).topic(topic());
133    }
134
135    /**
136     * Returns the proffered {@link Triple}s with the context of this RdfStream.
137     *
138     * @param stream
139     * @return proffered Triples with the context of this RDFStream
140     */
141    public <Tr extends Triple, T extends Iterable<Tr>> RdfStream withThisContext(final T stream) {
142        return new RdfStream(stream).namespaces(namespaces()).topic(topic());
143    }
144
145    /**
146     * @param newTriples Triples to add.
147     * @return This object for continued use.
148     */
149    public RdfStream concat(final Iterator<? extends Triple> newTriples) {
150        triples = Iterators.concat(newTriples, triples);
151        return this;
152    }
153
154    /**
155     * @param newTriple Triples to add.
156     * @return This object for continued use.
157     */
158    public <T extends Triple> RdfStream concat(final T newTriple) {
159        triples = Iterators.concat(singletonIterator(newTriple), triples);
160        return this;
161    }
162
163    /**
164     * @param newTriples Triples to add.
165     * @return This object for continued use.
166     */
167    public <T extends Triple> RdfStream concat(@SuppressWarnings("unchecked")
168        final T... newTriples) {
169        triples = Iterators.concat(Iterators.forArray(newTriples), triples);
170        return this;
171    }
172
173    /**
174     * @param newTriples Triples to add.
175     * @return This object for continued use.
176     */
177    public RdfStream concat(final Collection<? extends Triple> newTriples) {
178        triples = Iterators.concat(newTriples.iterator(), triples);
179        return this;
180    }
181
182    /**
183     * As {@link Iterables#limit(Iterable, int)} while maintaining context.
184     *
185     * @param limit
186     * @return RDFStream
187     */
188    public RdfStream limit(final Integer limit) {
189        if (limit < 0) {
190            return this;
191        }
192        return withThisContext(Iterables.limit(this, limit));
193    }
194
195    /**
196     * As {@link Iterables#skip(Iterable, int)} while maintaining context.
197     *
198     * @param skipNum
199     * @return RDFStream
200     */
201    public RdfStream skip(final Integer skipNum) {
202        if (skipNum < 0) {
203            return this;
204        }
205        return withThisContext(Iterables.skip(this, skipNum));
206    }
207
208    /**
209     * As {@link Iterables#filter(Iterable, Predicate)} while maintaining context.
210     *
211     * @param predicate
212     * @return RdfStream
213     */
214    public RdfStream filter(final Predicate<? super Triple> predicate) {
215        return withThisContext(Iterables.filter(this, predicate));
216    }
217
218    /**
219     * As {@link Iterators#transform(Iterator, Function)}.
220     *
221     * @param f
222     * @return Iterator
223     */
224    public <ToType> Iterator<ToType> transform(final Function<? super Triple, ToType> f) {
225        return Iterators.transform(this, f);
226    }
227
228    /**RdfStream
229     * @param prefix
230     * @param uri
231     * @return This object for continued use.
232     */
233    public RdfStream namespace(final String prefix, final String uri) {
234        namespaces.put(prefix, uri);
235        return this;
236    }
237
238    /**
239     * @param nses
240     * @return This object for continued use.
241     */
242    public RdfStream namespaces(final Map<String, String> nses) {
243        namespaces.putAll(nses);
244        return this;
245    }
246
247    /**
248     * @return The {@link Session} in context
249     */
250    public Session session() {
251        return this.context;
252    }
253
254    /**
255     * Sets the JCR context of this stream
256     *
257     * @param session The {@link Session} in context
258     */
259    public RdfStream session(final Session session) {
260        this.context = session;
261        return this;
262    }
263
264    /**
265     * @return The {@link Node} topic in context
266     */
267    public Node topic() {
268        return this.topic;
269    }
270
271    /**
272     * Sets the topic of this stream
273     *
274     * @param topic The {@link Node} topic in context
275     */
276    public RdfStream topic(final Node topic) {
277        this.topic = topic;
278        return this;
279    }
280
281    /**
282     * WARNING!
283     *
284     * This method exhausts the RdfStream on which it is called!
285     *
286     * @return A {@link Model} containing the prefix mappings and triples in
287     *         this stream of RDF
288     */
289    public Model asModel() {
290        final Model model = createDefaultModel();
291        model.setNsPrefixes(namespaces());
292        for (final Triple t : this) {
293            model.add(model.asStatement(t));
294        }
295        return model;
296    }
297
298    /**
299     * @param model A {@link Model} containing the prefix mappings and triples to be put into
300     *         this stream of RDF
301     * @return RDFStream
302     */
303    public static RdfStream fromModel(final Model model) {
304        final Iterator<Triple> triples = Iterators.transform(model.listStatements(), statement2triple);
305        return new RdfStream(triples).namespaces(model.getNsPrefixMap());
306    }
307
308    private static Function<Statement, Triple> statement2triple = new Function<Statement, Triple>() {
309
310        @Override
311        public Triple apply(final Statement s) {
312            return s.asTriple();
313        }
314
315    };
316
317    @Override
318    protected Iterator<Triple> delegate() {
319        return triples;
320    }
321
322    @Override
323    public Iterator<Triple> iterator() {
324        return this;
325    }
326
327    /**
328     * @return Namespaces in scope for this stream.
329     */
330    public Map<String, String> namespaces() {
331        return namespaces;
332    }
333
334
335    private static <T extends Triple> Function<T, Triple> cast() {
336        return new Function<T, Triple>() {
337
338            @Override
339            public Triple apply(final T prototriple) {
340                return prototriple;
341            }
342
343        };
344    }
345
346    /*
347     * We ignore duplicated triples for equality.
348     *
349     * (non-Javadoc)
350     * @see java.lang.Object#equals(java.lang.Object)
351     */
352    @Override
353    public boolean equals(final Object o) {
354        if (o == null) {
355            return false;
356        }
357        if (o == this) {
358            return true;
359        }
360        if (!(o instanceof RdfStream)) {
361            return false;
362        }
363        final RdfStream rdfo = (RdfStream) o;
364
365        final boolean triplesEqual =
366            equal(copyOf(rdfo.triples), copyOf(this.triples));
367
368        final boolean namespaceMappingsEqual =
369            equal(rdfo.namespaces(), this.namespaces());
370
371        final boolean topicEqual =
372                equal(rdfo.topic(), this.topic());
373
374        return triplesEqual && namespaceMappingsEqual && topicEqual;
375
376    }
377
378    @Override
379    public int hashCode() {
380        return hash(namespaces(), triples, topic());
381    }
