71      @Test
72      public void testModifyMultipleDatastreams() throws Exception {
73          final String pid = randomUUID().toString();
74  
75          createObject(pid);
76          createDatastream(pid, "ds_void", "marbles for everyone");
77  
78          final HttpPost post =
79              new HttpPost(serverAddress
80                      + pid + "/fcr:batch");
81          final MultipartEntityBuilder multiPartEntityBuilder = MultipartEntityBuilder.create();
82          multiPartEntityBuilder.addTextBody(".", "INSERT { <> <http://purl.org/dc/elements/1.1/title> 'xyz' } WHERE { }",
83                                             ContentType.parse(contentTypeSPARQLUpdate));
84          multiPartEntityBuilder.addTextBody("obj1","<>  <http://purl.org/dc/elements/1.1/title> 'obj1-title' ",
85                                             ContentType.parse(contentTypeTurtle));
86          multiPartEntityBuilder.addTextBody("ds1","asdfg", TEXT_PLAIN);
87          multiPartEntityBuilder.addTextBody("ds2", "qwerty", TEXT_PLAIN);
88          multiPartEntityBuilder.addTextBody("delete[]", "ds_void");
89          post.setEntity(multiPartEntityBuilder.build());
90  
91          final HttpResponse postResponse = client.execute(post);
92  
93          assertEquals(201, postResponse.getStatusLine().getStatusCode());
94  
95          final HttpGet getDSesMethod =
96              new HttpGet(serverAddress + pid);
97          final GraphStore result = getGraphStore(getDSesMethod);
98  
99          final String subjectURI = serverAddress + pid;
100 
101         assertTrue("Didn't find the title! ", result.contains(ANY,
102                                                               createURI(subjectURI),
103                                                               createURI("http://purl.org/dc/elements/1.1/title"),
104                                                               createLiteral("xyz")));
105         assertTrue("Didn't find the object title! ", result.contains(ANY,
106                                                                      createURI(subjectURI + "/obj1"),
107                                                                      createURI("http://purl.org/dc/elements/1.1/title"),
108                                                                      createLiteral("obj1-title")));
109         assertTrue("Didn't find the first datastream! ", result.contains(ANY,
110                 createURI(subjectURI), ANY, createURI(subjectURI + "/ds1")));
111         assertTrue("Didn't find the second datastream! ", result.contains(ANY,
112                 createURI(subjectURI), ANY, createURI(subjectURI + "/ds2")));
113         assertFalse("Found the deleted datastream! ", result.contains(Node.ANY,
114                 createURI(subjectURI), ANY, createURI(subjectURI + "/ds_void")));
115 
116     }
