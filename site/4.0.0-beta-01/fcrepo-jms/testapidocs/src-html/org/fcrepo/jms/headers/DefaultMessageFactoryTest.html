016package org.fcrepo.jms.headers;
017
018import static java.util.Collections.singleton;
019import static javax.jcr.observation.Event.NODE_ADDED;
020import static org.fcrepo.jms.headers.DefaultMessageFactory.BASE_URL_HEADER_NAME;
021import static org.fcrepo.jms.headers.DefaultMessageFactory.EVENT_TYPE_HEADER_NAME;
022import static org.fcrepo.jms.headers.DefaultMessageFactory.IDENTIFIER_HEADER_NAME;
023import static org.fcrepo.jms.headers.DefaultMessageFactory.PROPERTIES_HEADER_NAME;
024import static org.fcrepo.jms.headers.DefaultMessageFactory.TIMESTAMP_HEADER_NAME;
025import static org.fcrepo.kernel.RdfLexicon.REPOSITORY_NAMESPACE;
026import static org.junit.Assert.assertEquals;
027import static org.mockito.Mockito.when;
028import static org.mockito.MockitoAnnotations.initMocks;
029
030import java.io.IOException;
031import java.util.Set;
032
033import javax.jcr.RepositoryException;
034import javax.jms.JMSException;
035import javax.jms.Message;
036import javax.jms.Session;
037
038import org.apache.activemq.command.ActiveMQObjectMessage;
039import org.fcrepo.kernel.observer.FedoraEvent;
040import org.fcrepo.kernel.utils.EventType;
041import org.junit.Before;
042import org.junit.Test;
043import org.mockito.Mock;
044
045/**
046 * <p>DefaultMessageFactoryTest class.</p>
047 *
048 * @author ajs6f
049 */
050public class DefaultMessageFactoryTest {
051
052    @Mock
053    private Session mockSession;
054
055    @Mock
056    private FedoraEvent mockEvent;
057
058    private DefaultMessageFactory testDefaultMessageFactory;
059
060    @Before
061    public void setUp() throws JMSException {
062        initMocks(this);
063        when(mockSession.createMessage()).thenReturn(
064                new ActiveMQObjectMessage());
065        testDefaultMessageFactory = new DefaultMessageFactory("base-url");
066    }
067
068    @Test
069    public void testBuildMessage() throws RepositoryException, IOException,
070                                  JMSException {
071        final Long testDate = 46647758568747L;
072        when(mockEvent.getDate()).thenReturn(testDate);
073        final String testPath = "super/calli/fragi/listic";
074        when(mockEvent.getPath()).thenReturn(testPath);
075        final Set<Integer> testTypes = singleton(NODE_ADDED);
076        final String testReturnType =
077            REPOSITORY_NAMESPACE + EventType.valueOf(NODE_ADDED).toString();
078        when(mockEvent.getTypes()).thenReturn(testTypes);
079        final String prop = "test-property";
080        when(mockEvent.getProperties()).thenReturn(singleton(prop));
081        final Message testMessage =
082            testDefaultMessageFactory.getMessage(mockEvent, mockSession);
083        assertEquals("Got wrong date in message!", testDate, (Long) testMessage
084                .getLongProperty(TIMESTAMP_HEADER_NAME));
085        assertEquals("Got wrong identifier in message!", testPath, testMessage
086                .getStringProperty(IDENTIFIER_HEADER_NAME));
087        assertEquals("Got wrong type in message!", testReturnType, testMessage
088                .getStringProperty(EVENT_TYPE_HEADER_NAME));
089        assertEquals("Got wrong base-url in message", "base-url", testMessage.getStringProperty(BASE_URL_HEADER_NAME));
090        assertEquals("Got wrong property in message", prop, testMessage.getStringProperty(PROPERTIES_HEADER_NAME));
091    }
092
093}
