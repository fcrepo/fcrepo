016package org.fcrepo.kernel.impl;
017
018import org.slf4j.Logger;
019import org.springframework.util.ClassUtils;
020
021import javax.jcr.Session;
022import javax.jcr.lock.LockManager;
023import java.lang.reflect.InvocationHandler;
024import java.lang.reflect.InvocationTargetException;
025import java.lang.reflect.Method;
026
027import static java.lang.reflect.Proxy.newProxyInstance;
028import static org.slf4j.LoggerFactory.getLogger;
029
030/**
031 * Wraps a session to ensure that all lock tokens are removed from the session
032 * when logout() is invoked.
033 *
034 * The JCR spec isn't clear about what should happen when a session ends that
035 * is the owner of open-scoped lock tokens.  The current ModeShape implementation
036 * does not release those token upon logout, leaving them unable to be used
037 * by other sessions.
038 *
039 * TODO: Remove this class When modeshape is updated to handle this. https://www.pivotaltracker.com/story/show/69574878
040 *
041 * @author Mike Durbin
042 */
043
044public class LockReleasingSession implements InvocationHandler {
045
046    private static final Logger LOGGER = getLogger(LockReleasingSession.class);
047
048    private final Session session;
049
050    /**
051     * Constructor that wraps a session.
052     */
053    public LockReleasingSession(final Session session) {
054        this.session = session;
055    }
056
057    /**
058     * Wrap a JCR session with this dynamic proxy
059     */
060    public static Session newInstance(final Session session) {
061        LOGGER.trace("Wrapping session {} ({}).", session, session.getClass().getName());
062        return (Session) newProxyInstance(session.getClass().getClassLoader(),
063                ClassUtils.getAllInterfaces(session),
064                new LockReleasingSession(session));
065    }
066
067    /**
068     * Gets the wrapped session.
069     */
070    public Session getWrappedSession() {
071        return this.session;
072    }
073
074    @Override
075    public Object invoke(final Object proxy, final Method method, final Object[] args) throws Throwable {
076        final String name = method.getName();
077
078        if (name.equals("logout")) {
079            LOGGER.trace("Intercepted session.logout()");
080            if (session.isLive()) {  // this may be false if logout is called more than once on a session
081                final LockManager lockManager = session.getWorkspace().getLockManager();
082                for (String token : lockManager.getLockTokens()) {
083                    LOGGER.trace("Removing lock token {}.", token);
084                    lockManager.removeLockToken(token);
085                }
086            }
087        }
088        return delegate(method, args);
089    }
090
091    private Object delegate(final Method method, final Object[] args) throws Throwable {
092        final Object invocationResult;
093        try {
094            invocationResult = method.invoke(session, args);
095        } catch (final InvocationTargetException e) {
096            throw e.getCause();
097        }
098        return invocationResult;
099    }
100}
