047/**
048 * Service for managing access to Fedora 'nodes' (either datastreams or objects,
049 * we don't care.)
050 *
051 * @author Chris Beer
052 * @since May 9, 2013
053 */
054@Component
055public class NodeServiceImpl extends AbstractService implements NodeService {
056
057    private static final Logger LOGGER = getLogger(NodeServiceImpl.class);
058
059    /**
060     * Find or create a new Fedora resource at the given path
061     *
062     * @param session a JCR session
063     * @param path a JCR path
064     * @return FedoraResource
065     * @throws RepositoryException
066     */
067    @Override
068    public FedoraResource findOrCreateObject(final Session session, final String path) throws RepositoryException {
069        return new FedoraResourceImpl(findOrCreateNode(session, path));
070    }
071
072    /**
073     * Retrieve an existing Fedora resource at the given path
074     *
075     * @param session a JCR session
076     * @param path a JCR path
077     * @return Fedora resource at the given path
078     * @throws RepositoryException
079     */
080    @Override
081    public FedoraResource getObject(final Session session, final String path) throws RepositoryException {
082        return new FedoraResourceImpl(session.getNode(path));
083    }
084
085    /**
086     * Get an existing Fedora resource at the given path with the given version
087     * label
088     *
089     * @param session a JCR session
090     * @param path a JCR path
091     * @param versionId a JCR version label
092     * @return Fedora resource at the given path with the given version label
093     * @throws RepositoryException
094     */
095    @Override
096    public FedoraResource getObject(final Session session, final String path, final String versionId)
097        throws RepositoryException {
098        final VersionHistory versionHistory = getVersionHistory(session, path);
099
100        if (versionHistory == null) {
101            return null;
102        }
103
104        if (!versionHistory.hasVersionLabel(versionId)) {
105            return null;
106        }
107
108        final Version version = versionHistory.getVersionByLabel(versionId);
109        return new FedoraResourceImpl(version.getFrozenNode());
110    }
111
112    /**
113     * @return A Set of object names (identifiers)
114     * @throws RepositoryException
115     */
116    @Override
117    public Set<String> getObjectNames(final Session session, final String path) throws RepositoryException {
118        return getObjectNames(session, path, null);
119    }
120
121    /**
122     * Get the list of child nodes at the given path filtered by the given mixin
123     *
124     * @param session
125     * @param path
126     * @param mixin
127     * @return the list of child nodes at the given path filtered by the given mixin
128     * @throws RepositoryException
129     */
130    @Override
131    public Set<String> getObjectNames(final Session session, final String path, final String mixin)
132        throws RepositoryException {
133
134        final Node objects = session.getNode(path);
135        final ImmutableSet.Builder<String> b = builder();
136        final NodeIterator i = objects.getNodes();
137
138        while (i.hasNext()) {
139            final Node n = i.nextNode();
140            LOGGER.trace("Child of type {} is named {} at {}", n.getPrimaryNodeType(), n.getName(), n.getPath());
141            if (mixin == null || n.isNodeType(mixin)) {
142                b.add(n.getName());
143            }
144        }
145
146        return b.build();
147    }
148
149    /**
150     * Delete an existing object from the repository at the given path
151     *
152     * @param session
153     * @param path
154     * @throws RepositoryException
155     */
156    @Override
157    public void deleteObject(final Session session, final String path) throws RepositoryException {
158        final Node obj = session.getNode(path);
159        final PropertyIterator inboundProperties = new PropertyIterator(obj.getReferences());
160
161        for (final Property inboundProperty : inboundProperties) {
162            inboundProperty.remove();
163        }
164
165        obj.remove();
166    }
167
168    /**
169     * Copy an existing object from the source path to the destination path
170     *
171     * @param session
172     * @param source
173     * @param destination
174     * @throws RepositoryException
175     */
176    @Override
177    public void copyObject(final Session session, final String source, final String destination)
178        throws RepositoryException {
179        session.getWorkspace().copy(source, destination);
180    }
181
182    /**
183     * Move an existing object from the source path to the destination path
184     *
185     * @param session
186     * @param source
187     * @param destination
188     * @throws RepositoryException
189     */
190    @Override
191    public void moveObject(final Session session, final String source, final String destination)
192        throws RepositoryException {
193        session.getWorkspace().move(source, destination);
194    }
195
196    /**
197     * Get the full list of node types in the repository
198     *
199     * @param session
200     * @return full list of node types in the repository
201     * @throws RepositoryException
202     */
203    @Override
204    public NodeTypeIterator getAllNodeTypes(final Session session) throws RepositoryException {
205        final NodeTypeManager ntmanager = (NodeTypeManager) session.getWorkspace().getNodeTypeManager();
206        return ntmanager.getAllNodeTypes();
207    }
208
209    /**
210     * @param session
211     * @return node types
212     * @throws RepositoryException
213     */
214    @Override
215    public RdfStream getNodeTypes(final Session session) throws RepositoryException {
216        return new NodeTypeRdfContext(session.getWorkspace().getNodeTypeManager());
217    }
218
219    /**
220     * @param session
221     * @param cndStream
222     * @throws RepositoryException
223     * @throws IOException
224     */
225    @Override
226    public void registerNodeTypes(final Session session, final InputStream cndStream) throws RepositoryException,
227        IOException {
228        final NodeTypeManager nodeTypeManager = (NodeTypeManager) session.getWorkspace().getNodeTypeManager();
229        nodeTypeManager.registerNodeTypes(cndStream, true);
230    }
231}
