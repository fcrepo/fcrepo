016package org.fcrepo.kernel.impl.utils.impl;
017
018import com.google.common.collect.ImmutableSet;
019import org.apache.commons.io.IOUtils;
020import org.fcrepo.kernel.utils.ContentDigest;
021import org.fcrepo.kernel.impl.utils.FixityInputStream;
022import org.fcrepo.kernel.utils.FixityResult;
023import org.fcrepo.kernel.impl.utils.FixityResultImpl;
024import org.fcrepo.kernel.impl.utils.infinispan.StoreChunkInputStream;
025import org.infinispan.Cache;
026import org.infinispan.CacheImpl;
027import org.infinispan.distexec.DistributedCallable;
028import org.infinispan.loaders.CacheLoaderManager;
029import org.infinispan.loaders.CacheStore;
030import org.infinispan.loaders.decorators.ChainingCacheStore;
031
032import java.io.InputStream;
033import java.io.Serializable;
034import java.net.URI;
035import java.security.MessageDigest;
036import java.util.Collection;
037import java.util.Set;
038
039import static org.apache.commons.io.output.NullOutputStream.NULL_OUTPUT_STREAM;
040
041/**
042 * Infinispan DistributedCallable for checking the fixity of a binary key in every cache loader
043 *
044 * @author cabeer
045 */
046public class DistributedFixityCheck implements DistributedCallable<String, byte[], Collection<FixityResult>>,
047                                                   Serializable {
048    private final String dataKey;
049    private Cache<String, byte[]> cache;
050
051    /**
052     *
053     * @param dataKey
054     */
055    public DistributedFixityCheck(final String dataKey) {
056        this.dataKey = dataKey;
057    }
058
059    @Override
060    public Collection<FixityResult> call() throws Exception {
061        final ImmutableSet.Builder<FixityResult> fixityResults = new ImmutableSet.Builder<>();
062
063        for (final CacheStore store : stores()) {
064
065            final String digest = ContentDigest.getAlgorithm(new URI("urn:sha1"));
066
067            FixityInputStream fixityInputStream;
068            final InputStream cacheLoaderChunkInputStream = new StoreChunkInputStream(store, dataKey);
069
070            fixityInputStream = new FixityInputStream(cacheLoaderChunkInputStream, MessageDigest.getInstance(digest));
071            IOUtils.copy(fixityInputStream, NULL_OUTPUT_STREAM);
072
073            final URI calculatedChecksum = ContentDigest.asURI(digest, fixityInputStream.getMessageDigest().digest());
074            fixityResults.add(
075                new FixityResultImpl(getExternalIdentifier(store), fixityInputStream.getByteCount(), calculatedChecksum)
076            );
077        }
078
079        return fixityResults.build();
080    }
081
082    private String getExternalIdentifier(final CacheStore store) {
083        final String address;
084
085        if (cache.getCacheManager().getAddress() != null) {
086            address = cache.getCacheManager().getAddress().toString();
087        } else {
088            address = "localhost";
089        }
090
091        return "infinispan-cache-loader:" + address + "/" + store.toString() + "#" +  dataKey;
092    }
093
094    @Override
095    public void setEnvironment(final Cache<String, byte[]> cache, final Set<String> inputKeys) {
096        this.cache = cache;
097    }
098
099    private Set<CacheStore> stores() {
100        final CacheLoaderManager cacheLoaderManager
101            = ((CacheImpl) cache).getComponentRegistry().getLocalComponent(CacheLoaderManager.class);
102        if (cacheLoaderManager.getCacheLoader() instanceof ChainingCacheStore) {
103            return ((ChainingCacheStore)cacheLoaderManager.getCacheLoader()).getStores().keySet();
104        } else {
105            return ImmutableSet.of(cacheLoaderManager.getCacheStore());
106        }
107    }
108}
