054/**
055 * This class extends the {@link FileSystemConnector} to enable the autocreation of Fedora-specific datastream and
056 * content properties.
057 *
058 * @author Andrew Woods
059 *         Date: 1/30/14
060 */
061public class FedoraFileSystemConnector extends FileSystemConnector {
062
063    private static final Logger LOGGER = LoggerFactory.getLogger(FedoraFileSystemConnector.class);
064
065    private static final String DELIMITER = "/";
066    private static final String JCR_CONTENT = "jcr:content";
067    private static final String JCR_CONTENT_SUFFIX = DELIMITER + JCR_CONTENT;
068
069    /**
070     * The string path for a {@link File} object that represents the top-level directory in which properties are
071     * stored.  This is optional for this connector, but if set allows properties to be cached (greatly
072     * improving performance) for even read-only connectors.  When this property is specified the extraPropertiesStore
073     * should be null (not specified) as it would be overridden by this.
074     */
075    private String propertiesDirectoryPath;
076    private File propertiesDirectory;
077
078    @Override
079    public void initialize(final NamespaceRegistry registry,
080                           final NodeTypeManager nodeTypeManager) throws RepositoryException, IOException {
081        super.initialize(registry, nodeTypeManager);
082
083        if (!isReadonly()) {
084            throw new RepositoryException("The " + getClass().getName() + " must have \"readonly\" set to true!");
085        }
086
087        if (propertiesDirectoryPath != null) {
088           propertiesDirectory = new File(propertiesDirectoryPath);
089            if (!propertiesDirectory.exists() || !propertiesDirectory.isDirectory()) {
090                throw new RepositoryException("Configured \"propertiesDirectory\", " + propertiesDirectoryPath
091                        + ", does not exist or is not a directory.");
092            }
093            if (extraPropertiesStore() != null) {
094                LOGGER.warn("Extra properties store was specified but won't be used!");
095            }
096            setExtraPropertiesStore(new ExternalJsonSidecarExtraPropertyStore(this, translator(), propertiesDirectory));
097        }
098    }
099
100    /**
101     * This method returns the object/document for the node with the federated arg 'id'.
102     * <p/>
103     * Additionally, this method adds Fedora datastream and content properties to the result of the parent class
104     * implementation.
105     */
106    @Override
107    public Document getDocumentById(final String id) {
108        LOGGER.debug("Getting Federated document: {}", id);
109        if (null == id || id.isEmpty()) {
110            LOGGER.warn("Can not get document with null id");
111            return null;
112        }
113
114        final Document doc = super.getDocumentById(id);
115
116        final DocumentReader docReader = readDocument(doc);
117        final DocumentWriter docWriter = writeDocument(doc);
118
119        final String primaryType = docReader.getPrimaryTypeName();
120
121        if (!docReader.getMixinTypeNames().contains(FEDORA_RESOURCE)) {
122            LOGGER.trace("Adding mixin: {}, to {}", FEDORA_RESOURCE, id);
123            docWriter.addMixinType(FEDORA_RESOURCE);
124        }
125
126        // Is Fedora Datastream?
127        if (primaryType.equals(NT_FILE)) {
128            decorateDatastreamNode(docReader, docWriter);
129
130        // Is Fedora Content?
131        } else if (primaryType.equals(NT_RESOURCE)) {
132            decorateContentNode(docReader, docWriter, fileFor(id));
133        }
134
135        // Persist new properties (if allowed)
136        if (shouldCacheProperties()) {
137            saveProperties(docReader);
138        }
139
140        return docWriter.document();
141    }
142
143    /**
144     * Checks whether internally managed properties can and should be stored to
145     * an ExtraPropertiesStore.
146     */
147    protected boolean shouldCacheProperties() {
148        return extraPropertiesStore() != null && (!isReadonly() || this.propertiesDirectory != null);
149    }
150
151    @Override
152    public String sha1(final File file) {
153        final String cachedSha1 = getCachedSha1(file);
154        if (cachedSha1 == null) {
155            return computeAndCacheSha1(file);
156        } else {
157            return cachedSha1;
158        }
159    }
160
161
162    private String getCachedSha1(final File file) {
163        final String id = idFor(file) + JCR_CONTENT_SUFFIX;
164        if (extraPropertiesStore() != null) {
165            final Map<Name, Property> extraProperties = extraPropertiesStore().getProperties(id);
166            final Name digestName = nameFrom(CONTENT_DIGEST);
167            if (extraProperties.containsKey(digestName)) {
168                if (!hasBeenModifiedSincePropertiesWereStored(file, extraProperties.get(nameFrom(JCR_CREATED)))) {
169                    LOGGER.trace("Found sha1 for {} in extra properties store.", id);
170                    final String uriStr = ((URI) extraProperties.get(digestName).getFirstValue()).toString();
171                    return uriStr.substring(uriStr.indexOf("sha1:") + 5);
172                }
173            }
174        } else {
175            LOGGER.trace("No cache configured to contain object hashes.");
176        }
177        return null;
178    }
179
180    private String computeAndCacheSha1(final File file) {
181        final String id = idFor(file) + JCR_CONTENT_SUFFIX;
182        LOGGER.trace("Computing sha1 for {}.", id);
183        final String sha1 = super.sha1(file);
184        if (shouldCacheProperties()) {
185            final Map<Name, Property> updateMap = new HashMap<Name, Property>();
186            final Property digestProperty = new BasicSingleValueProperty(nameFrom(CONTENT_DIGEST),
187                    asURI("SHA-1", sha1));
188            final Property digestDateProperty = new BasicSingleValueProperty(nameFrom(JCR_CREATED),
189                    factories().getDateFactory().create(file.lastModified()));
190            updateMap.put(digestProperty.getName(), digestProperty);
191            updateMap.put(digestDateProperty.getName(), digestDateProperty);
192            extraPropertiesStore().updateProperties(id, updateMap);
193        }
194        return sha1;
195    }

199    private static void decorateDatastreamNode(final DocumentReader docReader, final DocumentWriter docWriter) {
200        if (!docReader.getMixinTypeNames().contains(FEDORA_DATASTREAM)) {
201            LOGGER.trace("Adding mixin: {}, to {}", FEDORA_DATASTREAM, docReader.getDocumentId());
202            docWriter.addMixinType(FEDORA_DATASTREAM);
203        }
204    }
205
206    private static void decorateContentNode(final DocumentReader docReader,
207                                            final DocumentWriter docWriter,
208                                            final File file) {
209        if (!docReader.getMixinTypeNames().contains(FEDORA_BINARY)) {
210            LOGGER.trace("Adding mixin: {}, to {}", FEDORA_BINARY, docReader.getDocumentId());
211            docWriter.addMixinType(FEDORA_BINARY);
212        }
213
214        if (null == docReader.getProperty(CONTENT_DIGEST)
215                || hasBeenModifiedSincePropertiesWereStored(file, docReader.getProperty(JCR_CREATED))) {
216            final BinaryValue binaryValue = getBinaryValue(docReader);
217            final String dsChecksum = binaryValue.getHexHash();
218            final String dsURI = asURI("SHA-1", dsChecksum).toString();
219
220            LOGGER.trace("Adding {} property of {} to {}", CONTENT_DIGEST, dsURI, docReader.getDocumentId());
221            docWriter.addProperty(CONTENT_DIGEST, dsURI);
222        }
223
224        if (null == docReader.getProperty(CONTENT_SIZE)) {
225            final long binarySize = file.length();
226            LOGGER.trace("Adding {} property of {} to {}", CONTENT_SIZE, binarySize, docReader.getDocumentId());
227            docWriter.addProperty(CONTENT_SIZE, binarySize);
228        }
229
230        LOGGER.debug("Decorated data property at path: {}", docReader.getDocumentId());
231    }
232
233    private static boolean hasBeenModifiedSincePropertiesWereStored(final File file, final Property lastModified) {
234        if (lastModified == null) {
235            LOGGER.trace("Hash for {} has not been computed yet.", file.getName());
236            return true;
237        } else {
238            final DateTime datetime = (DateTime) lastModified.getFirstValue();
239            if (datetime.toDate().equals(new Date(file.lastModified()))) {
240                return false;
241            } else {
242                LOGGER.trace("{} has been modified ({}) since hash was last computed ({}).", file.getName(),
243                        new Date(file.lastModified()), datetime.toDate());
244                return true;
245            }
246        }
247    }
248
249    private static BinaryValue getBinaryValue(final DocumentReader docReader) {
250        final Property binaryProperty = docReader.getProperty(JCR_DATA);
251        return (BinaryValue) binaryProperty.getFirstValue();
252    }
253
254    private void saveProperties(final DocumentReader docReader) {
255        LOGGER.trace("Persisting properties for {}", docReader.getDocumentId());
256        final Map<Name, Property> properties = docReader.getProperties();
257        final ExtraProperties extraProperties = extraPropertiesFor(docReader.getDocumentId(), true);
258        extraProperties.addAll(properties).except(JCR_PRIMARY_TYPE, JCR_DATA, CONTENT_SIZE);
259        extraProperties.save();
260    }
