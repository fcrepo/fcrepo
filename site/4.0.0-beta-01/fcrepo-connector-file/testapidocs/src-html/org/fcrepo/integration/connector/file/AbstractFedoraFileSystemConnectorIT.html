066 * An abstract suite of tests that should work against any configuration

122    /**
123     * Sets a system property and ensures artifacts from previous tests are
124     * cleaned up.
125     */
126    @BeforeClass
127    public static void setSystemPropertiesAndCleanUp() {
128
129        // Instead of creating dummy files over which to federate,
130        // we configure the FedoraFileSystemFederation instances to
131        // point to paths within the "target" directory.
132        final File testDir1 = new File("target/test-classes/config/testing");
133        cleanUpJsonFilesFiles(testDir1);
134        setProperty(PROP_TEST_DIR1, testDir1.getAbsolutePath());
135
136        final File testDir2 = new File("target/test-classes/spring-test");
137        cleanUpJsonFilesFiles(testDir2);
138        setProperty(PROP_TEST_DIR2, testDir2.getAbsolutePath());
139
140        final File testPropertiesDir = new File("target/test-classes-properties");
141        if (testPropertiesDir.exists()) {
142            cleanUpJsonFilesFiles(testPropertiesDir);
143        } else {
144            testPropertiesDir.mkdir();
145        }
146        setProperty(PROP_EXT_TEST_DIR, testPropertiesDir.getAbsolutePath());
147    }
148
149    @AfterClass
150    public static void unsetSystemPropertiesAndCleanUp() {
151        clearProperty(PROP_TEST_DIR1);
152        clearProperty(PROP_TEST_DIR2);
153        clearProperty(PROP_EXT_TEST_DIR);
154    }
155
156    protected static void cleanUpJsonFilesFiles(final File directory) {
157        final WildcardFileFilter filter = new WildcardFileFilter("*.modeshape.json");
158        final Collection<File> files = FileUtils.listFiles(directory, filter, TrueFileFilter.INSTANCE);
159        final Iterator<File> iterator = files.iterator();
160
161        // Clean up files persisted in previous runs
162        while (iterator.hasNext()) {
163            if (!iterator.next().delete()) {
164                fail("Unable to delete work files from a previous test run");
165            }
166        }
167    }

169    @Test
170    public void testGetFederatedObject() throws RepositoryException {
171        final Session session = repo.login();
172
173        final FedoraObject object = objectService.getObject(session, testFilePath());
174        assertNotNull(object);
175
176        final Node node = object.getNode();
177        final NodeType[] mixins = node.getMixinNodeTypes();
178        assertEquals(2, mixins.length);
179
180        boolean found = false;
181        for (final NodeType nodeType : mixins) {
182            if (nodeType.getName().equals(FEDORA_DATASTREAM)) {
183                found = true;
184            }
185        }
186        assertTrue("Mixin not found: " + FEDORA_DATASTREAM, found);
187
188        session.save();
189        session.logout();
190    }

192    @Test
193    public void testGetFederatedContent() throws RepositoryException {
194        final Session session = repo.login();
195
196        final Node node = datastreamService.getDatastreamNode(session, testFilePath() + "/jcr:content");
197        assertNotNull(node);
198
199        final NodeType[] mixins = node.getMixinNodeTypes();
200        assertEquals(2, mixins.length);
201
202        boolean found = false;
203        for (final NodeType nodeType : mixins) {
204            if (nodeType.getName().equals(FEDORA_BINARY)) {
205                found = true;
206            }
207        }
208        assertTrue("Mixin not found: " + FEDORA_BINARY, found);
209
210        final Property size = node.getProperty(CONTENT_SIZE);
211
212        final File file = fileForNode(node);
213        assertTrue(file.getAbsolutePath(), file.exists());
214        assertEquals(file.length(), size.getLong());
215
216        session.save();
217        session.logout();
218    }

230    @Test
231    public void testChangedFileFixity() throws RepositoryException, IOException, NoSuchAlgorithmException {
232        final Session session = repo.login();
233
234        final Node node = datastreamService.getDatastreamNode(session, testFilePath() + "/jcr:content");
235
236        final String originalFixity = checkFixity(node);
237
238        final File file = fileForNode(node);
239        appendToFile(file, " ");
240
241        final String newFixity = checkFixity(node);
242
243        assertNotEquals("Checksum is expected to have changed!", originalFixity, newFixity);
244
245        session.save();
246        session.logout();
247    }

249    private void appendToFile(final File f, final String data) throws IOException {
250        final FileOutputStream fos = new FileOutputStream(f, true);
251        try {
252            fos.write(data.getBytes("UTF-8"));
253        } finally {
254            fos.close();
255        }
256    }
257
258    private String checkFixity(final Node node) throws IOException, NoSuchAlgorithmException, RepositoryException {
259        assertNotNull(node);
260
261        final File file = fileForNode(node);
262        final byte[] hash = getHash(SHA_1, file);
263
264        final URI calculatedChecksum = asURI(SHA_1.toString(), hash);
265
266        final Collection<FixityResult> results = datastreamService.getFixity(node, calculatedChecksum, file.length());
267        assertNotNull(results);
268
269        assertFalse("Found no results!", results.isEmpty());
270
271        final Iterator<FixityResult> resultIterator = results.iterator();
272        while (resultIterator.hasNext()) {
273            final FixityResult result = resultIterator.next();
274            assertTrue(result.isSuccess());
275        }
276        return calculatedChecksum.toString();
277    }

279    protected File fileForNode(final Node node) {
280        return new File(getFederationRoot(), testFilePath().replace(federationName(), ""));
